אמבה אלגי
אמבה אלגי הוּא אמב (מִבְצָר הֲרָרִי הַמִּסְתַּמֵּךְ עַל תְּצוּרָה גאולוגית שְׁכִיחָה בְּאֶתְיוֹפְּיָה, בְּצוּרַת הַר שׁוּלְחָן הַמּוּקָּף במצוקים תְּלוּלִים), בְּצָפוּן אֶתְיוֹפְּיָה, בַּמָּחוֹז תיגראי. האמב שׁוֹלֵט עַל הַדֶּרֶךְ בֵּין הָעִיר מְקִלָּה דָּרוֹמָה לָעִיר מאיצ'ו. בַּשֵּׁל מיקומו הָאַסְטְרָטֶגִי הָיָה האמב זִירָה לְמִסְפָּר קְרָבוֹת חֲשׁוּבִים בַּהִיסְטוֹרְיָה האתיופית.
הִיסְטוֹרְיָה
אזכורים רִאשׁוֹנִים
הַפַּעַם הָרִאשׁוֹנָה בָּהּ מוֹפִיעַ שְׁמוֹ שֶׁל הָהָר בַּכְּתוּבִים הָיְיתָה בַּמַּהֲלָךְ הַמֵּאָה ה-15 כַּאֲשֶׁר קֵיסָר אֶתְיוֹפְּיָה באדה מריאם הָרִאשׁוֹן הוֹבִיל מַסָּע צְבָאִי כְּנֶגֶד שֵׁבֶט בַּשֵּׁם דובאה שֶׁשָּׁלַט בָּאֵזוֹר וְהִפְרִיעַ לִנְתִיבֵי הַתַּחְבּוּרָה. בִּתְחִילַּת הַמֵּאָה ה-19 הִשְׁתַּמֵּשׁ הראס וולדה סלאסי בַּמְּצוּדָה שֶׁעַל הָהָר עַל מְנַת לִכְלוֹא אֶת גָּבְרָה גּוּרוּ, אֶחָיו שֶׁל סבגדיס שֶׁמָּרַד כְּנֶגְדּוֹ. הַדֶּרֶךְ שֶׁסָּלְלוּ הַבְּרִיטִים בְּמַהֲלַךְ הַמִּשְׁלַחַת הַצְּבָאִית לְאֶתְיוֹפְּיָה (1868), מֵהַחוֹף אֶל אמבה מריאם, עָבְרָה לְיַד אמבה אלגי. בְּפֶבְּרוּאָר 1889 לָחַם ראס אלולה אנגדה בראס סִיּוּם גָּבְרָה, וּבִקְרָב זֶה נִפְצַע אנגדה.
קְּרַב אמבה אלגי (1895)
עֵרֶךְ מוּרְחָב - קְּרַב אמבה אלגי (1895)
הַקְּרָב הִתְרַחֵשׁ בְּמַהֲלַךְ הַמִּלְחָמָה הָאִיטַלְקִית-אתיופית הָרִאשׁוֹנָה בְּדֵצֶמְבֶּר 1895. ב-7 בְּדֵצֶמְבֶּר הֵבִיסוּ האתיופים כּוֹחַ אִיטַלְקִי שֶׁל 2,350 חַיָּילִים שֶׁהוּבַל עַל יְדֵי פייטרו טוסלי, וב-8 בְּדֵצֶמְבֶּר, גירש הַקֵּיסָר האתיופי מנליק הַשֵּׁנִי אֶת כּוֹחוֹתָיו שֶׁל הַגֵּנֵרָל הָאִיטַלְקִי אורסטה ברטיירי. לְאַחֵר הַתְּבוּסָה נָסוֹגוּ הָאִיטַלְקִים מאמבה אלגי וּמְגוּנְדָּר כּוּלָּהּ, לְמִבְצָר מַקְלָהּ שֶׁעָלָיו צַר מנליק בְּמֶשֶׁךְ חֲמִישָּׁה עָשָׂר יָמִים.
קְּרַב אמבה אלגי (1941)
הַקְּרָב הַשֵּׁנִי שֶׁל אמבה אלגי נֵעָרֵךְ בַּמִּסְגֶּרֶת הַמַּעֲרָכָה בַּמִּזְרָח אַפְרִיקָה בְּמִלְחֶמֶת הָעוֹלָם הַשְּׁנִיָּיה. מִתְּחִילַּת שְׁנַת 1941 פֶּתַח הַצָּבָא הַבְּרִיטִי בְּאֶמְצָעוּת יְחִידוֹת מִצָּבָא הוֹדוּ הַבְּרִיטִית וְעַמָּם כּוֹחוֹת מִצְּבָא דָּרוֹם אַפְרִיקָה, צָבָא צָרְפַת הֶחָופְשִׁית וִיחִידוֹת נוֹסָפוֹת שֶׁל לוחמי גרילה אתיופים הַמְּכוּנִּים כּוֹחַ גִּדְעוֹן בַּמַּעֲרָכָה כְּנֶגֶד הָאִיטַלְקִים שֶׁהֶחֱזִיקוּ בְּאֶתְיוֹפְּיָה, סומליה ואריתריאה. בִּתְנוּעַת מֶלְקָחַיִים איגפו כּוֹחוֹתָיו שֶׁל הַגֵּנֵרָל ויליאם פלאט אֶת הָאִיטַלְקִים מַצְפּוּן, וְכוֹחוֹתָיו הַדָּרוֹם אפריקאים שֶׁל הַגֵּנֵרָל אלן קנינגהם אֶת הָאִיטַלְקִים מִדָּרוֹם. הָאִיטַלְקִים נָסוֹגוּ וְהִתְבַּצְּרוּ בַּמִּבְצָר אמבה אלגי, שָׂם חצבו מְצוּדָה בְּמִנְהָרוֹת בָּאֶבֶן הַטִּבְעִית. תַּחַת פִּיקּוּדוֹ שֶׁל הַמִּשְׁנָה לַמֶּלֶךְ הָאִיטַלְקִי, הַנָּסִיךְ אמדאו הַדּוּכָּס הַשְּׁלִישִׁי מאאוסטה נִיהֲלוּ הָאִיטַלְקִים בְּמָקוֹם קָרֵב מגננה. בִּתְחִילַּת מַאי הֵחֵלּוּ הַבְּרִיטִים לִתְקוֹף מַצְפּוּן, וְהִקִּיפוּ אֶת הָהָר מִצִּידּוֹ הַמִּזְרָחִי וְהַמַּעֲרָבִי. ב-14 בְּמַאי הִצְטָרְפוּ אֲלֵיהֶם הַכּוֹחוֹת הַדָּרוֹם אפריקאים מִדָּרוֹם, וְהָהָר הוּקַּף מִכָּל עבריו. הַבְּרִיטִים תִּכְנְנוּ לִתְקוֹף אֶת הַמְּצוּדָה בְּאוֹפֶן חֲזִיתִי, אַךְ פָּגָז שֶׁפָּגַע פְּגִיעָה ממוזלת בְּמִצְבּוֹר דֶּלֶק, הֵבִיא לְזִיהוּם מֵי הַשְּׁתִיָּיה בְּדֶלֶק, וְלִכְנִיעַת מִגַּנֵּי הָהָר ב-15 בְּמַאי.
קָרֵב אמבה אלגי הַשְּׁלִישִׁי
ב-1943 הִתְקַיֵּים לְיַד אמבה אלגי הַקְּרָב הָאַחֲרוֹן בַּ"מֶּרֶד וויאנה" שֶׁכֻּוַּון כְּנֶגֶד שִׁלְטוֹנוֹ שֶׁל היילה סלאסי. הַמּוֹרְדִים כָּבְשׁוּ אֶת מַקְלָהּ וְהִמְשִׁיכוּ דָּרוֹמָה אֶל הָהָר, בּוֹ נִתְקְלוּ בַּעֲשֶׂרֶת אֲלָפִים חַיָּילִים שֶׁל מֶמְשֶׁלֶת אֶתְיוֹפְּיָה. בֵּין 18 בְּסֶפְּטֶמְבֶּר ל-6 בְּאוֹקְטוֹבֶּר 1943 נִיסּוּ הַמּוֹרְדִים, בְּהַנְהָגַת בלאטה היילה מריאם רָדָה לִכְבּוֹשׁ אֶת עֲמָדוֹת חַיָּילִי הַמֶּמְשָׁלָה בָּהָר, אַךְ לֹא עָמְדוּ בִּפְנֵי הַפְגָּזוֹת הארטילריה שֶׁל הַצָּבָא הַבְּרִיטִי שֶׁסִּיֵּיעַ לסלאסי, וּבְסוֹפוֹ שֶׁל דָּבָר הִתְפַּזְּרוּ.
