סוֹף הַחַיִּים
כָּאן תִּמְצְאוּ מֵידָע לְגַבֵּי הֶבֵּטִים שׁוֹנִים שֶׁל אָדָם בְּסוֹף חַיָּיו
בַּחֲבֵרָה הַמַּעֲרָבִית, שִׂיחַ סְבִיב הַמָּוֶת, הַכּוֹלֵל אֶת הָאֹפֶן בּוֹ אָדָם מְבַקֵּשׁ לְסַיֵּם אֶת חַיָּיו וּלְהִפָּרֵד מֵהַקְּרוֹבִים לוֹ, אֵינֶנּוּ דָּבָר שֶׁבַּשִּׁגְרָה. עִם זֹאת, נִתָּן לְהַבְחִין בְּשִׁנּוּי בְּיַחַס הַחֲבֵרָה לַמָּוֶת וּבִנְכוֹנוּת לָדוּן בּוֹ, בְּעִקָּר בְּכָל הַנּוֹגֵעַ לִזְכוּת הַבְּחִירָה שֶׁל הַחוֹלִים לָמוּת בְּכָבוֹד וּלְהַעֲבִיר אֶת סוֹף יְמֵיהֶם בְּאֹפֶן בּוֹ הֵם מְבַקְּשִׁים.
בְּפּוֹרְטָל זֶה, רִכַּזְנוּ אֶת הַהֶבֵּטִים הַשּׁוֹנִים שֶׁל זְכֻיּוֹת סְבִיב סוֹף הַחַיִּים - הָחֵל מֵאֹפֶן קַבָּלַת הַהַחְלָטוֹת לְגַבֵּי הַטִּפּוּל הָרְפוּאִי, דֶּרֶךְ הַקְּבוּרָה, עַד לִתְרוּמַת אֵיבָרִים וְעוֹד
מַדְרִיכִים וּמֵידָע חָשׁוּב:
-מַדְרִיךְ הֵעָרְכוּת לְסוֹף הַחַיִּים
-מַדְרִיךְ לִבְנֵי מִשְׁפָּחָה הַמִּתְמוֹדְדִים עִם מָוֶת (פְּטִירָה) וּקְבוּרָה
-קְבוּרַת הַמֵּת (הַנִּפְטָר)
-זכותון לַתְּקוּפָה שֶׁלְּאַחַר מוֹת (פְּטִירַת) בֶּן מִשְׁפָּחָה
-הַסְדָּרַת חוֹבוֹת אָדָם שֶׁנִּפְטַר
-טִפּוּל בַּכְּסָפִים וּבָרְכוּשׁ שֶׁל בֶּן מִשְׁפָּחָה לְאַחַר מוֹתוֹ
-חוֹבוֹת הַמִּשְׁפָּחָה לְעוֹבֵד זָר בַּסִּעוּד שֶׁמַּעֲסִיקוֹ נִפְטַר
-פִּצּוּיֵי פִּטּוּרִים לְעוֹבֵד שֶׁמַּעֲסִיקוֹ נִפְטַר
-פִּצּוּיֵי פִּטּוּרִים לִשְׁאֵרָיו שֶׁל עוֹבֵד שֶׁנִּפְטַר
-הַחוֹלֶה הַנּוֹטֶה לָמוּת
-יִפּוּי כּוֹחַ מְשֻׁלָּב לְטִפּוּל רְפוּאִי כְּלָלִי וְלַחוֹלֶה הַנּוֹטֶה לָמוּת
-רִשּׁוּם הַנְחָיוֹת רְפוּאִיּוֹת מַקְדִּימוֹת
-יִפּוּי כּוֹחַ לַהַחְלָטוֹת עַל טִפּוּל רְפוּאִי
-חוֹלֶה הַנּוֹטֶה לָמוּת
-הַזְּכוּת לָמוּת בְּכָבוֹד
-הַזְּכוּת לָמוּת בְּכָבוֹד שֶׁל קְטִין
-הַזְּכוּת לָמוּת בְּכָבוֹד שֶׁל חָסוּי
-וַעֲדָה מוֹסָדִית בְּעִנְיַן חוֹלֶה הַנּוֹטֶה לָמוּת
-וַעֲדָה אַרְצִית בְּעִנְיַן חוֹלֶה הַנּוֹטֶה לָמוּת
-טִפּוּל תּוֹמֵךְ
-הוֹסְפִּיס
-טִפּוּל בֵּיתִי תּוֹמֵךְ (טִפּוּל פַּלְיָאטִיבִי)
-טִפּוּל בֵּיתִי תּוֹמֵךְ לְחוֹלֵי סַרְטָן מִטַּעַם הָאֲגֻדָּה לַמִּלְחָמָה בַּסַּרְטָן
-קַבָּלַת קָנַבִּיס רְפוּאִי (מָרִיחוּאָנָה רְפוּאִית, גְּרָאס רְפוּאִי)
-אַמְבּוּלַנְס הַמִּשְׁאָלוֹת לַחוֹלִים בְּמַחֲלָה קָשָׁה אוֹ סוֹפָנִית
-תִּכְנוּן מִשְׁפָּטִי וְכַלְכָּלִי
-צַוָּאוֹת, יְרֻשּׁוֹת וְעִזְבוֹנוֹת
-תְּרוּמַת אֵיבָרִים מִן הַמֵּת (כַּרְטִיס אֶדִי)
-מַתַּן הַנְחָיוֹת מַקְדִּימוֹת לְמִנּוּי אַפּוֹטְרוֹפּוֹס
-הַרְשָׁאָה מֵרֹאשׁ לְבִצּוּעַ פְּעֻלּוֹת עֲתִידִיּוֹת בְּחֶשְׁבּוֹן בַּנְק מְשֻׁתָּף לְאַחַר מוֹתוֹ שֶׁל אֶחָד מִבַּעֲלֵי הַחֶשְׁבּוֹן (טֹפֶס אֲרִיכוּת יָמִים, סְעִיף הִוָּתְרוּת בַּחַיִּים)
-יִפּוּי כּוֹחַ מִתְמַשֵּׁךְ לַמַּצָּבִים שֶׁל אִי-יְכֹלֶת לְקַבֵּל הַחְלָטוֹת בֶּעָתִיד
-הֶחָבוּת הַכַּלְכָּלִית לְעוֹבֵד זָר בַּסִּעוּד בְּמִקְרֵה פְּטִירָה
-מַדְרִיךְ הֵעָרְכוּת לְסוֹף הַחַיִּים
-נְתִיחָה שֶׁלְּאַחַר הַמָּוֶת

-הִתְנַגְּדוּת בְּנֵי מִשְׁפָּחָה לַנְּתִיחָה שֶׁלְּאַחַר הַמָּוֶת
-פְּתִיחַת קֶבֶר לְצֹרֶךְ נְתִיחָה שֶׁלְּאַחַר הַמָּוֶת

-שְׂרֵפַת גּוּפָה

-קְבוּרָה בַּאֲתַר הַשֵּׁרוּתִים וְהַמֵּידָע הַמֶּמְשַׁלְתִּי
-מִמּוּן הַקְּבוּרָה וַהֲקָמַת מַצֵּבָה

-דְּמֵי קְבוּרָה לְתוֹשַׁב יִשְׂרָאֵל שֶׁנִּפְטַר בחו"ל
-תַּשְׁלוּם דְּמֵי קְבוּרָה לְנִפְטָר לֹא יְהוּדִי
-מַעֲנַק מַצֵּבָה לְנִצּוֹלֵי שׁוֹאָה
