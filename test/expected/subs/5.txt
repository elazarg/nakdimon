‫מַשְׁקֶה רִאשׁוֹן עָלַיי.
‫אָמַרְתָּ שֶׁאָנוּ כָּאן בִּגְלַל הַחֲקִירָה.
‫נָכוֹן. בָּאנוּ לְנַקּוֹת אֶת הָרֹאשׁ, לְשַׁחְרֵר קִיטוֹר.
‫זֶה הַמַּפְתֵּחַ לְפִעְנוּחַ כָּל תִּיק.
‫לֹא אֵצֵא אִיתְּךָ לַמּוֹעֲדוֹן.
‫-לָמָּה לֹא?
‫אַתָּה גֶּבֶר בֶּן 60, ‫וְזֶה יוֹם שְׁלִישִׁי.
‫חוֹק רִאשׁוֹן. כָּל אִישָּׁה שֶׁמִּתְלַבֶּשֶׁת יָפֶה ‫וְיוֹצֵאת בִּשְׁלִישִׁי בְּעֶרֶב, זוֹ אִישָּׁה כִּלְבָבִי.
‫אֵיךְ אַתָּה חוֹשֵׁב שֶׁפָּגַשְׁתִּי אֶת אִמְּךָ?
‫-בְּסֵדֶר, דַּי עִם הַחוּקִּים. יוֹדֵעַ מָה? הָלַכְתִּי.
‫סָלְחוּ לִי, רַבּוֹתַיי. זֶה מִשְּׁתֵּי הַבָּנוֹת שָׁם.
‫חוֹק שֵׁנִי.
‫אָסוּר לְסָרֵב לִדְרִינְק בְּחִינָּם מֵאִישָּׁה יָפָה.
‫הֵיי, שָׁאפְט. נִרְאֶה טוֹב.
‫מָתַי נִיתֵּן לְךָ שְׁתַּיִים-בִּמְחִיר-אַחַת?
‫מִיהוּ הֶחָבֵר שֶׁלְּךָ?
‫-בָּנוֹת, זֶה הַבֵּן שֶׁלִּי.
‫ג'וּנְיוֹר.
‫-אֱלוֹהִים. יֵשׁ לְךָ בֵּן?
‫אֲנִי תָּמִיד מִתְבַּלְבֵּל בֵּינֵיכֶן.
‫בֶּייְבִּי.
‫-וְשׁוּגֵר.
‫שׁוּגֵר.
‫-בּוֹא.
‫בְּסֵדֶר, קָדִימָה.
‫אֵלּוּ בְּוַודַּאי לֹא הַשֵּׁמוֹת הָאֲמִיתִּיִּים שֶׁלָּכֶן.
‫-בֶּטַח אֵין שׁוּם דָּבָר קָטָן בַּשָּׁאפְט הַזֶּה.
‫חֲבָל שֶׁלֹּא תִּזְכִּי לְגַלּוֹת, ‫כִּי הוּא בְּדִיּוּק עָמַד לָלֶכֶת.
‫מָה? לֹא. תִּישָּׁאֵר וְתִשְׁתֶּה אִיתָּנוּ קְצָת.
‫אַתָּה יוֹדֵעַ שֶׁאַתָּה רוֹצֶה.
‫אוּלַי עוֹד דְּרִינְק אֶחָד.
‫-טוֹב, מַגְנִיב.
‫קָדִימָה. אָמַרְתִּי לָךְ לָזוּז.
‫זוּזוּ מִכָּאן. תּוֹדָה.
