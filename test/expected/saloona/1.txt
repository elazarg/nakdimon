יֵשׁ לְצַיֵּין כִּי זוֹ לֹא פַּעַם רִאשׁוֹנָה שֶׁמִּשְׂחַק לִכְאוֹרָה תָּמִים מִתְבָּרֵר כִּמְסוּכָּן: רַק לִפְנֵי חוֹדֶשׁ פֶסְטִיבָלֵי מוּזִיקָה בְּאַנְגְּלִיָּה אָסְרוּ עַל שִׁימּוּשׁ בְּנַצְנַצִּים: וְלֹא רַק הֵם, מְקוֹמוֹת רַבִּים בָּעוֹלָם כְּבָר מְבִינִים שֶׁחֲתִיכוֹת הַפְּלַסְטִיק וְהַמַּתֶּכֶת הַקְּטַנּוֹת מְזַהֲמוֹת אֶת הַסְּבִיבָה, אַחֲרֵי כַּמָּה שָׁעוֹת שֶׁל הִשָּׁאֲרוּת עַל הַגּוּף הֵן נִשְׁטָפוֹת אֶל תּוֹךְ מֵי הַבִּיּוּב וְהַיְישֵׁר לָאוֹקְיָינוֹסִים, שֶׁסּוֹבְלִים מִזִּיהוּם גַּם בְּלִי הַתּוֹסֶפֶת הַמְּיוּתֶּרֶת הַזּוֹ, וְנִשְׁאָרוֹת שָׁם אַלְפֵי שָׁנִים עַד לְפֵירוּק מָלֵא.
בַּמַּיִם חֲתִיכוֹת הַפְּלַסְטִיק אוֹסְפוֹת סְבִיבָם כִּימִיקָלִים רְעִילִים שֶׁנִּכְנָסִים אֶל תּוֹךְ הַדָּגִים שֶׁאוֹכְלִים אוֹתָן, וְאָז אֶל בְּנֵי הָאָדָם שֶׁאוֹכְלִים אוֹתָם.
בַּפּוֹסְט שֶׁכְּבָר זָכָה לְמֵעַל 3,000 שִׁיתּוּפִים, יָעֵל מְבַקֶּשֶׁת מִכּוּלָּנוּ לִלְמוֹד מִמָּה שֶׁקָּרָה לָהּ וּלְהַעֲלוֹת אֶת הַמּוּדָעוּת לְסַכָּנַת הַסְּלַיְים.
חִיפּוּשׂ קָצָר בְּגוּגֶל מְסַפֵּק אֶת רְשִׁימַת הָרְכִיבִים שֶׁל הַדָּבִיק-הַדָּבִיק הַזֶּה:
דֶּבֶק פְּלַסְטִי (דֶּבֶק לָבָן- C4H6O2) - מְיוּצָּר מִפּוֹלִימֵר סִינְתֵּטִי צְמִיגִי וּמְשַׁמֵּשׁ לָרוֹב לְהַדְבָּקַת מוּצָרֵיי נְיָיר וְעֵץ, בּוֹרַאקְס (Na2B4O7·10H2O) - תִּרְכּוֹבֶת כִּימִית שֶׁל מֶלַח נַתְרָן וְחוּמְצַת בּוֹר, וּסְלַיְים עָשׂוּי לְהָכִיל לְעִיתִּים גַּם פֶּרְסִיל-גֵ'ל, הַמְּשַׁמֵּשׁ גֵ'ל לִכְבִיסָה, שֶׁכֵּן גַּם הוּא מֵכִיל בּוֹרַאקְס.
מְסוּכָּן בְּיוֹתֵר.
אָנָּא מִכֶּם, פִּתְחוּ אֶת הָעֵינַיִים וְשִׁמְרוּ עַל יַלְדֵיכֶם וְעַל עַצְמְכֶם.
בּוֹאוּ נִמְנַע אֶת הָאָסוֹן הַבָּא.
