הָאֱמֶת הִיא שֶׁכֹּהֵן הַגָּדוֹל נִכְנָס בַּמְּבוּכָה בנוגע לַשְּׁאֵלָה מָה יַעֲשֶׂה כַּאֲשֶׁר ייכנסו לָאוּלָם הַיְּדִידִים הָאֲמִיתִּיִּים שֶׁלּוֹ וְשֶׁל "הָאִישׁ שֶׁבְּקִיר", הֲלֹא הֵם יִוָּסֵף מֶלֶךְ הַפָּלָאפֶל, מֹשֶׁה הַקַּצָּב, אֱלִיהוּ הַסַּנְדְּלָר וּמוּטֶּה צ'לבין חַלְפָן הַדּוֹלָרִים.
הוּא שׁוֹבֵר אֶת רֹאשׁוֹ כֵּיצַד יַכְרִיז עַל בואם.
זֹאת מִשּׁוּם שֶׁאֲפִילּוּ הוּא מֵבִין, כִּי אוֹתוֹ סַם ליבהר עָלוּל לְתָמוּהַּ, אִם בֵּין רוֹאֶה הַחֶשְׁבּוֹן לָרַב הַשְּׁכוּנָה יוּכְרַז עַל בּוֹאוּ שֶׁל "יוֹסֵף בַּעַל דּוּכָן הַפָּלָאפֶל" אוֹ "יוז'י הַקַּבְּצָן".
אוּלָם שָׁעָה לִפְנֵי תְּחִילַּת הַטֶּקֶס, מַנִּיחַ לֵוִי הַקָּטָן בְּחֵיקוֹ דַּף נְיָיר קָטַן וּבוֹ שֵׁמוֹת וְתַפְקִידִים שׁוֹנִים, וּמוֹרֶה לוֹ בְּדִיּוּק אֵיזוֹ דָּרַך לִנְקוֹט בַּפָּרָשָׁה.
לְמַעֲשֶׂה, כַּאֲשֶׁר מַבִּיט כֹּהֵן הַגָּדוֹל בִּרְשִׁימָה, הוּא סָבוּר כִּי לֵוִי הַקָּטָן מַגְזִים עַד למאד בִּנְקוּדָּה זוֹ, אוּלָם מוּבָן שֶׁאֵין הוּא מַשְׁמִיעַ אֶת הַשְׁקָפָתוֹ זוֹ לְלֵוִי הַקָּטָן עָצְמוּ, שָׁכַן זֶה עָלוּל לִרְאוֹת בַּדָּבָר מֵעֵין בִּיקּוֹרֶת עַל תִּפְקוּדוֹ.
אִם כֵּן, לְאַחֵר כְּנִיסָתָם שֶׁל אַנְשֵׁי הַהוֹן הַחֲשׁוּבִים וְהִכְרַזְתָּ שָׁמַם וְתַפְקִידָם הָרָם, מַכְרִיז כֹּהֵן הַגָּדוֹל: "אָנוּ מְקַדְּמִים בִּבְרָכָה אַתְּ בּוֹאוּ שֶׁל סְגָן רֹאשׁ הָעִירִיָּה מַר אֱלִיהוּ בַּר-עַם".
וְאֶל הַחֶדֶר פּוֹסֵעַ מְעַדְּנוֹת יִוָּסֵף אִישׁ הַפָּלָאפֶל בַּמֵּיטָב מחלצותיו.
לְמַעֲשֶׂה, זָקוּק יוֹסֵף זֶה בְּדִיּוּק לְאוֹתוֹ תַּפְקִיד כְּדֵי להנפיק לְעַצְמוֹ אִישּׁוּרִים וְרִשְׁיוֹנוֹת שׁוֹנִים בַּמַּטָּרָה לְהַרְחִיב אֶת דּוּכָן הַפָּלָאפֶל שָׁלֵו, וְהוּא דַּי מְרוּצֶּה מֵהַתַּפְקִיד שֶׁמּוּלְבָּשׁ עָלָיו בְּאוֹתוֹ הָרֶגַע.
