70 אֶלֶף אִישׁ בַּהַפְגָּנָה נֶגֶד הַתָּכְנִית הַכַּלְכָּלִית שֶׁל הַמֶּמְשָׁלָה
עַשְׂרוֹת אַלְפֵי מַפְגִּינִים הִגִּיעוּ מֵרַחֲבֵי הָאָרֶץ לְהַפְגִּין נֶגֶד שַׂר הָאוֹצָר וְהַתָּכְנִית שֶׁלּוֹ * עָמִיר פָּרֶץ, יוֹ"ר הַהִסְתַּדְּרוּת: "לֹא נָתַן לְמִשְׂרַד הָאוֹצָר לִשְׁדּוֹד אוֹתָנוּ". * תְּקִיעוֹת שׁוֹפֵר בַּהַפְגָּנָה נִפְרֶדֶת שֶׁל הַחֲרֵדִים נֶגֶד פְּגִיעָה בְּקִצְבָּאוֹת הַיְּלָדִים * הַאִם עָמִיר פֶּרַץ רוֹצֶה לִהְיוֹת רֹאשׁ הַמֶּמְשָׁלָה? *
בְּיוֹם שְׁלִישִׁי שֶׁעָבַר, נֶעֶרְכָה בִּירוּשָׁלַיִם הַפְגָּנַת עֲנָק נֶגֶד הַתָּכְנִית הַכַּלְכָּלִית שֶׁל הַמֶּמְשָׁלָה. מַפְגִּינִים מִכָּל רַחֲבֵי הָאָרֶץ, הוּבְאוּ בְּאוֹטוֹבּוּסִים עַל יְדֵי הַהִסְתַּדְּרוּת, שֶׁרוֹצֶה "לְהָגֵן עַל הָעוֹבְדִים".
חֵלֶק מֵהַמַּפְגִּינִים נָשְׂאוּ שְׁלָטִים נֶגֶד שַׂר הָאוֹצָר בִּיבֵּי נְתַנְיָהוּ וְנֶגֶד הַשִּׁנוּיִים בְּפֶנְסִיָּה. 
לְמַעֲשֶׂה, בִּירוּשָׁלַיִם נֶעֶרְכוּ שְׁתֵּי הַפְגָנוֹת נִפְרָדוֹת בְּהִשְׁתַּתְּפוּת 70 אֶלֶף אִישׁ. בַּהַפְגָּנָה הַמֶּרְכָּזִית הִשְׁתַּתֵּף יוֹ"ר הַהִסְתַּדְּרוּת, עָמִיר פָּרֶץ, שֶׁאָמַר: "אֲנִי נִשְׁבַּע שֶׁלֹּא נָתַן לְמִשְׂרַד הָאוֹצָר לִשְׁדּוֹד אוֹתָנוּ".
כָּתַב "יַנְשׁוֹף" מֵצַיָּן, שֶׁעָמִיר פֶּרֶץ זָכָה לִקְרִיאוֹת עִידּוּד רַבּוֹת. חֵלֶק מֵהַצִּבּוּר אַף קָרָא לוֹ "רֹאשׁ הַמֶּמְשָׁלָה הַבָּא". פַּרְשָׁנִים חוֹשְׁבִים, שֶׁלַּעָמִיר פָּרַץ יֵשׁ תָּכְנִיּוֹת לְהִתְמוֹדֵד עַל רָאשׁוּת הַמֶּמְשָׁלָה, אִם בְּמִסְגֶּרֶת מִפְלֶגֶת הָעֲבוֹדָה, וְאִם בְּמִסְגֶּרֶת אַחֶרֶת.
בְּהַפְגָּנָה אַחֶרֶת, קְטַנָּה יוֹתֵר, הִשְׁתַּתְּפוּ חֲרֵדִים שֶׁבִּקְשׁוּ לִמְחוֹת נֶגֶד כּוָנַת הַמֶּמְשָׁלָה לִפְגּוֹעַ בְּקִצְבָאוֹת הַיְּלָדִים. הַמַּפְגִּינִים קָרְאוּ פִּרְקֵי תְּהִלִים וְתָקְעוּ בַּשׁוֹפְרוֹת.
