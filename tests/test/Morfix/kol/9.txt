חַיָּילִים, צָבָא וּבִיטְחוֹן
הַפּוֹרְטָל מֶרְכַּז מֵידָע עַל זְכוּיּוֹת חַיָּילֵי צה"ל בְּשֵׁירוּת סָדִיר (חוֹבָה וְקֶבַע) וּמִילּוּאִים, זְכוּיּוֹת הַחַיָּילִים לְאַחַר שִׁחְרוּרָם וּזְכוּיּוֹתֵיהֶם שֶׁל נְכֵי צה"ל.
בְּנוֹסָף, הַפּוֹרְטָל כּוֹלֵל מֵידָע רִאשׁוֹנִי לְגַבֵּי זְכוּיּוֹתֵיהֶם שֶׁל בְּנֵי הַמִּשְׁפָּחוֹת הַשַּׁכּוּלוֹת. אֶת הַמֵּידָע הַמַּקִּיף נִיתָּן לִמְצוֹא בַּאֲתַר מִשְׁפָּחוֹת וְהַנְצָחָה שֶׁל מִשְׂרַד הַבִּיטָּחוֹן.
חַיָּילִים בְּשֵׁירוּת חוֹבָה:
-הַתְאָמוֹת לְבַעֲלִי לְקוּיוֹת לְמִידָה בַּמִּבְחָנִים בְּלִשְׁכוֹת הַגִּיּוּס
-פִּיצּוּיֵי פִּיטּוּרִים לְעוֹבֵד שֶׁהִתְפַּטֵּר עֵקֶב גִּיּוּס לצה"ל
-אִיסּוּר אַפְלָיָה שֶׁל חַיָּילִים בִּמְקוֹמוֹת צִיבּוּרִיִּים
-חַיָּילִים וְחַיָּילוֹת בְּשֵׁירוּת חוֹבָה (צה"ל)
-הִתְנַדְּבוּת לְצָבָא וְאוּכְלוּסִיּוֹת מְיוּחָדוֹת
-חַיָּילִים בּוֹדְדִים
חַיָּילִים בְּשֵׁירוּת קֶבַע:
-בִּיטּוּחַ בְּרִיאוּת לְחַיָּילֵי קֶבַע
-הַגָּשַׁת עֲרָר לְוועֲדַת עֲרָר לַחַיָּילִים בְּשֵׁירוּת קֶבַע
-הִשְׁתַּתְּפוּת צה"ל בְּהוֹצָאוֹת קַיְיטָנָה לְיַלְדֵי חַיָּילִים בְּשֵׁירוּת קֶבַע
-תְּבִיעָה לַהַכָּרָה בַּפְּגִיעָה בָּעֲבוֹדָה לַחַיָּילִים בְּשֵׁירוּת קֶבַע
-תְּבִיעָה לַהַכָּרָה בְּלִיקּוּי רְפוּאִי אוֹ מַחֲלַת מִקְצוֹעַ לַחַיָּילִים בְּשֵׁירוּת קֶבַע
-אִיסּוּר אַפְלָיָה שֶׁל חַיָּילִים בִּמְקוֹמוֹת צִיבּוּרִיִּים
-אִיסּוּר פִּיטּוּרֵי חַיֶּילֶת בַּהֵרָיוֹן מִשֵּׁירוּת קֶבַע
-אִיסּוּר פִּיטּוּרֵי חַיָּיל/ת מִשֵּׁירוּת קֶבַע בְּמַהֲלַךְ חוּפְשַׁת לֵידָה אוֹ לְאַחֲרֶיהָ
-אִיסּוּר פִּיטּוּרֵי חַיָּיל/ת מִשֵּׁירוּת קֶבַע בְּמַהֲלַךְ טִיפּוּלֵי פּוֹרִיּוּת
-פָּטוּר לִמְשָׁרְתֵי קֶבַע מֵהֲלִיכֵי מִכְרָז לַתַּפְקִידִים בְּשֵׁירוּת הַמְּדִינָה
לוֹחֲמִים:
-קַבָּלַת תְּעוּדַת לוֹחֵם
-זְכוּיּוֹת וַהֲטָבוֹת לְלוֹחֲמֵי חוֹד בַּאֲתַר צה"ל
-חַיָּילִים מְשׁוּחְרָרִים וְחַיָּילֵי מִילּוּאִים
-אִיסּוּר אַפְלָיָה שֶׁל חַיָּילִים בִּמְקוֹמוֹת צִיבּוּרִיִּים
-רִישּׁוּם לְקוּפַּת חוֹלִים לְאַחַר הַשִּׁחְרוּר מצה"ל
-דְּמֵי לֵידָה לְחַיָּילוֹת מְשׁוּחְרָרוֹת שֶׁהֵחֵלּוּ לַעֲבוֹד
-זְכוּיּוֹת חַיָּילִים מְשׁוּחְרָרִים
-מִילּוּאִים
-גִּמְלַאי צה"ל
-הַגְדָּלַת קִצְבַּת גִּמְלַאי צה"ל לְנִיצּוֹלֵי שׁוֹאָה
-דְּחִיַּית שֵׁירוּת
-מְכִינָה קְדַם צְבָאִית וּשְׁנַת שֵׁירוּת
נְכֵי צה"ל וְנִפְגָּעִים:
-נְכֵי צה"ל
-תְּבִיעָה לַהַכָּרָה בַּפְּגִיעָה בָּעֲבוֹדָה לַחַיָּילִים בְּשֵׁירוּת קֶבַע
-תְּבִיעָה לַהַכָּרָה בְּלִיקּוּי רְפוּאִי אוֹ מַחֲלַת מִקְצוֹעַ לַחַיָּילִים בְּשֵׁירוּת קֶבַע
-תְּבִיעָה לַהַכָּרָה בַּפְּגִיעָה בָּעֲבוֹדָה לְאַנְשֵׁי כּוֹחוֹת הַבִּיטָּחוֹן
-תְּבִיעָה לַהַכָּרָה בְּלִיקּוּי רְפוּאִי אוֹ מַחֲלַת מִקְצוֹעַ לְאַנְשֵׁי כּוֹחוֹת הַבִּיטָּחוֹן
-חוֹבֶרֶת מֵידָע לִפְצוֹעַ (בְּשָׂפוֹת שׁוֹנוֹת) בַּאֲתַר אֲגַף שִׁיקּוּם נָכִים בְּמִשְׂרַד הַבִּטָּחוֹן
מִשְׁפָּחוֹת שַׁכּוּלוֹת:
-הֲנָחָה בָּאַרְנוֹנָה לְמִשְׁפָּחָה שַׁכּוּלָה
-הֲנָחָה בְּמַס רְכִישָׁה לִבְנֵי מִשְׁפָּחוֹת שַׁכּוּלוֹת
-מִימּוּן אִבְחוּנִי לְקוּיוֹת לְמִידָה לִיתוֹמִים בְּנֵי מִשְׁפָּחוֹת שַׁכּוּלוֹת
-מִימּוּן שִׁיעוּרֵי הוֹרָאָה מְתַקֶּנֶת לִיתוֹמִים בְּנֵי מִשְׁפָּחוֹת שַׁכּוּלוֹת
-מִימּוּן לִימּוּדֵי תְּעוּדָה מִקְצוֹעִית וְתוֹאַר רִאשׁוֹן לְאַחִים שַׁכּוּלִים
-הֵיעָדְרוּת מֵהָעֲבוֹדָה בְּיוֹם הַזִּיכָּרוֹן
-הֵיעָדְרוּת מִתַּעֲסוּקָה לְמִשְׁתַּקֵּם בְּיוֹם הַזִּיכָּרוֹן
-שְׁעוֹת סִיעוּד לְנִיצּוֹלֵי שׁוֹאָה הַזַּכָּאִים לְ"עֶזְרָה לַזּוּלָת" מִמִּשְׂרַד הַבִּיטָּחוֹן
-אִיסּוּר פִּיטּוּרִים בְּגִיל פְּרִישָׁה שֶׁל אַלְמְנוּת צה"ל וְהוֹרִים שַׁכּוּלִים שֶׁיַּלְדָּם נִפְטַר עֵקֶב שֵׁירוּת צְבָאִי
פְּדוּיֵי שֶׁבִי:
-קִצְבָּה לִמְשָׁרְתִים בְּכוֹחוֹת הַבִּיטָּחוֹן שֶׁנָּפְלוּ בַּשֶּׁבִי
-תַּשְׁלוּם לַחֲבֵרִים בְּאִרְגּוּנֵי מַחְתֶּרֶת שֶׁנֶּאֶסְרוּ בִּתְקוּפַת הַמַּנְדָּט הַבְּרִיטִי
-קִצְבָּה לְמִתְנַדְּבֵי הַצָּבָא הַבְּרִיטִי שֶׁנָּפְלוּ בַּשֶּׁבִי הַנָּאצִי
מֶרְכְּזֵי שֵׁירוּת צִיבּוּרִיִּים:
-מוֹקְדִים
