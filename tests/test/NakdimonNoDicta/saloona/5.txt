הַסִּיפּוּר הַזֶּה תָּפַס אוֹתִי לַחֲלוּטִין לֹא מוּכָנָה וּבִנְקוּדַּת זְמַן מְאוֹד רְגִישָׁה.
יֵשׁ לָנוּ שְׁתֵּי בָּנוֹת, וּבְאוֹתָהּ תְּקוּפָה חָשַׁבְנוּ עַל יֶלֶד נוֹסָף וְהִנֵּה פִּתְאוֹם אֲנִי כְּבָר לֹא "אִילָנִי מָגֵן" אֶלָּא: "אִילָנִי מֵגֵן סִיכּוּן גָּבוֹהַּ".
וּמִי בִּכְלָל מוּכָן לְדַבֵּר אִתְּךָ עַל יְלָדִים.
הִתְחִילָה שִׁגְרָה שֶׁל בְּדִיקוֹת וּמַעֲקָב: כָּל 3 חוֹדָשִׁים בְּדִיקוֹת אֵצֶל כִּירוֹרְג, בְּדִיקוֹת MRI, אוּלְטְרְסָאוּנְד, וְעִם כָּל זֶה, תְּקוּפָה אֲרוּכָּה עֲדַיִין חָיִיתִי בְּהַדְחָקָה וּבְהַכְחָשָׁה כְּשֶׁהַכּוֹל מְרַחֵף בָּאֲוִויר, לֹא נוֹגֵעַ בִּי.
עִם הַזְּמַן נֶחְשַׂפְתִּי לְמֵידָע.
שָׁמַעְתִּי שֶׁסַּרְטַן הַשָּׁד נִיתָּן לְרִיפּוּי בְּגִילּוּי מוּקְדָּם, שֶׁיֵּשׁ לוֹ טִיפּוּל.
לְגַבֵּי סַרְטַן הַשְּׁחָלוֹת אֵין גִּילּוּי מוּקְדָּם, וּמֵרֶגַע הַגִּילּוּי יֵשׁ מַמָּשׁ זְמַן קָצָר עַד שֶׁ"מֻזְדְּכִים עַל הַחַיִּים".
הֶחְלַטְתִּי לַעֲבוֹר נִיתּוּחַ כְּרִיתַת שְׁחָלוֹת וַחֲצוֹצְרוֹת, וּבַדַקָּה הַ-90 קִיבַּלְתִּי רַגְלַיִים קָרוֹת וּבָרַחְתִּי..
נְקוּדַּת הַמַּפְנֶה מִבְּחִינָתִי הָיְתָה סַדְנָא בָּהּ הִשְׁתַּתַּפְתִּי בְּהַמְלָצַת פְּרוֹפֵסוֹר פְרִידְמָן, מְנַהֵל הַיְּחִידָה הָאוּנְקוֹגֶנְטִית בְּתֵל הַשּׁוֹמֵר וד"ר שָׁחַר לֵב אֲרִי, מִבֵּית הַחוֹלִים אִיכִילוֹב שֶׁהֵקִימוּ קְבוּצַת תְּמִיכָה לַנַּשָּׂאִיּוֹת.
הַסֶּדְנָא פָּעֲלָה לְפִי עֶקְרוֹנוֹת "הָעֲבוֹדָה" שֶׁל בְּיירוּן קַיְיטִי.
נִפְגַּשְׁנוּ בְּמֶשֶׁךְ שְׁלוֹשָׁה חֳוֹדָשִׁים, בְּכָל יוֹם שִׁישִּׁי לַמִפְגָּשִׁים בְּנֵי חָמֵשׁ שָׁעוֹת.
עֶצֶם הַמִּפְגָּשׁ עִם נְשָׂאִיּוֹת כָּמוֹנִי, שֶׁרוּבָּן כְּבָר עָבְרוּ נִיתּוּחִים מְנִיעָתִיִים בֵּין שֶׁל כְּרִיתַת שְׁחָלוֹת וַחֲצוֹצְרוֹת וּבֵין כְּרִיתַת שָׁדַיִים הָיָה חָשׁוּב וּמַרְגִּיעַ - נוֹכַחְתִּי שֶׁיֵּשׁ חַיִּים אַחֲרֵי הַכְּרִיתָה.
מַטְּרַת הַסְּדָנָה הַמּוּצְהֶרֶת הָיְיתָה לְנַטְרֵל פְּחָדִים וַחֲרָדוֹת, שֶׁהֵם כְּשֶׁלְּעַצְמָם יְכוֹלִים לְזָרֶז הוֹפָעַת הַמַּחֲלָה.
יָצָאתִי מֵהַסְּדָנָה מְחוּזֶּקֶת וְהַפַּעַם נִיגַּשְׁתִּי בְּאוֹמֶץ לְנִיתּוּחַ כְּרִיתַת שְׁחָלוֹת וַחֲצוֹצְרוֹת.
לֹא אוֹמֵר שֶׁזֶּה הָיָה קַל, אֲבָל תּוֹדָה לְאֶל הַיּוֹם זֶה מֵאֲחוֹרַיי.
הַנִּיתּוּחַ מְלַוֶּוה בְּתוֹפָעוֹת לְוַואי שֶׁל גִּיל הַמַּעֲבָר, שֶׁתּוֹפְסוֹת אוֹתְךָ בְּבַת אַחַת - גַּלֵּי חוֹם מְטוֹרָפִים, וְעִמָּן הַפָּן הָרִגְשִׁי: הַפְסָקַת הַמַּחְזוֹר הַחוֹדְשִׁי, וְהַהַשְׁלָמָה עִם כָּךְ שֶׁהִפְסַקְתִּי לִהְיוֹת פּוֹרְיָה בְּגִיל כָּל כָּךְ צָעִיר לֹא הָיְיתָה קַלָּה.
וּבְכָל זֹאת, הַיּוֹם אֲנִי בְּמָקוֹם אַחֵר לְגַמְרֵי.
הֶחְלַטְתִּי שֶׁאֵין טַעַם לְהִתְוַוכֵּחַ עִם הַמְּצִיאוּת.
הִיא פֹּה, וְצָרִיךְ לְהִסְתַּכֵּל לָהּ בָּעֵינַיִים.
