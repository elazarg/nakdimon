הַמְּעַרְעֶרֶת הוֹסִיפָה וְטַעֲנָה כִּי הַמֵּשִׁיב הֵפֵר אֶת חוֹבַת תּוֹם הַלֵּב בְּנִיהוּל מַשָּׂא וּמַתָּן, וְלָכֵן יֵשׁ לְקַיֵּים אֶת הַהֶסְכֵּם, אַךְ יַחַד עִם זֹאת לִקְבּוֹעַ כִּי מַגִּיעִים לָהּ סְכוּמִים נוֹסָפִים שֶׁעַל הַמֵּשִׁיב לְהַעֲבִיר לָהּ לְאַחַר שֶׁיֵּיעָרֵךְ אִיזּוּן מַשְׁאַבִּים מָלֵא, שֶׁיִּתְיַיחֵס לִכְלַל הָרְכוּשׁ.
הַמְּעַרְעֶרֶת עָתְרָה לְכָךְ, שֶׁבֵּית הַמִּשְׁפָּט יַצְהִיר כִּי הַמֵּשִׁיב הִטְעָה אוֹתָהּ, הוֹלִיךְ אוֹתָהּ שׁוֹלֵל, הִצִּיג מֻצָּג שָׁוְוא בְּיַחַס לִרְכּוֹשׁוֹ וּנְכָסָיו לִפְנֵי שֶׁנִּכְרַת הַהֶסְכֵּם וְהֵפֵר אֶת הַחוֹבוֹת הַמּוּגְבָּרוֹת הַמּוּטָּלוֹת עָלָיו.
מִכָּאן עֲתִירָתָהּ, כִּי יִיקָּבַע שֶׁהַוִּויתּוּרִים שֶׁעָשְׁתָה אֵינָם חָלִים, וּבְכָל מִקְרֶה אֵינָם חוֹסְמִים אוֹתָהּ מְקַבֶּלֶת מְלוֹא הַכְּסָפִים הַמַּגִּיעִים לָהּ מִמִּכְלוֹל הָרְכוּשׁ, שֶׁהָיָה לְמֵשִׁיב בְּמוֹעֵד עֲרִיכַת הַהֶסְכֵּם.
לְטַעֲנָתָהּ, עַל הַמֵּשִׁיב לִמְסוֹר לָהּ אֶת הַפְּרָטִים הָרֶלֵבַנְטִיִּים בְּיַחַס לַהֲלִיךְ שֶׁהִתְקַיֵּים, לְטַעֲנָתָהּ, בְּמַס הַהַכְנָסָה וְיֵשׁ לְחַיֵּיבוֹ לְשַׁלֵּם לָהּ מַחֲצִית מִכָּל הַזְּכוּיוֹת שֶׁיִּתְבָּרְרוּ בַּהֲלִיךְ, לְאַחַר שֶׁהַמֵּשִׁיב יַחְשׂוֹף אֶת מְלוֹא הָרְכוּשׁ שֶׁהָיָה לוֹ בַּמוֹעֵד הַקּוֹבֵעַ.
עוֹד טַעֲנָה הַמְּעַרְעֶרֶת כִּי לֹא הָיְיתָה לָהּ אֶפְשָׁרוּת לַעֲרוֹךְ בֵּירוּר עוֹבַדְתִּי בְּיַחַס לְנְכָסֵי הַמֵּשִׁיב, כַּאֲשֶׁר זֶה רִימָּה בְּהַצְלָחָה אֶת שִׁלְטוֹנוֹת הַמַּס, הֲרֵי יֵשׁ לְהַנִּיחַ כִּי גַּם הִיא לֹא הָיְיתָה מַצְלִיחָה.
לְטַעֲנָתָהּ, הִיא לֹא יָדְעָה כִּי יֵשׁ לְמֵשִׁיב רְכוּשׁ רַב, אֶלָּא רַק חָשְׁדָה בְּכָךְ, וְזֹאת שָׁעָה שֶׁהַמְּשִׁיב לֹא סִיפֵּק לָהּ מֵידָע אֶלָּא הֶעָלִים זֹאת מִמֶּנָּה כְּמוֹ מַרְשׁוּיוֹת הַמַּס.
