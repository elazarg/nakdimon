בְּעִקְבוֹת הַפְרָשַׁת הַמֵּלָטוֹנִין, מוֹצִיא הַמּוֹחַ פְּקוּדּוֹת לַגּוּף, לְהִתְכּוֹנֵן לְשֵׁינָה.
זוֹ הַסִּיבָּה שֶׁבְּטִיסוֹת אֲנָשִׁים שָׂמִים מַסֵּכַת עֵינַיִים כֵּהָה שֶׁמַּחֲשִׁיכָה לָהֶם הַכֹּל וְנִרְדָּמִים דֵּי מַהֵר.
זוֹ גַּם הַסִּיבָּה שֶׁבְּיָמֵינוּ חֲבָרוֹת שֶׁל מַעַרְכוֹת הַפְעָלָה לְטֵלֵפוֹנִים חֲכָמִים הוֹסִיפוּ אֶפְשָׁרוּת שֶׁל שִׁינּוּי עוֹצְמַת וַאֲפִילּוּ צֶבַע הַתְּאוּרָה, לִקְרַאת הַחֲשֵׁיכָה.
כָּךְ הֵן מְעוּנְיָינוֹת לְסַיֵּיעַ לַמִּשְׁתַּמֵּשׁ לִישׁוֹן טוֹב וּמַסְפִּיק, כִּי אַחֶרֶת הָאוֹר הֶחָזָק הַבּוֹקֵעַ מֵהַמָּסָךְ רַק יְעוֹרֵר אוֹתוֹ עַד שָׁעוֹת מְאוּחָרוֹת.
הַדְלָקַת הָאוֹר בְּמַהֲלַךְ הַלַּיְלָה, מַעֲבִירָה לַמּוֹחַ מֵידָע שָׁגוּי וְהָאָדָם נוֹטֶה לְהִתְעוֹרֵר.
הַסִּיבָּה הִיא שֶׁמִּבְּחִינַת הַמּוֹחַ הָאוֹר אוֹמֵר שֶׁהִגִּיעַ הַיּוֹם וְיִיצּוּר הַמֵּלָטוֹנִין מוּפְסָק.
לָכֵן, יוֹדְעִים כּוֹל מִי שֶׁקָּמִים בְּמַהֲלַךְ הַלַּיְלָה "לַעֲשׂוֹת פִּיפִּי", שֶׁלֹּא כְּדַאי לְהַדְלִיק אוֹרוֹת בַּדֶּרֶךְ.
בְּכָל בּוֹקֶר, אוֹ כְּשֶׁיֵּשׁ אוֹר חָזָק מִסָּבִיב, נוֹצֶרֶת תּוֹפָעָה הֲפוּכָה לְמָה שֶׁקּוֹרֶה לָנוּ עִם חֲשֵׁיכָה.
אוֹר הַבּוֹקֶר, אוֹ תְּאוּרָה חֲזָקָה כָּלְשֶׁהִי, גּוֹרְמִים לָנוּ לָחוּשׁ שֶׁהִגִּיעַ יוֹם חָדָשׁ.
מִבְּחִינַת הַמּוֹחַ זֶה מְחַיֵּיב פְּעוּלָּה וְהַגּוּף שֶׁלָּנוּ מַפְסִיק מִיָּידִית לְיַיצֵּר מֵלָטוֹנִין, כְּדֵי לִגְרוֹם לָנוּ לְהִתְעוֹרֵר.
אַגַּב, הַמֵּלָטוֹנִין הוּא גַּם הַהֶסְבֵּר לָעֲיֵיפוּת שֶׁלָּנוּ בַּחוֹרֶף.
הֲרֵי בַּחוֹרֶף הַיָּמִים קְצָרִים הַרְבֵּה יוֹתֵר וְלָכֵן שְׁעוֹת הָאוֹר מְעַטּוֹת וְהַשֶּׁמֶשׁ שׁוֹקַעַת מוּקְדָּם.
הַתּוֹצָאָה? - בַּמּוֹחַ נִקְלֶטֶת הַשֶּׁמֶשׁ הַשּׁוֹקַעַת מוּקְדָּם, מַתְחִיל יִיצּוּר הַמֵּלָטוֹנִין, שֶׁהוּא "הַפִּיגָ'מָה שֶׁל הַמּוֹחַ", וְאָנוּ חָשִׁים עֲיֵיפוּת מוּקְדָּם יוֹתֵר מֵאֲשֶׁר בַּקַּיִץ.
