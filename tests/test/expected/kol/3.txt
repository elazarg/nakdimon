יוֹצְאֵי אֶתְיוֹפְּיָה
פּוֹרְטָל זֶה מְרַכֵּז מֵידָע, זְכוּיוֹת וַהֲטָבוֹת שֶׁעֲשׂוּיִים לְעַנְיֵין יִשְׂרְאֵלִים שֶׁמּוֹצָאָם מֵאֶתְיוֹפְּיָה
הָעֲלִיָּה מֵאֶתְיוֹפְּיָה אֲשֶׁר נִפְרֶשֶׂת עַל פְּנֵי יוֹתֵר מִ-30 שָׁנָה, יָצְרָה קְהִילָּה שֶׁחֶלְקָהּ מְעוֹרֶה לַחֲלוּטִין בַּחֶבְרָה וְחֶלְקָהּ עֲדַיִין בְּתַהֲלִיךְ קְלִיטָה.
בְּפוֹרְטָל זֶה תִּמְצְאוּ מֵידָע עַל תָּכְנִיּוֹת מְיוּחָדוֹת לְיוֹצְאֵי אֶתְיוֹפְּיָה, סִיּוּעַ בְּדִיּוּר, מִלְגוֹת לִימּוּדִים, הַכְוָונָה תַּעֲסוּקָתִית, חוֹבַת הַיִּיצּוּג הַהוֹלֵם בְּשֵׁירוּת הַמְּדִינָה וְעוֹד; כְּמוֹ גַּם חוּקִּים כְּלָלִיִּים הָאוֹסְרִים עַל אַפְלָיָה בְּשֶׁל מוֹצָא. בְּהֶמְשֵׁךְ הָעַמּוּד תִּמְצְאוּ רְשִׁימַת אִרְגּוּנֵי סִיּוּעַ וּמוֹקְדִים מֶמְשַׁלְתִּיִּים, שֶׁמַּטְּרָתָם לְשַׁלֵּב אֶת יוֹצְאֵי אֶתְיוֹפְּיָה בַּחֶבְרָה וְלִמְנוֹעַ מִקְרִים שֶׁל אַפְלָיָה וְגִזְעָנוּת
- חִינּוּךְ - יְלָדִים וְנוֹעַר
- תַּלְמִידִים עוֹלִים וְתוֹשָׁבִים חוֹזְרִים
- אִיסּוּר אַפְלָיָה בְּחִינּוּךְ
- סִיּוּעַ לְאִמָּהוֹת שֶׁהֵן עוֹלוֹת חֲדָשׁוֹת בְּמִימּוּן מְעוֹנוֹת יוֹם וּמִשְׁפַּחְתּוֹנִים
- תַּשְׁלוּם אַגְרַת חִינּוּךְ (סַל קְלִיטָה) לְתַלְמִידִים יוֹצְאֵי אֶתְיוֹפְּיָה
- הוֹרָאַת הָעִבְרִית כְּשָׂפָה שְׁנִיָּיה וְהַקְצָאַת שָׁעוֹת לְתַלְמִידִים עוֹלִים מֵאֶתְיוֹפְּיָה
- הֲקַלּוֹת בְּבַגְרוּת לְעוֹלִים חֲדָשִׁים
- הֲקַלּוֹת בְּבַגְרוּת לְעוֹלִים וָתִיקִים
- בְּחִינַת בַּגְרוּת בִּשְׂפַת אֵם כְּשָׂפָה זָרָה
- לִימּוּדִים - בּוֹגְרִים
- סִיּוּעַ לִסְטוּדֶנְטִים יוֹצְאֵי אֶתְיוֹפְּיָה
- מִלְגוֹת קִיּוּם לִסְטוּדֶנְטִים יוֹצְאֵי אֶתְיוֹפְּיָה מִטַּעַם אִרְגּוּן צְפוֹן אָמֵרִיקָה לְמַעַן יְהוּדֵי אֶתְיוֹפְּיָה
- תָּכְנִית יוֹאֵל - יוֹצְאֵי אֶתְיוֹפְּיָה לִמְצוּיָּנוּת בָּאָקָדֶמְיָה
- "עֲתִידִים לְתַשְׁתִּיּוֹת תַּחְבּוּרָה" - תָּכְנִית צוֹעֲרִים לִמְהַנְדְּסֵי תַּחְבּוּרָה
- "סְטַארְט אַפּ לְהַצְלָחָה" - לִיוּוּי סְטוּדֶנְטִים יוֹצְאֵי אֶתְיוֹפְּיָה בִּבְאֵר שֶׁבַע
- הַתְאָמוֹת בִּבְחִינַת הַתֵּיאוֹרְיָה שֶׁל מִשְׂרַד הָרִישּׁוּי
- מְלָגוֹת מִטַּעַם קֶרֶן גְּרוֹס לְחַיָּילִים מְשׁוּחְרָרִים יוֹצְאֵי אֶתְיוֹפְּיָה
- תַּרְגּוּם וְגִישּׁוּר
- נוֹכְחוּת מְתוּרְגְּמָן בְּדִיּוּנִים עַל הַרְחָקַת תַּלְמִידִים
- זַכָּאוּת לִמְתוּרְגְּמָן בְּמִסְגֶּרֶת בֵּירוּר בַּקָּשָׁה לְהֶיתֵּר לְפִיטּוּרִים לְפִי חוֹק עֲבוֹדַת נָשִׁים
- קוֹל לַבְּרִיאוּת - תַּרְגּוּם טֵלֵפוֹנִי בְּטִיפּוּלִים רְפוּאִיִּים לְיוֹצְאֵי אֶתְיוֹפְּיָה
- מוֹקֵד הַתַּרְגּוּם הָרְפוּאִי בְּבָתֵּי הַחוֹלִים הַמֶּמְשַׁלְתִּיִּים וְלִשְׁכוֹת מִשְׂרַד הַבְּרִיאוּת
- הַתְאָמוֹת בִּבְחִינַת הַתֵּיאוֹרְיָה שֶׁל מִשְׂרַד הָרִישּׁוּי
- גִּישּׁוּר בְּנוֹשֵׂא זְכוּיוֹת עוֹבְדֵי קַבְּלָן לִבְנֵי הָעֵדָה הָאֶתְיוֹפִּית
- שֵׁירוּתֵי גִּישּׁוּר לְתוֹשָׁבֵי לוֹד
- שֵׁירוּתֵי גִּישּׁוּר לְתוֹשָׁבֵי רְחוֹבוֹת
- זְכוּיוֹת עוֹבְדִים
- אִיסּוּר אַפְלָיַית עוֹבֵד בְּשֶׁל אֶרֶץ מוֹצָאוֹ
- גִּישּׁוּר בְּנוֹשֵׂא זְכוּיוֹת עוֹבְדֵי קַבְּלָן לִבְנֵי הָעֵדָה הָאֶתְיוֹפִּית
- חַג הַסִּיגְד
- נְקוּדּוֹת זִיכּוּי מִמַּס הַכְנָסָה לְעוֹלֶה חָדָשׁ
- מַעֲנַק עֲבוֹדָה (מַעֲנַק הַכְנָסָה, מַס הַכְנָסָה שְׁלִילִי)
- פּוֹרְטָל תַּעֲסוּקָה וּזְכוּיוֹת עוֹבְדִים
- הַבְטָחַת הַכְנָסָה
- הַכְשָׁרָה וְקִידּוּם תַּעֲסוּקָה
- תָּכְנִית תל"מ לְשִׁילּוּב צְעִירִים יוֹצְאֵי אֶתְיוֹפְּיָה בְּתַעֲסוּקָה
- תָּכְנִית שׁוֹבָרִים לְהַכְשָׁרָה מִקְצוֹעִית
- תּוֹכְנִית "לַמֶּרְחָק" - הַכְוָונָה תַּעֲסוּקָתִית לְיוֹצְאֵי אֶתְיוֹפְּיָה
- אֵשֶׁת חַיִל
- יוֹזְמִים עֵסֶק - תָּכְנִית יַזָּמוּת עִסְקִית לְיוֹצְאֵי אֶתְיוֹפְּיָה
- סַדְנָאוֹת לְהַעֲצָמָה תַּעֲסוּקָתִית לְיוֹצְאֵי אֶתְיוֹפְּיָה
- תָּכְנִיּוֹת הַכְשָׁרָה מִטַּעַם טֶק-קַרְיֵירָה
- תָּכְנִית מְצוּיָּינוֹת וּמַנְהִיגוּת לְקִידּוּם אָקָדֵמָאִים יוֹצְאֵי אֶתְיוֹפְּיָה בְּתַעֲסוּקָה
- מֶרְכַּז הִזְדַּמְּנוּת - מַעְגַּל הַחַיִּים וְהַתַּעֲסוּקָה בְּאֵר שֶׁבַע
- סְטַארְט אַפּ לְקַרְיֵירָה - תָּכְנִית לְתַעֲסוּקַת צְעִירִים יוֹצְאֵי אֶתְיוֹפְּיָה בִּבְאֵר שֶׁבַע
- סִיּוּעַ לְעוֹלִים וְתוֹשָׁבִים חוֹזְרִים בְּמִימּוּן קוּרְס הַכְשָׁרָה בְּסִיסִית לְיִישּׂוּמֵי מַחְשֵׁב
- יִיצּוּג הוֹלֵם בִּמְקוֹמוֹת עֲבוֹדָה
- יִיצּוּג הוֹלֵם לְיוֹצְאֵי אֶתְיוֹפְּיָה בַּשֵּׁירוּת הַצִּיבּוּרִי
- יִיצּוּג הוֹלֵם לְיוֹצְאֵי אֶתְיוֹפְּיָה בְּקֶרֶב הָעוֹבְדִים בְּשֵׁירוּת הַמְּדִינָה
- רְשִׁימַת מִשְׂרוֹת פְּנוּיוֹת בְּשֵׁירוּת הַמְּדִינָה הַמְּיוֹעָדוֹת לְיוֹצְאֵי אֶתְיוֹפְּיָה
