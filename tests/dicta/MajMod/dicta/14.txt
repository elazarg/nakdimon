אָלֵכְּס היגינס אָלֶכְּסַנְדֶּר גּוֹרְדּוֹן "אָלֵכְּס" היגינס (18 בְּמֵרְץ 1949 - 24 בְּיוּלִי 2010), יָדוּעַ גַּם בַּכִּינּוּי הוריקן היגנס, הָיָה שַׂחְקָן סנוקר מקצועני צְפוֹן אִירִי. היגינס זָכָה פַּעֲמַיִים בְּאַלִּיפוּת הָעוֹלָם בסנוקר בַּשָּׁנִים 1972 וֹ-1982, וְהָיָה פַּעֲמַיִים סְגַן אַלוּף הָעוֹלָם בסנוקר בַּשָּׁנִים 1976 וֹ-1980. בְּנוֹסָף, היגינס זָכָה בְּאַלִּיפוּת הָעוֹלָם בסנוקר לְזוּגוֹת, יַחַד עִם גִּ'ימִי וייט הָאַנְגְּלִי בִּשְׁנַת 1984, וְכֵן זָכָה שָׁלוֹשׁ פְּעָמִים בִּרְצִיפוּת בִּגְבִיעַ הָעוֹלָם בסנוקר לנבחרות, עִם נִבְחֶרֶת אִירְלַנְד בַּשָּׁנִים 1985, 1986 וֹ-1987. בְּשִׂיאוֹ דּוֹרַג היגינס בַּמָּקוֹם הַשֵּׁנִי בָּעוֹלָם בַּדֵּירוּג שַׂחְקָנֵי הסנוקר הָעוֹלָמִי. זָכָה לַכִּינּוּי "הוריקן" בִּזְכוּת סִגְנוֹן מִשְׂחֲקוֹ הַמָּהִיר. קַרְיֵירָה היגינס הֵחֵל לְשַׂחֵק סנוקר מקצועני בְּעוֹנַת 1971/1972, בְּגִיל 22. הוּא זָכָה בְּאַלִּיפוּת הָעוֹלָם בסנוקר כְּבָר בְּנִיסְּיוֹנוֹ הָרִאשׁוֹן בִּשְׁנַת 1972, בְּגִיל 22, לְאַחַר שניצח בַּגְּמָר אֶת גּ'וֹן ספנסר הָאַנְגְּלִי בתוצאה 32-37, וּבְכָךְ נִהְיֶה לְאַלּוּף הָעוֹלָם בסנוקר הַצָּעִיר בְּיוֹתֵר אִי פַּעַם (שִׂיא שֶׁנִּשְׁבַּר כַּעֲבוֹר 18 שָׁנִים עַל יְדֵי סְטִיבֶן הנדרי כשזכה בְּאַלִּיפוּת הָעוֹלָם בְּגִיל 21). לְאַחַר מִכֵּן הִגִּיעַ היגינס לִגְמַר אַלִּיפוּת הָעוֹלָם בַּשָּׁנִים 1976 וֹ-1980, אַךְ הִפְסִיד. בִּשְׁנַת 1982, זָכָה היגינס בַּפַּעַם הַשְּׁנִיָּיה בְּאַלִּיפוּת הָעוֹלָם, לְאַחַר שניצח בַּגְּמָר אֶת רִיֵּי רירדון הולשי בתוצאה 15-18, וְאִם לֹא הָיָה נִקְנַס בהורדת נְקוּדּוֹת דֵּירוּג עֵקֶב עֲבֵירוֹת מִשְׁמַעַת, הָיָה מְדוֹרָג בְּעוֹנַת 1982/1983 בַּמָּקוֹם הָרִאשׁוֹן בָּעוֹלָם. היגינס, עִם סִגְנוֹן מִשְׂחֲקוֹ, שֶׁהָיָה מַלְהִיב, הָיָה אַחְרַאי בְּמִידָּה רַבָּה לַעֲלִיָּיה בפופולריות שֶׁל עָנָף הסנוקר בְּסוֹף שְׁנוֹת הַשִּׁבְעִים. הוּא הָיָה שַׂחְקָן מָהִיר מְאוֹד, ובראיונות הוּא אַף הִסְבִּיר שֶׁהַסִּיבָּה לְכָךְ הָיְיתָה פָּשׁוּט חוֹסֶר סַבְלָנוּת. רַבִּים מחשיבים אוֹתוֹ לְאֶחָד מִשַּׂחְקָנֵי הסנוקר הַטּוֹבִים בְּיוֹתֵר בְּכָל הַזְּמַנִּים. עִם זֹאת, נוֹדַע גַּם בָּאוֹפִי וּמֶזֶג בְּעָיָיתִיִּים. עַל פִּי הַעֲרָכוֹת, כְּ-50 תְּלוּנוֹת רִשְׁמִיּוֹת הוגשו נֶגֶד אָלֵכְּס היגינס עַל יְדֵי הִתְאַחֲדוּת הסנוקר הָעוֹלָמִית - יוֹתֵר מִכָּל שַׂחְקָן סנוקר אַחֵר בַּהִיסְטוֹרְיָה שֶׁל הַמִּשְׂחָק.[1] בְּמַהֲלַךְ הַקַּרְיֵירָה, זָכָה היגינס בְּ-30 תְּאָרִים מקצועניים (26 תּוֹאֲרֵי יְחִידִים וֹ-4 תְּאָרִים קבוצתיים), בָּהֶם אַלִּיפוּת בְּרִיטַנְיָה בִּשְׁנַת 1983, טורניר המאסטרס בַּשָּׁנִים 1978 וֹ-1981, והמאסטרס האירי בִּשְׁנַת 1989, בּוֹ נִיצֵּחַ בְּגִיל 40 אֶת סְטִיבֶן הנדרי הַצָּעִיר.
