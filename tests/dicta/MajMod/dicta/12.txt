אלדד פֶּלֶד אלדד פֶּלֶד הוּא קָצִין צַהַ"ל בַּמִּילּוּאִים בְּדַרְגַּת אַלוּף מְשַׁנֶּה. שֵׁירוּת צְבָאִי עִם גיוסו לְצַהַ"ל שׁוּבַּץ פֶּלֶד בַּחֲטִיבַת גבעתי וְעָבַר מַסְלוּל הַכְשָׁרָה כְּלוֹחֵם, קוּרְס מ"כים חִי"ר וקורס קציני חִי"ר. עִם סִיּוּם הקורס שִׁימֵּשׁ כִּמְפַקֵּד מַחְלָקָה ופלוגה בחטיבה. בַּהֶמְשֵׁךְ שִׁימֵּשׁ כִּמְפַקֵּד פלחה"ן גבעתי וְכִמְפַקֵּד סיירת גבעתי. בַּשָּׁנִים 1999-2001 שִׁימֵּשׁ כִּמְפַקֵּד גְּדוּד שקד וְלָקַח חֵלֶק בִּפְעִילוּת המבצעית כְּנֶגֶד הַטֵּרוֹר הַפָלַסְטִינִי בִּרְצוּעַת עַזָּה ובלחימה בִּדְרוֹם לְבָנוֹן. לְאַחַר מִכֵּן שִׁימֵּשׁ כִּמְפַקֵּד בא"ח גבעתי[1] וכסגן מְפַקֵּד החטיבה[2]. בַּשָּׁנִים 2006-2009 שִׁימֵּשׁ כִּמְפַקֵּד מַעֲרַךְ מגל[3]. לְאַחַר שִׁחְרוּרוֹ פָּנָה לַעֲבוֹד כְּעוֹרֵךְ דִּין[4].
