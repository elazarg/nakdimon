איירין לדרמן איירין לדרמן הִיא אתלטית יהודייה-בריטית בֶּעָבָר, אלופה ושיאנית המכביה בַּעֲלַת 6 מֵדַלְיוֹת, מֵהֶן 4 מֵדַלְיוֹת זָהָב (הַשְּׁלִישִׁית 1950 והרביעית 1953) בריצת 100 מֶטֶר, בריצת 200 מֶטֶר ובקפיצה לרוחק. בִּיּוֹגְרַפְיָה לדרמן גָּדְלָה בְּלוֹנְדוֹן. אלופה ושיאנית המכביה בתחרויות הָאַתְלֵטִיקָה שֶׁל המכביה הַשְּׁלִישִׁית שֶׁנֶּעֶרְכוּ בִּשְׁנַת 1950 זָכְתָה בְּ-3 מֵדַלְיוֹת, אַחַת מִכָּל סוּג. מֵדַלְיַית זָהָב, בריצת 100 מֶטֶר בְּשִׂיא מכביה חָדָשׁ שֶׁל 12.8 שְׁנִיּוֹת, לִפְנֵי עדנה פְרִידְמָן מִדְּרוֹם אַפְרִיקָה (12.8 שְׁנִיּוֹת) ולאה הורביץ מִיִּשְׂרָאֵל (13.3 שְׁנִיּוֹת), מֵדַלְיַית כֶּסֶף, בריצת 60 מֶטֶר בתוצאה שֶׁל 8.3 שְׁנִיּוֹת, אַחֲרֵי עדנה פְרִידְמָן 8.2 שְׁנִיּוֹת וְלִפְנֵי ון דר שטאם מהולנד (8.3 שְׁנִיּוֹת) ומדליית אָרָד, בריצת 80 מֶטֶר משוכות בתוצאה שֶׁל 15.6 שְׁנִיּוֹת, אַחֲרֵי רוּת איילון (13.8 שְׁנִיּוֹת) וְשָׁרָה מיזרחי (15.4 שְׁנִיּוֹת), שְׁתֵּיהֶן מִיִּשְׂרָאֵל.[1][2] כּוֹכֶבֶת המכביה הָרְבִיעִית בתחרויות הָאַתְלֵטִיקָה שֶׁל המכביה הָרְבִיעִית שֶׁנֶּעֶרְכוּ בִּשְׁנַת 1953 זָכְתָה פְרִידְמָן בְּ-3 מֵדַלְיוֹת זָהָב יוֹתֵר מִכָּל סְפּוֹרְטָאי אַחֵר, בריצת 100 מֶטֶר בתוצאה שֶׁל 12.7 שְׁנִיּוֹת, לִפְנֵי רודה אייבל מִדְּרוֹם אַפְרִיקָה (13.0 שְׁנִיּוֹת) ודלית קליינמן מִיִּשְׂרָאֵל (13.2 שְׁנִיּוֹת) וּבְכָךְ הֲגָנָה עַל תוארה, בריצת 200 מֶטֶר בתוצאה שֶׁל 26.7 שְׁנִיּוֹת, לִפְנֵי רודה אייבל (27.3 שְׁנִיּוֹת) ורות הרשלר מִיִּשְׂרָאֵל (27.7 שְׁנִיּוֹת) ובקפיצה לרוחק בְּשִׂיא מכביה חָדָשׁ שֶׁל 5.36 מֶטֶר, לִפְנֵי תָּמָר מיטל-שומכר (4.87 מֶטֶר) וּמֵרִים טפר (4.86 מֶטֶר), שְׁתֵּיהֶן מִיִּשְׂרָאֵל.[3]
