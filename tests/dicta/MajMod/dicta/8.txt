אוֹרֶן מיילי אוֹרֶן מיילי (מְכוֹנָה אוֹרֶן הַקָּטָן; נוֹלַד בִּשְׁנַת 1978) הוּא קוֹסֵם ואמן טלפתיה יִשְׂרְאֵלִי, שֶׁהִתְפַּרְסֵם בְּיִשְׂרָאֵל בִּשְׁנוֹת ה-80 ובתחילת שְׁנוֹת ה-90 שֶׁל הַמֵּאָה ה-20. בִּיּוֹגְרַפְיָה מיילי נוֹלַד וְגָדַל בִּשְׁכוּנַת פרדס כָּץ בִּבְנֵי בְּרַק. הוּא הֵחֵל אֶת דַּרְכּוֹ בַּפִּינָּה שהוקצתה לוֹ אֵצֶל הבדרן מֵאִיר רֶמֶז. אֶת הוֹפָעוֹתָיו הָרִאשׁוֹנוֹת הֵחֵל בְּגִיל פָּחוֹת מֵעֶשֶׂר, וּמִכָּאן כינויו "אוֹרֶן הַקָּטָן". בִּשְׁנוֹת ה-80 ובתחילת שְׁנוֹת ה-90 שֶׁל הַמֵּאָה ה-20 הוֹפִיעַ במופעים ובתוכניות טֵלֵוִויזְיָה יַחַד עִם אָבִיו, רוֹנִּי מיילי. בְּאֶמְצַע שְׁנוֹת הַתִּשְׁעִים עָזַב אֶת עוֹלָם הַבִּידּוּר וְהֵחֵל לְסַיֵּיעַ לְאָבִיו בָּעֵסֶק לְיִיצּוּר שִׁלְטֵי פִּרְסוּם. כַּיּוֹם הוּא מוֹפִיעַ לִפְנֵי וַעֲדֵי עוֹבְדִים בהנחיית אָבִיו. בְּ-2006 זומן לַמִּשְׁטָרָה, בִּגְלַל בִּלְבּוּל שֶׁהָיָה בֵּינוֹ לְבֵין אוֹרֶן זריף, בַּחֲשָׁד שהונה אֲנָשִׁים שֶׁבָּאוּ לְטִיפּוּל אֶצְלוֹ.[1] בְּ-2013 הִשְׁתַּתְּפוּ בַּתּוֹכְנִית "הָאָח הַגָּדוֹל" אָבִיו שֶׁל אוֹרֶן, רוֹנִּי מיילי, וְאָחִיו גילי, שמופיעים לְעִיתִּים לְצִדּוֹ בהופעותיו.[2] מיילי נָשׂוּי וְאָב לִשְׁלוֹשָׁה יְלָדִים.
