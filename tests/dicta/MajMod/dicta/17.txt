אסוד אסוד הָיָה חֵיל משמר וִיחִידָה צְבָאִית שֶׁל חַיָּילִים מִמּוֹצָא אלאני. שָׁם הַשֵּׁבֶט המונגולי הוּא צוּרַת הָרַבִּים שֶׁל אֵס, שְׁמָם שֶׁל האלאנים בְּעַרְבִית. לְאַחַר הַפְּלִישָׁה המונגולית לרוס (רוֹס שֶׁל קייב) נִכְנְעוּ רַבִּים מהאלאנים לָאִימְפֶּרְיָה המונגולית. חֵלֶק מֵהֶם הִמְשִׁיךְ לְהִתְנַגֵּד לאורדת הַזָּהָב. לוֹחֲמִים רַבִּים הוּעַבְרוּ מִצְּפוֹן הקווקז למונגוליה. הוֹעַלְתָה גַּם הַשְׁעָרָה כִּי הֵם עָזְרוּ לאדוניהם הַחֲדָשִׁים במלחמתם כְּנֶגֶד הצ'רקסים וְכִי הֵם הִשְׁתַּתְּפוּ בַּפְּלִישָׁה המונגולית לְאֵירוֹפָּה. תַּחַת שִׁלְטוֹנוֹ שֶׁל מונגקה חאן הֵם נִלְחֲמוּ כְּנֶגֶד שׁוֹשֶׁלֶת סונג הַסִּינִית וממלכת ננזאו ביונאן (1258-1259). הַמִּשְׁמָר הַקֵּיסָרִי האלאני הָיָה מְחוּלָּק לִשְׁנֵי יְחִידוֹת שהמפקדה שֶׁלָּהֶם הָיְיתָה בחרחורין. לְאַחַר הכתרתו שֶׁל קובלאי חאן, הִשְׁתַּתְּפוּ האלאנים בְּמַעֲרָכָה כְּנֶגֶד אָרִיק בוקה, אֲחֵיהֶם שֶׁל מונגקה חאן וקובלאי חאן והטוען לכתר המונגולי, וּמְאוּחָר יוֹתֵר הִשְׁתַּתְּפוּ בְּמַעֲרָכָה כְּנֶגֶד קאידו תַּחַת שׁוֹשֶׁלֶת יואן. הֵם סָבְלוּ אבידות כְּבֵדוֹת מההתנגדות שֶׁל סונג כַּאֲשֶׁר שרתו תַּחַת הנהגתם שֶׁל באיאן מבארין[1] ואג'ו. בִּסְבִיבוֹת 1271 אירגן אוֹתָם קובלאי חאן כ"משמר גאסין" ("משמר אלאני") יַחַד עִם חֵלֶק מהקיפצקים. הַמִּשְׁמָר האלאני הִגִּיעַ לשיאו בִּתְקוּפַת שִׁלְטוֹנוֹ שֶׁל הַקֵּיסָר הַסִּינִי גיאטו חאן. הֵם נוֹתְרוּ בַּעֲלֵי הַשְׁפָּעָה במונגוליה עַד לְמֵאָה ה-15, כַּאֲשֶׁר האוירטים כָּבְשׁוּ אֶת מונגוליה. כַּיּוֹם חָיָה קְבוּצָה קְטַנָּה שֶׁל שֵׁבֶט האסוד במונגוליה הַפְּנִימִית.
