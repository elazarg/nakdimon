אוּזְבֵּקִיסְטָן בַּמִּשְׂחָקִים האולימפיים אוּזְבֵּקִיסְטָן מִשְׁתַּתֶּפֶת בַּמִּשְׂחָקִים האולימפיים מֵאָז אוֹלִימְפִּיָּאדַת אטלנטה (1996). עַד 1991 הָיְיתָה אוּזְבֵּקִיסְטָן חֵלֶק מִבְּרִית הַמּוֹעָצוֹת, וספורטאיה הִתְחָרוּ בְּמִסְגֶּרֶת הַמִּשְׁלַחַת הַסּוֹבְיֵיטִית. בְּאוֹתָהּ שָׁנָה התפרקה בְּרִית הַמּוֹעָצוֹת ל-15 מְדִינוֹת נִפְרָדוֹת, וֹ-12 מֵהֶן, בָּהֶן אוּזְבֵּקִיסְטָן, שָׁלְחוּ מִשְׁלַחַת מְאוּחֶדֶת לאולימפיאדת בַּרְצְלוֹנָה (1992). בְּאַרְבַּע האולימפיאדות בָּהֶן השתתפה אוּזְבֵּקִיסְטָן, צברו הַסְּפּוֹרְטָאִים האוזבקים 32 מֵדַלְיוֹת, מֵהֶן 9 מֵדַלְיוֹת זָהָב. הָעֲנָפִים הַבּוֹלְטִים בְּיוֹתֵר הֵם הֵיאָבְקוּת ואיגרוף. הִיסְטוֹרְיָה הַמִּשְׁלַחַת הָעַצְמָאִית הָרִאשׁוֹנָה שֶׁל אוּזְבֵּקִיסְטָן, בְּאוֹלִימְפִּיָּאדַת אטלנטה, כָּלְלָה 71 סְפּוֹרְטָאִים. שְׁנַיִים מֵהֶם זָכוּ במדליות הָרִאשׁוֹנוֹת בְּתוֹלְדוֹתֶיהָ: הג'ודוקא ארמן בגדסרוב זָכָה בְּמֵדַלְיַית כֶּסֶף והמתאגרף כרים טולגנוב זָכָה בְּמֵדַלְיַית אָרָד. בְּאוֹלִימְפִּיָּאדַת סִידְנִי (2000) נטלו חֵלֶק 70 סְפּוֹרְטָאִים אוזבקים. המתאגרף מחמטקודיר עבדולאייב זָכָה בְּמֵדַלְיַית הַזָּהָב הָרִאשׁוֹנָה בְּתוֹלְדוֹת אוּזְבֵּקִיסְטָן (בְּמִשְׂחֲקֵי הַקַּיִץ). שְׁנֵי מתאגרפים נוֹסָפִים, סרגיי מיחאילוב ורוסטם סאידוב, זָכוּ במדליות אָרָד, ומתאגרף רְבִיעִי, רוסלאן צ'גאייב, העפיל לְשַׁלֵּב רֶבַע גְּמַר. בהיאבקות בָּלַט אַרְתוּר טאימזוב, שֶׁזָּכָה בְּמֵדַלְיַית כֶּסֶף. גַּם בְּאוֹלִימְפִּיָּאדַת אָתוּנָה (2004) הִשְׁתַּתְּפוּ 70 סְפּוֹרְטָאִים אוזבקים. טאימזוב זָכָה בְּמֵדַלְיַית זָהָב, וְכָמוֹהוּ גַּם המתאבק אָלֶכְּסַנְדֶּר דוקטורשווילי. מִתְאַבֵּק שְׁלִישִׁי, מגומד איברגימוב זָכָה בְּמֵדַלְיַית כֶּסֶף. והמתאגרפים בחודירג'ון סולטונוב ואוטקירבק חיידרוב זָכוּ במדליות אָרָד, וּשְׁלוֹשָׁה מתאגרפים אוזבקים נוֹסָפִים, טולאשבוי דוניורוב, דילשוד מחמודוב ושרזוד חוסאנוב, הֶעְפִּילוּ לְשַׁלֵּב רֶבַע הַגְּמָר. בהרמת משקולות בָּלַט אָלֶכְּסַנְדֶּר אורינוב, שֶׁסִּייֵּם בַּמָּקוֹם הַשְּׁמִינִי. עַל השתתפותה שֶׁל אוּזְבֵּקִיסְטָן בַּמִּשְׂחָקִים אֵלֶּה העיבה פסילתה שֶׁל מיידה הַפַּטִּישׁ אולגה שצ'וקינה, בְּשֶׁל שִׁימּוּשׁ בְּסַמִּים. לאולימפיאדת בֶּיְיגִ'ינְג (2008) שָׁלְחָה אוּזְבֵּקִיסְטָן מִשְׁלַחַת קְטַנָּה יוֹתֵר, בַּת 58 סְפּוֹרְטָאִים. המתאבק אַרְתוּר טאימזוב זָכָה בְּמֵדַלְיַית זָהָב שְׁנִיָּיה בִּרְצִיפוּת, ועמיתו סוסלן טיגייב זָכָה בְּמֵדַלְיַית כֶּסֶף, אַךְ זוֹ נשללה מִמֶּנּוּ בְּ-2016 בְּשֶׁל הַרְשָׁעָה באמרוץ. דילשוד מנסורוב סִייֵּם בַּמָּקוֹם הַחֲמִישִׁי וקורבן קורבנוב בַּמָּקוֹם הַשְּׁבִיעִי. הג'ודוקא עבדולו טנגרייב זָכָה בְּמֵדַלְיַית כֶּסֶף, בְּעוֹד שרישוד סובירוב זָכָה בְּמֵדַלְיַית אָרָד. ג'ודוקא נוֹסָף, מיראלי שריפוב, סִייֵּם בַּמָּקוֹם הַחֲמִישִׁי. בְּנוֹסָף, שְׁנֵי מִתְעַמְלִים אוזבקים זָכוּ לָרִאשׁוֹנָה במדליות אָרָד - אַנְטוֹן פוקין בתרגיל המקבילים ויקטרינה חִילְּקוּ, שֶׁהָיְיתָה לספורטאית האוזבקית הָרִאשׁוֹנָה, וְהַיְּחִידָה עַד כֹּה, שזוכה במדליה הָאוֹלִימְפִּית בְּמִשְׂחֲקֵי הַקַּיִץ - בטרמפולינה. חוֹתֵר הקאנו ואדים מנקוב הָיָה קָרוֹב לִזְכִיָּיה רִאשׁוֹנָה במדליה גַּם בַּעֲנַף זֶה, אַךְ סִייֵּם בַּמָּקוֹם הָרְבִיעִי במקצה הספרינט ל-1,000 מֶטֶר. בַּעֲנַף הטאקוונדו בָּלַט אקמאל אירגשב, שֶׁסִּייֵּם בַּמָּקוֹם הַחֲמִישִׁי. המתאגרפים האוזבקים לֹא הִצְלִיחוּ בַּמִּשְׂחָקִים אֵלּוּ לִזְכּוֹת במדליות, אַךְ שְׁנַיִים מֵהֶם, דילשוד מחמודוב ואלשוד רסולוב, הֶעְפִּילוּ לְשַׁלֵּב רֶבַע הַגְּמָר. בְּאוֹלִימְפִּיָּאדַת לוֹנְדוֹן (2012) הִשְׁתַּתְּפוּ 54 סְפּוֹרְטָאִים אוזבקים. טאימזוב זָכָה בְּמֵדַלְיַית זָהָב שְׁלִישִׁית בִּרְצִיפוּת, שֶׁהָיְיתָה גַּם מֵדַלְיַית הַזָּהָב הַיְּחִידָה לִזְכוּתָהּ שֶׁל הַמִּשְׁלַחַת. מִתְאַבֵּק נוֹסָף, סוסלאן טיגייב, זָכָה בְּמֵדַלְיַית אָרָד, אַךְ זוֹ נשללה מִמֶּנּוּ חוֹדָשִׁים סְפוּרִים לְאַחַר הַמִּשְׂחָקִים, לְאַחַר שֶׁנִּמְצָא כִּי נֵטֶל סַמִּים. שְׁנֵי סְפּוֹרְטָאִים אוזבקים נוֹסָפִים שֶׁזָּכוּ במדליות הָיוּ הג'ודוקא רישוד סובירוב והמתאגרף אבוס אטויב, שֶׁזָּכוּ במדליות אָרָד. אַרְבָּעָה מתאגרפים נוֹסָפִים הודחו בְּשָׁלָב רֶבַע הַגְּמָר. סְפּוֹרְטָאִים אוזבקים נוֹסָפִים שבלטו בְּלוֹנְדוֹן כָּלְלוּ אֶת חוֹתֵר הקאנו ואדים מנקוב (מָקוֹם רְבִיעִי במקצה ל-1000 מֶטְרִים), מֵרִים המשקולות רוסלן נורודינוב (מָקוֹם הָרְבִיעִי), רוֹכֵב הָאוֹפַנַּיִים סרגיי לגוטין (מָקוֹם חֲמִישִׁי בְּמֵרוֹץ אוֹפְנֵי כְּבִישׁ), הג'ודוקא רמזידין (מָקוֹם חֲמִישִׁי), המתאבק קוֹרְבָּן קורבנוב (מָקוֹם חֲמִישִׁי), מֵרִים המשקולות איוואן אפרמוב (מָקוֹם חֲמִישִׁי); האתלטית סבטלנה רדזיוויל (מָקוֹם שביעי בקפיצה לְגוֹבַהּ) והג'ודוקא דילשוד צ'ורייב (מָקוֹם שביעי). אוֹלִימְפִּיָּאדַת רִיּוֹ דֶּה ז'ניירו (2016) הָיְיתָה הַמּוּצְלַחַת בְּיוֹתֵר בְּתוֹלְדוֹת אוּזְבֵּקִיסְטָן, עִם 13 מֵדַלְיוֹת, מֵהֶן 4 מֵדַלְיוֹת זָהָב, הֶישֵּׂג דּוֹמֶה להישגי הַמִּשְׁלָחוֹת האוזבקיות בְּשָׁלוֹשׁ האולימפיאדות הַקּוֹדְמוֹת גַּם יַחַד. האיגרוף הָיָה הענף הַבּוֹלֵט בְּיוֹתֵר. המתאגרפים הִתְחָרוּ בְּכָל עֶשֶׂר קָטֵגוֹרְיוֹת הַמִּשְׁקָל לִגְבָרִים, ובשבע מֵהֶן זָכוּ במדליות. חסנבוי דוסמטוב, שאחובידין זוירוב ופזלידין גאיבנזארוב זָכוּ במדליות זָהָב, שחראם גיאסוב ובקטמיר מליקוזייב זָכוּ במדליות כֶּסֶף, ומורוג'ון אחמדלייב ורוסטם טולגנוב זָכוּ במדליות אָרָד. בְּנוֹסָף, סִייְּמוּ הורשיד טוז'יבייב ובחודיר ז'לולוב בַּמָּקוֹם הַחֲמִישִׁי. בְּמֵדַלְיַית הַזָּהָב הָרְבִיעִית לִזְכּוֹת האוזבקים זָכָה מֵרִים המשקולות רוסלאן נורודינוב. איוואן אפרמוב סִייֵּם בַּמָּקוֹם הַחֲמִישִׁי ורוסטם ג'נגבייב בַּמָּקוֹם הַשִּׁישִּׁי. בהיאבקות זָכוּ שְׁלוֹשָׁה אוזבקים במדליות אָרָד: איחטיור נברוזוב, מגומד איברגימוב ואלמורט טסמורדוב. בקזוד עבדורחמונוב סִייֵּם בַּמָּקוֹם הַחֲמִישִׁי. הג'ודוקא רישוד סובירוב זָכָה בְּמֵדַלְיַית אָרָד שְׁלִישִׁית בִּרְצִיפוּת, וְגַם דיורבק אורוזבוייב זָכָה בְּמֵדַלְיַית אָרָד. עבדולו טנגרייב סִייֵּם בַּמָּקוֹם הַחֲמִישִׁי. סְפּוֹרְטָאִים אוזבקים נוֹסָפִים שבלטו בריו דֶּה ז'ניירו כָּלְלוּ אֶת לוֹחֵם הטאקוונדו דמיטרי שוקין שֶׁסִּייֵּם בַּמָּקוֹם הַחֲמִישִׁי, אֶת חוֹתֵר הקאנו גראסים קוצ'נב שֶׁסִּייֵּם בַּמָּקוֹם הַשְּׁבִיעִי במקצה ל-1000 מֶטֶר וּבִמְקוֹם הַשְּׁמִינִי במקצה הַזּוּגוֹת ל-1000 מֶטֶר, יַחַד עִם סריק מירבקוב, וְאֶת המתעמלת הַוָותִיקָה אוקסנה צ'וסוביטינה בַּת ה-41, שֶׁסִּייְּמָה בַּמָּקוֹם הַשְּׁבִיעִי בתרגיל הקפיצות. מִשְׂחֲקֵי הַחוֹרֶף אוֹלִימְפִּיָּאדַת הַחוֹרֶף הָרִאשׁוֹנָה בָּהּ השתתפה אוּזְבֵּקִיסְטָן הָיְיתָה אוֹלִימְפִּיָּאדַת לילהאמר (1994). הָיְיתָה זוֹ אוֹלִימְפִּיָּאדַת הַחוֹרֶף הַיְּחִידָה עַד כֹּה בָּהּ זָכָה סְפּוֹרְטָאי אוזבקי במדליה - לִינָה צ'ריאזובה זָכְתָה בְּמֵדַלְיַית זָהָב בסקי חוֹפְשִׁי. פְּרָט לְהֶישֵּׂג זֶה, הַהֶישֵּׂג הַגָּבוֹהַּ הַיָּחִיד הַנּוֹסָף שֶׁל הַמִּשְׁלַחַת האוזבקית נִרְשַׁם בְּאוֹלִימְפִּיָּאדַת נאגנו (1998), אָז דורגה המחליקה האומנותית טטיאנה מלינינה בַּמָּקוֹם הַשְּׁמִינִי. מֵדַלְיוֹת לְפִי אולימפיאדה באולימפיאדות הַחוֹרֶף: לְפִי עָנָף
