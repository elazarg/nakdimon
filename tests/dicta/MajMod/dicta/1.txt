"דֶּלֶק" חֶבְרַת הַדֶּלֶק הַיִּשְׂרְאֵלִית "דֶּלֶק" חֶבְרַת הַדֶּלֶק הַיִּשְׂרְאֵלִית בע"מ, הַיְּדוּעָה יוֹתֵר בְּשֵׁם דֶּלֶק, הִיא אַחַת מֵאַרְבַּע חֲבֵרוֹת הַדֶּלֶק הַגְּדוֹלוֹת בְּיִשְׂרָאֵל. בבעלותה מֵאוֹת תחנות דֶּלֶק בְּרַחֲבֵי הָאָרֶץ, בְּחֶלְקָן גַּם חֲנוּיוֹת נוֹחוּת תַּחַת הַמּוּתָג "מנטה" ומתחמים קמעונאיים. בְּבַעֲלוּת הַחֶבְרָה מִסְפַּר חברות-בנות בִּתְחוּמֵי הָאֵנֶרְגִּיָּה והלוגיסטיקה, המספקות שֵׁירוּתֵי שִׁינּוּעַ והובלה שֶׁל דֶּלֶק, מַעַרְכוֹת לוֹגִיסְטִיּוֹת, יִיצּוּר שְׁמֵנִים וחומרי איטום, וְשֵׁירוּתֵי נִיפּוּק לאוניות בִּנְמָלִים בָּאָרֶץ וּבָעוֹלָם. הִיסְטוֹרְיָה הַחֶבְרָה הוּקְמָה עַל יְדֵי מֶמְשֶׁלֶת יִשְׂרָאֵל בְּדֵצֶמְבֶּר 1951, וּבְדוֹמֶה לַחֲבָרוֹת הָאֵנֶרְגִּיָּה הָאֲחֵרוֹת (פָּז וסונול) הָיְיתָה נְתוּנָה בָּעֲשׂוֹרִים הָרִאשׁוֹנִים לְפִיקּוּחַ מֶמְשַׁלְתִּי. עַד 1960 התרחבה הַחֶבְרָה ופרשה 44 תחנות דֶּלֶק בְּרַחֲבֵי הָאָרֶץ; בְּמַקְבִּיל הִצְטַמְצֵם חֶלְקָן בַּשּׁוּק שֶׁל חֲבֵרוֹת הַדֶּלֶק הַזָּרוּת, שֶׁפָּעֲלוּ בָּאָרֶץ מֵאָז תְּקוּפַת הַמַּנְדָּט הַבְּרִיטִי. בִּתְקוּפָה זוֹ נוֹסְדָה גַּם חֶבְרַת הַשְּׁמֵנִים "דלקול". בְּ-1988 בּוּצְּעָה רפורמה בַּשּׁוּק הַדֶּלֶק בָּאָרֶץ, והממשלה הפחיתה אֶת הַפִּיקּוּחַ עַל הַשּׁוּק בְּאוֹפֶן מַשְׁמָעוּתִי; בְּעִקְבוֹת זֹאת גָּבְרָה הַתַּחֲרוּת בֵּין שָׁלוֹשׁ חֲבֵרוֹת הַדֶּלֶק הַגְּדוֹלוֹת. בְּ-1998 רָכַשׁ אֶת הַשְּׁלִיטָה בַּחֶבְרָה אִישׁ הָעֲסָקִים יִצְחָק תְּשׁוּבָה, וְהַחֶבְרָה הָפְכָה לְחֵלֶק מִקְּבוּצַת דֶּלֶק שבשליטתו, חֶבְרַת אחזקות מהגדולות בְּיִשְׂרָאֵל. בִּתְחִילַּת שְׁנוֹת ה-2000 יָצְאָה הַחֶבְרָה עִם מיתוג חָדָשׁ וְהֵחֵלָה לִפְרוֹשׁ אֶת רֶשֶׁת חֲנוּיוֹת הַנּוֹחוּת "מנטה" בְּחֵלֶק מִתַּחֲנוֹת הַדֶּלֶק שֶׁלָּהּ. בְּיוּנִי 2007 זָכְתָה הַחֶבְרָה בְּמִכְרָז להפרטת מסופי הניפוק והאחסון שֶׁל פִּי גלילות בְּאַשְׁדוֹד, באר-שבע וִירוּשָׁלַיִם; הָעִסְקָה, בְּשֹׁוִוי 800 מִילְיוֹן ש"ח, אִפְשְׁרָה לַחֶבְרָה כּוֹשֶׁר אִחְסוּן שֶׁל 600 מִילְיוֹן לִיטֶר (40% מכושר הָאִחְסוּן שֶׁל מֶשֶׁק הַדֶּלֶק בָּאָרֶץ) וכושר נִיפּוּק המספיק לצריכת הַדֶּלֶק הַכּוֹלֶלֶת שֶׁל הַמְּדִינָה, דָּבָר שֶׁהֶעֱנִיק לַחֶבְרָה יִתְרוֹן מַשְׁמָעוּתִי בִּתְחוּם ייבוא הדלקים בָּאָרֶץ. בְּמַקְבִּיל רכשה חֶבְרַת "דֶּלֶק אֵירוֹפָּה", שֶׁבָּהּ שׁוּתָּפָהּ חֶבְרַת דֶּלֶק, 869 תחנות דֶּלֶק (כּוֹלֵל 400 חֲנוּיוֹת נוֹחוּת), בְּהוֹלַנְד, לוקסמבורג וּבֶלְגִּיָּה, מִידַּי חֶבְרַת הָאֵנֶרְגִּיָּה "שברון טקסקו". בְּאוֹגוּסְט 2007 הונפקו לָרִאשׁוֹנָה מְנָיוֹת הַחֶבְרָה בַּבּוּרְסָה לניירות עֵרֶךְ בְּתֵל אָבִיב, וְדֶלֶק הָפְכָה לַחֶבְרָה צִיבּוּרִית, וּבַהֶמְשֵׁךְ נִכְלְלָה בְּמַדַּד תֵּל אָבִיב 100. בְּאַפְּרִיל 2013 רכשה "דֶּלֶק פטרוליום", חֶבְרַת הַבַּת שֶׁל קְבוּצַת דֶּלֶק שדרכה הֶחֱזִיקָה זוֹ בְּחֶבְרַת "דֶּלֶק", אֶת יִתְרַת 12.1% הַמְּנָיוֹת שֶׁהָיוּ בִּידֵי הַצִּיבּוּר, וְדֶלֶק הָפְכָה לַחֶבְרָה פְּרָטִית בְּבַעֲלוּת מְלֵאָה שֶׁל דֶּלֶק פטרוליום, ומניות הַחֶבְרָה נמחקו מהבורסה בת"א[1]. כְּשָׁנָה לְאַחַר מִכֵּן מָחֲקָה קְבוּצַת דֶּלֶק אֶת חֶבְרַת דֶּלֶק פטרוליום כְּחֵלֶק מִתַּהֲלִיךְ "שיטוח פירמידה", כָּךְ שֶׁלְּאַחַר תַּהֲלִיךְ זֶה הָפְכָה דֶּלֶק לַחֶבְרָה בְּבַעֲלוּת מְלֵאָה וישירה שֶׁל קְבוּצַת דֶּלֶק[2]. דֶּגֶל הַחֶבְרָה בַּכְּנִיסָה לְכָל תחנות הַדֶּלֶק שֶׁל הַחֶבְרָה מונף דֶּגֶל הַחֶבְרָה. הַדֶּגֶל, הַקּוֹדֵם והנוכחי כְּאֶחָד, הוּא מְקוֹרִי בְּעִיצּוּבוֹ וּבְכָךְ בּוֹלֵט בְּאוֹפֶן חָרִיג מִשְּׁאָר הדגלים המסחריים בָּאָרֶץ [3].
