"דֶּלֶק" חֶבְרַת הַדֶּלֶק הַיִּשְׂרְאֵלִית
"דֶּלֶק" חֶבְרַת הַדֶּלֶק הַיִּשְׂרְאֵלִית בע"מ, הַיְּדוּעָה יוֹתֵר בְּשֵׁם דֶּלֶק, הִיא אַחַת מֵאַרְבַּע חֶבְרוֹת הַדֶּלֶק הַגְּדוֹלוֹת בְּיִשְׂרָאֵל. בְּבַעֲלוּתָהּ מֵאוֹת תַּחֲנוֹת דֶּלֶק בְּרַחֲבֵי הָאָרֶץ, בְּחֶלְקָן גַּם חֲנוּיוֹת נוֹחוֹת תַּחַת הַמּוּתָג "מִנָּטָה" וּמִתְחָמִים קִמְעוֹנָאִיִּים. בְּבַעֲלוּת הַחֶבְרָה מִסְפַּר חֲבָרוֹת-בָּנוֹת בִּתְחוּמֵי הָאֵנֶרְגִּיָּה וְהַלּוֹגִיסְטִיקָה, הַמְּסַפְּקוֹת שֵׁירוּתֵי שִׁינּוּעַ וְהוֹבָלָה שֶׁל דֶּלֶק, מַעֲרָכוֹת לוֹגִיסְטִיּוֹת, יִיצּוּר שְׁמָנִים וְחוֹמְרֵי אִיטּוּם, וְשֵׁירוּתֵי נִיפּוּק לָאוֹנִיּוֹת בִּנְמָלִים בְּאֶרֶץ וּבָעוֹלָם.
הִיסְטוֹרְיָה
הַחֶבְרָה הוּקְמָה עַל יְדֵי מֶמְשֶׁלֶת יִשְׂרָאֵל בְּדֵצֶמְבֶּר 1951, וּבְדוֹמֶה לְחֶבְרוֹת הָאֵנֶרְגִּיָּה הָאֲחֵרוֹת (פָּז וְסוֹנוֹל) הָיְיתָה נְתוּנָה בָּעֲשׂוֹרִים הָרִאשׁוֹנִים לְפִיקּוּחַ מֶמְשַׁלְתִּי. עַד 1960 הִתְרַחֲבָה הַחֶבְרָה וּפָרְשָׁה 44 תַּחֲנוֹת דֶּלֶק בְּרַחֲבֵי הָאָרֶץ; בְּמַקְבִּיל הִצְטַמְצֵם חֶלְקָן בַּשּׁוּק שֶׁל חֶבְרוֹת הַדֶּלֶק הַזָּרוֹת, שֶׁפָּעֲלוּ בָּאָרֶץ מֵאָז תְּקוּפַת הַמַּנְדָּט הַבְּרִיטִי. בִּתְקוּפָה זוֹ נוֹסְדָה גַּם חֶבְרַת הַשְּׁמָנִים "דִּלְקוֹל". בְּ-1988 בּוּצְּעָה רֵפוֹרְמָה בְּשׁוּק הַדֶּלֶק בָּאָרֶץ, וְהַמֶּמְשָׁלָה הִפְחִיתָה אֶת הַפִּיקּוּחַ עַל הַשּׁוּק בְּאוֹפֶן מַשְׁמָעוּתִי; בְּעִקְבוֹת זֹאת גָּבְרָה הַתַּחֲרוּת בֵּין שָׁלוֹשׁ חֶבְרוֹת הַדֶּלֶק הַגְּדוֹלוֹת.
בְּ-1998 רָכַשׁ אֶת הַשְּׁלִיטָה בַּחֶבְרָה אִישׁ הָעֲסָקִים יִצְחָק תְּשׁוּבָה, וְהַחֶבְרָה הָפְכָה לְחֵלֶק מִקְּבוּצַת דֶּלֶק שֶׁבְּשַׁלִּיטָתוֹ, חֶבְרַת אַחְזָקוֹת מֵהַגְּדוֹלוֹת בְּיִשְׂרָאֵל. בִּתְחִילַּת שְׁנוֹת הַ-2000 יָצְאָה הַחֶבְרָה עִם מִיתּוּג חָדָשׁ וְהֵחֵלָּה לִפְרוֹשׁ אֶת רֶשֶׁת חֲנוּיוֹת הַנּוֹחוּת "מִנָּטָה" בְּחֵלֶק מִתַּחֲנוֹת הַדֶּלֶק שֶׁלָּהּ. בְּיוּנִי 2007 זָכְתָה הַחֶבְרָה בְּמִכְרָז לְהַפְרָטַת מְסוֹפֵי הַנִּיפּוּק וְהָאִחְסוּן שֶׁל פִּי גְּלִילוֹת בְּאַשְׁדּוֹד, בְּאֵר-שֶׁבַע וִירוּשָׁלַיִם; הָעִסְקָה, בְּשֹׁוִוי 800 מִילְיוֹן ש"ח, אִפְשְׁרָה לַחֶבְרָה כּוֹשֶׁר אִחְסוּן שֶׁל 600 מִילְיוֹן לִיטֶר (40% מִכּוֹשֶׁר הָאִחְסוּן שֶׁל מֶשֶׁק הַדֶּלֶק בָּאָרֶץ) וְכוֹשֶׁר נִיפּוּק הַמַּסְפִּיק לִצְרִיכַת הַדֶּלֶק הַכּוֹלֶלֶת שֶׁל הַמְּדִינָה, דָּבָר שֶׁהֶעֱנִיק לַחֶבְרָה יִתְרוֹן מַשְׁמָעוּתִי בִּתְחוּם יִיבּוּא הַדְּלָקִים בָּאָרֶץ. בְּמַקְבִּיל רָכְשָׁה חֶבְרַת "דֶּלֶק אֵירוֹפָּה", שֶׁבָּהּ שׁוּתָּפָה חֶבְרַת דֶּלֶק, 869 תַּחֲנוֹת דֶּלֶק (כּוֹלֵל 400 חֲנוּיוֹת נוֹחוֹת), בְּהוֹלַנְד, לוּקְסֶמְבּוּרְג וּבֶלְגִּיָּה, מִיְּדֵי חֶבְרַת הָאֵנֶרְגִּיָּה "שִׁבְרוֹן טַקְסְקוֹ".
בְּאוֹגוּסְט 2007 הוּנְפְּקוּ לָרִאשׁוֹנָה מְנָיוֹת הַחֶבְרָה בַּבּוּרְסָה לִנְיָירוֹת עֵרֶךְ בְּתֵל אָבִיב, וְדֶלֶק הָפְכָה לְחֶבְרָה צִיבּוּרִית, וּבַהֶמְשֵׁךְ נִכְלְלָה בְּמַדַּד תֵּל אָבִיב 100.
בְּאַפְּרִיל 2013 רָכְשָׁה "דֶּלֶק פֶּטְרוֹלְיוּם", חֶבְרַת הַבַּת שֶׁל קְבוּצַת דֶּלֶק שֶׁדַּרְכָּהּ הֶחֱזִיקָה זוֹ בְּחֶבְרַת "דֶּלֶק", אֶת יִתְרַת 12.1% הַמְּנָיוֹת שֶׁהָיוּ בִּידֵי הַצִּיבּוּר, וְדֶלֶק הָפְכָה לְחֶבְרָה פְּרָטִית בְּבַעֲלוּת מְלֵאָה שֶׁל דֶּלֶק פֶּטְרוֹלְיוּם, וּמְנָיוֹת הַחֶבְרָה נִמְחֲקוּ מֵהַבּוּרְסָה בת"א[1].
כְּשָׁנָה לְאַחַר מִכֵּן מָחֲקָה קְבוּצַת דֶּלֶק אֶת חֶבְרַת דֶּלֶק פֶּטְרוֹלְיוּם כְּחֵלֶק מִתַּהֲלִיךְ "שִׁיטּוּחַ פִּירָמִידָה", כָּךְ שֶׁלְּאַחַר תַּהֲלִיךְ זֶה הָפְכָה דֶּלֶק לְחֶבְרָה בְּבַעֲלוּת מְלֵאָה וִישִׁירָה שֶׁל קְבוּצַת דֶּלֶק[2].
דֶּגֶל הַחֶבְרָה
בַּכְּנִיסָה לְכָל תַּחֲנוֹת הַדֶּלֶק שֶׁל הַחֶבְרָה מוּנָף דֶּגֶל הַחֶבְרָה. הַדֶּגֶל, הַקּוֹדֵם וְהַנּוֹכְחִי כְּאֶחָד, הוּא מְקוֹרִי בְּעִיצּוּבוֹ וּבְכָךְ בּוֹלֵט בְּאוֹפֶן חָרִיג מִשְּׁאַר הַדְּגָלִים הַמִּסְחָרִיִּים בָּאָרֶץ
[3].
