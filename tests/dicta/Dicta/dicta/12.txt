אֶלְדָּד פֶּלֶד
אֶלְדָּד פֶּלֶד הוּא קְצִין צַהַ"ל בְּמִילּוּאִים בְּדַרְגַּת אַלּוּף מִשְׁנֶה.
שֵׁירוּת צְבָאִי
עִם גִּיּוּסוֹ לְצַהַ"ל שׁוּבַּץ פֶּלֶד בַּחֲטִיבַת גִּבְעָתִי וְעָבַר מַסְלוּל הַכְשָׁרָה כְּלוֹחֵם, קוּרְס מִ"כִּים חִי"ר וְקוּרְס קְצִינֵי חִי"ר. עִם סִיּוּם הַקּוּרְס שִׁימֵּשׁ כִּמְפַקֵּד מַחְלָקָה וּפְלוּגָּה בַּחֲטִיבָה. בְּהֶמְשֵׁךְ שִׁימֵּשׁ כִּמְפַקֵּד פַּלְחָה"ן גִּבְעָתִי וְכִמְפַקֵּד סַיֶּירֶת גִּבְעָתִי. בַּשָּׁנִים 1999-2001 שִׁימֵּשׁ כִּמְפַקֵּד גְּדוּד שָׁקֵד וְלָקַח חֵלֶק בַּפְּעִילוּת הַמִּבְצָעִית כְּנֶגֶד הַטֵּרוֹר הַפָלַסְטִינִי בִּרְצוּעַת עַזָּה וּבִלְחִימָה בִּדְרוֹם לְבָנוֹן. לְאַחַר מִכֵּן שִׁימֵּשׁ כִּמְפַקֵּד בא"ח גִּבְעָתִי[1] וְכִסְגַן מְפַקֵּד הַחֲטִיבָה[2]. בַּשָּׁנִים 2006-2009 שִׁימֵּשׁ כִּמְפַקֵּד מַעֲרַךְ מַגָּל[3]. לְאַחַר שִׁחְרוּרוֹ פָּנָה לַעֲבוֹד כְּעוֹרֵךְ דִּין[4].
