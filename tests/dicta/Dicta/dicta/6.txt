אוֹזְבְּקִים
אוֹזְבְּקִים הֵם קְבוּצָה אֶתְנִית בְּמֶרְכַּז אַסְיָה. רוֹבֶה שֶׁל קְבוּצָה זוֹ נִמְצָא בְּאוּזְבֵּקִיסְטָן, אֲבָל נִיתָּן לִמְצוֹא אוֹזְבְּקִים גַּם בִּמְדִינוֹת אֲחֵרוֹת כְּגוֹן טָגִ'יקִיסְטָן וְאַפְגָּנִיסְטָן. הָאוּכְלוּסִיָּיה כּוּלָּהּ מוּנָּה בֵּין 28 לְ-30 מִילְיוֹן אִישׁ.
מְקוֹר הַשֵּׁם
מְקוֹר הַשֵּׁם 'אוֹזְבֵּקִים' לֹא יָדוּעַ, אֲבָל יֶשְׁנָן הַשְׁעָרוֹת בְּנוֹגֵעַ לְכָךְ. הַשְׁעָרָה אַחַת אוֹמֶרֶת שֶׁהֵם נִקְרְאוּ עַל שֵׁם אוּזְבַּג חָאן.[1] הַשְׁעָרָה נוֹסֶפֶת הִיא שֶׁמַּשְׁמָעוּת הַמִּילָּה הִיא "עַצְמָאוּת" בְּשָׂפָה טוּרְקִית עַתִּיקָה.
שָׂפָה
הַשָּׂפָה הָאוּזְבֵּקִית הִיא שָׂפָה טוּרְקִית. אוּזְבֵּקִית מוֹדֶרְנִית נִכְתֶּבֶת ב כַּמָּה סוּגִים שֶׁל אוֹתִיּוֹת: לָטִינִיּוֹת, קִירִילִיּוֹת וְעַרְבִיּוֹת. לְאַחַר שֶׁאוּזְבֵּקִיסְטָן קִיבְּלָה עַצְמָאוּת מִבְּרִית הַמּוֹעָצוֹת הֶחֱלִיטוּ לְהוֹרִיד אֶת הָאוֹפְּצִיָּה לִכְתּוֹב בְּאַלְפָאבִּיתִּית קִירִילִי.
דָּת
מַרְבִּית הָאוֹזְבְּקִים הֵם מוּסְלְמִים סוּנִים מֵהָאַסְכּוֹלָה הַחַנֵּפִית[2], אַךְ קַיָּימִים כָּאֵלֶּה שֶׁלֹּא מְשׁוּיָּכִים לָאִסְלָאם. הָאוֹזְבְּקִים בִּמְדִינוֹת שֶׁלֹּא מְעוֹדֶדֶת קִיצוֹנִיּוּת אִסְלָאמִית (כְּמוֹ אוּזְבֵּקִיסְטָן וְרוּסְיָה) לֹא קִיצוֹנִיִּים כְּלָל. לְעוּמַּת זֹאת, מוּסְלְמִים בִּמְדִינוֹת אִסְלָאמִיּוֹת קִיצוֹנִיּוֹת (כְּמוֹ אַפְגְּאָנִיסְטָן) יוֹתֵר קִיצוֹנִיִּים וְשׁוֹמְרִים בִּקְפִידוֹת עַל הַדָּת. בִּתְקוּפַת בְּרִית הַמּוֹעָצוֹת הַדָּת לֹא נִשְׁאֲרָה חֲשׁוּבָה וְהוּזְנְחָה, אַךְ כַּאֲשֶׁר בְּרִית הַמּוֹעָצוֹת הִתְפָּרְקָה הִתְרַחֲשָׁה תְּחִיָּיה לַדָּת. תַּהֲלִיךְ זֶה נִמְשָׁךְ עַד לְמַצָּב כַּיּוֹם, שֶׁבּוֹ 96.3% מֵאוּכְלוּסִיַּית אוּזְבֵּקִיסְטָן הֵם אוּזְבֵּקִים מוּסְלְמִים.[3]
