אוּזְבֵּקִיסְטָן בְּמִשְׂחָקִים הָאוֹלִימְפִּיִּים
אוּזְבֵּקִיסְטָן מִשְׁתַּתֶּפֶת בַּמִּשְׂחָקִים הָאוֹלִימְפִּיִּים מֵאָז אוֹלִימְפְּיָאדַת אַטְלַנְטָה (1996). עַד 1991 הָיְיתָה אוּזְבֵּקִיסְטָן חֵלֶק מִבְּרִית הַמּוֹעָצוֹת, וּסְפּוֹרְטָאֶיהָ הִתְחָרוּ בְּמִסְגֶּרֶת הַמִּשְׁלַחַת הַסּוֹבְיֵיטִית. בְּאוֹתָהּ שָׁנָה הִתְפָּרְקָה בְּרִית הַמּוֹעָצוֹת לְ-15 מְדִינוֹת נִפְרָדוֹת, וְ-12 מֵהֶן, בָּהֶן אוּזְבֵּקִיסְטָן, שָׁלְחוּ מִשְׁלַחַת מְאוּחֶדֶת לְאוֹלִימְפְּיָאדַת בַּרְצְלוֹנָה (1992). בְּאַרְבַּע הָאוֹלִימְפִּיָּאדוֹת בָּהֶן הִשְׁתַּתְּפָה אוּזְבֵּקִיסְטָן, צָבְרוּ הַסְּפּוֹרְטָאִים הָאוֹזְבְּקִים 32 מֵדַלְיוֹת, מֵהֶן 9 מֵדַלְיוֹת זָהָב. הָעֲנָפִים הַבּוֹלְטִים בְּיוֹתֵר הֵם הֵיאָבְקוּת וְאִיגְרוּף.
הִיסְטוֹרְיָה
הַמִּשְׁלַחַת הָעַצְמָאִית הָרִאשׁוֹנָה שֶׁל אוּזְבֵּקִיסְטָן, בְּאוֹלִימְפְּיָאדַת אַטְלַנְטָה, כָּלְלָה 71 סְפּוֹרְטָאִים. שְׁנַיִים מֵהֶם זָכוּ בְּמֵדַלְיוֹת הָרִאשׁוֹנוֹת בְּתוֹלְדוֹתֶיהָ: הַגּ'וּדוֹקָא אֶרְמֵן בַּגְדַּסְרוֹב זָכָה בְּמֵדַלְיַית כֶּסֶף וְהַמִּתְאַגְרֵף כָּרִים טוֹלְגְּנוֹב זָכָה בְּמֵדַלְיַית אָרָד. בְּאוֹלִימְפְּיָאדַת סִידְנִי (2000) נָטְלוּ חֵלֶק 70 סְפּוֹרְטָאִים אוֹזְבְּקִים. הַמִּתְאַגְרֵף מֶחְמַטְּקוֹדִיר עבדולאייב זָכָה בְּמֵדַלְיַית הַזָּהָב הָרִאשׁוֹנָה בְּתוֹלְדוֹת אוּזְבֵּקִיסְטָן (בְּמִשְׂחֲקֵי הַקַּיִץ). שְׁנֵי מִתְאַגְרְפִים נוֹסָפִים, סֶרְגֵיי מִיחָאִילוֹב וְרוֹסְטָם סאידוב, זָכוּ בְּמֵדַלְיוֹת אָרָד, וּמִתְאַגְרֵף רְבִיעִי, רוֹסְלָאן צ'גאייב, הֶעְפִּיל לְשַׁלֵּב רֶבַע גְּמָר. בְּהֵיאָבְקוּת בָּלַט אַרְתוּר טִאימְזוֹב, שֶׁזָּכָה בְּמֵדַלְיַית כֶּסֶף.
גַּם בְּאוֹלִימְפְּיָאדַת אָתוּנָה (2004) הִשְׁתַּתְּפוּ 70 סְפּוֹרְטָאִים אוֹזְבְּקִים. טִאימְזוֹב זָכָה בְּמֵדַלְיַית זָהָב, וְכָמוֹהוּ גַּם הַמִּתְאַבֵּק אָלֶכְּסַנְדֶּר דּוֹקְטוֹרְשְׁוִוילִי. מִתְאַבֵּק שְׁלִישִׁי, מְגוּמָּד אִיבְרָגִימוֹב זָכָה בְּמֵדַלְיַית כֶּסֶף. וְהַמִּתְאַגְרְפִים בְּחוּדִירְג'וֹן סוֹלְטוֹנוֹב וְאוֹטְקִירְבֵק חָיִידְרוֹב זָכוּ בְּמֵדַלְיוֹת אָרָד, וּשְׁלוֹשָׁה מִתְאַגְרְפִים אוֹזְבָּקִים נוֹסָפִים, טוֹלְאַשְׁבּוֹי דוניורוב, דִּילְשׁוֹד מחמודוב וְשֶׁרְזוֹד חוֹסָאנוֹב, הֶעְפִּילוּ לִשְׁלַב רֶבַע הַגְּמָר. בַּהֲרָמַת מִשְׁקוֹלוֹת בָּלֵט אָלֶכְּסַנְדֶּר אוֹרִינוֹב, שֶׁסִּייֵּם בַּמָּקוֹם הַשְּׁמִינִי. עַל הִשְׁתַּתְּפוּתָהּ שֶׁל אוּזְבֵּקִיסְטָן בְּמִשְׂחָקִים אֵלֶּה הֵעִיבָה פְּסִילָתָהּ שֶׁל מְיַידֶּה הַפַּטִּישׁ אוֹלְגָּה שֶׁצּ'וּקִינָה, בְּשֶׁל שִׁימּוּשׁ בְּסַמִּים.
לְאוֹלִימְפְּיָאדַת בֵּייגִ'ינְג (2008) שָׁלְחָה אוּזְבֵּקִיסְטָן מִשְׁלַחַת קְטַנָּה יוֹתֵר, בַּת 58 סְפּוֹרְטָאִים. הַמִּתְאַבֵּק אַרְתוּר טְאִימָזוֹב זָכָה בְּמֵדַלְיַית זָהָב שְׁנִיָּיה בִּרְצִיפוּת, וַעֲמִיתוֹ סוֹסְלָן טִיגַיְיב זָכָה בְּמֵדַלְיַית כֶּסֶף, אַךְ זוֹ נִשְׁלְלָה מִמֶּנּוּ בְּ-2016 בְּשֶׁל הַרְשָׁעָה בְּאֶמְרוֹץ. דִּילְשׁוֹד מַנְסוֹרוֹב סִייֵּם בַּמָּקוֹם הַחֲמִישִׁי וְקוֹרְבָּן קוֹרְבָּנוֹב בַּמָּקוֹם הַשְּׁבִיעִי. הַגּ'וּדוֹקָא עַבְדּוּלוֹ טַנְגְּרְיֶיב זָכָה בְּמֵדַלְיַית כֶּסֶף, בְּעוֹד שֶׁרִישּׁוּד סוֹבִירוֹב זָכָה בְּמֵדַלְיַית אָרָד. ג'וּדוֹקָא נוֹסָף, מִירֵאלִי שְׁרִיפּוֹב, סִייֵּם בַּמָּקוֹם הַחֲמִישִׁי. בְּנוֹסָף, שְׁנֵי מִתְעַמְּלִים אוֹזְבְּקִים זָכוּ לָרִאשׁוֹנָה בְּמֵדַלְיוֹת אָרָד - אַנְטוֹן פּוּקִין בְּתַרְגִּיל הַמַּקְבִּילִים וִיקְטָרִינָה חִילְּקוּ, שֶׁהָיְיתָה לַסְּפּוֹרְטָאִית הָאוּזְבֵּקִית הָרִאשׁוֹנָה, וְהַיְּחִידָה עַד כֹּה, שֶׁזּוֹכֶה בְּמֵדַלְיָה הָאוֹלִימְפִּית בְּמִשְׂחֲקֵי הַקַּיִץ - בִּטְרַמְפּוֹלִינָה. חוֹתֵר הֵקֵאנוּ וָאדִים מִנָּקוּב הָיָה קָרוֹב לִזְכִיָּיה רִאשׁוֹנָה בְּמֵדַלְיָה גַּם בְּעָנָף זֶה, אַךְ סִייֵּם בַּמָּקוֹם הָרְבִיעִי בְּמִקְצֶה הַסְּפְּרִינְט לְ-1,000 מֶטֶר. בַּעֲנַף הֲטַאקְוַונְדּוֹ בָּלַט אֶקְמָאַל אֵירֶגְשָׁב, שֶׁסִּייֵּם בַּמָּקוֹם הַחֲמִישִׁי. הַמִּתְאַגְרְפִים הָאוֹזְבְּקִים לֹא הִצְלִיחוּ בְּמִשְׂחָקִים אֵלּוּ לִזְכּוֹת בְּמֵדַלְיוֹת, אַךְ שְׁנַיִים מֵהֶם, דִּילְשׁוּד מֵחֲמוֹדוֹב וְאֶלְשׁוֹד רֵסוֹלוֹב, הֶעְפִּילוּ לִשְׁלַב רֶבַע הַגְּמָר.
בְּאוֹלִימְפְּיָאדַת לוֹנְדוֹן (2012) הִשְׁתַּתְּפוּ 54 סְפּוֹרְטָאִים אוֹזְבְּקִים. טִאימְזוֹב זָכָה בְּמֵדַלְיַית זָהָב שְׁלִישִׁית בִּרְצִיפוּת, שֶׁהָיְיתָה גַּם מֵדַלְיַית הַזָּהָב הַיְּחִידָה לִזְכוּתָהּ שֶׁל הַמִּשְׁלַחַת. מִתְאַבֵּק נוֹסָף, סוֹסְלָאן טִיגַיְיב, זָכָה בְּמֵדַלְיַית אָרָד, אַךְ זוֹ נִשְׁלְלָה מִמֶּנּוּ חוֹדָשִׁים סְפוּרִים לְאַחַר הַמִּשְׂחָקִים, לְאַחַר שֶׁנִּמְצָא כִּי נָטַל סַמִּים. שְׁנֵי סְפּוֹרְטָאִים אוֹזְבְּקִים נוֹסָפִים שֶׁזָּכוּ בְּמֵדַלְיוֹת הָיוּ הַגּ'וּדוֹקָא רִישּׁוּד סוֹבִירוֹב וְהַמִּתְאַגְרֵף אֵבוּס אֲטוּיַּב, שֶׁזָּכוּ בְּמֵדַלְיוֹת אָרָד. אַרְבָּעָה מִתְאַגְרְפִים נוֹסָפִים הוּדְּחוּ בִּשְׁלַב רֶבַע הַגְּמָר. סְפּוֹרְטָאִים אוֹזְבְּקִים נוֹסָפִים שֶׁבָּלְטוּ בְּלוֹנְדוֹן כָּלְלוּ אֶת חוֹתֵר הֵקֵאנוּ וָאדִים מִנָּקוּב (מָקוֹם רְבִיעִי בְּמִקְצֶה לְ-1000 מֶטְרִים), מִרְיָם הַמִּשְׁקוֹלוֹת רוֹסְלָן נוֹרוֹדִינוֹב (מָקוֹם הָרְבִיעִי), רוֹכֵב הָאוֹפַנַּיִים סֶרְגֵיי לגוטין (מָקוֹם חֲמִישִׁי בְּמֵרוֹץ אוֹפְנֵי כְּבִישׁ), הַגּ'וּדוֹקָא רַמְזִידִין (מָקוֹם חֲמִישִׁי), הַמִּתְאַבֵּק קוֹרְבָּן קורבנוב (מָקוֹם חֲמִישִׁי), מִרְיָם הַמִּשְׁקוֹלוֹת אִיווֹאן אַפַּרְמוֹב (מָקוֹם חֲמִישִׁי); הָאַתְלֵטִית סְבֶטְלָנָה רַדְזִיוִויל (מָקוֹם שְׁבִיעִי בִּקְפִיצָה לַגּוֹבַהּ) וְהַגּ'וּדוֹקָא דִּילְשׁוֹד צ'וּרְיָיב (מָקוֹם שְׁבִיעִי).
אוֹלִימְפְּיָאדַת רִיּוֹ דֶּה זָ'נֵיירוֹ (2016) הָיְיתָה הַמּוּצְלַחַת בְּיוֹתֵר בְּתוֹלְדוֹת אוּזְבֵּקִיסְטָן, עִם 13 מֵדַלְיוֹת, מֵהֶן 4 מֵדַלְיוֹת זָהָב, הֶישֵּׂג דּוֹמֶה לְהֶישֵּׂגֵי הַמִּשְׁלָחוֹת הָאוֹזְבְּקִיּוֹת בִּשְׁלוֹשׁ הָאוֹלִימְפִּיָּאדוֹת הַקּוֹדְמוֹת גַּם יַחַד. הָאִיגְרוּף הָיָה הֶעָנָף הַבּוֹלֵט בְּיוֹתֵר. הַמִּתְאַגְרְפִים הִתְחָרוּ בְּכָל עֶשֶׂר קָטֵגוֹרְיוֹת הַמִּשְׁקָל לִגְבָרִים, וּבְשֶׁבַע מֵהֶן זָכוּ בְּמֵדַלְיוֹת. חֶסְנְבּוֹי דּוֹסְמְטוֹב, שָׁאחוּבִיְּדִין זוירוב וּפַזְלִידִין גְּאַיְבַּנְזָארוֹב זָכוּ בְּמֵדַלְיוֹת זָהָב, שֶׁחָרָאם גְּיָאסוֹב וּבְקָטָמִיר מִלִּיקּוּזָיֶיב זָכוּ בְּמֵדַלְיוֹת כֶּסֶף, וּמוֹרוֹג'וֹן אַחְמְדָלִיְּיב וְרוֹסְטָם טוֹלְגְנוֹב זָכוּ בְּמֵדַלְיוֹת אָרָד. בְּנוֹסָף, סִייְּמוּ הוֹרְשִׁיד טוֹזִ'יבְּיָיב וּבַחוֹדֵיר זְ'לוֹלוֹב בַּמָּקוֹם הַחֲמִישִׁי. בְּמֵדַלְיַית הַזָּהָב הָרְבִיעִית לִזְכוּת הָאוֹזְבְּקִים זָכָה מִרְיָם הַמִּשְׁקוֹלוֹת רוֹסְלָאן נוֹרוֹדִינוֹב. אִיווֹאן אַפַּרְמוֹב סִייֵּם בַּמָּקוֹם הַחֲמִישִׁי וְרוֹסְטָם גֶ'נְגַּבַּיְיב בַּמָּקוֹם הַשִּׁישִּׁי. בְּהֵיאָבְקוּת זָכוּ שְׁלוֹשָׁה אוֹזְבֵּקִים בְּמֵדַלְיוֹת אָרָד: אִיחֶטְיוֹר נִבְרוֹזוֹב, מְגוּמָּד אִיבְרָגִימוֹב וְאַלְמוֹרְט טַסְמוֹרְדוֹב. בְּקָזוּד עבדורחמונוב סִייֵּם בַּמָּקוֹם הַחֲמִישִׁי. הַגּ'וּדוֹקָא רִישּׁוּד סוֹבִירוֹב זָכָה בְּמֵדַלְיַית אָרָד שְׁלִישִׁית בִּרְצִיפוּת, וְגַם דְּיוֹרְבַּק אוֹרוֹזְבּוֹיֶיְב זָכָה בְּמֵדַלְיַית אָרָד. עַבְדּוּלוֹ טֶנְגְּרַיְיב סִייֵּם בַּמָּקוֹם הַחֲמִישִׁי. סְפּוֹרְטָאִים אוֹזְבְּקִים נוֹסָפִים שֶׁבָּלְטוּ בְּרִיּוֹ דֶּה זָ'נֵיירוֹ כָּלְלוּ אֶת לוֹחֵם הֲטַאקְוַונְדּוֹ דְּמִיטְרִי שַׁוְּקֵין שֶׁסִּייֵּם בַּמָּקוֹם הַחֲמִישִׁי, אֶת חוֹתֵר הֵקֵאנוּ גְּרָאסִים קוֹצְ'נֵב שֶׁסִּייֵּם בַּמָּקוֹם הַשְּׁבִיעִי בַּמִּקְצֶה לְ-1000 מֶטֶר וּבַמָּקוֹם הַשְּׁמִינִי בְּמִקְצֶה הַזּוּגוֹת לְ-1000 מֶטֶר, יַחַד עִם סָרִיק מִירְבְּקוֹב, וְאֶת הַמִּתְעַמֶּלֶת הַוּוֹתִיקָה אוֹקְסָנָה צ'וֹסוֹבִיטִינָה בַּת הַ-41, שֶׁסִּייְּמָה בַּמָּקוֹם הַשְּׁבִיעִי בְּתַרְגִּיל הַקְּפִיצוֹת.
מִשְׂחֲקֵי הַחוֹרֶף
אוֹלִימְפְּיָאדַת הַחוֹרֶף הָרִאשׁוֹנָה בָּהּ הִשְׁתַּתְּפָה אוּזְבֵּקִיסְטָן הָיְיתָה אוֹלִימְפְּיָאדַת לִילְהָאמֶר (1994). הָיְיתָה זוֹ אוֹלִימְפְּיָאדַת הַחוֹרֶף הַיְּחִידָה עַד כֹּה בָּהּ זָכָה סְפּוֹרְטַאי אוּזְבֵּקִי בְּמֵדַלְיָה - לִינָה צֶ'רְיַאֲזוֹבָה זָכְתָה בְּמֵדַלְיַית זָהָב בִּסְקִי חוֹפְשִׁי. פְּרָט לְהֶישֵּׂג זֶה, הַהֶישֵּׂג הַגָּבוֹהַּ הַיָּחִיד הַנּוֹסָף שֶׁל הַמִּשְׁלַחַת הָאוּזְבֵּקִית נִרְשַׁם בְּאוֹלִימְפְּיָאדַת נַאגְנוֹ (1998), אָז דּוֹרְגָה הַמַּחְלִיקָה הָאוֹמָּנוּתִית טַטְיָאנָה מִלִּינִינָה בַּמָּקוֹם הַשְּׁמִינִי.
