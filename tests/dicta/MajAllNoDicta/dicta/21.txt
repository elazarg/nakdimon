אוּזְבֵּקִיסְטָן בַּמִּשְׂחָקִים האולימפיים אוּזְבֵּקִיסְטָן מִשְׁתַּתֶּפֶת בַּמִּשְׂחָקִים האולימפיים מֵאָז אוֹלִימְפִּיָּאדַת אַטְלַנְטָה (1996). עַד 1991 הָיְיתָה אוּזְבֵּקִיסְטָן חֵלֶק מִבְּרִית הַמּוֹעָצוֹת, וספורטאיה הִתְחָרוּ בְּמִסְגֶּרֶת הַמִּשְׁלַחַת הַסּוֹבְיֵיטִית. בְּאוֹתָהּ שָׁנָה הִתְפָּרְקָה בְּרִית הַמּוֹעָצוֹת ל-15 מְדִינוֹת נִפְרָדוֹת, וֹ-12 מֵהֶן, בָּהֶן אוּזְבֵּקִיסְטָן, שָׁלְחוּ מִשְׁלַחַת מְאוּחֶדֶת לְאוֹלִימְפְּיָאדַת בַּרְצְלוֹנָה (1992). בְּאַרְבַּע האולימפיאדות בָּהֶן הִשְׁתַּתְּפָה אוּזְבֵּקִיסְטָן, צָבְרוּ הַסְּפּוֹרְטָאִים הָאוּזְבָּקִים 32 מֵדַלְיוֹת, מֵהֶן 9 מֵדַלְיוֹת זָהָב. הָעֲנָפִים הַבּוֹלְטִים בְּיוֹתֵר הֵם הֵיאָבְקוּת ואיגרוף. הִיסְטוֹרְיָה הַמִּשְׁלַחַת הָעַצְמָאִית הָרִאשׁוֹנָה שֶׁל אוּזְבֵּקִיסְטָן, בְּאוֹלִימְפִּיָּאדַת אַטְלַנְטָה, כָּלְלָה 71 סְפּוֹרְטָאִים. שְׁנַיִים מֵהֶם זָכוּ בְּמֵדַלְיוֹת הָרִאשׁוֹנוֹת בְּתוֹלְדוֹתֶיהָ: הג'ודוקא ארמן בגדסרוב זָכָה בְּמֵדַלְיַית כֶּסֶף והמתאגרף כָּרִים טולגנוב זָכָה בְּמֵדַלְיַית אֵרֵד. בְּאוֹלִימְפִּיָּאדַת סִידְנִי (2000) נָטְלוּ חֵלֶק 70 סְפּוֹרְטָאִים אוּזְבֵּקִים. הַמִּתְאַגְרֵף מחמטקודיר עבדולאייב זָכָה בְּמֵדַלְיַית הַזָּהָב הָרִאשׁוֹנָה בְּתוֹלְדוֹת אוּזְבֵּקִיסְטָן (בְּמִשְׂחֲקֵי הַקַּיִץ). שְׁנֵי מִתְאַגְרְפִים נוֹסָפִים, סרגיי מיחאילוב וְרוּסְטַם סאידוב, זָכוּ בְּמֵדַלְיוֹת אֵרֵד, ומתאגרף רְבִיעִי, רוסלאן צ'גאייב, הֶעְפִּיל לְשַׁלֵּב רֶבַע גָּמַר. בְּהֵיאָבְקוּת בַּלָּט אַרְתוּר טאימזוב, שֶׁזָּכָה בְּמֵדַלְיַית כֶּסֶף. גַּם בְּאוֹלִימְפִּיָּאדַת אַתּוּנָה (2004) הִשְׁתַּתְּפוּ 70 סְפּוֹרְטָאִים אוּזְבֵּקִים. טאימזוב זָכָה בְּמֵדַלְיַית זָהָב, וְכָמוֹהוּ גַּם המתאבק אָלֶכְּסַנְדֶּר דוקטורשווילי. מִתְאַבֵּק שְׁלִישִׁי, מגומד איברגימוב זָכָה בְּמֵדַלְיַית כֶּסֶף. והמתאגרפים בחודירג'ון סולטונוב ואוטקירבק חיידרוב זָכוּ בְּמֵדַלְיוֹת אֵרֵד, וּשְׁלוֹשָׁה מִתְאַגְרְפִים אוּזְבֵּקִים נוֹסָפִים, טולאשבוי דוניורוב, דילשוד מחמודוב ושרזוד חוסאנוב, הֶעְפִּילוּ לְשַׁלֵּב רֶבַע הַגְּמָר. בַּהֲרָמַת מִשְׁקוֹלוֹת בַּלָּט אָלֶכְּסַנְדֶּר אורינוב, שֶׁסִּייֵּם בִּמְקוֹם הַשְּׁמִינִי. עַל השתתפותה שֶׁל אוּזְבֵּקִיסְטָן בַּמִּשְׂחָקִים אֵלֶּה הֵעִיבָה פסילתה שֶׁל מְיַידֶּה הַפַּטִּישׁ אוֹלְגָּה שצ'וקינה, בְּשֶׁל שִׁימּוּשׁ בְּסַמִּים. לְאוֹלִימְפְּיָאדַת בֶּיְיגִ'ינְג (2008) שָׁלְחָה אוּזְבֵּקִיסְטָן מִשְׁלַחַת קְטַנָּה יוֹתֵר, בַּת 58 סְפּוֹרְטָאִים. המתאבק אַרְתוּר טאימזוב זָכָה בְּמֵדַלְיַית זָהָב שְׁנִיָּיה בִּרְצִיפוּת, ועמיתו סוסלן טיגייב זָכָה בְּמֵדַלְיַית כֶּסֶף, אַךְ זוֹ נשללה מִמֶּנּוּ בְּ-2016 בְּשֶׁל הָרִשְׁעָה באמרוץ. דילשוד מנסורוב סִייֵּם בִּמְקוֹם הַחֲמִישִׁי וְקוֹרְבָּן קורבנוב בִּמְקוֹם הַשְּׁבִיעִי. הג'ודוקא עבדולו טנגרייב זָכָה בְּמֵדַלְיַית כֶּסֶף, בְּעוֹד שרישוד סובירוב זָכָה בְּמֵדַלְיַית אֵרֵד. ג'ודוקא נוֹסָף, מיראלי שריפוב, סִייֵּם בִּמְקוֹם הַחֲמִישִׁי. בְּנוֹסָף, שְׁנֵי מִתְעַמְלִים אוּזְבֵּקִים זָכוּ לָרִאשׁוֹנָה בְּמֵדַלְיוֹת אֵרֵד - אַנְטוֹן פוקין בְּתַרְגִּיל הַמַּקְבִּילִים ויקטרינה חִילְּקוּ, שֶׁהָיְיתָה לספורטאית הָאוּזְבֵּקִית הָרִאשׁוֹנָה, וְהַיְּחִידָה עַד כֹּה, שֶׁזּוֹכֶה במדליה הָאוֹלִימְפִּית בְּמִשְׂחֲקֵי הַקַּיִץ - בטרמפולינה. חוֹתֵר הֵקֵאנוּ וְאֵדִים מנקוב הָיָה קָרוֹב לִזְכִיָּיה רִאשׁוֹנָה במדליה גַּם בַּעֲנַף זֶה, אַךְ סִייֵּם בִּמְקוֹם הָרְבִיעִי בַּמֻּקְצֶה הספרינט ל-1,000 מֶטֶר. בַּעֲנַף הטאקוונדו בַּלָּט אקמאל אירגשב, שֶׁסִּייֵּם בִּמְקוֹם הַחֲמִישִׁי. הַמִּתְאַגְרְפִים הָאוּזְבָּקִים לֹא הִצְלִיחוּ בַּמִּשְׂחָקִים אֵלּוּ לִזְכּוֹת בְּמֵדַלְיוֹת, אַךְ שְׁנַיִים מֵהֶם, דילשוד מחמודוב ואלשוד רסולוב, הֶעְפִּילוּ לְשַׁלֵּב רֶבַע הַגְּמָר. בְּאוֹלִימְפִּיָּאדַת לוֹנְדוֹן (2012) הִשְׁתַּתְּפוּ 54 סְפּוֹרְטָאִים אוּזְבֵּקִים. טאימזוב זָכָה בְּמֵדַלְיַית זָהָב שְׁלִישִׁית בִּרְצִיפוּת, שֶׁהָיְיתָה גַּם מֵדַלְיַית הַזָּהָב הַיְּחִידָה לִזְכוּתָהּ שֶׁל הַמִּשְׁלַחַת. מִתְאַבֵּק נוֹסָף, סוסלאן טיגייב, זָכָה בְּמֵדַלְיַית אֵרֵד, אַךְ זוֹ נשללה מִמֶּנּוּ חוֹדָשִׁים סְפוּרִים לְאַחַר הַמִּשְׂחָקִים, לְאַחַר שֶׁנִּמְצָא כִּי נָטַל סַמִּים. שְׁנֵי סְפּוֹרְטָאִים אוּזְבֵּקִים נוֹסָפִים שֶׁזָּכוּ בְּמֵדַלְיוֹת הָיוּ הג'ודוקא רישוד סובירוב והמתאגרף אֵבוּס אטויב, שֶׁזָּכוּ בְּמֵדַלְיוֹת אֵרֵד. אַרְבָּעָה מִתְאַגְרְפִים נוֹסָפִים הוּדְּחוּ בְּשָׁלָב רֶבַע הַגְּמָר. סְפּוֹרְטָאִים אוּזְבֵּקִים נוֹסָפִים שֶׁבָּלְטוּ בְּלוֹנְדוֹן כְּלָלוֹ אֶת חוֹתֵר הֵקֵאנוּ וְאֵדִים מנקוב (מָקוֹם רְבִיעִי בַּמֻּקְצֶה ל-1000 מֶטְרִים), מִרְיָם הַמִּשְׁקוֹלוֹת רוסלן נורודינוב (מָקוֹם הָרְבִיעִי), רוֹכֵב הָאוֹפַנַּיִים סרגיי לגוטין (מָקוֹם חֲמִישִׁי בְּמֵרוֹץ אוֹפְנֵי כְּבִישׁ), הג'ודוקא רמזידין (מָקוֹם חֲמִישִׁי), המתאבק קוֹרְבָּן קורבנוב (מָקוֹם חֲמִישִׁי), מִרְיָם הַמִּשְׁקוֹלוֹת אִיוָואן אפרמוב (מָקוֹם חֲמִישִׁי); האתלטית סְבֶטְלָנָה רדזיוויל (מָקוֹם שְׁבִיעִי בִּקְפִיצָה לְגוֹבַהּ) והג'ודוקא דילשוד צ'ורייב (מָקוֹם שְׁבִיעִי). אוֹלִימְפִּיָּאדַת רִיּוֹ דֶּה ז'ניירו (2016) הָיְיתָה הַמּוּצְלַחַת בְּיוֹתֵר בְּתוֹלְדוֹת אוּזְבֵּקִיסְטָן, עִם 13 מֵדַלְיוֹת, מֵהֶן 4 מֵדַלְיוֹת זָהָב, הֶישֵּׂג דּוֹמֶה לְהֶישֵּׂגֵי הַמִּשְׁלָחוֹת הָאוּזְבָּקִיּוֹת בְּשָׁלוֹשׁ האולימפיאדות הַקּוֹדְמוֹת גַּם יַחַד. האיגרוף הָיָה הֶעָנָף הַבּוֹלֵט בְּיוֹתֵר. הַמִּתְאַגְרְפִים הִתְחָרוּ בְּכָל עָשָׂר קָטֵגוֹרְיוֹת הַמִּשְׁקָל לִגְבָרִים, וּבְשֶׁבַע מֵהֶן זָכוּ בְּמֵדַלְיוֹת. חסנבוי דוסמטוב, שאחובידין זוירוב ופזלידין גאיבנזארוב זָכוּ בְּמֵדַלְיוֹת זָהָב, שחראם גיאסוב ובקטמיר מליקוזייב זָכוּ בְּמֵדַלְיוֹת כֶּסֶף, ומורוג'ון אחמדלייב וְרוּסְטַם טולגנוב זָכוּ בְּמֵדַלְיוֹת אֵרֵד. בְּנוֹסָף, סִייְּמוּ הורשיד טוז'יבייב ובחודיר ז'לולוב בִּמְקוֹם הַחֲמִישִׁי. בְּמֵדַלְיַית הַזָּהָב הָרְבִיעִית לִזְכּוֹת הָאוּזְבָּקִים זָכָה מִרְיָם הַמִּשְׁקוֹלוֹת רוסלאן נורודינוב. אִיוָואן אפרמוב סִייֵּם בִּמְקוֹם הַחֲמִישִׁי וְרוּסְטַם ג'נגבייב בִּמְקוֹם הַשִּׁישִּׁי. בְּהֵיאָבְקוּת זָכוּ שְׁלוֹשָׁה אוּזְבֵּקִים בְּמֵדַלְיוֹת אֵרֵד: איחטיור נברוזוב, מגומד איברגימוב ואלמורט טסמורדוב. בקזוד עבדורחמונוב סִייֵּם בִּמְקוֹם הַחֲמִישִׁי. הג'ודוקא רישוד סובירוב זָכָה בְּמֵדַלְיַית אֵרֵד שְׁלִישִׁית בִּרְצִיפוּת, וְגַם דיורבק אורוזבוייב זָכָה בְּמֵדַלְיַית אֵרֵד. עבדולו טנגרייב סִייֵּם בִּמְקוֹם הַחֲמִישִׁי. סְפּוֹרְטָאִים אוּזְבֵּקִים נוֹסָפִים שֶׁבָּלְטוּ בָּרְיוֹ דֶּה ז'ניירו כְּלָלוֹ אֶת לוֹחֵם הטאקוונדו דְּמִיטְרִי שוקין שֶׁסִּייֵּם בִּמְקוֹם הַחֲמִישִׁי, אֶת חוֹתֵר הֵקֵאנוּ גראסים קוצ'נב שֶׁסִּייֵּם בִּמְקוֹם הַשְּׁבִיעִי בַּמֻּקְצֶה ל-1000 מֶטֶר וּבִמְקוֹם הַשְּׁמִינִי בַּמֻּקְצֶה הַזּוּגוֹת ל-1000 מֶטֶר, יַחַד עִם סריק מירבקוב, וְאֶת המתעמלת הַוּוֹתִיקָה אוקסנה צ'וסוביטינה בַּת ה-41, שֶׁסִּייְּמָה בִּמְקוֹם הַשְּׁבִיעִי בְּתַרְגִּיל הַקְּפִיצוֹת. מִשְׂחֲקֵי הַחוֹרֶף אוֹלִימְפִּיָּאדַת הַחוֹרֶף הָרִאשׁוֹנָה בָּהּ הִשְׁתַּתְּפָה אוּזְבֵּקִיסְטָן הָיְיתָה אוֹלִימְפִּיָּאדַת לילהאמר (1994). הָיְיתָה זוֹ אוֹלִימְפִּיָּאדַת הַחוֹרֶף הַיְּחִידָה עַד כֹּה בָּהּ זָכָה סְפּוֹרְטַאי אוּזְבֵּקִי במדליה - לִינָה צ'ריאזובה זָכְתָה בְּמֵדַלְיַית זָהָב בִּסְקִי חוֹפְשִׁי. פְּרָט לְהֶישֵּׂג זֶה, הַהֶישֵּׂג הַגָּבוֹהַּ הַיָּחִיד הַנּוֹסָף שֶׁל הַמִּשְׁלַחַת הָאוּזְבֵּקִית נִרְשַׁם בְּאוֹלִימְפִּיָּאדַת נאגנו (1998), אָז דורגה המחליקה האומנותית טַטְיָאנָה מלינינה בִּמְקוֹם הַשְּׁמִינִי. מֵדַלְיוֹת לְפִי אוֹלִימְפִּיָּאדָה באולימפיאדות הַחוֹרֶף: לְפִי עָנָף
