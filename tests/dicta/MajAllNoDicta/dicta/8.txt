אוֹרֶן מיילי אוֹרֶן מיילי (מְכוֹנָה אוֹרֶן הַקָּטָן; נוֹלַד בִּשְׁנַת 1978) הוּא קוֹסֵם וְאָמֵן טלפתיה יִשְׂרְאֵלִי, שֶׁהִתְפַּרְסֵם בְּיִשְׂרָאֵל בִּשְׁנוֹת ה-80 וּבִתְחִילַּת שְׁנוֹת ה-90 שֶׁל הַמֵּאָה ה-20. בִּיּוֹגְרַפְיָה מיילי נוֹלַד וְגָדַל בִּשְׁכוּנַת פַּרְדֵּס כַּץ בִּבְנֵי בָּרָק. הוּא הֵחֵל אֶת דַּרְכּוֹ בַּפִּינָּה שהוקצתה לוֹ אֵצֶל הַבַּדְּרָן מֵאִיר רֶמֶז. אֶת הוֹפָעוֹתָיו הָרִאשׁוֹנוֹת הֵחֵל בְּגִיל פָּחוֹת מַעֲשֵׂר, וּמִכָּאן כִּינּוּיוֹ "אוֹרֶן הַקָּטָן". בִּשְׁנוֹת ה-80 וּבִתְחִילַּת שְׁנוֹת ה-90 שֶׁל הַמֵּאָה ה-20 הוֹפִיעַ בַּמּוֹפָעִים וּבְתוֹכְנִיּוֹת טֵלֵוִויזְיָה יַחַד עִם אָבִיו, רוֹנִּי מיילי. בְּאֶמְצַע שְׁנוֹת הַתִּשְׁעִים עָזַב אֶת עוֹלָם הַבִּידּוּר וְהֵחֵל לְסַיֵּיעַ לְאָבִיו בְּעֵסֶק לְיִיצּוּר שִׁלְטֵי פִּרְסוּם. כַּיּוֹם הוּא מוֹפִיעַ לִפְנֵי וְעֵדֵי עוֹבְדִים בְּהַנְחָיַית אָבִיו. בְּ-2006 זוּמַּן לַמִּשְׁטָרָה, בִּגְלַל בִּלְבּוּל שֶׁהָיָה בֵּינוֹ לְבֵין אוֹרֶן זריף, בַּחֲשָׁד שֶׁהוֹנָה אֲנָשִׁים שֶׁבָּאוּ לְטִיפּוּל אֶצְלוֹ.[1] בְּ-2013 הִשְׁתַּתְּפוּ בַּתּוֹכְנִית "הָאָח הַגָּדוֹל" אָבִיו שֶׁל אוֹרֶן, רוֹנִּי מיילי, וְאָחִיו גִּילִי, שֶׁמּוֹפִיעִים לְעִיתִּים לְצִדּוֹ בְּהוֹפָעוֹתָיו.[2] מיילי נָשׂוּי וְאָב לִשְׁלוֹשָׁה יְלָדִים.
