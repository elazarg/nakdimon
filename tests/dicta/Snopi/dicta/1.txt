"דֶּלֶק" חֲבֵרַת הַדֶּלֶק הַיִּשְׂרְאֵלִית
"דֶּלֶק" חֲבֵרַת הַדֶּלֶק הַיִּשְׂרְאֵלִית בע"מ, הַיְּדוּעַה יוֹתֵר בַּשֵּׁם דָּלַק, הִיא אַחַת מֵאַרְבַּע חֶבְרוֹת הַדֶּלֶק הַגְּדוֹלוֹת בְּיִשְׂרָאֵל. בבעלותה מֵאוֹת תַּחֲנוֹת דֶּלֶק בְּרַחֲבֵי הָאָרֶץ, בְּחֶלְקָן גַּם חֲנוּיּוֹת נוֹחוֹת תַּחַת הַמּוּתָג "מנטה" ומתחמים קמעונאיים. בְּבַעֲלוּת הַחֶבְרָה מְסַפֵּר חֲבֵרוּת-בָּנוֹת בִּתְחוּמִי הָאֶנֶרְגְּיָה וְהַלּוֹגִיסְטִיקָה, הַמְּסַפְּקוֹת שֵׁירוּתִי שֶׁיָּנוּעַ וְהוֹבָלָה שֶׁל דֶּלֶק, מַעַרְכוֹת לוגיסטיות, יִיצּוּר שְׁמָנִים וְחוֹמְרֵי אִיטּוּם, וְשֵׁירוּתִי ניפוק לָאֳונִיּוֹת בִּנְמָלִים בָּאָרֶץ וּבָעוֹלָם.
הִיסְטוֹרְיָה
הַחֶבְרָה הוּקְמָה עַל יְדֵי מֶמְשֶׁלֶת יִשְׂרָאֵל בְּדֵצֶמְבֶּר 1951, וּבְדוֹמֶה לַחֲבֵרוּת הָאֶנֶרְגְּיָה הָאֲחֵרוֹת (פָּז וסונול) הָיְיתָה נתונה בֶּעָשׂוֹרִים הָרִאשׁוֹנִים לְפִיקּוּחַ מֶמְשַׁלְתִּי. עַד 1960 הִתְרַחֲבָה הַחֶבְרָה וּפָרְשָׂה 44 תַּחֲנוֹת דָּלַק בְּרַחֲבֵי הָאָרֶץ; בַּמַּקְבִּיל הִצְטַמְצֵם חֶלְקָן בַּשּׁוּק שֶׁל חֶבְרוֹת הַדֶּלֶק הַזָּרוֹת, שֶׁפָּעֲלוּ בָּאָרֶץ מֵאָז תְּקוּפַת הַמַּנְדָּט הַבְּרִיטִי. בִּתְקוּפָה זוֹ נוֹסְדָה גַּם חֲבֵרַת הַשְּׁמָנִים "דלקול". ב-1988 בּוּצְּעָה רֵפוֹרְמָה בַּשּׁוּק הַדֶּלֶק בָּאָרֶץ, וְהַמֶּמְשָׁלָה הִפְחִיתָה אֶת הַפִּיקּוּחַ עַל הַשּׁוּק בְּאוֹפֶן מַשְׁמָעוּתִי; בָּעֲקֵבוֹת זֹאת גָּבְרָה הִתְחָרוּת בֵּין שָׁלוֹשׁ חֶבְרוֹת הַדֶּלֶק הַגְּדוֹלוֹת.
ב-1998 רָכַשׁ אֶת הַשְּׁלִיטָה בַּחֶבְרָה אִישׁ הָעֲסָקִים יִצְחַק תְּשׁוּבָה, וְהַחֶבְרָה הָפְכָה לַחֵלֶק מִקְּבוּצַת דָּלַק שבשליטתו, חָבַרְתָּ אַחְזָקוֹת מֵהַגְּדוֹלוֹת בְּיִשְׂרָאֵל. בִּתְחִילַּת שְׁנוֹת ה-2000 יָצְאָה הַחֶבְרָה עִם מיתוג חָדָשׁ וְהֵחֵלָּה לַפָּרוּשׂ אֶת רֶשֶׁת חֲנוּיּוֹת הַנּוֹחוּת "מנטה" בַּחֵלֶק מִתַּחֲנוֹת הַדֶּלֶק שֶׁלָּהּ. בְּיוֹנִי 2007 זָכְתָה הַחֶבְרָה בַּמִּכְרָז לְהַפְרָטַת מִסּוֹפִי הניפוק וְהָאִחְסוּן שֶׁל פִּי גלילות בְּאַשְׁדּוֹד, בְּאֵר-שָׂבֵעַ וִירוּשָׁלַיִם; הַעֲסָקָה, בַּשֹּׁוִוי 800 מִילְיוֹן שָׂ"ח, אִפְשְׁרָה לְחֶבְרָה כּוֹשֶׁר אִחְסוּן שֶׁל 600 מִילְיוֹן לִיטֶר (40% מִכּוֹשֶׁר הָאִחְסוּן שֶׁל מֶשֶׁק הַדֶּלֶק בָּאָרֶץ) וְכוֹשֶׁר ניפוק הַמַּסְפִּיק לִצְרִיכַת הַדֶּלֶק הַכּוֹלֶלֶת שֶׁל הַמְּדִינָה, דָּבָר שֶׁהֶעֱנִיק לְחֶבְרָה יִתְרוֹן מַשְׁמָעוּתִי בַּתְּחוּם ייבוא הדלקים בָּאָרֶץ. בַּמַּקְבִּיל רָכְשָׁה חָבַרְתָּ "דֶּלֶק אֵירוֹפָּה", שֶׁבָּהּ שׁוּתָּפָה חֲבֵרַת דֶּלֶק, 869 תַּחֲנוֹת דֶּלֶק (כּוֹלֵל 400 חֲנוּיּוֹת נוֹחוֹת), בְּהוֹלַנְד, לוקסמבורג וּבֶלְגִיָה, מִידֵּי חֲבֵרַת הָאֶנֶרְגְּיָה "שֶׁבָּרוֹן טקסקו".
בְּאוֹגוּסְט 2007 הונפקו לְרִאשׁוֹנָה מְנָיוֹת הַחֶבְרָה בַּבּוּרְסָה לִנְיָירוֹת עֵרֶךְ בְּתֵל אָבִיב, וְדֶלֶק הָפְכָה לְחֶבְרָה צִיבּוּרִית, וּבַהֶמְשֵׁךְ נִכְלָלָה בְּמַדָּד תֵּל אָבִיב 100.
בְּאַפְּרִיל 2013 רָכְשָׁה "דֶּלֶק פטרוליום", חֲבֵרַת הַבַּת שֶׁל קְבוּצַת דָּלַק שֶׁדַּרְכָּהּ הֶחֱזִיקָה זוֹ בַּחֲבֵרַת "דֶּלֶק", אֶת יתרת 12.1% הַמְּנָיוֹת שֶׁהָיוּ בִּידֵי הַצִּיבּוּר, וְדֶלֶק הָפְכָה לְחֶבְרָה פְּרָטִית בְּבַעֲלוּת מְלֵאָה שֶׁל דֶּלֶק פטרוליום, וּמְנָיוֹת הַחֶבְרָה נִמְחֲקוּ מֵהַבּוּרְסָה בת"א[1].
כְּשָׁנָה לְאַחַר מִכֶּן מְחַקֶּה קְבוּצַת דָּלַק אֶת חֲבֵרַת דֶּלֶק פטרוליום כְּחֵלֶק מִתַּהֲלִיךְ "שיטוח פִּירָמִידָה", כָּךְ שֶׁלְּאַחֵר תַּהֲלִיךְ זֶה הָפְכָה דֶּלֶק לְחֶבְרָה בְּבַעֲלוּת מִלֵּאָה וִישִׁירָה שֶׁל קְבוּצַת דֶּלֶק[2].
דֶּגֶל הַחֶבְרָה
בַּכְּנִיסָה לְכָל תַּחֲנוֹת הַדֶּלֶק שֶׁל הַחֶבְרָה מוּנָף דֶּגֶל הַחֶבְרָה. הַדֶּגֶל, הַקּוֹדֶם וְהַנּוֹכְחִי כְּאֶחָד, הוּא מְקוֹרִי בעיצובו וּבַכָּךְ בּוֹלֵט בְּאוֹפֶן חָרִיג מִשְּׁאַר הַדְּגָלִים הַמִּסְחָרִיִּים בָּאָרֶץ
[3].
