אַיְּירִין לְדַרְמֵן
אַיְּירִין לְדַרְמָן הִיא אֶתְלֵטִית יְהוּדִיָּיה-בְּרִיטִית בֶּעָבָר, אַלּוּפָה וְשִׂיאָנִית הַמְּכַבִּיָה בַּעֲלַת 6 מֶדַלְיוֹת, מֵהֶן 4 מֵדַלְיוֹת זָהָב (הַשְּׁלִישִׁית 1950 וְהָרְבִיעִית 1953) בְּרִיצַת 100 מֶטֶר, בְּרִיצַת 200 מֶטֶר וּבִקְפִיצָה לְרוֹחֵק.
בִּיּוֹגְרַפְיָה
לְדַרְמָן גָּדְלָה בְּלוֹנְדוֹן.
אַלּוּפָה וְשִׂיאָנִית הַמְּכַבִּיָה
בְּתַחֲרוּיוֹת הָאַתְלֵטִיקָה שֶׁל הַמֶּכַבִּיָה הַשְּׁלִישִׁית שֶׁנֶּעֶרְכוּ בִּשְׁנַת 1950 זָכְתָה בְּ-3 מֶדַלְיוֹת, אַחַת מִכָּל סוּג. מֵדַלְיַית זָהָב, בְּרִיצַת 100 מֶטֶר בְּשִׂיא מְכַּבִּיָהָ חָדָשׁ שֶׁל 12.8 שְׁנִיּוֹת, לִפְנֵי עֶדְנָה פְרִידְמָן מִדְּרוֹם אַפְרִיקָה (12.8 שְׁנִיּוֹת) וְלְאָה הוֹרְבִיץ מִיִּשְׂרָאֵל (13.3 שְׁנִיּוֹת), מֵדַלְיַית כֶּסֶף, בְּרִיצַת 60 מֶטֶר בְּתוֹצָאָה שֶׁל 8.3 שְׁנִיּוֹת, אַחֲרֵי עֶדְנָה פְרִידְמָן 8.2 שְׁנִיּוֹת וְלִפְנֵי וָן דָּר שֶׁטָּאם מְהוֹלַנְד (8.3 שְׁנִיּוֹת) וּמֵדַלְיַית אָרָד, בְּרִיצַת 80 מֶטֶר מְשׂוּכוֹת בְּתוֹצָאָה שֶׁל 15.6 שְׁנִיּוֹת, אַחֲרֵי רוּת אַיְילוֹן (13.8 שְׁנִיּוֹת) וְשָׁרָה מִיזְרְחִי (15.4 שְׁנִיּוֹת), שְׁתֵּיהֶן מִיִּשְׂרָאֵל.[1][2]
כּוֹכֶבֶת הַמֶּכַבִּיָה הָרְבִיעִית
בְּתַחֲרוּיוֹת הָאַתְלֵטִיקָה שֶׁל הַמֶּכַבִּיָה הָרְבִיעִית שֶׁנֶּעֶרְכוּ בִּשְׁנַת 1953 זָכְתָה פְרִידְמָן בְּ-3 מֵדַלְיוֹת זָהָב יוֹתֵר מִכָּל סְפּוֹרְטָאי אַחֵר, בְּרִיצַת 100 מֶטֶר בְּתוֹצָאָה שֶׁל 12.7 שְׁנִיּוֹת, לִפְנֵי רוֹדָה אַיְיבֵל מִדְּרוֹם אַפְרִיקָה (13.0 שְׁנִיּוֹת) וְדָלִית קְלַיְינְמָן מִיִּשְׂרָאֵל (13.2 שְׁנִיּוֹת) וּבְכָךְ הֲגָנָה עַל תּוֹאֲרָה, בְּרִיצַת 200 מֶטֶר בְּתוֹצָאָה שֶׁל 26.7 שְׁנִיּוֹת, לִפְנֵי רוֹדָה אַיְיבֵל (27.3 שְׁנִיּוֹת) וְרוּת הָרַשְׁלָר מִיִּשְׂרָאֵל (27.7 שְׁנִיּוֹת) וּבִקְּפִיצָה לְרוֹחָק בְּשִׂיא מְכַּבִּיָהָ חָדָשׁ שֶׁל 5.36 מֶטֶר, לִפְנֵי תָּמָר מֵיטַל-שׁוּמְכָּר (4.87 מֶטֶר) וּמֵרִים טֵפֶר (4.86 מֶטֶר), שְׁתֵּיהֶן מִיִּשְׂרָאֵל.[3]
