הָעֶרֶב שֶׁלִּי נֶהֱרַס.
שְׁנָתִי נְדַדֶּה.
אִילּוּ הִתְרַחֵשׁ הַדָּבָר בְּפָרִיז, הוּא הָיָה מַדְאִיג אוֹתִי בָּהּ בְּמִידָּה אֵבֶל מְזַעְזֵעַ פָּחוֹת.
הֶחְרִידָה אוֹתִי הָעוּבְדָּה כִּי הַשִּׂנְאָה הִגִּיעָה אֶל עָרֵי הַשָּׂדֶה, אֵל פשוטי הֵעֵם הַתְּמִימִים וְטוֹבֵי הַלֵּב, וְנִנְעְצָה עָמוֹק בְּלִבָּם.
בַּיָּמִים הַבָּאִים סִיפַּרְתִּי אֶת האפיזודה לַיְּדִידִים.
רוּבָּם לֹא הִתְיַיחֲסוּ אֵלֶיהָ בְּכוֹבֶד רֹאשׁ: "כַּמָּה לָעַגְנוּ אֲנַחְנוּ, הַצָּרְפָתִים, לְמַלְכָּה הַשְׁמָנָה ויקטוריה, וְכַעֲבוֹר שְׁנָתַיִים כָּרַתְנוּ בְּרִית עַם אַנְגְּלִיָּה.
אֵינְךָ מַכִּיר אֶת הַצָּרְפָתִים, הַפּוֹלִיטִיקָה אֵינָהּ חוֹדֶרֶת עָמוֹק לְנַפְשָׁם." רַק רולאן רָאָה זֹאת בְּעֵינַיִים אֲחֵרוֹת: "כְּכָל שֶׁהָעָם תָּם יוֹתֵר, כֵּן קַל יוֹתֵר לרמותו.
מֵאָז נִבְחַר פואנקרה, הורע הַמַּצָּב.
הַמַּסָּע שֶׁלּוֹ לפטרבורג לֹא יִהְיֶה מַסַּע תַּעֲנוּגוֹת." הֶאֱרַכְנוּ בַּשִּׂיחָה עַל קוֹנְגְּרֶס הסוציאליסטים הַבֵּינְלְאוּמִּי שֶׁיִּתְכַּנֵּס בַּקַּיִץ בווינה, אֲבָל גַּם לגביו הָיָה רולאן סַפְקָן מֵהָאֲחֵרִים: "כַּמָּה מֵהֵם יַחֲזִיקוּ מַעֲמָד כְּשֶׁיּוּדְבְּקוּ צַוִּוי הַגִּיּוּס, מִי יוֹדֵעַ? אָנוּ חַיִּים בַּיָּמִים שֶׁל תְּחוּשׁוֹת הַמּוֹנִיוֹת, שֶׁל הִיסְטֶרִיּוֹת הֲמוֹנִיּוֹת, שֶׁאֵין לַחֲזוֹת כָּלַל מֵרֹאשׁ אֶת עוצמתן בִּימֵי מִלְחָמָה."
