יוּזְכַּר כִּי הערעורים הַנִּדּוֹנִים נוֹגְעִים לחיובים בְּכֶפֶל מַס עַל פִּי סְעִיף 50(א) לְחוֹק מַס עָרַךְ מוּסָף לְגַבִּי שְׁנֵי הַיְּחִידִים שבירו וְשֶׁפִּיר, וְכֵן שׁוּמוֹת תשומות וּקְנָסוֹת עַל אִי נִיהוּל סַפָּרִים לְגַבֵּי שְׁנֵי הַתַּאֲגִידִים.
בַּעֲקֵבוֹת הַכְרָעַת הַדִּין שֶׁנִּיתְּנָה עַל יְדֵי בֵּית מִשְׁפָּט הַשָּׁלוֹם בְּתֵל אָבִיב-יִפּוּ בת"פ 13-03-7455 (וְכֵן דחיית הערעורים עַל הַכְרָעַת הַדִּין בְּהַמְלָצַת בֵּית הַמִּשְׁפָּט הַמְּחוֹזִי בְּתֵל-אָבִיב-יִפּוּ בע"פ 15-09-23079) בִּיקֵּשׁ הַמֵּשִׁיב כִּי הַמִּמְצָאִים וְהַמַּסְקָנוֹת בִּפְסַק הַדִּין הַפְּלִילִי יִהְיוּ קְבִילִים כִּרְאָיוֹת לִכְאוֹרָה לָאָמוּר בָּהֶם בַּמִּסְגֶּרֶת הערעורים דנן, וְזֹאת מִכּוֹחַ סְעִיף 42א(א) לִפְקוּדַּת הָרְאָיוֹת [נוֹסַח חָדָשׁ], התשל"א-1971.
כְּמוֹ כֵן בִּיקֵּשׁ הַמֵּשִׁיב כִּי "הערעורים.... יִדְחוּ בַּסּוּגְיוֹת שהוכרעו בַּהֲלִיךְ הַפְּלִילִי" (סְעִיף 19(ב) לַבַּקָּשָׁה).
יֵשׁ לְצַיֵּין כִּי הַמֵּשִׁיב עָתַר לִפְסַק דִּין "חֶלְקִי" בִּלְבַד כִּי הֲרֵי לֹא קִיַּימְתָּ חֲפִיפָה מְלֵאָה בֵּין חשבוניות הַמַּס שֶׁנִּדּוֹנּוּ בַּהֲלִיךְ הַפְּלִילִי לְבֵין אֵלּוּ שֶׁהֵן מושא עִרְעוּרִי הַמַּס הָאֶזְרָחִיִּים.
לְעִנְיָין אַחֲרוֹן זֶה אָשׁוּב בַּהֶמְשֵׁךְ.
אֶתְיַיחֵס רָאשִׁית לחיובים שֶׁהוּטְּלוּ עַל הַיְּחִידִים לְפִי סְעִיף 50(א) לְחוֹק בַּקֶּשֶׁר לְהוֹצָאַת חשבוניות שֶׁלֹּא כְּדִין.
הָאִישּׁוּם הַפְּלִילִי בְּנוֹשֵׂא זֶה (לְאַחֵר צִמְצוּמוֹ) הִתְיַיחֵס ל-304 חשבוניות מַס אֲשֶׁר נִמְסְרוּ לְיָדִי שְׁלוֹשָׁה עָשָׂר גּוֹרְמִים שׁוֹנִים (ה"המקזזים" כִּלְשׁוֹן הַכְרָעַת הַדִּין), כַּאֲשֶׁר סְכוּם הַמָּעָ"מ הַכּוֹלֵל הַגָּלוּם בָּהֶן מַגִּיעַ לִכְדִי 8.13 מִילְיוֹן שָׂ"ח.
