אֲנִי בְּמַצָּב דּוֹמֶה..
טִיפּוּלִים לְיֶלֶד שְׁלִישִׁי
הַטִּיפּוּלִים לְיֶלֶד שְׁלִישִׁי לֹא נִכְלָלִים בְּסַל הַבְּרִיאוּת וְלָכֵן הָעֲלוּיוֹת תְּלוּיוֹת בקופ"ח שֶׁלָּךְ.
אֲנִי בַּכְּלָלִית וּמְטוּפֶּלֶת בְּאִיכִילוֹב.
אֲנִי מַנִּיחָה שֶׁתִּצְטָרְכִי לִפְתּוֹחַ תִּיק בְּכָל מָקוֹם שֶׁתֵּלְכִי אֵלָיו, עָדִיף לַעֲשׂוֹת אֶת כָּל הַבְּדִיקוֹת מֵרֹאשׁ (פְּרוֹפִיל הוֹרְמוֹנָלִי, מַחֲלוֹת מִין, כִירוּרְג שָׁד וְכוּ...)
בד"כ תּוֹר הוּא לִשְׁבוּעַיִים עַד חוֹדֶשׁ קָדִימָה וְלִפְעָמִים מִתְפַּנֶּה מַשֶּׁהוּ מוּקְדָּם יוֹתֵר..
לִפְגִישָׁה עִם הָרוֹפֵא כְּדֵי לְקַבֵּל פְּרוֹטוֹקוֹל אֶפְשָׁר לְנַסּוֹת לְקַבֵּל טוֹפֶס 17 אֲבָל בד"כ לֹא מְאַשְּׁרִים, אָז אֲנִי מְשַׁלֶּמֶת וַאֲנַסֶּה לְקַבֵּל הֶחְזֵרִים מֵהַמּוּשְׁלָם.
אַחֲרֵי שֶׁמְּקַבְּלִים פְּרוֹטוֹקוֹל צָרִיךְ לְקַבֵּל אִישּׁוּר מֵרֹאשׁ מֵהַבִּיטּוּחַ הַמַּשְׁלִים (כְּלָלִית מוּשְׁלָם) וְרַק אַחֲרֵי זֶה אֶפְשָׁר לְהַתְחִיל.
כְּשֶׁיֵּשׁ פְּרוֹטוֹקוֹל זֶה לוֹקֵחַ כַּמָּה יָמִים וְלֹא יוֹתֵר מִזֶּה.
מַחְזוֹר טִיפּוּל כּוֹלֵל הַבְּדִיקוֹת מַעֲקָב וְהַשְּׁאִיבָה עוֹלָה 1650 ש"ח לֹא כּוֹלֵל תְּרוּפוֹת (שֶׁיְּכוֹלוֹת לְהַגִּיעַ לְמֵעַל 1000 ש"ח תָּלוּי בְּאוֹרֶךְ הַטִּיפּוּל וּבַתְּרוּפָה).
הַחֲזָרָה שֶׁל מוּקְפָּאִים עוֹלֶה 330 שח.
מְקַוָוה שֶׁעָזַרְתִּי, בְּכָל מִקְרֶה, לְכָל מְכוֹן ivf יֵשׁ בַּאֲתַר אִינְטֶרְנֶט פֵּירוּט שֶׁל כָּל הַטְּפָסִים וְהַבְּדִיקוֹת הַנִּדְרָשׁוֹת וַאֲנִי בְּטוּחָה שֶׁבַּטֵּלֵפוֹן יַעֲנוּ לְךָ בְּסַבְלָנוּת לְכָל שְׁאֵלָה..
