1. זֶה נָכוֹן בְּאוֹפֶן כְּלָלִי. חִידּוּד לְגַבֵּי קַבָּלוֹת - קַבָּלָה אַתָּה מוֹצִיא רַק כַּאֲשֶׁר אַתָּה מְקַבֵּל כֶּסֶף (מְזוּמָּן, צֵ'ק, הַעֲבָרָה וְכוּ') וְעַל הַסְּכוּם שֶׁאַתָּה מְקַבֵּל.
2. אִי אֶפְשָׁר לַעֲשׂוֹת אֶת זֶה בְּאֶקְסֵל אוֹ מַשֶּׁהוּ כָּזֶה. עִם זֹאת זֶה לֹא חַיָּיב לִהְיוֹת עִם פִּנְקָס. יֵשׁ כָּל מִינֵי תּוֹכְנוֹת שֶׁמְּאוּשָּׁרוֹת עַל יְדֵי מַס הַכְנָסָה שֶׁאֶפְשָׁר לְהָפִיק אִיתָּן אֶת הַמִּסְמָכִים וּלְנַהֵל אֶת הַכֹּל דַּרְכָּן אִם זֶה יוֹתֵר נוֹחַ לָךְ.
3. לֹא הֶגְיוֹנִי. זֶה צָרִיךְ לִהְיוֹת תּוֹאֵם (לְמַעֵט הֶפְרֵשִׁים שֶׁנּוֹבְעִים מִשִּׁינּוּיִים בְּשַׁעַר הַדּוֹלָר - מִתְקַשֵּׁר לִסְעִיף 5).
אַתָּה צָרִיךְ לְהוֹצִיא הַזְמָנָה וְחֶשְׁבּוֹן עִסְקָה לְפִי מָה שֶׁאַתָּה אָמוּר לְקַבֵּל - לֹא לְפִי מָה שֶׁעֵסֶק ב' אָמוּר לְקַבֵּל.
4. צָרִיךְ גַּם סֵפֶר תִּקְבּוּלִים תַּשְׁלוּמִים (לְרִישּׁוּם מְרוּכָּז שֶׁל קַבָּלוֹת וְהוֹצָאוֹת).
אִם יֵשׁ לְךָ מְלַאי אָז גַּם סֵפֶר תְּנוּעוֹת מְלַאי.
5. הַקַּבָּלָה בְּשַׁ"ח כַּמּוּבָן לְפִי מָה שֶׁאַתָּה מְקַבֵּל.
הַשְּׁאָר הָיִיתִי מוֹצִיא בְּדוֹלָרִים כִּי אִי אֶפְשָׁר לָדַעַת מָה יִהְיֶה הַשַּׁעַר כְּשֶׁתְּקַבֵּל אֶת הַכֶּסֶף.
בְּהַצְלָחָה.
