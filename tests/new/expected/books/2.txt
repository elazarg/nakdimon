אֲנִי נִכְנָס לַחֶדֶר וּמִתְיַישֵּׁב עַל הַמִּיטָּה וּפוֹתֵחַ אֶת קוּפְסַת הַצְּבָעִים וְאֶת חוֹבֶרֶת הַצְּבִיעָה וּמוֹצִיא אֶת הָאָדוֹם הַמְּסַחְרֵר מֵהַקּוּפְסָה וּמַתְחִיל לִקְרוֹא.
בִּתְחִילַּת הַחוֹבֶרֶת מְסוּפָּר הַסִּיפּוּר שֶׁל ג'וֹ.
ג'וֹ הוּא שַׁתְיָין שֶׁמְּאַבֵּד אֶת אִשְׁתּוֹ, אֶת הָעֲבוֹדָה שֶׁלּוֹ, אֶת כָּל כַּסְפּוֹ, וּבְסוֹפוֹ שֶׁל דָּבָר מַגִּיעַ לָרְחוֹב וְשָׁם הוּא שׁוֹתֶה יַיִן זוֹל יָשָׁר מֵהַבַּקְבּוּק.
לְאוֹרֶךְ כָּל הַתְּקוּפָה הַזֹּאת מְסָרֵב ג'וֹ לְהוֹדוֹת שֶׁיֵּשׁ לוֹ בְּעָיָה וְשֶׁהוּא אִיבֵּד שְׁלִיטָה.
הַסִּיפּוּר מְסוּפָּר בְּמִילִּים פְּשׁוּטוֹת וּבְצִיּוּרִים פְּשׁוּטִים, רִישּׁוּמִים בְּקַו שֶׁל דְּמוּיוֹת וּמְקוֹמוֹת, וְאֶת אֵלֶּה צָרִיךְ לְמַלֵּא בְּצֶבַע.
אִם אֲנִי מֵבִין נָכוֹן, הָרַעְיוֹן הַבְּסִיסִי הוּא שֶׁבִּזְמַן שֶׁאֲנִי מַשְׁקִיעַ זְמַן בִּצְבִיעַת הַתְּמוּנוֹת, אֲנִי אָמוּר לְהַפְנִים אֶת הַסִּיפּוּר הַזְּוָועָתִי שֶׁל ג'וֹ וּלְשַׁיֵּיךְ אֶת הַזְּוָועָה הַזֹּאת לְמַצָּבִים שׁוֹנִים בַּחַיִּים שֶׁלִּי.
אִם ג'וֹ אִיבֵּד שְׁלִיטָה, הֲרֵי גַּם אֲנִי אִיבַּדְתִּי שְׁלִיטָה.
אִם ג'וֹ מַגִּיעַ לָרְחוֹב, הֲרֵי כְּדַאי לִי לְהִיזָּהֵר, אַחֶרֶת גַּם אֲנִי אֶהְיֶה הוֹמְלֵס.
סוֹף הַסִּיפּוּר שֶׁל ג'וֹ אוֹפְּטִימִי לְמַדַּי: הוּא מוֹדֶה שֶׁאִיבֵּד שְׁלִיטָה וּמִצְטָרֵף לְאַלְכּוֹהוֹלִיסְטִים אָנוֹנִימִים.
אַחַר כָּךְ יֵשׁ שְׁאֵלוֹן וּבוֹ עֶשְׂרִים וְשֶׁבַע שְׁאֵלוֹת שֶׁעוֹסְקוֹת בְּהֶרְגֵּלֵי הַשְּׁתִיָּיה שֶׁל הַנִּבְדָּק.
הַשְּׁאֵלוֹת פְּשׁוּטוֹת, וְכוּלָּן מְחַיְּיבוֹת תְּשׁוּבוֹת שֶׁל כֵּן וְלֹא.
הַאִם יֵשׁ מַצָּבִים שֶׁאֵינְךָ מַרְגִּישׁ בָּהֶם בְּנוֹחַ אִם אֵין בְּנִמְצָא אַלְכּוֹהוֹל? כֵּן.
הַאִם קָרָה פַּעַם שֶׁהִתְעוֹרַרְתָּ בַּבּוֹקֶר לְאַחַר לַיְלָה שֶׁל שְׁתִיָּיה וְלֹא זָכַרְתָּ אֵיפֹה הָיִיתָ בָּעֶרֶב הַקּוֹדֵם? כֵּן.
כְּשֶׁאַתָּה פִּיכֵּחַ, הַאִם אַתָּה מִתְחָרֵט עַל דְּבָרִים שֶׁעָשִׂיתָ בִּהְיוֹתְךָ שִׁיכּוֹר? כֵּן.
הַאִם הַיָּדַיִים שֶׁלְּךָ רוֹעֲדוֹת בַּבּוֹקֶר שֶׁלְּאַחַר לַיְלָה שֶׁל שְׁתִיָּיה? כֵּן.
הַאִם קוֹרֶה שֶׁאַתָּה שָׁתוּי בְּמֶשֶׁךְ כַּמָּה יָמִים רְצוּפִים? כֵּן, כֵּן, כֵּן, כֵּן.
אֲנִי עוֹנֶה בְּחִיּוּב לְכָל הַשְּׁאֵלוֹת, וְעַל פִּי הַמַּפְתֵּחַ שֶׁבְּסוֹף הַשְּׁאֵלוֹן, פֵּירוּשׁ הַדָּבָר שֶׁאֲנִי נִמְצָא בַּשְּׁלַבִּים הָאַחֲרוֹנִים שֶׁל אַלְכּוֹהוֹלִיזְם כְרוֹנִי וּמְסוּכָּן.
מָה אַתֶּם סָחִים.
