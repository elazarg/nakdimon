יוּזְכַּר כִּי הָעִרְעוּרִים הַנְּדוֹנִים נוֹגְעִים לְחִיּוּבִים בְּכֶפֶל מַס עַל פִּי סְעִיף 50(א) לְחוֹק מַס עֵרֶךְ מוּסָף לְגַבֵּי שְׁנֵי הַיְּחִידִים שְׁבִּירוֹ וְשָׁפִיר, וְכֵן שׁוּמוֹת תְּשׂוּמוֹת וּקְנָסוֹת עַל אִי נִיהוּל סְפָרִים לְגַבֵּי שְׁנֵי הַתַּאֲגִידִים.
בְּעִקְבוֹת הַכְרָעַת הַדִּין שֶׁנִּיתְּנָה עַל יְדֵי בֵּית מִשְׁפַּט הַשָּׁלוֹם בְּתֵל אָבִיב-יָפוֹ בת"פ 13-03-7455 (וְכֵן דְּחִיַּית הָעִרְעוּרִים עַל הַכְרָעַת הַדִּין בְּהַמְלָצַת בֵּית הַמִּשְׁפָּט הַמְּחוֹזִי בְּתֵל-אָבִיב-יָפוֹ בע"פ 15-09-23079) בִּיקֵּשׁ הַמֵּשִׁיב כִּי הַמִּמְצָאִים וְהַמַּסְקָנוֹת בִּפְסַק הַדִּין הַפְּלִילִי יִהְיוּ קְבִילִים כִּרְאָיוֹת לִכְאוֹרָה לָאָמוּר בָּהֶם בְּמִסְגֶּרֶת הָעִרְעוּרִים דְּנַן, וְזֹאת מִכּוֹחַ סְעִיף 42א(א) לִפְקוּדַּת הָרְאָיוֹת [נוֹסַח חָדָשׁ], התשל"א-1971.
כְּמוֹ כֵן בִּיקֵּשׁ הַמֵּשִׁיב כִּי "הָעִרְעוּרִים.... יִדְחוּ בַּסּוּגְיוֹת שֶׁהוּכְרְעוּ בַּהֲלִיךְ הַפְּלִילִי" (סְעִיף 19(ב) לַבַּקָּשָׁה).
יֵשׁ לְצַיֵּין כִּי הַמֵּשִׁיב עָתַר לִפְסַק דִּין "חֶלְקִי" בִּלְבַד כִּי הֲרֵי לֹא קַיֶּימֶת חֲפִיפָה מְלֵאָה בֵּין חֶשְׁבּוֹנִיּוֹת הַמַּס שֶׁנָּדוֹנוּ בַּהֲלִיךְ הַפְּלִילִי לְבֵין אֵלּוּ שֶׁהֵן מוּשַּׂא עִרְעוּרֵי הַמַּס הָאֶזְרָחִיִּים.
לְעִנְיָין אַחֲרוֹן זֶה אָשׁוּב בַּהֶמְשֵׁךְ.
אֶתְיַיחֵס רֵאשִׁית לַחִיּוּבִים שֶׁהוּטְּלוּ עַל הַיְּחִידִים לְפִי סְעִיף 50(א) לַחוֹק בְּקֶשֶׁר לְהוֹצָאַת חֶשְׁבּוֹנִיּוֹת שֶׁלֹּא כַּדִּין.
הָאִישּׁוּם הַפְּלִילִי בְּנוֹשֵׂא זֶה (לְאַחַר צִמְצוּמוֹ) הִתְיַיחֵס לְ-304 חֶשְׁבּוֹנִיּוֹת מַס אֲשֶׁר נִמְסְרוּ לִידֵי שְׁלוֹשָׁה עָשָׂר גּוֹרְמִים שׁוֹנִים (הַ"הַמְּקַזְּזִים" כִּלְשׁוֹן הַכְרָעַת הַדִּין), כַּאֲשֶׁר סְכוּם הַמַּעַ"מ הַכּוֹלֵל הַגָּלוּם בָּהֶן מַגִּיעַ לִכְדֵי 8.13 מִילְיוֹן שַׁ"ח.
