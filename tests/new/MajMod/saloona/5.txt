הַסִּיפּוּר הַזֶּה תָּפַס אוֹתִי לַחֲלוּטִין לֹא מוּכָנָה ובנקודת זְמַן מְאוֹד רְגִישָׁה. יֵשׁ לָנוּ שְׁתֵּי בְּנוֹת, וּבְאוֹתָהּ תְּקוּפָה חָשַׁבְנוּ עַל יֶלֶד נוֹסָף וְהִנֵּה פִּתְאוֹם אֲנִי כְּבָר לֹא "אילני מֵגֵן" אֶלָּא: "אילני מֵגֵן סִיכּוּן גָּבוֹהַּ". וּמִי בִּכְלָל מוּכָן לְדַבֵּר אִתְּךָ עַל יְלָדִים. הִתְחִילָה שֶׁגָּרָה שֶׁל בְּדִיקוֹת וּמַעֲקָב: כָּל 3 חוֹדָשִׁים בְּדִיקוֹת אֵצֶל כירורג, בְּדִיקוֹת MRI, אולטרסאונד, וְעִם כָּל זֶה, תְּקוּפָה אֲרוּכָּה עֲדַיִין חָיִיתִי בהדחקה ובהכחשה כשהכול מרחף בָּאֲוִויר, לֹא נוֹגֵעַ בִּי. עִם הַזְּמַן נֶחְשַׂפְתִּי לַמֵּידָע. שָׁמַעְתִּי שסרטן הַשֵּׁד נִיתָּן לריפוי בגילוי מוּקְדָּם, שֶׁיֵּשׁ לוֹ טִיפּוּל. לְגַבֵּי סַרְטָן השחלות אֵין גִּילּוּי מוּקְדָּם, ומרגע הַגִּילּוּי יֵשׁ מַמָּשׁ זְמַן קָצָר עַד ש"מזדכים עַל הַחַיִּים". הֶחְלַטְתִּי לַעֲבוֹר נִיתּוּחַ כריתת שחלות וחצוצרות, וּבָדְקָה ה-90 קִיבַּלְתִּי רַגְלַיִים קָרוֹת וברחתי.. נְקוּדַּת הַמִּפְנֶה מִבְּחִינָתִי הָיְתָה סדנא בָּהּ הִשְׁתַּתַּפְתִּי בהמלצת פְּרוֹפֵסוֹר פְרִידְמָן, מְנַהֵל הַיְּחִידָה האונקוגנטית בְּתֵל הַשּׁוֹמֵר וד"ר שַׁחַר לֵב אֲרִי, מִבֵּית הַחוֹלִים אִיכִילוֹב שֶׁהֵקִימוּ קְבוּצַת תְּמִיכָה לנשאיות. הסדנא פְּעֻלָה לְפִי עֶקְרוֹנוֹת "הָעֲבוֹדָה" שֶׁל ביירון קייטי. נִפְגַּשְׁנוּ בְּמֶשֶׁךְ שְׁלוֹשָׁה חוֹדָשִׁים, בְּכָל יוֹם שִׁישִּׁי למפגשים בְּנֵי חָמֵשׁ שָׁעוֹת. עֶצֶם הַמִּפְגָּשׁ עִם נשאיות כָּמוֹנִי, שרובן כְּבָר עָבְרוּ נִיתּוּחִים מניעתיים בֵּין שֶׁל כריתת שחלות וחצוצרות וּבֵין כריתת שדיים הָיָה חָשׁוּב ומרגיע - נוכחתי שֶׁיֵּשׁ חַיִּים אַחֲרֵי הכריתה. מַטְּרַת הסדנה המוצהרת הָיְיתָה לְנַטְרֵל פְּחָדִים וחרדות, שֶׁהֵם כְּשֶׁלְּעַצְמָם יְכוֹלִים לְזָרֶז הוֹפָעַת הַמַּחֲלָה. יָצָאתִי מהסדנה מחוזקת וְהַפַּעַם נִיגַּשְׁתִּי באומץ לְנִיתּוּחַ כריתת שחלות וחצוצרות. לֹא אוֹמֵר שֶׁזֶּה הָיָה קַל, אֲבָל תּוֹדָה לְאֵל הַיּוֹם זֶה מֵאֲחוֹרַיי. הַנִּיתּוּחַ מְלַוֶּוה בְּתוֹפָעוֹת לְוַואי שֶׁל גִּיל הַמַּעֲבָר, שתופסות אוֹתְךָ בְּבַת אַחַת - גַּלֵּי חוֹם מְטוֹרָפִים, ועמן הַפָּן הרגשי: הַפְסָקַת הַמַּחְזוֹר החודשי, וההשלמה עִם כָּךְ שהפסקתי לִהְיוֹת פוריה בְּגִיל כָּל כָּךְ צָעִיר לֹא הָיְיתָה קַלָּה. וּבְכָל זֹאת, הַיּוֹם אֲנִי בַּמָּקוֹם אַחֵר לְגַמְרֵי. הֶחְלַטְתִּי שֶׁאֵין טַעַם לְהִתְוַוכֵּחַ עִם הַמְּצִיאוּת. הִיא פֹּה, וְצָרִיךְ לְהִסְתַּכֵּל לָהּ בָּעֵינַיִים.
