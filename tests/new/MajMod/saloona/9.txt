בְּכָל הִזְדַּמְּנוּת שֶׁהָיְיתָה לוֹ סִיפֵּר שִׁמְעוֹן פְּרַס שֶׁהוּא שׁוֹתֶה מִיץ מִלִּימוֹן טָרִי מִדַּי בּוֹקֶר. זֶה הָיָה הַסּוֹד הַגָּלוּי שֶׁלּוֹ לאריכות יָמִים. אָז אָמְנָם אַף אֶחָד לֹא חַי לְנַצֵּחַ, אֲבָל כּוּלָּנוּ אוֹהֲבִים לִחְיוֹת אֶת הַחַיִּים שֶׁלָּנוּ טוֹב יוֹתֵר, בָּרִיא יוֹתֵר, יָפֶה יוֹתֵר - וְאִם זֶה לֹא דּוֹרֵשׁ מַאֲמָץ כְּבִיר, אֲנַחְנוּ בִּכְלָל מבסוטים. מִכֵּיוָון שֶׁרוּבֵּנוּ בְּנוֹת תְּמוּתָה רְגִילוֹת, אֲפִילּוּ לֹא נִשְׁלַח אוֹתְךָ לסחוט לִימוֹן ולבלוע אֶת המיץ הֶחָמוּץ כְּאִילּוּ שֶׁמְּדוּבָּר בַּכַּפִּית דְּבַשׁ, אֶלָּא נִסְתַּפֵּק בהמלצה לַחֲלוֹט שְׁתֵּי פְּרוּסוֹת לִימוֹן בְּכוֹס מַיִם חַמִּים מִדַּי בּוֹקֶר, וְאֶפְשָׁר אֲפִילּוּ לְהַמְתִּיק בִּמְעַט דְּבַשׁ. אָז אֵיךְ כּוֹס מִי לִימוֹן בַּבּוֹקֶר תַּחֲסוֹךְ לְךָ בַּתַּקְצִיב הביקורים אֵצֶל הקוסמטיקאית וְלָמָּה הַגִּ'ינְס שֶׁלְּךָ יָמוּת עַל זֶה? בְּעֶזְרַת עדי מגד, דִּיאֵטָן במכבי שֵׁירוּתֵי בְּרִיאוּת, אספנו 7 סִיבּוֹת טוֹבוֹת לְהַתְחִיל לִשְׁתּוֹת מִי לִימוֹן כָּל יוֹם, עַל הַבּוֹקֶר. 1. שִׁיפּוּר פְּעוּלַּת מַעֲרֶכֶת הָעִיכּוּל עדי מַדְגִּישׁ שֶׁחָשׁוּב לִשְׁתּוֹת אֶת מִי הלימון דָּבָר רִאשׁוֹן בַּבּוֹקֶר. הַסִּיבָּה ששותים אֶת מִי הלימון בַּבּוֹקֶר הִיא שֶׁאֵין כָּמוֹהוּ להנעת מַעֲרֶכֶת הָעִיכּוּל בַּבּוֹקֶר, וְכוּלָּנוּ יוֹדְעִים שֶׁאִם הִיא מִתְעוֹרֶרֶת עַל צַד שְׂמֹאל כָּל הַיּוֹם ייראה כָּךְ. הפקטין שבלימון מְשַׁמֵּשׁ כזרז להתנעת הַמַּעֲרֶכֶת בַּבּוֹקֶר, וּלְאוֹרֶךְ הַיּוֹם אוֹתוֹ פקטין תּוֹרֵם לִתְחוּשַׁת שובע, מזרז אֶת חִילּוּף הַחוֹמָרִים ומסייע לִירִידָה בְּמִשְׁקָל אוֹ שְׁמִירָה עַל מִשְׁקָל גּוּף תַּקִּין. אָמַרְתָּ לִימוֹן, אָמַרְתָּ גִ'ינְס. 2. הזעת? שְׁתֵּי לִימוֹן בִּפְעִילוּת גּוּפָנִית אֲנַחְנוּ מְאַבְּדִים נוֹזְלִים וּמְלָחִים וּמִינֵרָלִים, וּפְעָמִים רַבּוֹת אֲפִילּוּ לֹא מוּדָעִים לְכָךְ עַד שמתייבשים. הַסִּיבָּה לְכָךְ הִיא ששתיית מַיִם נטולי מלחים מדללת אֶת המאגרים בַּגּוּף. הוֹסָפַת כַּמָּה פְּרוּסוֹת לִימוֹן לבקבוק הַמַּיִם שמתלווה אֵלַיִיךְ לִפְעִילוּת תעזור לַגּוּף לְהָאִיץ אֶת חִידּוּשׁ המאגרים וְלִמְנוֹעַ הִתְיַיבְּשׁוּת. 3. שְׁפַן, אוֹהֵב לִימוֹן? נָכוֹן שכילדים סִיפְּרוּ לָנוּ שגזר בְּיוֹם יִשְׁמוֹר נַפְשֵׁנוּ ממשקפיים, אֲבָל בְּגִילֵנוּ משקפיים כְּבָר אֵינָם דָּבָר מַפְחִיד כָּל כָּךְ, אָז בּוֹאִי נְדַבֵּר עַל הָעֵינַיִים שֶׁלָּנוּ. שיבושי רְאִייָּה יְכוֹלִים לְהִימָּנַע ע"י שְׁתִיַּית לִימוֹן, שֶׁמֵּכִיל הֲמוֹן ויטמין סי ואנטי-אוקסידנטים, שהוכחו כמעכבי קטרקט ותהליכים ניווניים שֶׁל הרשתית. לָכֵן מוּמְלָץ כְּבָר מִגִּיל צָעִיר לְהַצִּיעַ גַּם לשפנים, סְלִיחָה, לִילָדִים.
