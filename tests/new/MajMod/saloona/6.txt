כָּךְ יָצָאנוּ שׁוּב לְפָחוֹת מִ-24 שָׁעוֹת שֶׁל תַּעֲנוּגוֹת וְהַפַּעַם נִבְחַר אֵזוֹר מַטָּה יְהוּדָה. הַמָּלוֹן הַנִּבְחָר הָיָה מָלוֹן נְוֵוה אִילָן שֶׁל רֶשֶׁת סי הוטל - ממוקם בַּסְּבִיבָה פסטורלית ובמרחק שֶׁל 15 דַּקּוֹת מִיְּרוּשָׁלַיִם. הִגַּעְנוּ לְאַחַר צָהֳרַיִים שֶׁל חֲוָויוֹת בַּמּוֹעָצָה אֲזוֹרִית מַטָּה יְהוּדָה בְּסוֹף חוֹדֶשׁ יָנוּאָר הַחוֹרְפִּי. כָּךְ הִתְחִיל מַסַּע הֶיכֵּרוּת אִינְטֶנְסִיבִי עִם מָה שֶׁיֵּשׁ לְמַטָּה יְהוּדָה לְהַצִּיעַ לָנוּ בניצוחם שֶׁל צֶוֶות היח"צ המעולה שֶׁל הַמַּטֶּה כְּשֶׁכָּל תַּחֲנָה מְלַוֶּוה אֵיךְ לֹא, בשולחנות מְלֵאִים בְּכָל טוֹב. עָלֵינוּ למיניבוס תַּיָּירִים עִם הַצֶּוֶות החינני וקולגות בָּרֵי מַזָּל נוֹסָפִים ושמנו פעמינו ליקב פלאם האלוהי לצהרי טְעִימוֹת יַיִן וּגְבִינוֹת. שרית שֶׁלּוֹ, מְנַהֶלֶת השיווק שֶׁל היקב פָּתְחָה לָנוּ שׁוּלְחַן עִם פינוקי גַּן עֵדֶן שֶׁחַיָּיבִים לְהַחֲלִיק לקיבתנו בְּסִיּוּעַ יינות שׁוֹנִים מִתּוֹצֶרֶת הייקב. תּוֹךְ כְּדֵי עֲלִייַּת מִפְלַס הָאַלְכּוֹהוֹל קִיבַּלְנוּ סֶקֶר מַקִּיף עַל תולדות היקב, הכרמים, סוּגֵי הענבים והטרואר המאפיין שֶׁל יְהוּדָה (לבורים מבינכם כִּי אֲנַחְנוּ כְּבָר מְבִינִים, הַכַּוָּונָה לְסוּג הָאֲדָמָה ומרכיביה וְכֵיצַד הִיא מוּשְׁפַּעַת מִמֶּזֶג הָאֲוִויר ומשפיעה חָזְרָה עַל גידולי הַזָּנִים. הַהַגְדָּרָה הַזּוֹ גובשה לְאוֹרֶךְ מֵאוֹת שָׁנִים בְּצָרְפַת וְאֶצְלֵנוּ מְנַסִּים לאפיין אוֹתָהּ אינסטנט בראי תַּעֲשִׂיַּית הַיַּיִן הַצְּעִירָה). אַחֲרֵי כְּשָׁעָה וָחֵצִי התנדנדנו חָזְרָה למיניבוס לְהֶמְשֵׁךְ הַדֶּרֶךְ. אוֹי בוי.
