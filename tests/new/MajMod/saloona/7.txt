הִגַּעְתִּי להרצאתה שֶׁל רוּת דַּיָּין וולפנר סַקְרָנִית, לֹא יוֹדַעַת וְלֹא מַכִּירָה. רֵאשִׁית הִתְרַשַּׁמְתִּי מאישה נעימת הֲלִיכוֹת בַּעֲלַת מַרְאֶה אֵלֵגַנְטִי. אָהַבְתִּי אֶת הַשְּׁלֵוָוה וְהַשֶּׁקֶט שֶׁבְּאוֹפֶן דִּיבּוּרָהּ וְאֶת הַשָּׂפָה הרהוטה והעדינה. עֲדַיִין לֹא הֵבַנְתִּי אֵיךְ אוֹתוֹ מוּשָּׂג מִיסְטִי מִתְחַבֵּר להרצאה וּלְסַפֵּר שֶׁכָּתְבָה עוֹרֶכֶת דִּין. חָשַׁבְתִּי לְעַצְמִי שֶׁאֲנִי, כיועצת הוֹרוּת וּמִשְׁפָּחָה, שפוגשת הוֹרִים רַבִּים, עֲשׂוּיָה לִמְצוֹא עִנְיָין וְלִלְמוֹד מדבריה שֶׁל מִי שמלווה וליוותה זוּגוֹת רַבִּים. רוּת דַּיָּין וולפנר עוֹמֶדֶת בָּרֹאשׁ אֶחָד ממשרדי עוֹרְכֵי הַדִּין הַמַּצְלִיחִים בָּאָרֶץ, מעסיקה סוֹלְלַת עורכות דִּין ומתמחה בדיני מִשְׁפָּחָה. שְׁנוֹת עֲבוֹדָה רַבּוֹת, נִיסָּיוֹן שֶׁנִּצְבַּר בְּלִיוּוּי זוּגוֹת בְּמַצְּבֵי מַשְׁבֵּר הוֹבִילוּ אוֹתָהּ לִכְתִיבַת סֵפֶר הַמֵּכִיל מקבץ מֵהַסִּיפּוּרִים בָּהֶם נִתְקְלָה בִּשְׁנוֹת עֲבוֹדָתָהּ. הַהִתְבּוֹנְנוּת והניתוח שֶׁל הַמִּקְרִים אוֹתָם לִיוְּותָה הֵבִיאוּ אוֹתָהּ להסקת תּוֹבָנוֹת אוֹדוֹת מוֹסַד הַנִּישּׂוּאִין. רוּת בנתה תְּפִיסָה לְפִיהָ בְּכָל אָדָם קַיָּימוֹת מִפְלָצוֹת קְטַנּוֹת, הִיא פּוֹתַחַת אֶת דְּבָרֶיהָ בציטוט שֶׁל סְטִיבֶן קִינְג: "מִפְלָצוֹת קַיָּימוֹת בֶּאֱמֶת. גַּם רוּחוֹת רְפָאִים. הֵן חַיוֹת בֵּינֵינוּ. וְלִפְעָמִים הֵן מנצחות". רוּת מוּנָּה אֶת סוּגֵי הַמִּפְלָצוֹת וְכֵיצַד הֵן בָּאוֹת לִידֵי בִּיטּוּי בְּמַעֲרֶכֶת יְחָסִים שֶׁל בְּנֵי זוּג, מחבלות, מַשְׁפִּיעוֹת לְרָעָה ומקלקלות יְחָסִים.
