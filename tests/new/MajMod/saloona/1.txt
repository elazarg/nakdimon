יֵשׁ לְצַיֵּין כִּי זוֹ לֹא פַּעַם רִאשׁוֹנָה שֶׁמְּשַׂחֵק לִכְאוֹרָה תָּמִים מִתְבָּרֵר כמסוכן: רַק לִפְנֵי חוֹדֶשׁ פסטיבלי מוּזִיקָה בְּאַנְגְּלִיָּה אסרו עַל שִׁימּוּשׁ בנצנצים: וְלֹא רַק הֵם, מְקוֹמוֹת רַבִּים בָּעוֹלָם כְּבָר מְבִינִים שחתיכות הַפְּלַסְטִיק והמתכת הַקְּטַנּוֹת מזהמות אֶת הַסְּבִיבָה, אַחֲרֵי כַּמָּה שָׁעוֹת שֶׁל השארות עַל הַגּוּף הֵן נשטפות אֶל תּוֹךְ מִי הַבִּיּוּב והיישר לאוקיינוסים, שֶׁסּוֹבְלִים מִזִּיהוּם גַּם בְּלִי הַתּוֹסֶפֶת המיותרת הַזּוֹ, ונשארות שָׁם אַלְפֵי שָׁנִים עַד לפירוק מָלֵא. בְּמַיִם חֲתִיכוֹת הַפְּלַסְטִיק אוספות סְבִיבָם כימיקלים רְעִילִים שֶׁנִּכְנָסִים אֶל תּוֹךְ הַדָּגִים שֶׁאוֹכְלִים אוֹתָן, וְאָז אֶל בְּנֵי הָאָדָם שֶׁאוֹכְלִים אוֹתָם. בְּפוֹסְט שֶׁכְּבָר זָכָה למעל 3,000 שיתופים, יָעֵל מְבַקֶּשֶׁת מכולנו לִלְמוֹד מִמָּה שֶׁקָּרָה לָהּ וּלְהַעֲלוֹת אֶת הַמּוּדָעוּת לסכנת הסליים. חִיפּוּשׂ קָצָר בְּגוּגֶל מְסַפֵּק אֶת רְשִׁימַת הָרְכִיבִים שֶׁל הדביק-הדביק הַזֶּה: דָּבֵק פלסטי (דָּבֵק לָבָן- C4H6O2) - מְיוּצָּר מפולימר סינתטי צְמִיגִי וּמְשַׁמֵּשׁ לָרוֹב להדבקת מוצריי נְיָיר ועץ, בוראקס (Na2B4O7·10H2O) - תרכובת כִּימִית שֶׁל מֶלַח נתרן וחומצת בּוֹר, וסליים עָשׂוּי לְהָכִיל לְעִיתִּים גַּם פרסיל-ג'ל, הַמְּשַׁמֵּשׁ גֶ'ל לכביסה, שֶׁכֵּן גַּם הוּא מֵכִיל בוראקס. מְסוּכָּן בְּיוֹתֵר. אָנָּא מִכֶּם, פָּתְחוּ אֶת הָעֵינַיִים וְשִׁמְרוּ עַל יַלְדֵיכֶם וְעַל עַצְמְכֶם. בּוֹאוּ נִמְנָע אֶת הָאָסוֹן הַבָּא.
