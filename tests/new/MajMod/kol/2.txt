הַתַּחְבּוּרָה הַצִּיבּוּרִית הִיא שֶׁל כּוּלָּנוּ - אֵלֶּה הַזְּכוּיוֹת שֶׁחָשׁוּב לְהַכִּיר הַתַּחְבּוּרָה הַצִּיבּוּרִית בְּיִשְׂרָאֵל מְחוּיֶּבֶת לִמְנוֹעַ אפליה באספקת שֵׁירוּתֶיהָ, ולהנגיש אוֹתָם לִקְבוּצוֹת אוּכְלוּסִיָּיה שׁוֹנוֹת. בְּנוֹסָף, קַיָּימִים פְּטוּרִים מִתַּשְׁלוּם וְהַנָּחוֹת בִּמְחִיר הַנְּסִיעָה לציבורים מְסוּיָּמִים. פורטל זֶה מֶרְכָּז אֶת ההנחות וההטבות הַנִּיתָּנוֹת לאוכלוסיות שׁוֹנוֹת בַּתַּחְבּוּרָה הַצִּיבּוּרִית וְכֵן אֶת זְכוּיוֹתֵיהֶם הַשּׁוֹנוֹת שֶׁל הַמִּשְׁתַּמְשִׁים בָּהּ רִיכּוּז מֵידָע חָשׁוּב -פִּיצּוּי בְּגִין איחור אוֹ בִּיטּוּל רַכֶּבֶת -אִיסּוּר לִכְפּוֹת עַל נָשִׁים איזור יְשִׁיבָה בָּאוֹטוֹבּוּס -זכותון נְסִיעָה בַּמּוֹנִית מְיוּחֶדֶת (סְפֵּיְישֵׁל) -נְסִיעָה בָּאוֹטוֹבּוּס עִם מַכְשִׁיר כרטוס (מַכְשִׁיר תיקוף רַב קַו) שֶׁאֵינוֹ תַּקִּין -אֶזְרָחִים וָתִיקִים -הַנָּחָה לְאֶזְרָחִים וָתִיקִים באוטובוסים, בָּרַכֶּבֶת הַקַּלָּה ובמטרונית -הַנָּחָה בָּרַכֶּבֶת לְאֶזְרָחִים וָתִיקִים -אֲנָשִׁים בִּמְצוּקָה כַּלְכָּלִית -הַנָּחָה בַּנְּסִיעָה בַּתַּחְבּוּרָה צִיבּוּרִית לִמְקַבְּלֵי גמלת הַבְטָחַת הַכְנָסָה -הַנָּחָה בַּתַּחְבּוּרָה הַצִּיבּוּרִית לִמְקַבְּלֵי קִצְבַּת שאירים עִם הַשְׁלָמַת הַכְנָסָה -נָכִים וַאֲנָשִׁים עִם מוּגְבָּלוּת -הַנָּחָה לַאֲנָשִׁים עִם נָכוּת באוטובוסים, בָּרַכֶּבֶת הַקַּלָּה ובמטרונית -הַנָּחָה לנכי צַהַ"ל באוטובוסים, בָּרַכֶּבֶת הַקַּלָּה ובמטרונית -הַנָּחָה לנכים נִפְגְּעֵי פְּעוּלּוֹת איבה באוטובוסים, בָּרַכֶּבֶת הַקַּלָּה ובמטרונית -הַנָּחָה בָּרַכֶּבֶת לַאֲנָשִׁים עִם נָכוּת -הַנָּחָה בָּרַכֶּבֶת לנכי צַהַ"ל -הַנָּחָה בָּרַכֶּבֶת לנכים נִפְגְּעֵי פְּעוּלּוֹת איבה -פְּטוֹר מִתַּשְׁלוּם עַל הוֹבָלַת כִּיסֵּא גַּלְגַּלִּים בַּמּוֹנִית -אִיסּוּר אפליית אָדָם עִם מוּגְבָּלוּת בַּתַּחְבּוּרָה הַצִּיבּוּרִית -עיוורים ולקויי רְאִייָּה -פְּטוֹר מִתַּשְׁלוּם עַל נְסִיעָה באוטובוסים לעיוורים ולקויי רְאִייָּה -פְּטוֹר מִתַּשְׁלוּם עַל נְסִיעָה בָּרַכֶּבֶת לעיוורים ולקויי רְאִייָּה -נְגִישׁוּת לעיוורים ולקויי רְאִייָּה בַּתַּחְבּוּרָה הַצִּיבּוּרִית -נְגִישׁוּת לעיוורים ולקויי רְאִייָּה ברכבות -נְגִישׁוּת לנכים וּבַעֲלֵי מוּגְבָּלוּיוֹת -נְגִישׁוּת לַאֲנָשִׁים עִם מוּגְבָּלוּת ולנכים באוטובוסים -נְגִישׁוּת לַאֲנָשִׁים עִם מוּגְבָּלוּת ולנכים בְּתַחֲנוֹת אוֹטוֹבּוּס -נְגִישׁוּת לַאֲנָשִׁים עִם מוּגְבָּלוּת ולנכים ברכבות -כְּנִיסָה לרכבת בקלנועית -נְגִישׁוּת לעיוורים ולקויי רְאִייָּה בַּתַּחְבּוּרָה הַצִּיבּוּרִית -נְגִישׁוּת לעיוורים ולקויי רְאִייָּה ברכבות -שׁוֹטְרִים למתנדבי שְׁנַת שֵׁירוּת -פְּטוֹר מִתַּשְׁלוּם בָּאוֹטוֹבּוּס למתנדבי הַשֵּׁירוּת הלאומי-אזרחי -פְּטוֹר מִתַּשְׁלוּם בָּרַכֶּבֶת לחניכי מכינות קדם-צבאיות -פְּטוֹר מִתַּשְׁלוּם בָּרַכֶּבֶת למתנדבי
