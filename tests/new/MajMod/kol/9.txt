חַיָּילִים, צָבָא וּבִיטָּחוֹן הפורטל מֶרְכָּז מֵידָע עַל זְכוּיוֹת חַיָּילֵי צַהַ"ל בַּשֵּׁירוּת סָדִיר (חוֹבָה וְקָבַע) וּמִילּוּאִים, זְכוּיוֹת הַחַיָּילִים לְאַחַר שִׁחְרוּרָם וזכויותיהם שֶׁל נכי צַהַ"ל. בְּנוֹסָף, הפורטל כּוֹלֵל מֵידָע רִאשׁוֹנִי לְגַבֵּי זְכוּיוֹתֵיהֶם שֶׁל בְּנֵי הַמִּשְׁפָּחוֹת השכולות. אֶת הַמֵּידָע המקיף נִיתָּן לִמְצוֹא בָּאֲתָר מִשְׁפָּחוֹת והנצחה שֶׁל מִשְׂרַד הַבִּיטָּחוֹן. חַיָּילִים בַּשֵּׁירוּת חוֹבָה: -התאמות לְבַעֲלֵי לקויות לְמִידָה במבחנים בלשכות הַגִּיּוּס -פִּיצּוּיֵי פִּיטּוּרִים לָעוֹבֵד שהתפטר עֵקֶב גִּיּוּס לְצַהַ"ל -אִיסּוּר אפליה שֶׁל חַיָּילִים בִּמְקוֹמוֹת צִיבּוּרִיִּים -חַיָּילִים וחיילות בַּשֵּׁירוּת חוֹבָה (צַהַ"ל) -הִתְנַדְּבוּת לַצָּבָא ואוכלוסיות מְיוּחָדוֹת -חַיָּילִים בּוֹדְדִים חַיָּילִים בַּשֵּׁירוּת קָבַע: -בִּיטּוּחַ בְּרִיאוּת לחיילי קָבַע -הַגָּשַׁת ערר לוועדת ערר לְחַיָּילִים בַּשֵּׁירוּת קָבַע -הִשְׁתַּתְּפוּת צַהַ"ל בְּהוֹצָאוֹת קייטנה לילדי חַיָּילִים בַּשֵּׁירוּת קָבַע -תְּבִיעָה לְהַכָּרָה בִּפְגִיעָה בַּעֲבוֹדָה לְחַיָּילִים בַּשֵּׁירוּת קָבַע -תְּבִיעָה לְהַכָּרָה בליקוי רְפוּאִי אוֹ מַחְלַת מִקְצוֹעַ לְחַיָּילִים בַּשֵּׁירוּת קָבַע -אִיסּוּר אפליה שֶׁל חַיָּילִים בִּמְקוֹמוֹת צִיבּוּרִיִּים -אִיסּוּר פִּיטּוּרֵי חיילת בהריון מִשֵּׁירוּת קָבַע -אִיסּוּר פִּיטּוּרֵי חייל/ת מִשֵּׁירוּת קָבַע בְּמַהֲלַךְ חופשת לְיָדָהּ אוֹ לאחריה -אִיסּוּר פִּיטּוּרֵי חייל/ת מִשֵּׁירוּת קָבַע בְּמַהֲלַךְ טיפולי פוריות -פְּטוֹר למשרתי קָבַע מהליכי מכרז לתפקידים בַּשֵּׁירוּת הַמְּדִינָה לוֹחֲמִים: -קַבָּלַת תְּעוּדַת לוֹחֵם -זְכוּיוֹת והטבות ללוחמי חוד בָּאֲתָר צַהַ"ל -חַיָּילִים משוחררים וחיילי מִילּוּאִים -אִיסּוּר אפליה שֶׁל חַיָּילִים בִּמְקוֹמוֹת צִיבּוּרִיִּים -רִישּׁוּם לְקוּפַּת חוֹלִים לְאַחַר הַשִּׁחְרוּר מִצַּהַ"ל -דְּמֵי לְיָדָהּ לחיילות משוחררות שֶׁהֵחֵלּוּ לַעֲבוֹד -זְכוּיוֹת חַיָּילִים משוחררים -מִילּוּאִים -גמלאי צַהַ"ל -הַגְדָּלַת קִצְבַּת גמלאי צַהַ"ל לניצולי שׁוֹאָה -דְּחִיַּית שֵׁירוּת -מְכִינָה קְדַם צְבָאִית ושנת שֵׁירוּת נכי צַהַ"ל וְנִפְגָּעִים: -נכי צַהַ"ל -תְּבִיעָה לְהַכָּרָה בִּפְגִיעָה בַּעֲבוֹדָה לְחַיָּילִים בַּשֵּׁירוּת קָבַע -תְּבִיעָה לְהַכָּרָה בליקוי רְפוּאִי אוֹ מַחְלַת מִקְצוֹעַ לְחַיָּילִים בַּשֵּׁירוּת קָבַע -תְּבִיעָה לְהַכָּרָה בִּפְגִיעָה בַּעֲבוֹדָה לְאַנְשֵׁי כּוֹחוֹת הַבִּיטָּחוֹן -תְּבִיעָה לְהַכָּרָה בליקוי רְפוּאִי אוֹ מַחְלַת מִקְצוֹעַ לְאַנְשֵׁי כּוֹחוֹת הַבִּיטָּחוֹן -חוֹבֶרֶת מֵידָע לפצוע (בְּשָׂפוֹת שׁוֹנוֹת) בָּאֲתָר אֲגַף שיקום נָכִים בְּמִשְׂרַד הַבִּטָּחוֹן מִשְׁפָּחוֹת שכולות: -הַנָּחָה בארנונה לַמִּשְׁפָּחָה שכולה -הַנָּחָה בְּמַס רְכִישָׁה לִבְנֵי מִשְׁפָּחוֹת שכולות -מִימּוּן אבחוני לקויות לְמִידָה ליתומים בְּנֵי מִשְׁפָּחוֹת שכולות -מִימּוּן שִׁיעוּרֵי הוֹרָאָה מְתַקֶּנֶת ליתומים בְּנֵי מִשְׁפָּחוֹת שכולות -מִימּוּן לִימּוּדֵי תְּעוּדָה מִקְצוֹעִית ותואר רִאשׁוֹן לאחים שכולים -היעדרות מֵהָעֲבוֹדָה בְּיוֹם הַזִּיכָּרוֹן -היעדרות מתעסוקה למשתקם בְּיוֹם הַזִּיכָּרוֹן -שָׁעוֹת סיעוד לניצולי שׁוֹאָה הַזַּכָּאִים ל"עזרה לזולת" מִמִּשְׂרַד הַבִּיטָּחוֹן -אִיסּוּר פִּיטּוּרִים בְּגִיל פְּרִישָׁה שֶׁל אלמנות צַהַ"ל וְהוֹרִים שכולים שילדם נִפְטַר עֵקֶב שֵׁירוּת צְבָאִי פדויי שְׁבִי: -קִצְבָּה למשרתים בְּכוֹחוֹת הַבִּיטָּחוֹן שֶׁנָּפְלוּ בַּשְּׁבִי -תַּשְׁלוּם לַחֲבֵרִים בארגוני מחתרת שנאסרו בִּתְקוּפַת הַמַּנְדָּט הַבְּרִיטִי -קִצְבָּה למתנדבי הַצָּבָא הַבְּרִיטִי שֶׁנָּפְלוּ בַּשְּׁבִי הַנָּאצִי מֶרְכָּזִי שֵׁירוּת צִיבּוּרִיִּים: -מוקדים
