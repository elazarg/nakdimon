סוֹף הַחַיִּים כָּאן תִּמְצְאוּ מֵידָע לְגַבֵּי הֲיִבְטֵים שׁוֹנִים שֶׁל אָדָם בְּסוֹף חַיָּיו בַּחֶבְרָה הַמַּעֲרָבִית, שִׂיחַ סְבִיב הַמָּוֶות, הַכּוֹלֵל אֶת הָאוֹפַן בּוֹ אָדָם מְבַקֵּשׁ לְסַיֵּים אֶת חַיָּיו וּלְהִיפָּרֵד מהקרובים לוֹ, אֵינֶנּוּ דָּבָר שֶׁבַּשִּׁגְרָה. עִם זֹאת, נִיתָּן לְהַבְחִין בְּשִׁינּוּי בְּיַחַס הַחֶבְרָה לְמָוֶות ובנכונות לָדוּן בּוֹ, בְּעִיקָּר בְּכָל הַנּוֹגֵעַ לִזְכּוֹת הַבְּחִירָה שֶׁל הַחוֹלִים לָמוּת בְּכָבוֹד וּלְהַעֲבִיר אֶת סוֹף ימיהם בְּאוֹפֶן בּוֹ הֵם מְבַקְּשִׁים. בפורטל זֶה, ריכזנו אֶת הַהֶיבֵּטִים הַשּׁוֹנִים שֶׁל זְכוּיוֹת סְבִיב סוֹף הַחַיִּים - הֵחֵל מֵאוֹפֶן קַבָּלַת הַהַחְלָטוֹת לְגַבֵּי הַטִּיפּוּל הָרְפוּאִי, דֶּרֶךְ הקבורה, עַד לתרומת אֵיבָרִים וְעוֹד מדריכים וּמֵידָע חָשׁוּב: -מַדְרִיךְ היערכות לְסוֹף הַחַיִּים -מַדְרִיךְ לִבְנֵי מִשְׁפָּחָה הַמִּתְמוֹדְדִים עִם מָוֶות (פְּטִירָה) וקבורה -קבורת הַמֵּת (הנפטר) -זכותון לִתְקוּפָה שֶׁלְּאַחַר מוֹת (פְּטִירַת) בֶּן מִשְׁפָּחָה -הַסְדָּרַת חוֹבוֹת אָדָם שֶׁנִּפְטַר -טִיפּוּל בִּכְסָפִים וברכוש שֶׁל בֶּן מִשְׁפָּחָה לְאַחַר מוֹתוֹ -חוֹבוֹת הַמִּשְׁפָּחָה לָעוֹבֵד זָר בסיעוד שמעסיקו נִפְטַר -פִּיצּוּיֵי פִּיטּוּרִים לָעוֹבֵד שמעסיקו נִפְטַר -פִּיצּוּיֵי פִּיטּוּרִים לשאריו שֶׁל עוֹבֵד שֶׁנִּפְטַר -הַחוֹלֶה הנוטה לָמוּת -יִיפּוּי כּוֹחַ מְשׁוּלָּב לְטִיפּוּל רְפוּאִי כְּלָלִי ולחולה הנוטה לָמוּת -רִישּׁוּם הַנְחָיוֹת רְפוּאִיּוֹת מקדימות -יִיפּוּי כּוֹחַ להחלטות עַל טִיפּוּל רְפוּאִי -חוֹלֶה הנוטה לָמוּת -הַזְּכוּת לָמוּת בְּכָבוֹד -הַזְּכוּת לָמוּת בְּכָבוֹד שֶׁל קטין -הַזְּכוּת לָמוּת בְּכָבוֹד שֶׁל חסוי -וַעֲדָה מוסדית בָּעִנְיָין חוֹלֶה הנוטה לָמוּת -וַעֲדָה ארצית בָּעִנְיָין חוֹלֶה הנוטה לָמוּת -טִיפּוּל תּוֹמֵךְ -הוספיס -טִיפּוּל בֵּיתִי תּוֹמֵךְ (טִיפּוּל פליאטיבי) -טִיפּוּל בֵּיתִי תּוֹמֵךְ לחולי סַרְטָן מִטַּעַם הָאֲגוּדָּה לְמִלְחָמָה בְּסַרְטָן -קַבָּלַת קנביס רְפוּאִי (מָרִיחוּאָנָה רְפוּאִית, גְּרָאס רְפוּאִי) -אַמְבּוּלַנְס המשאלות לַחוֹלִים בַּמַּחֲלָה קָשֶׁה אוֹ סוֹפָנִית -תִּכְנוּן מִשְׁפָּטִי וכלכלי -צַוָּואוֹת, ירושות ועזבונות -תרומת אֵיבָרִים מִן הַמֵּת (כַּרְטִיס אֵדִי) -מַתַּן הַנְחָיוֹת מקדימות לְמִינּוּי אַפּוֹטְרוֹפּוֹס -הרשאה מֵרֹאשׁ לְבִיצּוּעַ פְּעוּלּוֹת עֲתִידִיּוֹת בְּחֶשְׁבּוֹן בַּנְק מְשׁוּתָּף לְאַחַר מוֹתוֹ שֶׁל אֶחָד מִבַּעֲלֵי הַחֶשְׁבּוֹן (טוֹפֶס אריכות יָמִים, סְעִיף היוותרות בַּחַיִּים) -יִיפּוּי כּוֹחַ מִתְמַשֵּׁךְ לְמַצָּבִים שֶׁל אי-יכולת לְקַבֵּל הַחְלָטוֹת בֶּעָתִיד -החבות הַכַּלְכָּלִית לָעוֹבֵד זָר בסיעוד בְּמִקְרֶה פְּטִירָה -מַדְרִיךְ היערכות לְסוֹף הַחַיִּים -נתיחה שֶׁלְּאַחַר הַמָּוֶות -קְבִיעַת מָוֶות -נתיחה שֶׁלְּאַחַר הַמָּוֶות -הִתְנַגְּדוּת בְּנֵי מִשְׁפָּחָה לנתיחה שֶׁלְּאַחַר הַמָּוֶות -פְּתִיחַת קבר לְצוֹרֶךְ נתיחה שֶׁלְּאַחַר הַמָּוֶות -קבורה -קבורת הַמֵּת (הנפטר) -רְכִישַׁת חֲלֻקַת קבר בַּחַיִּים -הוֹצָאַת הוֹדָעַת פְּטִירָה -הוֹצָאַת רִשְׁיוֹן קבורה -בְּחִירַת מָקוֹם הקבורה לנפטר -שְׁמִירַת גופת הנפטר בקירור לִפְנֵי קבורה -הוֹבָלַת הנפטר לקבורה -צוּרוֹת הקבורה הַיְּהוּדִית -קבורה אֶזְרָחִית -שריפת גּוּפָהּ -תרומת גּוּפָהּ לְמַדָּע -קבורה בָּאֲתָר הַשֵּׁירוּתִים וְהַמֵּידָע הַמֶּמְשַׁלְתִּי -מִימּוּן הקבורה וַהֲקָמַת מַצָּבָהּ -מִימּוּן שֵׁירוּתֵי קבורה (דְּמֵי קבורה) -מִימּוּן הוֹצָאוֹת קבורת תּוֹרֵם אֵיבָרִים לְאַחַר מוֹתוֹ -דְּמֵי קבורה לְתוֹשָׁב יִשְׂרָאֵל שֶׁנִּפְטַר בְּחוּ"ל -תַּשְׁלוּם דְּמֵי קבורה לנפטר לֹא יְהוּדִי -מענק מַצָּבָהּ לניצולי שׁוֹאָה -הִשְׁתַּתְּפוּת בַּהֲקָמַת מַצָּבָהּ לְנָכֶה הַמִּלְחָמָה בנאצים -הֲקָמַת מַצָּבָהּ לנפטרים גלמודים -מִשְׁפָּחוֹת נִפְגְּעֵי עֲבֵירוֹת המתה (רֶצַח והריגה) -נִפְגַּע עֲבֵירַת המתה -מֶרְכָּזִי סִיּוּעַ לְמִשְׁפָּחוֹת נִפְגְּעֵי עֲבֵירוֹת המתה -וַעֲדַת זַכָּאוּת לְנִפְגְּעֵי עֲבֵירוֹת המתה -סִיּוּעַ כַּסְפִּי לְמִשְׁפָּחוֹת -סִיּוּעַ מִשְׁפָּטִי לְמִשְׁפָּחוֹת -סִיּוּעַ רִגְשִׁי לְמִשְׁפָּחוֹת -תגמול לַיֶּלֶד שהתייתם בְּעִקְבוֹת רֶצַח בֵּין בְּנֵי זוּג -פְּנִיָּיה מְקֻוֶּונֶת שֶׁל נִפְגַּע עֲבֵירָה לפרקליטות הַמְּדִינָה -התארגנות הַמִּשְׁפָּחָה -פְּטִירָה, שארים ופיצויים בָּאֲתָר הַשֵּׁירוּתִים וְהַמֵּידָע הַמֶּמְשַׁלְתִּי -מֵידָע לְמִשְׁפַּחַת הנפטר בָּאֲתָר פוֹרוּם חֲבֵרוֹת הקדישא -יְמֵי אֲבָל -הוֹצָאַת תְּעוּדַת פְּטִירָה -טִיפּוּל בִּכְסָפִים וברכוש שֶׁל בֶּן מִשְׁפָּחָה לְאַחַר מוֹתוֹ -מִימּוּשׁ הַיְרוּשָּׁה -מענק פְּטִירָה לשאירים -תַּשְׁלוּם יִתְרַת קִצְבַּת זיקנה לשאיריו שֶׁל הזכאי -הַגָּשַׁת בַּקָּשָׁה לִדְחִיַּית דִּיּוּן בְּבֵית מִשְׁפָּט -מַדְרִיךְ לִבְנֵי מִשְׁפָּחָה הַמִּתְמוֹדְדִים עִם מָוֶות (פְּטִירָה) וקבורה -זכותון לִתְקוּפָה שֶׁלְּאַחַר מוֹת
