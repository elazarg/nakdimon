אֲנִי בְּמַצָּב דּוֹמֶה.. טיפולים לַיֶּלֶד שְׁלִישִׁי הַטִּיפּוּלִים לַיֶּלֶד שְׁלִישִׁי לֹא נִכְלָלִים בְּסַל הַבְּרִיאוּת וְלָכֵן הָעֲלוּיוֹת תְּלוּיוֹת בקופ"ח שֶׁלְּךָ. אֲנִי בכללית ומטופלת באיכילוב. אֲנִי מַנִּיחָה שֶׁתִּצְטָרְכִי לִפְתּוֹחַ תִּיק בְּכָל מָקוֹם שתלכי אֵלָיו, עָדִיף לַעֲשׂוֹת אֶת כָּל הַבְּדִיקוֹת מֵרֹאשׁ (פרופיל הוֹרְמוֹנָלִי, מַחֲלוֹת מִין, כירורג שֵׁד וְכוּ...) בד"כ תּוֹר הוּא לִשְׁבוּעַיִים עַד חוֹדֶשׁ קָדִימָה וְלִפְעָמִים מתפנה מַשֶּׁהוּ מוּקְדָּם יוֹתֵר.. לִפְגִישָׁה עִם הָרוֹפֵא כְּדֵי לְקַבֵּל פְּרוֹטוֹקוֹל אֶפְשָׁר לְנַסּוֹת לְקַבֵּל טוֹפֶס 17 אֲבָל בד"כ לֹא מְאֻשָׁרִים, אָז אֲנִי מֻשְׁלֶמֶת ואנסה לְקַבֵּל החזרים מהמושלם. אַחֲרֵי שֶׁמְּקַבְּלִים פְּרוֹטוֹקוֹל צָרִיךְ לְקַבֵּל אִישּׁוּר מֵרֹאשׁ מהביטוח הַמַּשְׁלִים (כְּלָלִית מוּשְׁלָם) וְרַק אַחֲרֵי זֶה אֶפְשָׁר לְהַתְחִיל. כְּשֶׁיֵּשׁ פְּרוֹטוֹקוֹל זֶה לוֹקֵחַ כַּמָּה יָמִים וְלֹא יוֹתֵר מִזֶּה. מַחְזוֹר טִיפּוּל כּוֹלֵל הַבְּדִיקוֹת מְעֲקֵב והשאיבה עוֹלֶה 1650 ש"ח לֹא כּוֹלֵל תְּרוּפוֹת (שיכולות לְהַגִּיעַ למעל 1000 ש"ח תָּלוּי בְּאוֹרֶךְ הַטִּיפּוּל ובתרופה). הַחֲזָרָה שֶׁל מוקפאים עוֹלֶה 330 שָׂח. מְקַוֶּוה שעזרתי, בְּכָל מִקְרֶה, לְכָל מְכוֹן ivf יֵשׁ בָּאֲתָר אִינְטֶרְנֵט פירוט שֶׁל כָּל הַטְּפָסִים והבדיקות הַנִּדְרָשׁוֹת וַאֲנִי בְּטוּחָה שבטלפון יַעֲנוּ לְךָ בְּסַבְלָנוּת לְכָל שָׁאֲלָה..
