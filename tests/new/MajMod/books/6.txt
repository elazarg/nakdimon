הָעֶרֶב שֶׁלִּי נֶהֱרַס. שְׁנָתִי נָדְדָה. אִילּוּ הִתְרַחֵשׁ הַדָּבָר בְּפָרִיז, הוּא הָיָה מדאיג אוֹתִי בָּהּ בְּמִידָּה אֲבָל מזעזע פָּחוֹת. החרידה אוֹתִי הָעוּבְדָּה כִּי הַשִּׂנְאָה הִגִּיעָה אֶל עָרֵי הַשָּׂדֶה, אֶל פשוטי הָעָם הַתְּמִימִים וטובי הַלֵּב, וננעצה עָמוֹק בְּלִבָּם. בְּיָמִים הַבָּאִים סִיפַּרְתִּי אֶת האפיזודה לידידים. רוּבָּם לֹא הִתְיַיחֲסוּ אֵלֶיהָ בְּכוֹבֶד רֹאשׁ: "כַּמָּה לעגנו אֲנַחְנוּ, הַצָּרְפָתִים, למלכה הַשְּׁמֵנָה וִיקְטוֹרְיָה, וְכַעֲבוֹר שְׁנָתַיִים כרתנו בְּרִית עִם אַנְגְּלִיָּה. אֵינֵךְ מַכִּיר אֶת הַצָּרְפָתִים, הַפּוֹלִיטִיקָה אֵינָהּ חוֹדֶרֶת עָמוֹק לנפשם." רַק רולאן רָאָה זֹאת בָּעֵינַיִים אֲחֵרוֹת: "כְּכָל שהעם תַּם יוֹתֵר, כֵּן קַל יוֹתֵר לרמותו. מֵאָז נִבְחַר פּוּאַנְקָרֶה, הורע הַמַּצָּב. הַמַּסָּע שֶׁלּוֹ לפטרבורג לֹא יִהְיֶה מַסַּע תַּעֲנוּגוֹת." הֶאֱרַכְנוּ בְּשִׂיחָה עַל קוֹנְגְּרֵס הסוציאליסטים הַבֵּינְלְאוּמִּי שיתכנס בַּקַּיִץ בְּוִוינָה, אֲבָל גַּם לְגַבָּיו הָיָה רולאן ספקן מֵהָאֲחֵרִים: "כַּמָּה מֵהֶם יחזיקו מַעֲמָד כשיודבקו צווי הַגִּיּוּס, מִי יוֹדֵעַ? אָנוּ חַיִּים בְּיָמִים שֶׁל תְּחוּשׁוֹת הַמּוֹנִיוֹת, שֶׁל היסטריות הַמּוֹנִיוֹת, שֶׁאֵין לַחֲזוֹת כְּלָל מֵרֹאשׁ אֶת עוֹצְמָתָן בִּימֵי מִלְחָמָה."
