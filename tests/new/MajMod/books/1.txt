חוֹקְרִים רַבִּים לֹא הִשְׁתַּכְנְעוּ ונותרו מבולבלים. לְאַחַר שמחקרו הָרִאשׁוֹן פּוּרְסַם, הִתְקַשְׁרָה אֶל שטרן אִישָּׁה שֶׁבַּעֲלָהּ, חָתָן פְּרַס נוֹבֵּל, סָבַל ממקרה חָמוּר שֶׁל אלצהיימר. "הִיא אָמְרָה, 'עַל מָה אַתָּה מְדַבֵּר, לַעֲזָאזֵל!'" אָמַר שטרן. אף-על-פי שֶׁאִישׁ לֹא טָעַן - אוֹ אוֹמֵר כַּיּוֹם - שהשכלה הִיא עֲרוּבָּה לַהֲגָנָה מִפְּנֵי שיטיון, הִתְקַשּׁוּ לְקַבֵּל אֶת הָרַעְיוֹן שֶׁמַּשֶּׁהוּ אמורפי כְּמוֹ הַשְׂכָּלָה יָכוֹל לְהָגֵן עַל הַמּוֹחַ מִפְּנֵי פְּגִיעָתָהּ הַפִיזִית הממשית שֶׁל מַחֲלָה קָשֶׁה. "פָּשׁוּט לֹא חָשַׁבְנוּ שהמוח עוֹבֵד כָּכָה," אוֹמֵר שטרן. אַךְ בְּמִידָּה הוֹלֶכֶת וְגוֹבֶרֶת נִרְאֶה שֶׁאָכֵן, כָּךְ הוּא עוֹבֵד. מֶחְקָרִים מהימנים נוֹסָפִים, כְּדוּגְמַת מֶחְקָר "המסדרים הַדָּתִיִּים" שֶׁל רֹאשׁ (עוֹד נזירות, וְגַם כְּמָרִים) בְּ-2004, חָשְׂפוּ גַּם שברמת חוּמְרָה נְתוּנָה, נִמְצְאוּ אֵצֶל המשכילים יוֹתֵר גושי סִיבִים ומשקעים - עוֹד הוֹכָחָה, אוּלַי, שבעלי הַהַשְׂכָּלָה הַרְבֵּה יוֹתֵר הָיוּ מוּגַנִּים לְמֶשֶׁךְ זְמַן רַב יוֹתֵר מִפְּנֵי הַשְׁפָּעָתָהּ הַקָּשָׁה בְּיוֹתֵר שֶׁל הַמַּחֲלָה. אוֹתָהּ קְבוּצָה גַּם מָצְאָה שֶׁלֹּא רַק רָמַת הַהַשְׂכָּלָה הָיְתָה קְשׁוּרָה לסכנת השיטיון, אֶלָּא גַּם "פְּעִילוּיוֹת מעוררות-קוגניציה". בַּמֶּחְקָר מכונן, שֶׁנָּתַן דְּחִיפָה לְתַעֲשִׂיַּית התשבצים, מָצְאָה קְבוּצַת רֹאשׁ שבמהלך תְּקוּפָה שֶׁל חָמֵשׁ שָׁנִים הָיָה לְמִי שֶׁהִרְבּוּ לְהַפְעִיל אֶת תָּאֵי הָעֶצֶב שֶׁלָּהֶם סִיכּוּי נָמוּךְ בַּחֲצִי לְפַתֵּחַ אלצהיימר. פְּעִילוּיוֹת מעוררות הוּגְדְּרוּ ככאלה, שֶׁבָּהֶן "חִיפּוּשׂ אוֹ עִיבּוּד מֵידָע" הָיָה חֵלֶק מֶרְכָּזִי, ושניתן לעשותן לְבַד (כְּדֵי לְנַטְרֵל אֶת הַשְׁפָּעַת הַחֲבָרוֹת, שתרומתה לַמּוֹחַ עֲשׂוּיָה לִהְיוֹת אַדִּירָה). כְּלוֹמַר: מִשְׂחָק ברידג' הָיָה מִחוּץ לִתְחוּם, אַךְ קְרִיאַת מגזינים ועיתונים, בִּיקּוּר בספריה, השתעשעות בְּמִשְׂחֲקֵי מִילִּים, הִשְׁתַּתְּפוּת בְּשִׁיעוּרֵי מוּזִיקָה אוֹ לְמִידַּת שָׂפָה זָרָה - כּוּלָּם נֶחְשְׁבוּ.
