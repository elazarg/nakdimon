אֲנִי נִכְנַס לַחֲדַר ומתיישב עַל הַמִּיטָּה וּפוֹתֵחַ אֶת קוּפְסַת הַצְּבָעִים וְאֶת חוֹבֶרֶת הִצְבִּיעָה ומוציא אֶת הָאָדוֹם המסחרר מהקופסה וּמַתְחִיל לִקְרוֹא. בִּתְחִילַּת הַחוֹבֶרֶת מְסוּפָּר הַסִּיפּוּר שֶׁל ג'וֹ. ג'וֹ הוּא שַׁתְיָין שמאבד אֶת אִשְׁתּוֹ, אֶת הָעֲבוֹדָה שֶׁלּוֹ, אֶת כָּל כַּסְפּוֹ, וּבְסוֹפוֹ שֶׁל דָּבָר מַגִּיעַ לָרְחוֹב וְשָׁם הוּא שׁוֹתֶה יַיִן זוֹל יָשָׁר מהבקבוק. לְאוֹרֶךְ כָּל הַתְּקוּפָה הַזֹּאת מְסָרֵב ג'וֹ לְהוֹדּוּת שֶׁיֵּשׁ לוֹ בְּעָיָה וְשֶׁהוּא אִיבֵּד שְׁלִיטָה. הַסִּיפּוּר מְסוּפָּר בְּמִילִּים פְּשׁוּטוֹת ובציורים פְּשׁוּטִים, רִישּׁוּמִים בְּקַו שֶׁל דְּמוּיוֹת ומקומות, וְאֶת אֵלֶּה צָרִיךְ לְמַלֵּא בְּצֶבַע. אִם אֲנִי מֵבִין נָכוֹן, הָרַעְיוֹן הַבְּסִיסִי הוּא שֶׁבִּזְמַן שֶׁאֲנִי מַשְׁקִיעַ זְמַן בצביעת הַתְּמוּנוֹת, אֲנִי אָמוּר לְהַפְנִים אֶת הַסִּיפּוּר הזוועתי שֶׁל ג'וֹ ולשייך אֶת הַזְּוָועָה הַזֹּאת לְמַצָּבִים שׁוֹנִים בַּחַיִּים שֶׁלִּי. אִם ג'וֹ אִיבֵּד שְׁלִיטָה, הֲרֵי גַּם אֲנִי אִיבַּדְתִּי שְׁלִיטָה. אִם ג'וֹ מַגִּיעַ לָרְחוֹב, הֲרֵי כְּדַאי לִי לְהִיזָּהֵר, אַחֶרֶת גַּם אֲנִי אֶהְיֶה הומלס. סוֹף הַסִּיפּוּר שֶׁל ג'וֹ אוֹפְּטִימִי לְמַדַּי: הוּא מוֹדֶה שֶׁאִיבֵּד שְׁלִיטָה ומצטרף לאלכוהוליסטים אנונימים. אַחֵר כָּךְ יֵשׁ שְׁאֵלוֹן וּבוֹ עֶשְׂרִים וְשֶׁבַע שְׁאֵלוֹת שעוסקות בהרגלי הַשְּׁתִיָּיה שֶׁל הנבדק. הַשְּׁאֵלוֹת פְּשׁוּטוֹת, וכולן מְחַיְּיבוֹת תְּשׁוּבוֹת שֶׁל כֵּן וְלֹא. הַאִם יֵשׁ מַצָּבִים שֶׁאֵינְךָ מַרְגִּישׁ בָּהֶם בְּנוֹחַ אִם אֵין בְּנִמְצָא אַלְכּוֹהוֹל? כֵּן. הַאִם קָרָה פַּעַם שהתעוררת בַּבּוֹקֶר לְאַחַר לַיְלָה שֶׁל שְׁתִיָּיה וְלֹא זכרת אֵיפֹה הָיִיתָ בָּעֶרֶב הַקּוֹדֵם? כֵּן. כְּשֶׁאַתָּה פִּיכֵּחַ, הַאִם אַתָּה מִתְחָרֵט עַל דְּבָרִים שֶׁעָשִׂיתָ בהיותך שִׁיכּוֹר? כֵּן. הַאִם הַיָּדַיִים שֶׁלְּךָ רוֹעֲדוֹת בַּבּוֹקֶר שֶׁלְּאַחַר לַיְלָה שֶׁל שְׁתִיָּיה? כֵּן. הַאִם קוֹרֶה שֶׁאַתָּה שתוי בְּמֶשֶׁךְ כַּמָּה יָמִים רְצוּפִים? כֵּן, כֵּן, כֵּן, כֵּן. אֲנִי עוֹנָה בחיוב לְכָל הַשְּׁאֵלוֹת, וְעַל פִּי הַמַּפְתֵּחַ שֶׁבְּסוֹף השאלון, פֵּירוּשׁ הַדָּבָר שֶׁאֲנִי נִמְצָא בַּשְּׁלַבִּים הָאַחֲרוֹנִים שֶׁל אלכוהוליזם כְּרוֹנִי וּמְסוּכָּן. מָה אַתֶּם סחים.
