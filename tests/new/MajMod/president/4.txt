נחמיה, הָרַב נחמיה. אֲנִי עוֹמֵד כָּאן, מוּל מיטתך, מַבִּיט- בילדיך הרכים, עִם הַשֵּׁמוֹת הַיָּפִים כָּל כָּךְ, באשתך, והלב מְסָרֵב לְהַאֲמִין. "הושע נָא, ההרוגה עָלֶיךָ" כָּךְ אָמַרְנוּ, הַבּוֹקֶר, בתפילת הושענא רַבָּה. שָׁנָה שָׁנָה, אָנוּ אוֹמְרִים, אֶת אוֹתָן הַמִּלִּים. אַךְ הַפַּעַם, הַמִּלִּים הַלָּלוּ, הָיוּ מָרוּת מַתְמִיד, כואבות, מטלטלות. כֵּיצַד קוֹרֶה, כָּךְ אָנוּ שׁוֹאֲלִים, שבליל הושענא רַבָּה, נִרְצָחִים יְהוּדִים בְּדַרְכָּם לכותל הַמַּעֲרָבִי. "אנכי אערבנו - מִידַּי תבקשנו" כָּךְ אָמַר יְהוּדָה עַל בִּנְיָמִין, וְכָךְ נהגת אַתָּה, נחמיה, כַּאֲשֶׁר יָצָאת מביתך הַמּוּגָן, לִשְׁמֹעַ זעקות הָעֶזְרָה. אַתָּה נהרגת לְמַעַן עַמְּךָ וּלְמַעַן מדינתך וּלְמַעַן ארצך וּלְמַעַן מולדתך. נָפַלְתָּ עַל משמרתך. הָאַחְרָיוּת, הָעַרְבוּת, מְסִירוּת הַנֶּפֶשׁ עֲבוּר הָאַחֵר, הָיוּ נֵר לרגלך כתלמיד חָכָם, כְּמוֹרֶה, כָּרַב, כחיל לוֹחֵם, וּכְאֵב. בִּמְסִירוּת הַנֶּפֶשׁ שֶׁלְּךָ, בהקרבה היוקדת, באהבת יִשְׂרָאֵל שפעמה בלבך, הִשְׁאַרְתְּ צוֹאָה - צוֹאָה שֶׁל אַהֲבָה והקרבה, לילדיך. גַּם כַּאֲשֶׁר לֹא תִּהְיֶה לְיָדָם, וְלֹא תספיג אֶת חינוכם בַּעֲרָכִים נשגבים, הֵם לְעוֹלָם לֹא ישכחו אֶת הַגְּבוּרָה שֶׁלְּךָ, אֶת הָאַהֲבָה שֶׁלְּךָ. גַּם אֲנַחְנוּ לֹא נִשְׁכַּח. יַקִּירִי, בְּנֵי הַמִּשְׁפָּחָה, תַּלְמִידִים, אָנוּ נִמְצָאִים בַּמַּאֲבָק קָשֶׁה, יומיומי וּמִתְמַשֵּׁךְ. כְּאֵבֵי הבנין - בְּנִין הָאָרֶץ, ובנין יְרוּשָׁלַיִם; כְּאֵבֵי הִגְדִּילָה; הָיוּ וְהֵם עֲדַיִן - כְּאֵבִים עַזִּים. לְעִתִּים, כְּמוֹ הַיּוֹם, אֵלּוּ כְּאֵבֵי תופת שֶׁל מַמָּשׁ. אויבינו, יוֹדְעִים לְהַכְאִיב לָנוּ, אַךְ לֹא לְנַצֵּחַ אוֹתָנוּ. הַמַּאֲבָק בַּטֵּרוֹר, דּוֹרֵשׁ נְחִישׁוּת, ותעצומות נֶפֶשׁ, וְאָנוּ נַגִּיעַ גַּם נַגִּיעַ, לרוצחיהם שֶׁל הַתְּמִימִים והטהורים, למסיתים ולשולחיהם, ונכה בם - מַכָּה קָשֶׁה. בשמכם, ובשם הָעָם כּוּלּוֹ, אֲנִי רוֹצֶה לְחַזֵּק אֶת יְדֵי כּוֹחוֹת הַבִּטָּחוֹן, הַפּוֹעֲלִים בְּשָׁעָה זוֹ, בְּרַחֲבֵי יְרוּשָׁלַיִם, וּבְרַחֲבֵי אֶרֶץ יִשְׂרָאֵל כּוּלָּהּ, בְּצוּרָה נְחוּשָׁה, נֶגֶד הַמְּרַצְּחִים. גָּדַלְתִּי בִּירוּשָׁלַיִם, בִּזְמַן שנאסר עָלֵינוּ בְּכוֹחַ הַזְּרוֹעַ, לְהַגִּיעַ אֶל הכתל הַמַּעֲרָבִי וְאַף הָיִיתִי יֶלֶד, בְּיָמִים, בָּהֶם תקיעת שופר בְּרַחֲבַת הכותל, הָיְתָה עִילָּה למעצר בִּידֵי הַמַּנְדָּט הַבְּרִיטִי; הַיָּמִים הָאֵלּוּ לֹא יָשׁוּבוּ. אָסוּר לָנוּ, בְּשׁוּם פָּנִים וָאוֹפֶן להרתע, מלהגיע לכותל הַמַּעֲרָבִי, שָׂרִיד בֵּית מקדשנו. זָכִיתִי בְּעַצְמִי הַשָּׁנָה, לְהַגִּיעַ לכותל, בְּמַעֲמָד הַקָּהָל - בְּנוֹכְחוּת הֲמוֹנֵי בֵּית יִשְׂרָאֵל. אֲנִי קוֹרֵא, לְכָל הָעוֹלִים לְרֶגֶל: אֶל תדירו רגליכם מֵהָעִיר הָעַתִּיקָה, עָלֵינוּ לִצְעוֹד בְּדַרְכָּם, שֶׁל הָרַב נחמיה וְשֶׁל אַהֲרֹן, ולהוכיח בְּמַעֲשִׂים, שהטרור לֹא יַצְלִיחַ לִפְגּוֹעַ באורחות חַיֵּינוּ.
