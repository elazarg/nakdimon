לִפְנֵי כְּחוֹדֶשׁ צִייְּנוּ מַנְהִיגֵי אֵירוֹפָּה מֵאָה שָׁנָה לַקְּרָב ורדן. הַקְּרָב הָאָרוֹךְ בְּיוֹתֵר בְּמִלְחֶמֶת הָעוֹלָם הָרִאשׁוֹנָה בּוֹ נִפְגְעוּ 900 אֶלֶף בְּנֵי אָדָם, מִתּוֹכָם 300 אֶלֶף הֲרוּגִים. בצבאות הַמְּדִינוֹת הַלּוֹחֲמוֹת הִשְׁתַּתְּפוּ כַּחֲצִי מִילְיוֹן חַיָּילִים יְהוּדִים שניצבו פְּעָמִים רַבּוֹת אֵלֶּה מוּל אֵלֶּה וירו אֵלֶּה בָּאֵלֶּה. אֶלָּא, שהקזת הַדָּם הפטריוטית הַזּוֹ לֹא הקנתה לְעַם הַיְּהוּדִי פריווילגיות, לְהֶפֶךְ. יְהוּדִים בְּנֵי כָּל הָאֲרָצוֹת הואשמו בריגול ובבגידה. בְּגֶרְמַנְיָה הֵחֵלָּה 'ספירת רָאשִׁים' שֶׁל חַיָּילִים יְהוּדִים, בְּמִזְרַח אֵירוֹפָּה חַיָּילִים יְהוּדִים הואשמו בבולשביקיות ובאנטי בולשביקיות. מַעֲרִיכִים כִּי לְבַד מִכְּ-170 אֶלֶף חַיָּילִים יְהוּדִים שֶׁנֶּהֶרְגוּ בְּמִלְחֶמֶת הָעוֹלָם הָרִאשׁוֹנָה, נִרְצְחוּ בְּמִזְרַח אֵירוֹפָּה עוֹד כְּ 100 אֶלֶף יְהוּדִים בפרעות לְאַחַר הַמִּלְחָמָה. הַמִּלְחָמָה הַנּוֹרָאָה הַהִיא, מִלְחֶמֶת הָעוֹלָם הָרִאשׁוֹנָה, הָיְיתָה הָאוֹת לאקסודוס, "יְצִיאַת אֵירוֹפָּה", שֶׁל הָעָם הַיְּהוּדִי. יְהוּדִים רַבִּים רָאוּ בְּאַרְצוֹת הַבְּרִית שֶׁל אָמֵרִיקָה אֶת בֵּיתָם הֶחָדָשׁ. אֲחֵרִים, מַנְהִיגֵי הַצִּיּוֹנוּת, שחוו טלטלה מעמידתם בִּשְׂדֵה הַקְּרָב אֶל מוּל אחיהם-אויביהם הַיְּהוּדִים, פָּעֲלוּ לְהַגְשָׁמַת הַחֲלוֹם שֶׁל בֵּית לְאוּמִּי לְעַם הַיְּהוּדִי בְמוֹלַדְתוֹ הָעַתִּיקָה והנצחית, בָּאָרֶץ יִשְׂרָאֵל. מִנְּקוּדַּת מבטנו הַיּוֹם, מִלְחֶמֶת הָעוֹלָם הָרִאשׁוֹנָה הָיְיתָה רַק קדימון לְמִלְחָמָה נוֹרָאָה עוֹד יוֹתֵר שֶׁהִגִּיעָה אַחֲרֶיהָ, מִלְחֶמֶת הָעוֹלָם הַשְּׁנִיָּיה. בַּמִּלְחָמָה הַשְּׁנִיָּיה, הַשְׁמָדַת הָעָם הַיְּהוּדִי הָיְיתָה כְּבָר יַעַד מוּגְדָּר שֶׁל מַעַצְמוֹת הַצִּיר ובראשן גֶּרְמַנְיָה הַנָּאצִית. יְצִיאַת אֵירוֹפָּה הַשְּׁנִיָּיה שֶׁל הָעָם הַיְּהוּדִי התרחשה עִם תבוסתו שֶׁל הָרַיְיךְ הַשְּׁלִישִׁי. זוֹ הָיְיתָה הגירה טראומתית שֶׁל פְּלִיטוֹת וּכְאֵב, שֶׁל תַּרְבּוּת שהוחרבה, שֶׁל עִם שֶׁאִיבֵּד שְׁלִישׁ מבניו ובנותיו.
