הַפּוֹעֵל נצרת עִילִּית שְׁמָהּ הָעֶרֶב (שֶׁ') בַּצַּד אֶת הַצָּרוֹת הַכַּלְכָּלִיּוֹת, ורשמה נִיצָּחוֹן 0:1 דְּרָמָטִי בַּמִּשְׂחָק הַבֵּיתִי מוּל מַכַּבִּי הֶרְצְלִיָּה. שַׁעַר שֶׁל אֶרֶז פינטו דקתיים לְסִיּוּם, הֶעֱנִיק אֶת שָׁלוֹשׁ הַנְּקוּדּוֹת לצפוניים, שֶׁחָזְרוּ למאבקי הַצַּמֶּרֶת. הַכֹּחַ עֲמִידָר רָמַת גַּן שׁוּב לֹא נִיצְּחָה, כַּאֲשֶׁר סִייְּמָה בְּ-1:1 בֵּיתִי מוּל סקציה נֵס צִיּוֹנָה. קוֹבִּי חסן כָּבַשׁ אֶת שְׁעָרוֹ הַשֵּׁנִי בְּמַדֵּי הסגולים תֵּשַׁע דַּקּוֹת לְסִיּוּם, יוֹסִי אסייג הישווה שֶׁבַע דַּקּוֹת לְאַחַר מִכֵּן. מִי שֶׁחָזְרָה לְחַיֵּיךְ הִיא הַפּוֹעֵל כְּפָר סַבָּא, שגברה בְּבֵיתָהּ 1:2 עַל הַפּוֹעֵל רִאשׁוֹן לְצִיּוֹן. חַמָאדֵה מרעי (דַּקָּה 10) וְנָתַן שקורי (40) הרשיתו עֲבוּר הַקְּבוּצָה שֶׁל תּוֹמֶר קשטן, אלירן אפריאט איזן זְמַנִּית בַּדַּקָּה ה-20. אַחֲרֵי אַרְבָּעָה נִיצְחוֹנוֹת רְצוּפִים, אֶחֱוֶוה עראבה נֶעֶצְרָה בִּבְאֵר שֶׁבַע אֵצֶל מַכַּבִּי הַמְּקוֹמִית. רוֹנֵן שוורצמן דַּוְוקָא הֶעֱלָה אֶת הַקְּבוּצָה שֶׁל אֱיָיל לחמן לְיִתְרוֹן בַּדַּקָּה ה-68, אָבִי מַלְכָּה (74) וינון בַּרְדָּה (80) דָּאֲגוּ לְמַהְפָּךְ מְאוּחָר. אֶתְמוֹל, המוליכה עִירוֹנִי קרית שְׁמוֹנֶה, קִיבְּלָה מָנָה נְדִיבָה שֶׁל מַזָּל בְּיוֹם שִׁישִּׁי. פַּעַם אַחַת כשהצליחה לְנַצֵּחַ בטדי אֶת הַפּוֹעֵל יְרוּשָׁלַיִם 2:4 עִם צֶמֶד שְׁעָרִים שֶׁל תּוֹמֶר סוויסה בַּדַּקּוֹת הָאַחֲרוֹנוֹת. פַּעַם שְׁנִיָּה כשיריבתה הַגְּדוֹלָה בְּצַמֶּרֶת, עִירוֹנִי רָמַת הַשָּׁרוֹן, הִפְסִידָה בְּבֵיתָהּ להפועל אשקלון 2:0. בְּנֵי לוֹד נִיצְּחָה בְּאוֹתָהּ תּוֹצָאָה אֶת הַפּוֹעֵל מרמורק בָּרְחוֹבוֹת.
