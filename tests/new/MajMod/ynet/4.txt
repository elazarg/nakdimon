האווירה הַחוֹרְפִּית נִמְשֶׁכֶת. מָחָר (יוֹם ד') יִהְיֶה גשום לִפְרָקִים, בְּעִיקָּר בִּצְפוֹן וּבַמֶּרְכָּז הָאָרֶץ. הַלַּיְלָה יֵירֵד גֶּשֶׁם מְקוֹמִי בְּעִיקָּר בְּמֶרְכַּז הָאָרֶץ, וְיִיתָּכֵן גַּם גֶּשֶׁם מְקוֹמִי קַל בִּצְפוֹן וּבִדְרוֹם. מָחָר ייחלשו הַגְּשָׁמִים ויתמעטו לִקְרַאת עֶרֶב. מַחְבֶּרֶת "מטאו טֵק" נִמְסַר כִּי תָּחוּל יְרִידָה קַלָּה בְּטֶמְפֵּרָטוּרוֹת. אפיק רדוד בַּקַּרְקַע עֲדַיִין מותיר אֶת האטמוספרה באזורנו בִּלְתִּי יַצִּיבָה עִם סִיכּוּי לגשמים מְקוֹמִיִּים, בְּעִיקָּר בִּצְפוֹן וּבַמֶּרְכָּז. הָרוּחַ בַּיָּם הַתִּיכוֹן תִּהְיֶה דְּרוֹמִית עַד דְּרוֹם מערבית, בִּמְהִירוּת 10 עַד 25 קָמָ"שׁ וְהַיָּם עַצְמוֹ יִהְיֶה גַּלֵּי עַד גַּבָּהּ גַּלִּים, בְּגוֹבַהּ 80-150 ס"מ. הַטֶּמְפֵּרָטוּרוֹת החזויות הַלַּיְלָה וּמָחָר: קִרְיַית שְׁמוֹנֶה 8-16, קצרין 7-14, צפת 6-13, טְבֶרְיָה 12-19, נצרת 9-14, חֵיפָה 12-19, תל-אביב 12-18, יְרוּשָׁלַיִם 9-14, אשקלון 12-21, עַיִן גֶּדִי 14-23, בְּאֵר שֶׁבַע 11-19, עֲרָד 10-16, מְצַפֶּה רמון 7-14, אילת 12-23. בְּיוֹם חֲמִישִׁי יִהְיֶה מעונן חֶלְקִית וְיִיתָּכֵן גֶּשֶׁם מְקוֹמִי בְּעִיקָּר בִּצְפוֹן הָאָרֶץ ומרכזה. תָּחוּל עֲלִיָּיה בְּטֶמְפֵּרָטוּרוֹת. יִיתָּכֵן גֶּשֶׁם מְקוֹמִי קַל בִּדְרוֹם. בְּיוֹם שִׁישִּׁי יֵירֵד גֶּשֶׁם לִפְרָקִים, מְלַוֶּוה בְּסוּפוֹת רְעָמִים בּוֹדְדוֹת, בְּעִיקָּר בִּצְפוֹן הָאָרֶץ. ינשבו רוּחוֹת דְּרוֹם מערביות עֵרוּת. בְּמֶרְכַּז וּבִדְרוֹם יִיתָּכֵן אובך וְאַף יִיתָּכְנוּ סוּפוֹת חוֹל מְקוֹמִיּוֹת שיתחוללו בִּדְרוֹם הָאָרֶץ. בְּשַׁבָּת יֵירֵד גֶּשֶׁם לִפְרָקִים מְלַוֶּוה בְּסוּפוֹת רְעָמִים בִּצְפוֹן וּבַמֶּרְכָּז הָאָרֶץ. ינשבו רוּחוֹת עֵרוּת. חֲשָׁשׁ לשיטפונות וְהַצָּפוֹת בִּמְקוֹמוֹת הַמּוֹעֲדִים. תָּחוּל יְרִידָה בְּטֶמְפֵּרָטוּרוֹת. בְּחֶרְמוֹן יִיתָּכֵן שֶׁלֶג.
