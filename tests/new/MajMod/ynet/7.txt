"בַּנְק יִשְׂרָאֵל מְצַפֶּה מהתאגידים הבנקאיים, להטמיע אֶת הַכֵּלִים לְצוֹרֶךְ מְנִיעָה ואיתור אֵירוּעִים שֶׁל התממשות סִיכּוּן, וּבְאוֹפֶן שיבטיח כִּי יישמר נָתִיב בִּיקּוֹרֶת שֶׁיְּאַפְשֵׁר אֶת שחזור הָאֵירוּעַ החריג, וקיום מַעֲרַךְ תְּגוּבָה וְדִיוֵּוחַ עַל אֵירוּעִים חֲרִיגִים", כָּךְ אָמְרָה רחל יעקבי, מְנַהֶלֶת יְחִידַת בִּיקּוֹרֶת מַעַרְכוֹת מֵידָע בפיקוח עַל הַבַּנְקִים בְּבַנְק יִשְׂרָאֵל. אֶת הַדְּבָרִים אָמְרָה יעקובי עַל רֶקַע הָאֵירוּעִים הפליליים שקרו לָאַחֲרוֹנָה בְּמַעֲרֶכֶת הַבַּנְקִים, בָּהֶם הִצְלִיחוּ כנופיות לְזַיֵּיף כַּרְטִיסֵי אַשְׁרַאי ומשכו כֶּסֶף מלקוחות תְּמִימִים. "יֵשׁ לִפְעוֹל לִמְנִיעַת אֶפְשָׁרוּת שֶׁבָּהּ תַּאֲגִיד בנקאי ייקלע לארוע חָרִיג, מִבְּלִי שֶׁבְּיָדָיו אֶמְצָעִים מְסַפְּקִים לְשַׁחְזֵר אֶת הָאֵירוּעַ ולהתחקות אַחֵר מאפייניו, עַל מְנַת לְהָפִיק לקחים אופרטיביים וְלִמְנוֹעַ הַיְּשָׁנוֹת שֶׁל אֵירוּעִים דּוֹמִים בֶּעָתִיד", אָמְרָה יעקובי. לִדְבָרֶיהָ, הַפִּיקּוּחַ עַל הַבַּנְקִים הִסְדִּיר לָאַחֲרוֹנָה אֶת תְּחוּם אַבְטָחַת הַמֵּידָע בְּמַעֲרֶכֶת הַבַּנְקָאִית עַל פִּי סטנדרטים מַחְמִירִים ומתקדמים עַל פִּי קָנָה מִידָּה בֵּינְלְאוּמִּי. בַּנְק יִשְׂרָאֵל שָׁם דָּגֵשׁ עַל חִיזּוּק מְעוֹרָבוּת הדירקטוריון וההנהלה בִּתְחוּם זֶה, "בְּנִיגּוּד לגישות שרווחו בֶּעָבָר ושגרסו כִּי הָאַחְרָיוּת בִּתְחוּמִים אֵלּוּ מוּטֶּלֶת אַךְ וְרַק עַל פונקציית טֶכְנוֹלוֹגְייַת הַמֵּידָע בתאגיד הַבַּנְקָאִי", אָמְרָה יעקבי. יעקבי צִייְּנָה בְּנוֹסָף אֶת חוֹבַת הַבַּנְקִים לְקַיֵּים הַעֲרָכַת סִיכּוּנִים עַל בָּסִיס מִתְמַשֵּׁךְ, לַעֲקוֹב אַחֵר הִתְפַּתְּחוּיוֹת טֶכְנוֹלוֹגְיוֹת וְלַעֲרוֹךְ סקרי בְּטִיחוּת וְנִיסְיוֹנוֹת חֲדִירָה מבוקרים מֵעֵת לְעֵת, בְּמַטָּרָה לְהַעֲרִיךְ אֶת האפקטיביות שֶׁל אֶמְצְעֵי הָאַבְטָחָה.
