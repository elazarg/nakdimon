דּוּ"חַ מְיֻחָד שֶׁל וַעֲדָה לִבְדִיקַת הַפְּעָרִים הַחֶבְרָתִיִּים בַּכְּנֶסֶת קוֹבֵעַ:
הַחֶבְרָה הַיִּשְׂרְאֵלִית בִּבְעָיָה קָשָׁה
מִסְפַּר הַמִּשְׁפָּחוֹת הָעֲנִיוֹת גָּדֵל בַּשָּׁנִים הָאַחֲרוֹנוֹת * בִּתְחוּם הַחִנּוּךְ יֵשׁ פְּעָרִים גְּדוֹלִים וּמִשְׂרַד הַחִנּוּךְ לֹא עוֹשָׂה הַרְבֵּה לִפְתּוֹר אֶת הַבְּעָיָה * הַנָּשִׂיא קֶצֶב הִבִּיעַ תִּקְוָה, שֶׁהַפְּעָרִים יִצְטַמַצְמוּ * יוֹ"ר הַכְּנֶסֶת: "אֲפִלוּ טַנְקִים לֹא יוּכְלוּ לְהָגֵן עַל חֶבְרָה שֶׁפְּגוּעָה מִבִּפְנִים" * 
וַעֲדַת חֲקִירָה שֶׁל הַכְּנֶסֶת, הַגִּישָׁה בַּשָּׁבוּעַ שֶׁעָבַר אֶת תּוֹצְאוֹת הַמֶּחְקָר בְּנוֹשֵׂא הַפְּעָרִים הַחֶבְרָתִיִּים בְּיִשְׂרָאֵל. מֵהַדּוּ"חַ, שֶׁהָגַּשׁ עַל יְדֵי חֲבֵר הַכְּנֶסֶת רָן כֹּהֵן עוֹלֶה, שֶׁהַפַּעַר בֵּין הָעֲנִיִּים לַעֲשִׁירִים גָּדֵל.
בִּתְחוּם הַחִנּוּךְ, קוֹבַעַת הַוַּעֲדָה שֶׁבַּאֲזוֹרִים שֶׁל מְצוּקָה כַּלְכָּלִית, הַחִנּוּךְ פָּחוֹת טוֹב. "מַעֲרֶכֶת הַחִנּוּךְ לֹא פּוֹתֶרֶת אֶת הַבְּעָיָה שֶׁל חֹסַר הַשִּׁוְיוֹן בְּחִנּוּךְ. בְּמִקְרִים רַבִּים הִיא אֲפִלוּ עוֹזֶרֶת בְּקִיּוּם אוֹתָם פְּעָרִים", נֶאֱמַר בַּדוּ"חַ הַוַּעֲדָה.
נְשִׂיא הַמְּדִינָה, מֹשֶׁה קֶצֶב, שֶׁקִבֵּל לְיָדָיו אֶת הַדּוּ"חַ הִבִּיעַ תִּקְוָה, שֶׁדְּבָרִים בַּמְּדִינָה יַתְחִילוּ לְהִשְׁתַּנּוֹת וְהַפְּעָרִים הַקָּשִׁים יַתְחִילוּ לְהִצְטַמְצֵם.
אַבְרָהָם בּוֹרֵג, יוֹ"ר הַכְּנֶסֶת אָמַר: "אֵין טַנְק אוֹ מָטוֹס בָּעוֹלָם שֶׁיָּכוֹל לְהָגֵן עַל חֶבְרָה שֶׁנִּפְגַּעַת מִבִּפְנִים. אֲנַחְנוּ מִתַּחַת לְקַו הָאָדָם בַּתְּחוּם הַחֶבְרָתִי וַאֲנַחְנוּ חַיָּבִים לְשַׁנּוֹת אֶת הַמַּצָּב".
הַחוֹקְרִים מָצְאוּ כִּי בָּאָרֶץ יֵשׁ מִסְפָּר קָטָן שֶׁל אֲנָשִׁים הַמַּחֲזִיק בָּרַב הַכְּסָפִים בַּמְּדִינָה. עוֹד קָבַע הַדּוּ"חַ, שֶׁבְּכָל שָׁנָה נוֹסָפוֹת עוֹד 50 אֶלֶף מִשְׁפָּחוֹת לְמַּעְגָּל הֶעָנִי.
