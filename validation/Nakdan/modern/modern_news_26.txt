סוֹנִי חָשְׂפָה אֶת מִפְרַט הַפְלֶיְיסְטֵיְישְׁן 5: אֵיךְ הִיא מִתְמוֹדֶדֶת מוּל הַיְּרִיבָה מִבֵּית מַיְקְרוֹסוֹפְט?
יוֹמַיִים אַחֲרֵי שֶׁמַּיְקְרוֹסוֹפְט חָשְׂפָה אֶת מִפְרַט קוֹנְסוּלַת הַאקְסְבוּקְס הַחֲדָשָׁה שֶׁלָּהּ, גַּם סוֹנִי הִצִּיגָה אֶת הַדּוֹר הַחֲמִישִׁי שֶׁל קוֹנְסוּלַת הַמִּשְׂחָקִים הַמַּצְלִיחָה שֶׁלָּהּ.
הָאֵם יַצְלִיחוּ לְהָבִיס אֶת מַחְשְׁבִי הַגֶּיְימִינְג הָעָוצְמָתִיִּים שֶׁנִּמְצָאִים כַּיּוֹם בַּשּׁוּק?

יוֹמַיִים אַחֲרֵי שֶׁמַּיְקְרוֹסוֹפְט חָשְׂפָה אֶת מִפְרַט קוֹנְסוּלַת הָאֶקְסְבּוּקְס Series X, גַּם סוֹנִי חָשְׂפָה אֶמֶשׁ (ד') אֶת הַיּוֹרֶשֶׁת הַבָּאָה לְבֵית פְּלֶיְיסְטֵיְישְׁן, הַ-5.
כָּאָמוּר, מְדוּבָּר בְּמַאֲבָק בַּפִּסְגָּה, הַמִּתְחָרָה הַשְּׁלִישִׁית, נִינְטֶנְדּוֹ, מְשַׂחֶקֶת בְּלִיגָה מִשֶּׁלָּהּ וְאֵינָהּ מִתְעָרֶבֶת בְּמִלְחֶמֶת סוֹנִי-מַיְקְרוֹסוֹפְט.
חֲשִׂיפַת הַקּוֹנְסוּלוֹת מִתְבַּצַּעַת בְּעוֹלָם גֶּיְימִינְג מִשְׁתַּנֶּה.
שֵׁירוּתֵי הַסְּטְרִימִינְג לַמִּשְׂחָקִים כְּגוֹן סְטַדְיָה שֶׁל גּוּגֶל אוֹ Now שֶׁל אַנְבִּידֶיהָ עֲדַיִין לֹא תָּפְסוּ מִסִּיבּוֹת שׁוֹנוֹת, וְהַקּוֹנְסוּלוֹת דּוֹר 5 כַּנִּרְאֶה כָּאן כְּדֵי לְהוֹבִיל אֶת שׁוּק הַגֶּיְימִינְג לְעוֹד כַּמָּה שָׁנִים טוֹבוֹת.

הַפְלֶיְיסְטֵיְישְׁן 5 עֲדַיִין לֹא הוּצְּגָה בִּמְלוֹא הֲדָרָה.
סוֹנִי רַק הֶרְאֲתָה דִּיאַגְרָמוֹת טֶכְנִיּוֹת וְצִילּוּמֵי שְׁבָבִים כְּדֵי לְהַמְחִישׁ אֶת הַיְּכוֹלוֹת שֶׁלָּהּ.
לֹא בָּרוּר אִם הַקּוֹנְסוּלָה עֲדַיִין לֹא מוּכָנָה מִבְּחִינַת עִיצּוּב אוֹ שֶׁסּוּנִי פָּשׁוּט רוֹצָה לִמְשׁוֹךְ עוֹד קְצָת אֶת הַצְּפִיָּיה שֶׁל חוֹבְבֵי הַמּוּתָג.
כָּךְ אוֹ כָּךְ, הַנְּתוּנִים שֶׁהִצִּיגָה הַחֶבְרָה מַסְפִּיק מְפוֹרָטִים תֵּיאוֹרֵטִית כְּדֵי שֶׁנּוּכַל לְהַצִּיג אוֹתָם.
כְּמוֹ עִם הָאַקְסְבּוּקְס, נְנַסֶּה לְהִימָּנַע מִסִּיפּוּרִים טֶכְנִיִּים וּלְהִתְמַקֵּד בְּמַשְׁמָעוּיוֹת מִבְּחִינַת הַגֶּיְימֵרִים.
עִם זֹאת חַיָּיבִים הַקְדָּמָה שֶׁתְּתָאֵר אֶת הַתְּכוּנוֹת.
 
הַפְלֶיְיסְטֵיְישְׁן 5 זוֹכֶה לְמִפְרָט מְאוֹד דּוֹמֶה לְזֶה שֶׁל הָאַקְסְבּוּקְס.
בְּחֵלֶק מֵהַתְּכוּנוֹת הִיא קְצָת פָּחוֹת מוֹבִילָה וּבְחֵלֶק הִיא כֵּן.
הַלֵּב שֶׁל הַקּוֹנְסוּלָה הוּא מְעַבֵּד AMD Zen 2 בְּדִיּוּק כְּמוֹ זֶה שֶׁמַּפְעִיל אֶת הָאַקְסְבּוּקְס, אַךְ בְּ-PS5 הוּא טִיפָּה פָּחוֹת מָהִיר עִם 3.5 גִּיגָה-הָרָץ מַקְסִימָלִיִּים לְכָל לִיבָּהּ (וְיֵשׁ 8 מֵהֶן).
גַּם הַשְּׁבָב הַגְּרָפִי הוּא אוֹתוֹ אֶחָד, AMD RDNA 2 וְגַם כָּאן הוּא מְעַט יוֹתֵר אִיטִּי מְאַשֵּׁר אֵצֶל הַיְּרִיבָה.
זִיכְרוֹן הַפְּעוּלָּה עוֹמֵד עַל 16 גִּיגָה-בַּיֵּיט GDDR6 כְּמוֹ בְּאַקְסְבּוֹקְס בְּדִיּוּק וּמֶרְחָב הָאִחְסוּן מַצִּיעַ כְּ-825 גִּיגָה-בַּיֵּיט - כְּ-175 גִּיגָה-בַּיֵּיט פָּחוֹת מֵאֲשֶׁר הַיְּרִיבָה - אֲבָל יֵשׁ כָּאן טְוִויסְט אֵלָיו אֶתְיַיחֵס מִיָּד.

גַּם מִבְּחִינַת הַתְּכוּנוֹת אֵין מַמָּשׁ הֶבְדֵּל, לִשְׁתֵּי הַקּוֹנְסוּלוֹת תְּמִיכָה בְּרֵזוֹלוּצְיוֹת גְּבוֹהוֹת 4K וְ-8K וְכֵן תְּמִיכָה בְּ-HDR לְשִׁיפּוּר הַנִּרְאוֹת שֶׁל הַתְּצוּגָה.
לִשְׁתֵּיהֶן חֲרִיץ הָרְחָבָה לְמֶרְחַב הָאִחְסוּן, אַךְ בְּעוֹד בְּאַקְסְבּוּקְס יוֹדְעִים לוֹמַר עַל חִיבּוּר USB לְכוֹנָנִים חִיצוֹנִיִּים, בְּסוֹנִי הִתְעַלְּמוּ מִכָּךְ בְּאֵלֵגַנְטִיּוּת.
אַךְ קָשֶׁה לְהַאֲמִין שֶׁסּוֹנִי לֹא תְּצַיֵּיד אֶת הַקּוֹנְסוּלָה שֶׁלָּהּ בִּלְפָחוֹת חִיבּוּר אֶחָד כָּזֶה וְלוֹ כְּדֵי לְחַבֵּר הֶתְקֵן חִיצוֹנִי כָּלְשֶׁהוּ.

וְעַכְשָׁיו נַעֲבוֹר לַהֶבְדֵּלִים וּנְנַסֶּה לְנַתֵּחַ אוֹתָם עַל בְּסִיס מָה שֶׁאֲנַחְנוּ יוֹדְעִים כָּרֶגַע.
בְּעוֹד הָאֲקַסְבּוּקְס נֶחְשְׂפָה בְּאוֹפֶן מִיַּקְרוֹסוֹפֵּטִי מְאוֹד עִם רַעַשׁ וְצִלְצוּלִים.
מְסִיבַּת הָעִיתּוֹנָאִים שֶׁל סוֹנִי הָיְתָה מִינוֹרִית וְהִתְמַקְּדָה בְּלַהַג טֶכְנִי.
וְהַסִּיבָּה לְכָךְ הִיא שֶׁהַקּוֹנְסוּלָה הַיַּפָּנִית מַצִּיגָה (לְפָחוֹת עַל הַנְּיָיר) יְכוֹלוֹת פְּחוּתוֹת מְאַשֵּׁר הַיְּרִיבָה הָאָמֵרִיקָאִית.
מְהִירוּת הַמְּעַבֵּד, מֶרְחַב הָאִחְסוּן וְעָוצְמַת הַשְּׁבָב הַגְּרָפִי, שֶׁלְּהַזְכִּירְכֶם הֵם מֵאוֹתוֹ הַמָּקוֹר בִּשְׁתֵּיהֶן, מְסַפְּקוֹת יִתְרוֹן מִסְפָּרִי בָּרוּר לְאֶקְסְבּוּקְס.
 
אָבֵל בְּסוֹנִי צִייְּדוּ אֶת הַ-PS5 בְּיִתְרוֹן אַדִּיר וּכְדֵי לְהַסְבִּיר אוֹתוֹ הָיוּ צְרִיכִים לְגַיֵּיס אֶת מְנַהֵל הַפִּיתּוּחַ הַטֶּכְנִי - דְּמוּת שֶׁבְּדֶרֶךְ כְּלָל אֵינָהּ מוֹפִיעָה בַּצִּיבּוּר - כְּדֵי לְהַסְבִּיר אוֹתוֹ.
הַפִּילוֹסוֹפְיָה שֶׁל סוֹנִי, הִיא לְהַפְחִית אֶת מִסְפַּר לִיבּוֹת הַחִישּׁוּב שֶׁל הַשְּׁבָב הַגְּרָפִי אַךְ לָתֵת לְכָל אַחַת מֵהֶן עָוצְמַת חִישּׁוּב הַרְבֵּה יוֹתֵר גְּבוֹהָה.
כָּךְ לְמָשָׁל בְּאֶקְסְבּוּקְס מְדוּבָּר בְּ-52 יְחִידוֹת חִישּׁוּב וּמְהִירוּת שֶׁל 1.825 גִּיגָה-הֵרְץ.
בְּ-PS5 מְדוּבָּר בְּ-36 יְחִידוֹת חִישּׁוּב שֶׁפּוֹעֲלוֹת בִּמְהִירוּת שֶׁל 2.23 גִּיגָה-הֵרְץ.
לֹא בָּרוּר עֲדַיִין מֵי תְּסַפֵּק אֶת הַסְּחוֹרָה יוֹתֵר בִּיעִילוּת וְנֶאֱלַץ לְהַמְתִּין וְלִרְאוֹת אֶת הַמִּשְׂחָקִים שֶׁיַּעֲשׂוּ שִׁימּוּשׁ בַּתְּכוּנוֹת הָאֵלֶּה כְּדֵי לְהָבִין זֹאת.
בֵּינְתַיִים וּלְפָחוֹת עַד סוֹף הַשָּׁנָה הַקְּרוֹבָה, אַף מִשְׂחָק לֹא יְנַצֵּל אֶת הַתְּכוּנוֹת הָאֵלֶּה.
חֲשִׂיפַת הַמִּפְרָט הַטֶּכְנִי שֶׁל הַסּוּנִי פְּלֶיְיסְטֵיְישְׁן 5

הַקְּלָף הֶחָבוּי הַנּוֹסָף הוּא בִּדְמוּת מְהִירוּת הַגִּישָׁה לְמֶרְחַב הָאִחְסוּן.
נָתוּן זֶה חָשׁוּב שֶׁכֵּן הוּא קוֹבֵעַ כַּמָּה זְמַן תַּמְתִּינוּ עַד שֶׁהַקּוֹנְסוּלָה תִּטְעַן אֶת הַמִּשְׂחָק אוֹ תַּעֲבִיר אֶתְכֶם מִמָּסַךְ לַמָּסָךְ.
זְמַנֵּי הַמְתָּנָה גְּבוֹהִים (שֶׁאֵצֶל גֶּיְימֵרִים נִמְדָּדִים בְּעֶשְׂרוֹת שְׁנִיּוֹת) מְיַיצְּרִים תִּסְכּוּל וַעֲשׂוּיִים לִפְגּוֹם בַּחֲוָויַית הַשִּׁימּוּשׁ.
אִם בְּאַקְסְבּוֹקְס, מַיְקְרוֹסוֹפְט "הִסְתַּפְּקָה" בִּמְהִירוּת גִּישָׁה שֶׁל 2.4 גִּיגָה-בִּיט לִשְׁנִיָּיה, בְּסוּנִי מַצִּיגִים נָתוּן יוֹתֵר מִכָּפוּל עִם 5.5 גִּיגָה-בִּיט לִשְׁנִיָּיה.
 
זֹאת וְעוֹד, בְּמַיְקְרוֹסוֹפְט מִשְׁתַּמְּשִׁים בְּפִתְרוֹן קִנְיָינִי שֶׁהַחֶבְרָה פִּיתְּחָה בִּמְיוּחָד לְאַקְסְבּוֹקְס, וְלָכֵן לַמְרוֹת שֶׁיֵּשׁ חֲרִיץ הָרְחָבָה הוּא תּוֹמֵךְ רַק בְּהֶתְקֵן אִחְסוּן הַמְּיוּצָּר עַל יְדֵי סִיגֵיְיט בְּשִׁיתּוּף עִם מַיְקְרוֹסוֹפְט.
כָּךְ שֶׁכְּכָל הַנִּרְאֶה שֶׁהוּא יִימָּכֵר בִּמְחִיר קָבוּעַ שֶׁמַּיְקְרוֹסוֹפְט תִּנְקוֹב - וְעַל פִּי הַנִּיסָּיוֹן הַתִּמְחוּר שֶׁלָּהּ נוֹטֶה לִהְיוֹת יָקָר בְּיַחַס לַשּׁוּק.
 
מִנֶּגֶד סוֹנִי לֹא הִמְצִיאָה אֶת הַגַּלְגַּל מֵחָדָשׁ.
הִיא הָלְכָה לַתֶּקֶן PCIe 4.0 הֶחָדָשׁ שֶׁמְּיוֹעָד גַּם לְהִיכָּנֵס לְשִׁימּוּשׁ בְּמַחְשְׁבִים.
הַתֶּקֶן הַזֶּה מָהִיר מְאוֹד אַךְ בְּעִיקָּר הוּא יְאַפְשֵׁר לַמִּשְׁתַּמְּשִׁים לְהַתְקִין כּוֹנָנִים מִיַּצְרָנִים שׁוֹנִים.
תַּחֲרוּת זוֹ גַּם תְּסַיֵּיעַ לִשְׁמוֹר אֶת הַמְּחִירִים נְמוּכִים יוֹתֵר.
וּבְהִתְחַשֵּׁב בְּכָךְ שֶׁבֶּעָתִיד הַקָּרוֹב מְשַׂחֲקִים כַּנִּרְאֶה יִדְרְשׁוּ מֶרְחַב אִחְסוּן זָמִין שֶׁל כְּ-100 גִּיגָה-בַּיֵּיט לְכָל אֶחָד מֵהֶם - אֵין מְדוּבָּר בְּלוּקְסוּס.
עִם זֹאת חָשׁוּב לְצַיֵּין שֶׁאֵין עֲדַיִין כּוֹנָנִים כָּאֵלֶּה בַּשּׁוּק כָּךְ שֶׁלֹּא יָדוּעַ כַּמָּה הֵם יַעֲלוּ בַּהַתְחָלָה, אַךְ כַּנִּרְאֶה שֶׁלֹּא פָּחוֹת מֵאֵלֶּה שֶׁל מַיְקְרוֹסוֹפְט, בַּהֶמְשֵׁךְ עִם הַתַּחֲרוּת הַמְּחִירִים כַּנִּרְאֶה יָרְדוּ יוֹתֵר מַהֵר בַּצַּד שֶׁל סוֹנִי.

חָשׁוּב לְצַיֵּין שֶׁבְּסוֹפוֹ שֶׁל דָּבָר לֹא הַנְּתוּנִים הַטֶּכְנִיִּים יַכְרִיעוּ אֶת הַנִּיצָּחוֹן שֶׁל צַד זֶה אוֹ אַחֵר.
הַמִּשְׁתַּמְּשִׁים בְּסוֹפוֹ שֶׁל דָּבָר יַעֲדִיפוּ תְּמוּרָה לַמְּחִיר - כְּמוֹ שֶׁסּוֹנִי סִיפְּקָה עִם הָ-PS4 שֶׁנִּמְכְּרָה בְּהַשָּׁקָתָהּ בְּכַ-100 דּוֹלָר פָּחוֹת מֵהָאַקְסְבּוֹקְס וָואן.
זוֹ אַחַת הַסִּיבּוֹת הָעִיקָּרִיּוֹת לַפַּעַר הַגָּדוֹל שֶׁסּוּנִי הִצְלִיחָה לְיַיצֵּר מוּל הַמִּתְחָרָה מִבֵּית מַיְקְרוֹסוֹפְט וְהָאַחֲרוֹנָה כַּנִּרְאֶה שֶׁתְּנַסֶּה לֹא לְשַׁחְזֵר אֶת הַטָּעוּת הַזֹּאת הַפַּעַם.
הַנְּקוּדָּה הַחֲשׁוּבָה הַשְּׁנִיָּיה הִיא הַתּוֹכֶן אוֹ הַמִּשְׂחָקִים.

כְּכָל שֶׁצַּד זֶה אוֹ אַחֵר יַצְלִיחַ לִסְגּוֹר יוֹתֵר עִסְקָאוֹת בִּלְעָדִיּוֹת מוּל מְפַתְּחֵי הַמִּשְׂחָקִים כָּךְ הַגֶּיְימֵרִים יַעֲנִיקוּ לוֹ אֶת הַעֲדָפָתָם.
מַיְקְרוֹסוֹפְט בֵּינְתַיִים לֹא הִצְלִיחָה לְיַיצֵּר לְהִיטִים כְּמוֹ אֵצֶל סוֹנִי וּבְעִיקָּר הִיא גַּם מַשִּׁיקָה לֹא מְעַט מְשַׂחֲקִים לְשׁוּק הַ-PC בְּמַקְבִּיל - כָּךְ שֶׁלֹּא מְדוּבָּר בְּכוֹתָרִים בִּלְעָדִיִּים שֶׁמַּכְרִיחִים אֶת הַלָּקוֹחוֹת לְהַשְׁקִיעַ גַּם בַּקּוֹנְסוּלָה.
מִנֶּגֶד סוֹנִי יָדְעָה לְיַיצֵּר לְעַצְמָהּ מַסְפִּיק מִשְׂחָקִים יִיחוּדִיִּים לִפְלַטְפוֹרְמָה שֶׁלָּהּ כְּדֵי שֶׁהַשַּׂחְקָנִים יַשְׁקִיעוּ וְיִרְכְּשׁוּ אֶת הַפְלֶיְיסְטֵיְישְׁן כְּדֵי לֵיהָנוֹת מֵהֶם.

הַמַּאֲבָק בֵּין הַשְּׁתַּיִים כַּנִּרְאֶה יִתְבַּטֵּא גַּם בְּכָךְ שֶׁבְּעוֹד מַיְקְרוֹסוֹפְט מְנַסֶּה לְיַיצֵּר פְּלַטְפוֹרְמַת תּוֹכֶן נִרְחֶבֶת שֶׁמַּקִּיפָה גַּם אֶת שֵׁירוּת הַגֶּיְימִינְג בֶּעָנָן, שֵׁירוּת Game Pass שֶׁמַּעֲנִיק מִשְׂחָקִים בְּחִינָּם לְמִנּוּיִים וּבִכְלָל גִּישָׁה שֶׁל שֵׁירוּת סְטְרִימִינְג, סוֹנִי לְעוּמָּתָהּ מְנַסָּה לִדְבּוֹק בְּיִיחוּדִיּוּת.
אָמְנָם גַּם סוֹנִי צְפוּיָה לְהַשִּׁיק שֵׁירוּת סְטְרִימִינְג, אַךְ זֶה כַּנִּרְאֶה יִפְעַל רַק עַל הַפְלֶיְיסְטֵיְישְׁן.
אַגַּב, בְּאוֹפֶן אִירוֹנִי מַשֶּׁהוּ, מַיְקְרוֹסוֹפְט הִיא זוֹ שֶׁתְּסַפֵּק לְסוֹנִי אֶת הַתַּשְׁתִּית לַשֵּׁירוּת הַזֶּה דֶּרֶךְ שֵׁירוּת הֶעָנָן שֶׁלָּהּ Azure.
בְּסוֹפוֹ שֶׁל דָּבָר יִהְיוּ אֵלֶּה הַמְּפַתְּחִים שֶׁיַּעֲנִיקוּ אֶת הַנִּיצָּחוֹן, מְשַׂחֲקִים שֶׁיַּעֲנִיקוּ חֲוָויַית מִשְׂחָק טוֹבָה יוֹתֵר עַל קוֹנְסוּלָה אַחַת עַל פְּנֵי הַשְּׁנִיָּיה תְּשַׁכְנֵעַ אֶת הַגֶּיְימֶר לִבְחוֹר בְּמוּתָג זֶה אוֹ אַחֵר.
וְכַמּוּבָן גַּם הַנֶּאֱמָנוּת לַמּוּתָג שֶׁאֵצֶל סוֹנִי, בֵּינְתַיִים, הִיא גְּבוֹהָה יוֹתֵר.
 
לְסִיכּוּם, נֶאֱלַץ לְהַמְתִּין בָּחֳודָשִׁים הַקְּרוֹבִים לַהִתְפַּתְּחוּיוֹת הַשּׁוֹנוֹת כְּדֵי לְהָבִין לְאָן נוֹשֶׁבֶת הָרוּחַ.
