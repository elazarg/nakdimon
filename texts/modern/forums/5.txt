מוֹדֶה מֵרֹאשׁ לְעוֹנִים וְלַמְּגִיבִים
מִקְוֶוה שֶׁלֹּא יָצָא מְבוּלְגָּן מִדַּי וְלֹא אָרוֹךְ מִדַּי פָּשׁוּט אֶכְתּוֹב וְאֶזְרוֹם עִם הַכְּתִיבָה בִּכְדֵי לְתָאֵר לָכֶם אֶת הַסִּיטוּאַצְיָה שֶׁאֲנִי מְצוּיָה בָּהּ מִזֶּה זְמַן רַב וְשֶׁלֹּא תְּבַיֵּישׁ סֶרֶט הוֹלִיוֹוּדִי...
נַתְחִיל מִזֶּה שֶׁאֲנִי מְאוֹהֶבֶת קָשׁוֹת בַּגֶּבֶר מַלְאָךְ מוּשְׁלָם נְשָׁמָה טוֹבָה וְצַדִּיקָהּ עִם לֵב מִיַּהֲלוֹמִים שֶׁמּוּכָן לְהַקְרִיב וְלִמְסוֹר אֶת נַפְשׁוֹ בִּכְדֵי לֹא לִפְגּוֹעַ בָּאִישׁ.
בְּוַודַּאי מְתָאֲרִים לָכֶם אֵיזֶה סָבַל אָדָם טוֹב שֶׁכְּזֶה חוֹוֶה... כִּי זוֹ דֶּרֶךְ הָעוֹלָם לְצַעֲרֵנוּ צַדִּיק וְרַע לוֹ רָשָׁע וְטוֹב לוֹ עַד אֲשֶׁר אֱלוֹהִים בָּא וְעוֹשֶׂה צֶדֶק , אֲהוּב לִיבִּי מְנַסָּה בְּכָל דֶּרֶךְ אֶפְשָׁרִית אַךְ לֹא פּוֹגְעָנִית לְהִשְׁתַּחְרֵר וּלְהִתְנַתֵּק מִקֶּשֶׁר קוֹדֶם אַךְ נִתְקַל בִּקְשָׁיִים אִיּוּמִים מִצַּד הַפַּרְטְנֵרִית שֶׁהִנֵּה אֵגוֹאִיסְטִית שְׁתַלְּטָנִית עַד כְּדֵי מַחֲלַת נֶפֶשׁ וְאוּמְלָלָה בְּיוֹתֵר שֶׁמּוּכָנָה לְהַחֲזִיק בַּשְּׁבִי גֶּבֶר שֶׁאֵינוֹ מְאוֹהָב בָּהּ וְאֵינוֹ רוֹצֶה לִחְיוֹת אֶת חַיָּיו לְצִידָּהּ.
לַמְרוֹת אוֹפְיָהּ הַשְּׁלִילִי הוּא מִתְקַשֶּׁה וְאֵינוֹ רוֹצֶה וּבְצֶדֶק לִפְגּוֹעַ בָּהּ.
הוּא מַמְתִּין וּמַעֲדִיף שֶׁהִיא תָּבִין אֶת הַמְּצִיאוּת שֶׁבָּהּ הִיא חָיָה לְבַד, מָה שֶׁלֹּא קוֹרֶה.
הוּא בִּמְצוּקָה נוֹרָאִית כִּי חַיָּיו לֹא חַיִּים וְהִנָּם סֵבֶל וְגֵיהִנּוֹם שֶׁלֹּא נִיתָּן לְתֵיאוּר אֲפִילּוּ.
לְמַרְאִית עַיִן זֶה נִרְאֶה הָפוּךְ הִיא מַצִּיגָה לְעוֹלָם זוּגִיּוּת וּמִשְׁפָּחָה מְפוֹאֶרֶת וְחַיֵּי גַּן עֵדֶן וְאֶת הָרִיקָּבוֹן לֹא רוֹאִים רַק אֲנָשִׁים חֲכָמִים עִם אִינְטֵלִיגֶנְצִיָּה רִגְשִׁית גְּבוֹהָה מְרִיחִים שֶׁמַּשֶּׁהוּ אֵינוֹ כַּשּׁוּרָה.
חוֹשְׁדִים שֶׁמִּתְרַחֵשׁ מַשֶּׁהוּ לֹא תַּקִּין וְלֹא נוֹרְמָלִי אַךְ הִיא דּוֹאֶגֶת בְּצוּרָה שִׁיטָתִית לְהָזֵים אֶת הַשְּׁמוּעוֹת וְהָרִיכוֹלִים אֲשֶׁר נַעֲשִׂים בְּאוֹפֶן קָבוּעַ עַל יְדֵי הַמְּקוֹרָבִים.
עָצוּב שֶׁאִישָּׁה זוֹ אֵינָהּ יוֹדַעַת מָה הִיא אָהֲבָה וְלוֹקַחַת מֵאֲהוּב לִיבִּי אֶת הַזְּכוּת לְמִימּוּשׁ אַהֲבָתוֹ.
הוּא מִתְיַיסֵּר יִיסּוּרִים גְּדוֹלִים מִפְּאַת לִיבּוֹ הַטּוֹב.
אַךְ הִיא דּוֹאֶגֶת וּמִטְפַּחַת בִּיסוֹדִיּוּת וּבִקְפִידָה רַבָּה אֶת אֵזוֹר הַנּוֹחוּת שֶׁלָּהּ וּמְנַסָּה לִתְקוֹעַ יְתֵדוֹת רַבִּים עָמוֹק כְּכָל שֶׁנִּיתַּן כְּדֵי לְהִישָּׁאֵר בָּאֵזוֹר הַזֶּה וּלְהַשְׁאִיר אֶת הַמַּלְאָךְ שָׁם.
מִבְּחִינָתוֹ הָאֵזוֹר הַזֶּה מִזְּמַן לֹא אֵזוֹר נוֹחוּת אֶלָּא אֵזוֹר סָבַל וְגַהֲנוּם.
הוּא אֵינוֹ מוֹצֵא אֶת מְקוֹמוֹ וְאֶת בֵּיתוֹ.
חַי כְּנַוַּוד כַּאֲשֶׁר הוּא כָּל הַיּוֹם וְכָל הַלַּיְלָה עָמָל כִּנְמָלָה חֲרוּצָה אַךְ לְלֹא נַחַת וְשַׁלְוָוה אֲמִיתִּיִּים שֶׁמַּגִּיעִים לְכָל אֶחָד וּשְׁנָתוֹ לֹא מְתוּקָה כְּפִי שֶׁהוּא רָאוּי.
אוֹתָהּ אַחַת הֵבִיאָה אֶת עַצְמָהּ לַמַּצָּב הַזֶּה בּוֹ הוּא חָדַל לְאוֹהֲבָהּ וְיֵיאָמֵר לִזְכוּתָהּ שֶׁהִיא נִלְחֲמָה וְחֶפְצָהּ בְּתִיקּוּן וְשִׁינּוּי פְּנֵי הַדְּבָרִים אַךְ לְלֹא הוֹעִיל וּלְלֹא הַצְלָחָה.
כַּנִּרְאֶה שֶׁזֶּה חָלַף וְנִגְמַר בֵּינֵיהֶם הִיא מְמָאֶנֶת בְּתוֹקֶף לְקַבֵּל זֹאת וְחָשָׁה כְּמֵי שֶׁמַּפְסִידָה בְּקֶרֶב וְאֵינָהּ יוֹדַעַת לְהַפְסִיד בְּכָבוֹד.
הוּא מִבְּחִינָתוֹ נִשְׁבַּע לָמוּת וּלְוַותֵּר עַל הָעוֹלָם הַזֶּה וְרַק שֶׁלֹּא לִפְגּוֹעַ בָּאִישׁ אוֹ לִגְרוֹם לְצַעַר.
מָה שֶׁבְּפוֹעַל לֹא בֶּאֱמֶת קוֹרֶה הַהֵיפֶךְ הַגָּמוּר נַעֲשֶׂה כָּל הַמְּעוֹרָבִים בַּפָּרָשָׁה נִפְגָּעִים חָזָק וּמִתְקַשִּׁים לְנַהֵל חַיִּים נוֹרְמָלִים.
הָעוֹלָם הַזֶּה אֵינוֹ הוֹגֵן.
הִיא מִשְׁתַּמֶּשֶׁת בְּכָל הַסִּיבּוֹת וְהַתֵּירוּצִים בָּעוֹלָם כְּדֵי לִקְשׁוֹר אוֹתוֹ אֵלֶיהָ, גַּם בִּדְבָרִים מְטוֹרָפִים וַהֲזוּיִים , לֹא אַחַת רָאָה אוֹתָהּ שֶׁמָּה בַּמַּשְׁקֶה שֶׁלּוֹ מִדָּם הַמַּחְזוֹר שֶׁלָּהּ וְהִשְׁקְתָה אוֹתוֹ בְּמִיץ אֲשֶׁר טַבְלָה בּוֹ אֶת שָׁדֶיהָ וְרָאָה כָּל מִינֵי דְּבָרִים לֹא קְשׁוּרִים בַּבַּיִת וְהֵבִין שֶׁהִיא פּוֹנָה לְעֶזְרַת כָּל מִינֵי מִיסְטִיקָנִים כְּדֵי לְהַשְׁפִּיעַ עַל מַהֲלַךְ חַיָּיו.
לֹא בָּרוּר אֵיךְ אוֹתָהּ אַחַת מְסוּגֶּלֶת לִחְיוֹת חַיֵּי שֶׁקֶר וּלְלֹא אַהֲבָה אֲמִיתִּית וּבַמִּלְחָמָה אֵינְסוֹפִית כְּדֵי לִשְׁמוֹר עַל הַשָּׁבוּי וְעַל אֵזוֹר הַנּוֹחוּת שֶׁלָּהּ שֶׁהָפַךְ לְלֹא נוֹחַ בַּעֲלִיל.
גַּם הַיְּלָדִים נַעֲשׂוּ תַּחַת הַשְׁפָּעוֹת וְלֹא מֵאַהֲבָה.
הֵם חַיִּים בִּלְשַׂחֵק אַהֲבָה וּמִשְׁפָּחָה מוּשְׁלֶמֶת כְּשֶׁלְּמַעֲשֶׂה אֵין שָׁם כְּלוּם מִלְּבַד קְשָׁיִים וְשִׂנְאָה וּכְאֵב.
אֲהוּבִי הַמּוּשְׁלָם מִתְחַנֵּן עַל נַפְשׁוֹ שֶׁלֹּא אֵלֵךְ עִם גְּבָרִים אֲחֵרִים וּמְאַיֵּים לָשִׂים קֵץ עַל חַיָּיו.
הוּא לֹא מְסוּגָּל לַחֲשׁוֹב עַל חַיָּיו בִּלְעָדַיי אֲבָל בָּרוּר שֶׁמַּצָּב זֶה לֹא נִיתָּן לְהַנְצָחָה.
אֲהוּבִי וַאֲנִי קְשׁוּרִים מְאוֹד מְאוֹהָבִים מְאוֹד רוֹצִים מְאוֹד לִהְיוֹת בְּיַחַד אַךְ נִבְצַר מֵאִיתָּנוּ לְקַיֵּים וּלְמַמֵּשׁ אֶת הָאַהֲבָה רַק כְּדֵי שֶׁלֹּא לִפְגּוֹעַ חָלִילָה בָּהּ.
לָאַחֲרוֹנָה הִתְחִילוּ לְקַנֵּן בִּי הַמַּחְשָׁבוֹת שֶׁבְּעֶצֶם אֵינֶנּוּ עוֹשִׂים דָּבָר טוֹב בַּהִתְנַהֲלוּת הַזּוֹ.
כִּי מוֹנְעִים מִמֶּנָּה חַיִּים שֶׁל אֱמֶת וְאוּלַי גַּם אַהֲבָה אֲמִיתִּית שֶׁמְּחַכָּה לָהּ מֵעֵבֶר לַפִּינָּה.
וְיֵשׁ תְּחוּשָׁה וּמַחֲשָׁבָה שֶׁבִּמְקוֹם מַעֲשֶׂה טוֹב יֵשׁ כָּאן מַעֲשֶׂה רַע.
הֲרֵי הַלְּבָבוֹת שֶׁלָּנוּ קְשׁוּרִים לָעַד.
נִשְׁבַּעְנוּ לֶאֱהוֹב לְנַצֵּחַ וּלְהִיפָּגֵשׁ בַּגִּלְגּוּל הַבָּא.
הוּא פִיזִית נִמְצָא בַּשֶּׁבִי שֶׁלָּהּ אַךְ לִיבּוֹ וּמַחְשְׁבוֹתָיו אִיתִּי וְשֶׁלִּי.
כָּל הַיּוֹם וְכָל הַלַּיְלָה אֲנַחְנוּ מְדַבְּרִים וּמַרְגִּישִׁים אַהֲבָה שֶׁלֹּא נִיתֶּנֶת לְמִימּוּשׁ.
גַּם אִם נִהְיֶה עִם אֲחֵרִים הַלֵּב שֶׁלָּנוּ שַׁיָּיךְ אֶחָד לַשְּׁנִיָּה לְעוֹלָמִים.
אַמְשִׁיךְ מְאוּחָר יוֹתֵר לִכְתּוֹב נֶאֱלֶצֶת לְסַיֵּים אֲהוּבִי הִתְקַשֵּׁר :)
