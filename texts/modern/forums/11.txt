בַּזְּמַן הָאַחֲרוֹן יֵשׁ לִי הִתְלַבְּטוּת שֶׁחוֹזֶרֶת עַל עַצְמָהּ דֵּי הַרְבֵּה, הָאֵם לִרְכֹּב לִי בַּדְּרָכִים הַקְּצָרוֹת אוֹ בַּדְּרָכִים הַיָּפוֹת.
כִּמְעַט תָּמִיד יֵשׁ לִי אוֹפְּצִיָּה שֶׁל דֶּרֶךְ אַחַת קְצָרָה וִיעִילָה יוֹתֵר וְדֶרֶךְ אַחַת שֶׁהִיא יָפָה וְנֶחְמָדָה יוֹתֵר, וְכָל פַּעַם מֵחָדָשׁ אֲנִי מִתְלַבֶּטֶת אֶת אֵיזוֹ מִבֵּין שְׁתֵּיהֶן לָקַחַת, אֵיזוֹ מִבֵּין שְׁתֵּיהֶן לִבְחוֹר, עַל אֵיזוֹ מִבֵּין שְׁתֵּיהֶן לַעֲלוֹת.
---
לַהוֹרִים שֶׁלִּי הָיָה פַּעַם וִיכּוּחַ קָבוּעַ כְּזֶה, אִם לְהַגִּיעַ לְקוֹקָה קוֹלָה דֶּרֶךְ רַבִּי עֲקִיבָא ("רַבִּי" בְּמִלְּעֵיל, עַל פִּי אַבָּא שֶׁלִּי) אוֹ דֶּרֶךְ חֲצִיַּית כָּל הָרְחוֹבוֹת הַקְּטַנִּים שֶׁל בְּנֵי בְּרַק ("בְּנֵי" בְּמִלְּרַע, עִם סֶגוֹלִים אוֹ צֵירֶהִים מִתַּחַת לַבֵּית וְלַנּוּן שֶׁל הַ"בְּנֵי", גַּם כֵּן עַל פִּי אַבָּא שֶׁלִּי).
אַבָּא שֶׁלִּי הֶעֱדִיף אֶת רַבִּי עֲקִיבָא הַיָּשָׁר אַךְ הַלְּעִתִּים פָּקוּק - אִמָּא שֶׁלִּי הֶעֱדִיפָה אֶת הַחֲצִיָּיה שֶׁל כָּל הָרְחוֹבוֹת הַקְּטַנִּים שֶׁל בְּנֵי בְּרַק, שֶׁאָמְנָם הִיא אֲרוּכָּה יוֹתֵר מֶרְחַקִית אֲבָל לָרוֹב כִּמְעַט וְלֹא עוֹצְרִים בָּהּ, בְּעִיקָּר רַק נוֹסְעִים לְאַט יוֹתֵר וּמַגְבִּירִים אֶת הַהֶאָטָה בֵּין פְּנִיָּיה לִפְנִיָּיה לִפְנִיָּיה לִפְנִיָּיה.
אֲנִי חוֹשֶׁבֶת שֶׁלִּפְנֵי שֶׁנּוֹלַד לִי הַזִּיכָּרוֹן הֵם הָיוּ רָבִים עַל זֶה, אֲבָל מֵאָז שֶׁאֲנִי זוֹכֶרֶת פָּשׁוּט הָיָה הֶסְכֵּם שֶׁהַנּוֹהֵג הוּא שֶׁקּוֹבֵעַ וְזֶהוּ זֶה. מִדֵּי פַּעַם הֵם הָיוּ שׁוֹכְחִים וְהָיָה מִתְפַלֵּק לָהֶם וִיכּוּחַ בַּנּוֹשֵׂא, כָּכָה כַּנִּרְאֶה לָמַדְתִּי בְּעֶצֶם עַל חִילּוּק הַדֵּעוֹת בְּנוֹשֵׂא הַסְּפֵּצִיפִי הַזֶּה.
---
בְּמִידָּה מְסוּיֶּמֶת וֵוייְז וּמָּפְּס הָרְגוּ חֵלֶק גָּדוֹל מֵהַוִּיכּוּחִים הָאֵלּוּ וְהַהֶסְכֵּמִים הָאֵלּוּ וְהַטָּעוּיוֹת וְחוֹסֶר הַנּוֹחוֹת הַנִּלְוִוים הָאֵלּוּ, וְיֵשׁ בָּזֶה מַשֶּׁהוּ סָבַּבָּאִי כְּזֶה אֲבָל יֵשׁ בְּזֶה גַּם מַשֶּׁהוּ לֹא סָבַּבָּאִי כָּזֶה.
לְמָשָׁל, לֹא מִזְּמַן חָזַרְתִּי מֵהַחֲתוּנָּה שֶׁל פְרֶד בְּנֵס צִיּוֹנָה, וּבְרֶגַע אַחֲרוֹן הֶחְלַטְתִּי שֶׁלֹּא לְהַפְעִיל אֶת הַוֵוייְז וְלָצֵאת לַדֶּרֶךְ מִחוּץ לָעִיר בְּרִיגּוּשׁ שֶׁל נְסִיעָה לְאִיבּוּד אוֹפְּצְיוֹנָלִי.
לֹא קָרָה שׁוּם דָּבָר מְיוּחָד בַּנְּסִיעָה הַזֹּאת אֲבָל עָצַרְתִּי בְּיֶלוֹאוּ, וּבְנִשְׁנוּשׁ לֵילִי מְרוּגָּשׁ, מַשְׁמִין וְטָעוּתִי מִיְּסוֹדוֹ, אָכַלְתִּי סֶנְדְּוִויץ' שְׁנִיצֵל דּוֹחֶה מְחוּמָּם בְּמִיקְרוֹ יֶלוֹאוּאִי, שָׁתִיתִי מִיץ תַּפּוּזִים, אָכַלְתִּי קִינּוּחַ (הָיָה דִּיל), שָׁתִיתִי קָפֶה שָׁחוֹר, חִיזַּקְתִּי בְּעוֹד קִינּוּחַ, עִישַּׁנְתִּי סִיגַרְיָה וְהִמְשַׁכְתִּי עִם מַיִם מִינֵרָלִים וְעִם בְּחִילָה יֶלוֹאוּאִית מְזַעְזַעַת כָּל הַדֶּרֶךְ הַבַּיְתָה. בַּסּוֹף בִּכְלָל לֹא נָסַעְתִּי לְאִיבּוּד אֲבָל הָיָה שָׁם מַשֶּׁהוּ סָבַּבָּאִי כָּזֶה בַּתְּחוּשָׁה.
--
כָּכָה בְּעֵרֶךְ אַבָּא שֶׁלִּי הָיָה לוֹקֵחַ אוֹתָנוּ פַּעַם לְטִיּוּלִים. הָיִינוּ נִכְנָסִים לָאוֹטוֹ עִם צֵידָנִיּוֹת (מְלֵאוֹת בְּ"אוֹכָלִים", עַל פִּי אַבָּא שֶׁלִּי) וְנוֹסְעִים לְצָפוֹן אוֹ לְכָל מִינֵי כִּיוּוּנִים אֲחֵרִים, אֲבָל בְּעִיקָּר לְצָפוֹן.
בְּשָׁלָב כָּלְשֶׁהוּ הָיִינוּ עוֹבְרִים לְיַד קִיבּוּץ "עֵין אִיבּוּד" אוֹ "נְוֵה כְּלוּם" וְאַבָּא שֶׁלִּי הָיָה מַצִּיעַ - "רוֹצִים שֶׁנִּיכָּנֵס נִרְאֶה מָה יֵשׁ פֹּה?", וַאֲנַחְנוּ לֹא הָיִינוּ רוֹצִים, כִּי כָּל הַמְּקוֹמוֹת הָאֵלּוּ הֵם אוֹתוֹ דָּבָר, וּבַחֲצִי כַּעַס הָיִינוּ אוֹמְרִים שֶׁבְּסֵדֶר ("חֲצִי כַּעַס" - כִּי רָצִינוּ אֶקְסְטְרִים, רָצִינוּ יָהוּדִיָּהּ אוֹ זָוִויתָן אוֹ בַּנְיָאס, וּ"בְּסֵדֶר" - כִּי הוּא אַבָּא שֶׁלָּנוּ).
כְּשֶׁהָיִינוּ שָׁם בִּפְנִים אַבָּא שֶׁלִּי הָיָה מְזַהֶה אֵיזוֹ וַרְדָּה אוֹ דַּלְיָה אוֹ שָׂרָה, אוֹ מַזְכִּירַת קִיבּוּץ גֵּנֵרִית אַחֶרֶת, הָיָה מְסַפֵּר לָהּ שֶׁאֲנַחְנוּ רוֹצִים קְצָת לִרְאוֹת אֶת הַמָּקוֹם וְשׁוֹאֵל אוֹתָהּ אֵם הִיא מוּכָנָה לָקַחַת אוֹתָנוּ לַסִּיבוּב וּלְהַרְאוֹת לָנוּ וּלְהַסְבִּיר.
וַרְדָּה אוֹ דַּלְיָה אוֹ שָׂרָה אוֹ מַזְכִּירַת קִיבּוּץ גֵּנֵרִית אַחֶרֶת בְּדֶרֶךְ כְּלָל שָׂמְחוּ מְאֹד שֶׁמִּישֶׁהוּ מִתְעַנְיֵין בְּחֶלְקַת הָאֱלוֹהִים וְהַקִּיּוּם הָאֲבוּדָה שֶׁלָּהֶן. הֵן הָיוּ לוֹקְחוֹת אוֹתָנוּ לְסִיבוּב רְפָתוֹת, מוֹסִיפוֹת פִּרְטֵי הִיסְטוֹרְיָה חוֹמְהוּמִגְדָּלִית וּמְתַבְּלוֹת בִּקְצָת רְכִילוּת מְקוֹמִית מְעַנְיֶינֶת. בְּתוֹר יְלָדִים עֲדַיִין הִמְשַׁכְנוּ קְצָת לִכְעוֹס אֲבָל הַיּוֹם אֲנִי חוֹשֶׁבֶת שֶׁהֵם הָיוּ אִיבּוּדִים נֶחְמָדִים מְאֹד.
