שֵׁירוּת כַּלְכָּלִיסְט פּוּרְסַם:  16.05.15 , 20:20
הִתְחַמְּמוּת בַּיְּחָסִים בֵּין הוֹדוֹ לְסִין.
בִּיקּוּרוֹ שֶׁל רֹאשׁ מֶמְשֶׁלֶת הוֹדּוּ נַרֶנְדְרָה מוֹדִי בְּסִין נֶחְתַּם הַיּוֹם בְּטֶקֶס חֲגִיגִי בְּשַׁנְגְּחַאי, שֶׁבּוֹ נֶחְתְּמוּ עִסְקָאוֹת בְּשֹׁוִוי 22 מִילְיַארְד דּוֹלָר בֵּין חֲבָרוֹת סִינִיּוֹת לַחֲבָרוֹת הוֹדִיּוֹת.

מוֹדִי פָּנָה לְאַנְשֵׁי הָעֲסָקִים הַסִּינִים שֶׁהִשְׁתַּתְּפוּ בַּכִּינּוּס, וְהִבְטִיחַ לָהֶם שֶׁהוֹדּוּ כְּבָר אֵינָהּ מְדִינָה שֶׁמַּעֲרִימָה קְשָׁיִים עַל מִי שֶׁמְּבַקֵּשׁ לְהַשְׁקִיעַ בָּהּ.
"הִגַּעְתִּי לְכָאן כְּדֵי לְסַפֵּר לָכֶם שֶׁהַסְּבִיבָה הָעִסְקִית בְּהוֹדּוּ הִשְׁתַּנְּתָה", הִכְרִיז מוֹדִי.

שׁוּרַת הָעִסְקָאוֹת שֶׁנֶּחְתְּמָה בְּעַנְפֵי הַפְּלָדָה, הַפִינַנְסִים, הָאֵנֶרְגִּיָּה הַסּוֹלָארִית, הַתַּחְבּוּרָה וְהַטֵּלֵקוֹמוּנִיקַצְיָה הֵחֵלָּה לְהִתְגַּבֵּשׁ בְּעֵת בִּיקּוּרוֹ שֶׁל נְשִׂיא סִין שִׁי גִ'ינְפִּינְג בְּהוֹדּוּ לִפְנֵי שְׂבֵעָה חֳודָשִׁים.
מוֹדִי פָּתַח אֶת בִּיקּוּרוֹ בְּסִין בָּעִיר שִׂיאָן, עִיר הַבִּירָה שֶׁל מְחוֹז שֶׁאֲנָשַׁי, שֶׁמִּמֶּנּוּ הִגִּיעָה מִשְׁפַּחְתּוֹ שֶׁל שִׁי.
בְּכָךְ הֵשִׁיב רֹאשׁ הַמֶּמְשָׁלָה הַהוֹדִי בְּמֶחְוָוה תַּחַת מֶחְוָוה, שֶׁכֵּן שִׁי פָּתַח אֶת בִּיקּוּרוֹ בְּהוֹדּוּ בַּשָּׁנָה שֶׁעָבְרָה בָּעִיר אַחְמַדְבָּאד, שֶׁבָּהּ הֵחֵל מוֹדִי אֶת הַקַּרְיֵירָה הַפּוֹלִיטִית שֶׁלּוֹ.

מוֹדִי הִמְשִׁיךְ אֶת בִּיקּוּרוֹ בְּבֶּיְיגִ'ינְג, שֶׁבָּהּ הוּא נִפְגַּשׁ עִם רֹאשׁ הַמֶּמְשָׁלָה הַסִּינִי לִי קָצִ'יאַנְג.
שְׁנֵי הַמַּנְהִיגִים אַף הִצְטַלְּמוּ לִתְמוּנַת סֶלְּפִי שֶׁמּוֹדֵי פִּרְסֵם בְּחֶשְׁבּוֹן הַטְּוִויטֶר שֶׁלּוֹ.
"הַשִּׂיחוֹת שֶׁלָּנוּ הָיוּ פְּתוּחוֹת, בּוֹנוֹת וִידִידוּתִיּוֹת.
נָגַעְנוּ בְּכָל הַנּוֹשְׂאִים", אָמַר מוֹדִי בִּמְסִיבַּת הָעִיתּוֹנָאִים שֶׁנֶּעֶרְכָה לְאַחַר הַפְּגִישָׁה.

בִּדְבָרָיו רָמַז מוֹדִי לְסִכְסוּךְ הַגְּבוּלוֹת הַמִּתְמַשֵּׁךְ בֵּין שְׁתֵּי הַמְּדִינוֹת.
"הַיְּחָסִים בֵּינֵינוּ הָיוּ סְבוּכִים בָּעֲשׂוֹרִים הָאַחֲרוֹנִים.
בִּיקַּשְׁתִּי מִסִּין לִשְׁקוֹל מֵחָדָשׁ אֶת גִּישָׁתָהּ בַּאֲשֶׁר לְחֵלֶק מֵהַנּוֹשְׂאִים שֶׁמּוֹנְעִים אֶת מִימּוּשׁ הַפּוֹטֶנְצְיָאל הַמָּלֵא שֶׁל הַשּׁוּתָּפוּת שֶׁלָּנוּ", אָמַר.
"יֵשׁ לָנוּ אַחְרָיוּת הִיסְטוֹרִית לַהֲפוֹךְ אֶת הַיְּחָסִים בֵּין סִין לְהוֹדּוּ לִמְקוֹר כּוֹחַ לְכָל אַחַת מֵהַמְּדִינוֹת, וּלְהַשְׁפָּעָה מֵיטִיבָה עַל הָעוֹלָם", צִייֵּן מוֹדִי.

הַסֶלְּפִי הַמְּשׁוּתָּף, רֹאשׁ מֶמְשֶׁלֶת הוֹדּוּ נַרֶנְדְרָה מוֹדִי וְרֹאשׁ מֶמְשֶׁלֶת סִין לִי קָצִ'יאַנְג
רֹאשׁ הַמֶּמְשָׁלָה הַסִּינִי לִי אָמַר בִּמְסִיבַּת הָעִיתּוֹנָאִים שֶׁ"שְׁתֵּי הַמְּדִינוֹת צְרִיכוֹת לִשְׁמוֹר עַל הַמּוֹמֶנְטוּם הַחִיּוּבִי בְּעֵת שִׂיחוֹת בְּדֶרֶג נָמוּךְ יוֹתֵר עַל הַמַּחֲלוֹקוֹת, וּלְהַגִּיעַ לְפִתְרוֹן הֲדָדִי.
אֵינֶנּוּ מַכְחִישִׁים שֶׁיֶּשְׁנָן אִי־הַסְכָּמוֹת בֵּינֵינוּ, אֲבָל הָאִינְטֵרֵס הַמְּשׁוּתָּף עוֹלֶה עֲלֵיהֶן".
 
הַמַּטָּרָה: לִסְגּוֹר אֶת הַפַּעַר מוּל סִין
סִין הִיא שׁוּתֶּפֶת הַסַּחַר הַגְּדוֹלָה בְּיוֹתֵר שֶׁל הוֹדוֹ.
בְּ-2014 הִגִּיעַ הֶיקֵּף תְּנוּעַת הַסְּחוֹרוֹת בֵּין שְׁתֵּי הַמַּעֲצָמוֹת לְ-71 מִילְיַארְד דּוֹלָר.
עִם זֹאת, הַגֵּירָעוֹן הַמִּסְחָרִי שֶׁל הוֹדוּ עִם סִין זִינֵּק מִמִּילְיַארְד דּוֹלָר בִּלְבַד בְּ-2001, לְ-38 מִילְיַארְד דּוֹלָר בְּ-2014.
 
הַנְּתוּנִים הַלָּלוּ מְעִידִים עַל הַפִּיגּוּר הֶעָמוֹק שֶׁבּוֹ נִמְצֵאת הוֹדוֹ בְּהַשְׁוָואָה לִקְפִיצָה שֶׁעָשְׁתָה סִין מִבְּחִינַת פִּיתּוּחַ.
מוֹדִי, שֶׁנִּבְחַר לְרָאשׁוּת הַמֶּמְשָׁלָה בְּדֶלְהִי לִפְנֵי שָׁנָה בְּדִיּוּק, שָׂם לוֹ לְמַטָּרָה לִסְגּוֹר אֶת הַפַּעַר שֶׁבֵּין הוֹדוּ לְשֶׁכַּנָּתָהּ הַגְּדוֹלָה - בְּיִיחוּד עַל רֶקַע הַהֶאָטָה בְּכַלְכָּלַת סִין, וְהַמַּעֲבָר שֶׁלָּהּ מִצְּמִיחָה מְבוּסֶּסֶת יְצוּא וְהַשְׁקָעָה בְּתַשְׁתִּיּוֹת לְכַלְכָּלָה מְבוּסֶּסֶת צְרִיכָה פְּרָטִית.

בַּחֲזִית הַמַּאֲבָק בְּשִׁינּוּי הָאַקְלִים הִסְכִּימוּ שְׁתֵּי הַמְּדִינוֹת עַל הֶאָצַת הַמַּאֲמַצִּים לְרִיסּוּן פְּלִיטוֹת גַּזִּי הַחֲמָמָה בְּתִחוּמָן.
עִם זֹאת, הַמַּנְהִיגִים קָרְאוּ לִמְדִינוֹת הַמַּעֲרָב לְסַיֵּיעַ לָהֶן וּלְאוּמּוֹת אֲחֵרוֹת בְּאַסְיָה לְקַדֵּם אֶת הַשָּׂגַת הַמַּטָּרָה הַזּוֹ בְּאֶמְצָעִים פִינַנְסִיִּים וְטֶכְנוֹלוֹגִיִּים.

הוֹדוּ וְסִין נֶחֱשָׁבוֹת לִשְׁתֵּי מְדִינוֹת הַמַּפְתֵּחַ לְהַשָּׂגַת אֲמָנָה בֵּינְלְאוּמִּית לְהַגְּבָּלַת פְּלִיטַת גַּזִּי חֲמָמָה בִּוְועִידַת פָּרִיז לְשִׁינּוּי אַקְלִים שֶׁתֵּיעָרֵךְ בְּסוֹף הַשָּׁנָה.
אַף שֶׁבִּיקּוּרוֹ שֶׁל מוֹדִי לֹא הֵבִיא עִמּוֹ הִתְחַיְּיבוּת קוֹנְקְרֵטִית לְרִיסּוּן הַפְּלִיטוֹת בִּשְׁתֵּי הַמְּדִינוֹת, הֵן הִתְחַיְּיבוּ לְפַרְסֵם אֶת תּוֹכְנִיּוֹתֵיהֶן בָּעִנְיָין לִקְרַאת הַוְּועִידָה.

בְּתוֹךְ כָּךְ, רֹאשׁ מֶמְשֶׁלֶת סִין לִי יַגִּיעַ מָחֳרָתַיִים לִבְרָזִיל וְיִיפָּגֵשׁ עִם הַנְּשִׂיאָה דִּילְמָה רוֹסֶף בִּבְּרָזִילְיָה הַבִּירָה.
לִי צָפוּי לְהַכְרִיז עַל הַשְׁקָעָה סִינִית שֶׁל 50 מִילְיַארְד דּוֹלָר בְּפִיתּוּחַ תַּשְׁתִּיּוֹת בַּמְּדִינָה הַדָּרוֹם־אֲמֵרִיקָאִית.
גּוּלַּת הַכּוֹתֶרֶת שֶׁל הַמֵּיזָמִים הָעֲתִידִיִּים הִיא בְּנִיַּית מְסִילַּת בַּרְזֶל שֶׁתְּחַבֵּר בֵּין חוֹף הָאוֹקְיָינוֹס הָאַטְלַנְטִי בִּבְּרָזִיל עַד לְחוֹף הָאוֹקְיָינוֹס הַשָּׁקֵט בְּפֵּרוּ.
מְסִילַּת הַבַּרְזֶל הַחֲדָשָׁה אֲמוּרָה לְהָקֵל אֶת יְצוּאָם שֶׁל בַּרְזֶל וְסוֹיָה בְּרָזִילָאִית לְסִין, שֶׁהִיא יַעַד הַיְּצוּא מִסְפַּר אַחַת שֶׁל בְּרָזִיל בְּמוּצָרִים אֵלּוּ.
