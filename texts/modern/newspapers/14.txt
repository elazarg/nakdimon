אִם תָּעִירוּ אֶת אִילָנִית לֵוִי בִּשְׁתַּיִים בַּלַּיְלָה הִיא קוֹדֶם כָּל תְּפַרְגֵּן.
זֶה לֹא אוֹטוֹמָטִי אֶלָּא אִינְסְטִינְקְטִיבִי.
נִרְאֶה שֶׁזֶּה פָּשׁוּט חָזָק מִמֶּנָּה.
כְּשֶׁהִיא מְדַבֶּרֶת עַל הַקּוֹלֵגָה לְהַנְחָיַית "גַּלִּית וְאִילָנִית", (יְמֵי שִׁישִּׁי, 16:00, קֶשֶׁת 12) הִיא מְסַפֶּרֶת עַד כַּמָּה הִיא לָמְדָה מִמֶּנָּה, כְּשֶׁנִּשְׁאֲלָה לְאוֹרֶךְ הַשָּׁנִים עַל בְּנוֹת הַזּוּג שֶׁל הַגָּרוּשׁ אֱיָיל גּוֹלָן, הִיא מְמַהֶרֶת לְהַחֲמִיא שֶׁהֵן מְהַמְּמוֹת.
גַּם כְּשֶׁאִשְׁתּוֹ הַטְּרִיָּיה שֶׁל אֱיָיל, דָּנִיאֵל גְּרִינְבֵּרְג, הִכְרִיזָה בְּאִינְסְטַגְרָם לִפְנֵי כְּחוֹדֶשׁ עַל הַהֵרָיוֹן, הִיא מִיהֲרָה לְבָרֵךְ וְהוֹסִיפָה לְבָבוֹת.
מִי שֶׁלֹּא מַכִּיר אוֹתָהּ, בִּכְלָל, עָשׂוּי לַחֲשׁוֹד בָּאַהֲבָה הַזֹּאת, לִרְמוֹז אֲפִילּוּ שֶׁהִיא לֹא אוֹתֶנְטִית, אֲבָל מִי שֶׁמַּכִּיר אוֹתָהּ יוֹדֵעַ שֶׁאֵין בָּהּ עֶצֶם לֹא אוֹתֶנְטִית בְּגוּפָהּ.
אַחֲרֵי עֶשְׂרִים שְׁנוֹת קַרְיֵירָה בְּתַעֲשִׂיַּית הַבִּידּוּר, וְאַחֲרֵי שֶׁיָּדְעָה פָּרָשׁוֹת וְצִיטּוּטִים, הַדּוּגְמָנִית וְהַמַּנְחָה הִתְעַיְּיפָה מֵהַטְּרִיקִים הַיַּחְצָנִיִּים וּכְלָלֵי הַמִּשְׂחָק.
אוּלַי זוֹ גַּם הַסִּיבָּה שֶׁהִיא לֹא מוּפְתַּעַת שֶׁהַשְּׁאֵלָה הָרִאשׁוֹנָה שֶׁלִּי מִתְיַיחֶסֶת לְגוֹלָן.
הִיא אֲפִילּוּ לֹא מְרִימָה גַּבָּה.

מְעַט מִן הָאוֹר
"הָאֱמֶת הִיא שֶׁכְּבָר הִתְרַגַּלְתִּי, שָׁאֲלוּ אוֹתִי הַכֹּל.
תִּרְאֶה, מִצַּד אֶחָד אֲנִי עֶשְׂרִים שָׁנָה בַּתְּחוּם וַאֲנִי מְבִינָה אֶת הָרָצוֹן וְאֶת הַהִתְעַנְייְנוּת.
מִצַּד שֵׁנִי, עָבְרוּ 12 שָׁנִים מֵאָז שֶׁהִתְגָּרַשְׁתִּי, 19 מֵאָז שֶׁהִתְחַתְּנוּ, זֶה הָיָה בְּעִידָּן אַחַר (הִיא צוֹחֶקֶת).
יֵשׁ לַאֲנָשִׁים כָּל מִינֵי פַנְטַזְיוֹת וּשְׁאֵלוֹת בָּרֹאשׁ.
אֲנָשִׁים לֹא מַסְכִּימִים לִשְׁכּוֹחַ אֶת הַזּוּגִיּוּת הַזּוֹ, לֹא מְבִינִים שֶׁהָיָה וְנִגְמַר.
אֲבָל בְּסֵדֶר, זֶה לֹא נוֹרָא.
אֲנִי מְקַבֶּלֶת אֶת זֶה בַּהֲבָנָה".

אָז בּוֹאִי נִסְגּוֹר אֶת הַפִּינָּה הַזֹּאת.
לֹא הִגַּעְתְּ לַחֲתוּנָּה שֶׁל אֱיָיל וְדָנִיאֵל, לַמְרוֹת שֶׁהוּזְמַנְתְּ.
לָמָּה? הָייְתָה הִתְלַבְּטוּת?
"זֶה הָיָה בָּרוּר לִי מֵהַתְחָלָה.
לִפְנֵי יְמֵי הַקּוֹרוֹנָה זוֹ הָייְתָה אֲמוּרָה לִהְיוֹת חֲתוּנָּה עֲנָקִית עִם לְמַעְלָה מֵאֶלֶף אִישׁ.
זֶה בֶּאֱמֶת לֹא הַמָּקוֹם שֶׁלִּי, זֶה יוֹם קָדוֹשׁ שֶׁל שְׁנֵי אֲנָשִׁים וְזֶה הִרְגִּישׁ לִי לָקַחַת פוֹקוּס בַּמָּקוֹם שֶׁהוּא לֹא לִי.
זֶה אוּלַי נִשְׁמָע קְצָת נַרְקִיסִיסְטִי, אֲבָל אֶפְשָׁר לְהָבִין לָמָּה אֲנִי מִתְכַּוֶּונֶת.
אֲנָשִׁים הָיוּ מַתְחִילִים לְהַגִּיד עִם מִי דִּיבַּרְתִּי, מְפָרְשִׁים כָּל מַבָּט.
רָצִיתִי שֶׁהַפוֹקוּס יִשָּׁאֵר עַל עַל אֱיָיל וְדָנִיאֵל, וְשֶׁלַּיְּלָדִים שֶׁלִּי יִהְיֶה נָעִים.
אִם הַיְּלָדִים הָיוּ אוֹמְרִים לִי, אֲנַחְנוּ צְרִיכִים אוֹתָךְ שָׁם, בָּא לָנוּ שֶׁתָּבוֹאִי, הָיִיתִי בָּאָה, בָּרוּר.
אֲבָל הֵם אָמְרוּ לִי, אִמָּא, זֶה הֲכִי מוּבָן".

אַתֶּם מְדַבְּרִים פָּתוּחַ עַל כָּל הַדְּבָרִים הָאֵלֶּה?
"בַּחַיִּים שֶׁלָּנוּ אִי אֶפְשָׁר אַחֶרֶת.
אִם אֲנִי לֹא אֶתֵּן לַיְּלָדִים שֶׁלִּי אֶת הַמֵּידָע, אֲנָשִׁים אֲחֵרִים יִמְלְאוּ לָהֶם אֶת הָרֹאשׁ בְּכָל מִינֵי דְּבָרִים - וְאָז זֶה מְסוּכָּן.
אֵיפֹה שֶׁיֵּשׁ חָלָל רֵיק יֵשׁ אֶת הַמַּחְשָׁבוֹת הֲכִי נוֹרָאִיּוֹת.
עָדִיף שֶׁאֲנִי אֲדַבֵּר אִיתָּם הֲכִי פָּתוּחַ".

אִם אֲנַחְנוּ כְּבָר בִּפְחָדִים, בּוֹאִי בְּכָל זֹאת נֵלֵךְ 12 שָׁנִים אָחוֹרָה.
לֹא הִפְחִיד אוֹתָךְ לַעֲזוֹב אֶת הָאַבָּא שֶׁל הַיְּלָדִים שֶׁלָּךְ, כְּשֶׁאַתְּ כָּל כָּךְ צְעִירָה וְהוּא כְּבָר עִם הַטַּייְטֶל "הַזַּמָּר הֲכִי מְפוּרְסָם בְּיִשְׂרָאֵל"?
"זוֹ הָייְתָה נְקוּדַּת הַתּוּרְפָּה שֶׁלִּי וְשֶׁל אֱיָיל, זֶה שֶׁהוּא הָיָה זַמָּר מְפוּרְסָם, זוֹ הַסִּיבָּה שֶׁבִּגְלָלָהּ לֹא רָצִיתִי יוֹתֵר אֶת הַקֶּשֶׁר הַזֶּה.
לֹא רָצִיתִי אֶת הַהֲמוּלָּה הַזֹּאת וְשֶׁכּוּלָּם יְדַבְּרוּ עַל חַיֵּי.
הַיּוֹם אֲנִי רוֹאֶה אֶת זֶה כַּמּוּבָן מֵאֵלָיו אֲבָל אָז זֶה לֹא הָיָה כָּכָה.
אָז לֹא, זֶה לֹא הִפְחִיד אוֹתִי לַעֲזוֹב מֵהַסִּיבּוֹת הָאֵלֶּה.
כְּאִמָּא לַיַּלְדָּה שֶׁיָּדַעְתִּי שֶׁיּוֹם אֶחָד הִיא תַּהֲפוֹךְ לִהְיוֹת אִישָּׁה, הָיָה לִי בָּרוּר שֶׁאֲנִי צְרִיכָה לְחַנֵּךְ אוֹתָהּ מָה זֶה לִהְיוֹת עַצְמָאִית וְלֹא לְפַחֵד מִשּׁוּם דָּבָר וּמֵאַף אֶחָד".

הַיְּלָדִים אָמְרוּ לְךָ פַּעַם 'אִמָּא, הַלְּוַואי שֶׁאַתְּ וְאַבָּא הֲיִיתֶם בְּיַחַד'?
"לְהֶפֶךְ.
הֵם אָמְרוּ לִי, 'אִמָּא, אֵיזֶה מוּזָר שֶׁאַתְּ וְאַבָּא פַּעַם גַּרְתֶּם בְּאוֹתוֹ בַּיִת'.
אָלִין הָייְתָה בַּת שָׁנָה כְּשֶׁהִתְגָּרַשְׁנוּ, אֵין לָהֶם שׁוּם זִיכָּרוֹן אַחֵר כִּמְעַט.
נָכוֹן שֶׁהָיוּ תְּקוּפוֹת טוֹבוֹת יוֹתֵר וְהָיוּ תְּקוּפוֹת טוֹבוֹת פָּחוֹת, אֲבָל אֱיָיל וַאֲנִי תָּמִיד שָׁמַרְנוּ עַל יְחָסִים טוֹבִים, וְלֹא רַק לְמַרְאִית עַיִן.
פוֹרְמָט שֶׁל גֵּירוּשִׁים מִבְּחִינָתֵנוּ הָיָה לְסַיֵּים אֶת הָרִיבִים, לֹא לְהַתְחִיל אוֹתָם מֵחָדָשׁ".

חוּץ מְלִיאָם וְאָלִין, יוֹצֵא לְךָ וְלֶאֱיָיל לְדַבֵּר עַל דְּבָרִים אֲחֵרִים מִדֵּי פַּעַם?
"הַשִּׂיחוֹת הֵם עַל לִיאָם וְאָלִין נָטוּ.
דָּנִיאֵל יְכוֹלָה לְסַמֵּס מִדֵּי פַּעַם עַל דְּבָרִים קְצָת אֲחֵרִים וַאֲנַחְנוּ מְדַבְּרוֹת בֵּינֵינוּ, אֲבָל לִיאָם וְאָלִין הֵם הַחוּט הַמְּקַשֵּׁר וְהַיָּחִיד בֵּינֵינוּ הַיּוֹם".

כְּשֶׁדָּנִיאֵל הוֹדִיעָה עַל הֵרָיוֹן בְּאִינְסְטַגְרָם מִיָּד בֵּירַכְתְּ אוֹתָהּ וְהַמְּגִיבִים כִּינּוּ אוֹתָךְ כְּאִישָּׁה הֲכִי מְפַרְגֶּנֶת שֶׁיֵּשׁ.
עַד כַּמָּה הַפִירְגּוּן הַזֶּה הוּא הַחְלָטָה מוּדַעַת?
"זֶה הָיָה הֲכִי טִבְעִי לִי בָּעוֹלָם לְהָגִיב אֶת זֶה.
מָה שֶׁלֹּא טִבְעִי לִי זֶה לְסַנֵּן אֶת מָה שֶׁאֲנִי חוֹשֶׁבֶת כִּי זֶה לֹא נִרְאֶה אָמִין.
אֶצְלִי בָּרֹאשׁ זֶה אָמוּר לְהִתְנַהֵל בְּדִיּוּק כָּכָה".

בַּמִּקְרֶה שֶׁל רוֹסְלָנָה כַּנִּרְאֶה קְצָת פָּחוֹת, אוּלַי זֶה הַסִּיבָּה שֶׁאֲנָשִׁים מוּפְתָּעִים מִמְּךָ.
"עִם רוֹסְלָנָה אֲנִי לֹא חוֹשֶׁבֶת שֶׁיֵּשׁ רִיב.
לִפְעָמִים דְּבָרִים יוֹצְאִים מֵהֶקְשֵׁרָם וְהַתִּקְשׁוֹרֶת מוֹסִיפָה תַּבְלִינִים, זֶה לֹא אוֹמֵר שֶׁזֶּה הַמַּצָּב בַּחַיִּים הָאֲמִיתִּיִּים.
אֲנִי אוֹהֶבֶת אוֹתָהּ וּמַכִּירָה אוֹתָהּ, הִיא גַּם לֹא אָדָם שֶׁל רִיבִים וּמִלְחָמוֹת, זֶה לֹא בֶּאֱמֶת הַמַּצָּב לְדַעְתִּי".


הִיא אָמְרָה בָּרֵאָיוֹן בֶּעָבָר שֶׁדָּנִיאֵל שָׂמָה עַיִן עַל כָּל הַאַקְסִים שֶׁלָּהּ וְשֶׁזֶּה קָרָה בְּאוֹפֶן סִדְרָתִי.
"מְרַאְייְנִים הֵם אֲנָשִׁים מְאוֹד חֲמוּדִים, אֲבָל לִפְעָמִים אַתֶּם מוֹשְׁכִים אוֹתָנוּ בַּלָּשׁוֹן.
מְרַאֲיֵין מְיַיצֵּר רֶגֶשׁ בַּמְּרוּאֲיָין, מִתְגָּרֶה בּוֹ, וְאָז הָרֶגֶשׁ הַזֶּה הוֹפֵךְ לְהִתְגּוֹנְנוּת וְרָצוֹן לָצֵאת חָזָק.
אֲנִי בֶּאֱמֶת לֹא חוֹשֶׁבֶת שֶׁבֶּאֱמֶת מְדוּבָּר בְּרִיב כְּמוֹ שֶׁמְּתָאֲרִים אוֹתוֹ".

"הַהוֹרוּת יָצְרָה אוֹתִי"
בְּמֶרְכַּז קַבָּלַת הַהַחְלָטוֹת שֶׁלָּהּ תָּמִיד נִיצָּבִים הַיְּלָדִים.
לֵוִי הִיא מָה שֶׁמְּכַנִּים אִמָּא לְבִיאָה.
הַחוֹסֶן הַנַּפְשִׁי שֶׁל לִיאָם (17) וְאָלִין (14) הוּא הַדָּבָר שֶׁהֲכִי חָשׁוּב לָהּ עַל רֶקַע הַפְּרוֹפִיל הֲתִקְשׁוֹרְתִּי הַנּוֹסֵק שֶׁל הַשְּׁנַיִים, וְכַמּוּבָן לְאוֹר פָּרָשׁוֹת הֶעָבָר.
בְּמִידָּה מְסוּיֶּמֶת הִיא מְפַצֶּה עַל כָּל מָה שֶׁנֶּחֱסָךְ מֵהֶם, כִּילָדָיו שֶׁל אֶחָד הַזַּמָּרִים הֲכִי מְפוּרְסָמִים בָּאָרֶץ.
אֶת הַבַּיִת בּוֹ הִיא מְגַדֶּלֶת אוֹתָם עִם בֶּן זוּגָהּ מִזֶּה עֶשֶׂר שָׁנִים, אֱלִירָז שָׂדֶה, הִיא מַגְדִּירָה כְּבֵית פָּתוּחַ.
חֲבוּרוֹת נְעָרִים וּנְעָרוֹת נִכְנָסִים וְיוֹצְאִים כָּל הַיּוֹם, וְהִיא מַכִּירָה אֶת כּוּלָּם.
מִצַּד שֵׁנִי, יָכוֹל לִהְיוֹת שֶׁזּוֹ פָּשׁוּט דֶּרֶךְ לִשְׁמוֹר אוֹתָם קְרוֹבִים אֵלֶיהָ, מוּגַנִּים.
תָּמִיד שָׁם, יַחַד אִיתָּהּ.

"הַהוֹרוּת לְלִיאָם וְאָלִין יָצְרָה אוֹתִי", הִיא אוֹמֶרֶת.
"זֶה אֲפִילּוּ לֹא לְגַלּוֹת מִי אֲנִי, אֲנִי נוֹלַדְתִּי אִיתָּם כְּאָדָם.
הַיְּלָדִים שֶׁלִּי זֶה הַדָּבָר הֲכִי טוֹב שֶׁאֱלוֹהִים נָתַן לִי, אֲנִי אוֹמֶרֶת עַל זֶה תּוֹדָה כָּל יוֹם.
הֵם עָשׂוּ אוֹתִי אָדָם שָׁפוּי, נוֹרְמָלִי וְהַכֹּל הֶבֶל הֲבָלִים.
בְּגִיל 21 קִיבַּלְתִּי פְּרוֹפּוֹרְצְיוֹת וְהֵבַנְתִּי שֶׁהַכֹּל הֶבֶל הֲבָלִים וְשֶׁהֵם הֲכִי חֲשׁוּבִים.
זֶה דָּבָר מַדְהִים".

וּכְכָל שֶׁהַיְּלָדִים גְּדֵלִים כָּךְ הֵם זוֹכִים לְיוֹתֵר הִתְעַנְייְנוּת.
כָּל הַמְּדִינָה מַכִּירָה אוֹתָם כְּבָר.
אֵיךְ אַתְּ מִתְמוֹדֶדֶת עִם זֶה?
"יְמֵי הַקּוֹרוֹנָה הֵם בְּדִיּוּק הַדֶּרֶךְ לְהַסְבִּיר אֶת מָה שֶׁאֲנִי מַרְגִּישָׁה.
בַּהַתְחָלָה כּוּלָּם אָמְרוּ 'אוֹי וַאֲבוֹי, אֵיךְ נִסְתַּדֵּר בְּלִי לָצֵאת מֵהַבַּיִת?', וְתוֹךְ שְׁבוּעַיִים כּוּלָּם הִתְרַגְּלוּ לְמַצֵּב וְלָצֵאת מֵהַבַּיִת נִרְאֶה כְּמוֹ דָּבָר מוּזָר הַיּוֹם.
זֶה מָה שֶׁאֲנִי מַכִּירָה.
אֲנִי לֹא מַכִּירָה הוֹרוּת אַחֶרֶת אוֹ יְלָדִים אֲחֵרִים, וְזֶה גַּם מָה שֶׁלֵּיאָם וְאָלִין מַכִּירִים.
לְזֶה הֵם נוֹלְדוּ וְעִם הַדָּבָר הַזֶּה אֲנִי הָפַכְתִּי לִהְיוֹת אִמָּא.
מִתּוֹךְ זֶה תָּמִיד הָיָה לִי חָשׁוּב לִי שֶׁתִּהְיֶה לָהֶם קַרְקַע יַצִּיבָה וְשֶׁהַחַיִּים שֶׁלָּהֶם יִהְיוּ רְגִילִים.
כָּל שְׁאָר הַדְּבָרִים הֵם סְתָם עוֹד נְיוּאַנְסִים בַּחַיִּים שֶׁלָּנוּ".

אַתְּ כִּמְעַט לֹא אוֹמֶרֶת לָהֶם לֹא, וְיַחַד עִם זֶה אֶת לֹא מְאַשֶּׁרֶת לְאָלִין לְאַשֵּׁר עוֹקְבִים זָרִים בְּאִינְסְטַגְרָם.
"זֶה הָיָה בָּרוּר שֶׁהָאִינְסְטַגְרָם שֶׁלָּהּ יִהְיֶה סָגוּר וּפְרָטִי, לֹא הָייְתָה עַל זֶה שִׂיחָה אֲפִילּוּ.
בְּטִיק טוֹק, לְעוּמַּת זֹאת, יֵשׁ לָהּ יוֹתֵר מִמֵּאָה אֶלֶף עוֹקְבִים.
אֲבָל זוֹ אַפְּלִיקַצְיָה שׁוֹנָה מֵאִינְסְטַגְרָם, אָלִין הִיא כִּישְׁרוֹנִית וְאוֹהֶבֶת לִרְקוֹד וְלָשִׁיר, אָז שָׁם זֶה בָּא לִידֵי בִּיטּוּי.
כָּל מָה שֶׁקּוֹרֶה אֶצְלָהּ אֲנִי רוֹאֶה וְעוֹקֶבֶת.
הַיְּלָדִים יוֹדְעִים שֶׁיֵּשׁ צַד מְאוֹד מְכוֹעָר וְאַלִּים לַאֲנָשִׁים יַחַד עִם כָּל הַמַּחְמָאוֹת שֶׁהֵם גַּם מְקַבְּלִים, אָז הֵם יוֹדְעִים אֵיפֹה לַעֲצוֹר בְּעַצְמָם.
כְּשֶׁאַתָּה הוֹרָה אַתָּה צָרִיךְ לִבְחוֹר אֶת הַמִּלְחָמוֹת שֶׁלְּךָ, אַחֶרֶת אַתָּה יָכוֹל לְהַפְסִיד בַּהַכֹּל".

הִשְׁלַמְתְּ עִם זֶה כְּבָר שֶׁהֵם כַּנִּרְאֶה יִהְיוּ כּוֹכָבִים עוֹד יוֹתֵר עֲנָקִיִּים בֶּעָתִיד? 
"אִם הֵם יִהְיוּ מְאוּשָּׁרִים וּבְרִיאִים, לֹא אִכְפַּת לִי כְּלוּם.
שֶׁיַּעֲשׂוּ כָּל מָה שֶׁעוֹשֶׂה לָהֶם טוֹב בַּחַיִּים הָאֵלֶּה.
אַחֲרֵי כָּל כָּךְ הַרְבֵּה דְּבָרִים שֶׁרָאִיתִי בַּתַּעֲשִׂיָּיה, כָּל כָּךְ הַרְבֵּה אֲנָשִׁים שֶׁעוֹלִים וְיוֹרְדִים, אֲנִי יְכוֹלָה לְהַגִּיד שֶׁזֶּה בֶּאֱמֶת מִקְצוֹעַ מְאוֹד קָשֶׁה.
אֲנָשִׁים שׁוֹאֲלִים אוֹתִי אֵיךְ אַתְּ שׁוֹמֶרֶת עַל עַצְמְךָ, וַאֲנִי מְבִינָה שֶׁהַקַּרְקַע הַיְּצִיבָה שֶׁלִּי זֶה מָה שֶׁשּׁוֹמֵר אוֹתִי.
עָשִׂיתִי אֶת זֶה בְּלִי לְהִתְכַּוֵּון, לֹא הֵבַנְתִּי מָה הַמַּשְׁמָעוּת שֶׁל לִיצוֹר לְעַצְמִי גַּם אֶת הַפִּינָּה הַשְּׁקֵטָה שֶׁלִּי.
עָשִׂיתִי דְּבָרִים כִּצְעִירָה בְּלִי לָדַעַת אֶת הַמַּשְׁמָעוּת שֶׁלָּהֶם - וְזֶה מָה שֶׁשָּׁמַר עָלַיי".

מִי מְאַתְגֵּר אוֹתָךְ יוֹתֵר כְּאִמָּא, לִיאָם אוֹ אָלִין?
"שְׁנֵיהֶם, בְּדַרְכָּם הַמּוּפְלָאָה, הַיִּיחוּדִית וְהַשּׁוֹנֶה.
זָכִיתִי בַּיְּלָדִים וּבַבַּיִת שֶׁאֲפִילּוּ בִּזְמַן הַקּוֹרוֹנָה מְאוֹד מְאוֹד קַל לָנוּ.
הַיְּלָדִים אוֹהֲבִים אֶת הַבַּיִת, הַחֲבֵרִים שֶׁלָּהֶם כָּאן כָּל הַיּוֹם, אָז עַכְשָׁיו רַק קָשֶׁה לָהֶם שֶׁלֹּא מַגִּיעִים אֲלֵיהֶם חֲבֵרִים.
הֵם יְלָדִים שֶׁל בַּיִת וַאֲנִי אוֹמֶרֶת עַל זֶה תּוֹדָה לֶאֱלוֹהִים כָּל הַזְּמַן, לַמְרוֹת שֶׁהֵם מְאַתְגְּרִים אוֹתִי עִם הַנִּיקְיוֹנוֹת".

וְאֵיךְ אַתְּ מִתְמוֹדֶדֶת עִם הַתְּקוּפָה הַזֹּאת?
הָאֱמֶת הַבִּידּוּד לֹא הִכְנִיס אוֹתִי לְשׁוֹק, אֲנִי בֶּאֱמֶת הַרְבֵּה בַּבַּיִת, אֲבָל הֵבַנְתִּי כַּמָּה דְּבָרִים חִיּוּנִיִּים יֵשׁ לִי שֶׁלֹּא הֵבַנְתִּי שֶׁהֵם חִיּוּנִיִּים בֶּעָבָר.
לָק גֵּ'ל, פֵּדִיקוּר, אִמָּא שֶׁלִּי, מוֹשִׁיקוֹ טָקוֹ שֶׁעוֹשֶׂה לִי גַּבּוֹת... זֶה מַצָּב הָזוּי".

אָז אֶפְשָׁר לְנַחֵשׁ מָה הַדָּבָר הָרִאשׁוֹן שֶׁתַּעֲשִׂי אַחֲרֵי שֶׁהַהֶסְגֵּר יֵירֵד.
"לָק גֵּ'ל".

"לָמָּה אֲנַחְנוּ לֹא מְבִיאִים יְלָדִים? אַתֶּם לֹא יוֹדְעִים לַעֲשׂוֹת אֶחָד וְעוֹד אֶחָד?"
לֹא רַק הַקּוֹרוֹנָה פָּגְעָה הַשָּׁנָה בַּפְּרִיוִֹוילֶגְיָה לֵיהָנוֹת מִשְּׁטוּיוֹת, מֵהַדְּבָרִים הַפְּשׁוּטִים.
לִפְנֵי כְּשָׁנָה, כְּשֶׁהוֹבִיל אֶת שִׁידּוּרֵי הַלִּיכּוּד TV, שָׂדֶה הֶעֱלָה לְשִׁידּוּר אֶת הַשַּׂחְקָן יוֹסִי וִידֶר, שְׁחִיקָה אֶת עִיתּוֹנָאי חֲדָשׁוֹת 12, אַמְנוֹן אַבְרָמוֹבִיץ'.
בְּרֵאָיוֹן הַמְּשׂוּחָק שָׁאַל וִידֶר, "אֲנִי נִרְאֶה טוֹב", וְשָׂדֶה הֵשִׁיב לוֹ - "אַתָּה נִרְאֶה טִיל כְּמוֹ תָּמִיד, אַמְנוֹן".
פְּלִיטַת הַפֶּה עוֹרְרָה בּוּז וְתַדְהֵמָה אַחֲרֵי שֶׁנִּתְפְּסָה כְּלַעַג לְאַבְרָמוֹבִיץ', שֶׁנִּכְוָוה בְּמִלְחֶמֶת יוֹם הַכִּיפּוּרִים.
שָׂדֶה הִתְנַצֵּל וְלֵוִי נִיצְּלָה אֶת הַבָּמָה הַטֵּלֵוִויזְיוֹנִית כְּדֵי לְהִתְיַיחֵס לַפָּרָשָׁה.
"קַמְתִּי לְבוֹקֶר שֶׁל עֲלֵיהוֹם בָּזוּי וּמַגְעִיל.", אָמְרָה אָז בְּתָכְנִית "גַּלִּית וְאִילָנִית".
"קָרָאתִי אֶת כָּל הַקְּלָלוֹת, עַל כָּל גִּוּוּנֵיהֶן.
דְּבָרִים עָלַיי וְעַל הַיְּלָדִים שֶׁלִּי.
אֶפְשָׁר גַּם לִטְעוֹת וּלְהַגִּיד סְלִיחָה, אֲבָל כָּכָה? בְּכָזֹאת צוּרָה גּוֹעֳלִית? פָּגְעוּ לִי בַּנֶּפֶשׁ הֲכִי הֲכִי טְהוֹרָה שֶׁאֲנִי מַכִּירָה.
זֶה לֹא פֵייְר".

מִי לָקַח יוֹתֵר קָשֶׁה אֶת הַפָּרָשָׁה עִם אַמְנוֹן אַבְרָמוֹבִיץ' - אֶת אוֹ הוּא?
"אֲנִי חוֹשֶׁבֶת שֶׁאֲנִי.
אֲנִי מַאֲמִינָה בָּאָדָם הַטּוֹב וְקָרָאתִי אֶת כָּל הַטּוֹקְבֵּקִים.
כּוּלָּנוּ רוֹצִים לִשְׁמוֹעַ מָה אוֹמְרִים עָלֵינוּ מֵאֲחוֹרֵי הַגַּב וּמִבְּחִינָתִי לִקְרוֹא טוֹקְבֵּקִים זֶה לָדַעַת מָה אוֹמְרִים עָלֶיךָ מֵאֲחוֹרֵי הַגַּב.
רַק בַּכַּתָּבוֹת עַל אָלִין וְלִיאָם אֲנִי לֹא קוֹרֵאת טוֹקְבֵּקִים.
עָשִׂיתִי אֶת זֶה פַּעַם אַחַת וְזוֹ הָייְתָה טָעוּת, עַכְשָׁיו גַּם עַל אֱלִירָז קָשֶׁה לִי לִקְרוֹא.
פַּעַם רִאשׁוֹנָה כָּעַסְתִּי.
אֲנִי לֹא מְגִיבָה עַל מָה שֶׁכּוֹתְבִים עָלֵינוּ, אַף פַּעַם כִּי זֶה לֹא בֶּאֱמֶת מַעֲסִיק אוֹתִי, אֲבָל הַפַּעַם זֶה כָּל כָּךְ כָּאַב לִי.
זוֹ בְּסַךְ הַכֹּל הָייְתָה טָעוּת שֶׁלֹּא הָייְתָה בָּהּ  כַּוָּונָה רָעָה לִפְגּוֹעַ וְהָייְתָה גַּם הִתְנַצְּלוּת.
אֲנִי קָלַטְתִּי אֶת הַמִּילָּה הַבְּעָיָיתִית הַזּוֹ וְיָשָׁר אָמַרְתִּי לוֹ 'מַאמִי?'.
הוּא הֵבִין מָה הַמַּשְׁמָעוּת וְתָפַס אֶת הָרֹאשׁ בַּבַּיִת.
אָמַרְתִּי לוֹ שֶׁיַּעֲלֶה פּוֹסְט לִפְנֵי שֶׁיַּתְחִיל כָּל הַבָּלָגָן וְכוּלָּם סְבִיבוֹ אָמְרוּ לוֹ 'אַל תַּעֲלֶה, אַל תָּעִיר מֵתִים'.
אֵיזֶה מֵתִים? יָדַעְתִּי יָשָׁר מָה הוֹלֵךְ לִקְרוֹת.
הִתְאַכְזַבְתִּי מֵהֲמוֹן אֲנָשִׁים, לֹא כָּאֵלֶּה שֶׁקְּרוֹבִים אֵלַיי בְּרָמָה יוֹם-יוֹמִית, אֲבָל מֵהַרְבֵּה אֲחֵרִים".

בְּמַהֲלַךְ הַזּוּגִיּוּת שֶׁלָּכֶם, אֱלִירָז הִסְפִּיק לַעֲבוֹר סוּג שֶׁל מַהְפָּךְ.
מִכּוֹכַב רֵיאָלֵיטִי הוּא פִּתְאוֹם הָפַךְ לְמִישֶׁהוּ שֶׁאֲנָשִׁים רוֹצִים לְהַקְשִׁיב לַדֵּעוֹת הַפּוֹלִיטִיּוֹת שֶׁלּוֹ.
אֵיךְ אֶת מַסְבִּירָה אֶת הַשִּׁינּוּי הַזֶּה?
"זֶה נֶחְמָד שֶׁאֲנָשִׁים קוֹרְאִים לְזֶה מַהְפָּךְ אֲבָל אֱלִירָז הָיָה כָּכָה תָּמִיד.
זֶה הוּא וְזֶה הָאָדָם שֶׁאֲנִי חָיָה אִיתּוֹ.
אֲנִי וּפוֹלִיטִיקָה זֶה סִיפּוּר אַהֲבָה-שִׂנְאָה.
מִצַּד אֶחָד אֲנִי רוֹדֶפֶת אַחֲרֵי מֵידָע וְקוֹרֵאת כָּל הַזְּמַן וּמִי עָשָׂה מָה בַּכְּנֶסֶת, אֲבָל כְּשֶׁזֶּה מַגִּיעַ אֵלַיי הַבַּיְתָה אֲנִי פָּחוֹת בָּעִנְיָין.
כְּשֶׁמִּישֶׁהוּ נִכְנָס לַפּוֹלִיטִיקָה יֵשׁ תְּפִיסָה שֶׁאֶפְשָׁר לִגְמוֹר עָלָיו עַד הַסּוֹף, אֵין חוּקִּים יוֹתֵר.
אֶת הָאֱמֶת? אֲנִי שְׂבֵעָה מִזֶּה.
יֵשׁ לָנוּ הֲמוֹן דִּיּוּנִים עַל זֶה, אֲנִי מְבִינָה אוֹתוֹ וְאֶת הָרָצוֹן שֶׁלּוֹ.
זֶה בּוֹעֵר בּוֹ כָּל חַיָּיו וּמִבְּחִינָתוֹ זוֹ שְׁלִיחוּת, וּבַשָּׁנָה הָאַחֲרוֹנָה גַּם הִגִּיעַ הַקַּמְפֵּייְן שֶׁל הַלִּיכּוּד וְכָל הַמְּעוֹרָבוּת שֶׁלּוֹ בָּזֶה.
רָאוּ אוֹתוֹ כְּפָנִים שֶׁל הַדָּבָר הַזֶּה וְזֶה הִבְעִיר בּוֹ עוֹד וְעוֹד אֶת הָרָצוֹן.
בּוֹא נִרְאֶה מָה יִהְיֶה".

עַד כַּמָּה זֶה בָּא לוֹ טִבְעִי לְהִיכָּנֵס לַזּוּגִיּוּת עִם מִישֶׁהִי שֶׁיֵּשׁ לָהּ 2 יְלָדִים?
"עָבַרְנוּ הֲמוֹן גִּלְגּוּלִים, אֲנַחְנוּ עָשׂוֹר בְּיַחַד תֵּיכֶף.
בּוֹא נַגִּיד שֶׁלֹּא סְתָם הִתְחַתַּנּוּ רַק לִפְנֵי שָׁנָה.
שָׁנִים לֹא רָאִינוּ אֶת עַצְמֵנוּ כְּזוּג שֶׁיּוֹם אֶחָד יַעֲשֶׂה מִשְׁפָּחָה.
הָיוּ לָנוּ כָּל כָּךְ הַרְבֵּה מַחְסוֹמִים, זֶה לֹא הָיָה דָּבָר פָּשׁוּט.
עִם כַּמָּה שֶׁהַיְּלָדִים שֶׁלִּי, חַמְסָה, הֵם בֶּאֱמֶת מַתָּנָה וְיֵשׁ בֵּינֵיהֶם מַעֲרֶכֶת יְחָסִים מַדְהִימָה שֶׁלֹּא קְשׁוּרָה אֵלַיי שֶׁיְּכוֹלָה לִגְרוֹם לִי לִבְכּוֹת - הַתַּהֲלִיךְ הַזֶּה הָיָה אָרוֹךְ.
אֱלִירָז וַאֲנִי עָבַרְנוּ חֲתִיכַת מַסָּע.
אִם אֵין תִּקְשׁוֹרֶת, אֵין מַצָּב לִשְׂרוֹד תְּהוֹמוֹת כָּאֵלֶּה.
לְמַזָּלִי, הוּא קוֹאצֵ'ר וְלֹא יַעֲזוֹב אוֹתְךָ עַד שֶׁלֹּא תַּגִּיד אֶת הָאֱמֶת, גַּם אִם הִיא לֹא נְעִימָה.
זָכִיתִי".

כְּשֶׁאַתֶּם רָבִים, אָלִין וְלִיאָם מְנַסִּים לְפַיֵּיס בֵּינֵיכֶם בַּבַּיִת?
"בֶּטַח! מָלֵא.
אָלִין עוֹשָׂה לִי פַּרְצוּפִים אִם אֲנִי מַגְזִימָה וְאוֹמֶרֶת לִי 'אִמָּא, דַּי'".

אָמַרְתְּ פַּעַם שֶׁהוּא אָמַר לְךָ לֹא לְהִתְלַבֵּשׁ חָשׂוּף.
"זֶה מַשֶּׁהוּ שֶׁבַּהַתְחָלָה הָיָה קָשֶׁה לְאִילָנִית הַנּוּדְנִיקִית, שֶׁלֹּא יְכוֹלָה לָתֵת לְמִישֶׁהוּ לְהַגִּיד לָהּ לַעֲשׂוֹת.
בְּאוֹפֶן טִבְעִי אֲנִי לֹא מִתְלַבֶּשֶׁת חָשׂוּף, אֲנִי אִמָּא לִילָדִים גְּדוֹלִים.
יֵשׁ לִי יֶלֶד בֵּן 17 עוֹד מְעַט, אֵין אֶצְלִי דְּבָרִים כָּאֵלֶּה בִּכְלָל.
זֶה לֹא הָיָה קָשֶׁה לִי בִּגְלַל שֶׁרָצִיתִי לְהִתְלַבֵּשׁ חָשׂוּף, אֶלָּא מִמָּקוֹם אַחֵר".
מְפוּרְסָמוֹת בָּאָרֶץ וּבָעוֹלָם מוֹבִילוֹת אֶת קַמְפֵּייְן 'אַל תִּכְנְסוּ לִי לְרַחֵם' וְגַם אוֹתָךְ לֹא מַפְסִיקִים לִשְׁאוֹל עַל יְלָדִים עִם אֱלִירָז.
זֶה מַכְעִיס אוֹתָךְ אוֹ שֶׁאַתְּ מְקַבֶּלֶת אֶת זֶה בַּהֲבָנָה?
"אֲנִי רוֹצֶה לְהַגִּיד מַשֶּׁהוּ לְאַחְיוֹתַיי: אֲנִי אִתְּכֶן, אֲבָל כּוּלָּם יַמְשִׁיכוּ לְהִיכָּנֵס לָכֵן לָרֶחֶם.
כּוּלָּם יַמְשִׁיכוּ לִשְׁאוֹל, רַק עַכְשָׁיו הֵם יוֹסִיפוּ אֶת הַמִּשְׁפָּט 'יֵשׁ אֶת עִנְיַין הָאַל תִּכָּנְסוּ לִי לָרֶחֶם'.
זֶה הַכֹּל.
מִזְּמַן הֵבַנְתִּי אֶת זֶה.
אֲנִי לֹא מִתְעַצְבֶּנֶת מִזֶּה.
בָּרוּר שֶׁאֲנִי וֶאֱלִירָז רוֹצִים יְלָדִים, שׁוֹאֲלִים אוֹתִי אֶת זֶה וַאֲנִי אוֹמֶרֶת 'לֹא, אֲנַחְנוּ בְּיַחַד עָשׂוֹר, הִתְחַתְּנוּ וַאֲנַחְנוּ עֲדַיִין בּוֹחֲנִים אִם אֲנַחְנוּ מַתְאִימִים'.
זֶה מַצְחִיק אוֹתִי כְּשֶׁשּׁוֹאֲלִים אִם אֲנַחְנוּ רוֹצִים יְלָדִים, מָה אַתֶּם חוֹשְׁבִים? שֶׁאֲנַחְנוּ לֹא רוֹצִים?".

וּבְכָל זֹאת.
"מָה אֲנָשִׁים מְצַפִּים שֶׁאֲנִי אֶעֱנֶה עַל זֶה? שֶׁאַגִּיד לָהֶם שֶׁדִּיבַּרְתִּי עִם הַבֵּיצִית שֶׁלִּי וְהִיא אָמְרָה לִי שֶׁהַיּוֹם אֵין לָהּ כּוֹחַ? אֲנִי בְּעִיקָּר צוֹחֶקֶת וְלֹא מְבִינָה מָה אֲמוּרָה לִהְיוֹת הַתְּשׁוּבָה שֶׁלִּי.
לָמָּה לִשְׁאוֹל מָה עִם יֶלֶד? אַתֶּם אֲמוּרִים לְהָבִין לְבַד.
בְּגִילֵנוּ לֹא הִתְחַתְּנוּ כְּדֵי לְהַעֲבִיר שְׁנָתַיִים בְּמַלְדִּיבִיִּים בְּיַחַד, אָז לָמָּה אֲנַחְנוּ לֹא מְבִיאִים יְלָדִים? אַתֶּם לֹא יוֹדְעִים לַעֲשׂוֹת אֶחָד וְעוֹד אֶחָד? אִי אֶפְשָׁר לְהָבִין אֶת הָרְגִישׁוּת? צָרִיךְ לְהָבִין שֶׁאוּלַי גַּם עוֹבֵר עָלַיי יוֹם מַסְרִיחַ בִּמְיוּחָד בַּדָּבָר הַסְּפֵּצִיפִי הַזֶּה.
בִּגְלַל זֶה יָצָא קַמְפֵּייְן 'צְאוּ לִי מֵהָרֶחֶם'.
הָיוּ מְפוּרְסָמוֹת שֶׁשָּׁאֲלוּ אוֹתָן מָה עִם יֶלֶד יוֹם אַחֲרֵי שֶׁהֵן עָבְרוּ הַפָּלָה.
לַמְרוֹת הַכֹּל, אֲנִי מְנַסֶּה לְהִתְמוֹדֵד עִם הַשְּׁאֵלוֹת הָאֵלֶּה בְּשָׁלוֹם".

אִם לֶאֱלִירָז הָיוּ גַּם יְלָדִים מִנִּישּׂוּאִים קוֹדְמִים, אֶת חוֹשֶׁבֶת שֶׁבְּכָל זֹאת הֲיִיתֶם רוֹצִים יֶלֶד מְשׁוּתָּף?
"הָאִינְסְטִינְקְט הָרִאשׁוֹנִי שֶׁלִּי בַּשְּׁאֵלָה הַזּוֹ זֶה לְהַגִּיד שֶׁכֵּן.
אֲנִי רוֹצָה יְלָדִים עִם אֱלִירָז, וְגַם אִם הָיוּ לוֹ יְלָדִים אֲנִי חוֹשֶׁבֶת שֶׁהָיִיתִי רוֹצֶה.
בָּרוּר שֶׁיֵּשׁ אֶת הָעוּבְדָּה הַבְּרוּרָה שֶׁלֶּאֱלִירָז אֵין יְלָדִים וְשֶׁבִּגְלַל זֶה הוּא רוֹצֶה, אֲבָל הַיּוֹם, אַחֲרֵי עָשׂוֹר בְּיַחַד, אֲנִי יְכוֹלָה לְהַגִּיד שֶׁלִּי יֵשׁ כְּמִיהָה לַיְּלָדִים אִיתּוֹ.
עַכְשָׁיו יֵשׁ לִי הֲבָנָה מְלֵאָה מָה זֶה לְרַצּוֹת יֶלֶד שֶׁהוּא פְּרִי שֶׁלָּנוּ, פְּרִי אַהֲבָה שֶׁלִּי וְשֶׁלּוֹ.
זוֹ הַסִּיבָּה שֶׁאֲנַחְנוּ רוֹצִים יְלָדִים הַיּוֹם, לֹא רַק בִּגְלַל שֶׁלּוֹ אֵין".

"אֲנִי מַצְבִּיעַת יָמִין וְגָלִית שְׂמֹאל וַעֲדַיִין, יֵשׁ הֲמוֹן קַוֵּוי דִּמְיוֹן"
וְיֵשׁ גַּם אֶת הֲשׁוּתָּפָהּ הַשְּׁנִיָּיה לַחַיִּים, גַּלִּית גּוּטְמָן, אִיתָּהּ הִיא חוֹלֶקֶת מָסָךְ מִדֵּי שִׁישִּׁי.
הַהֶבְדֵּל בֵּין הַשְּׁתַּיִים בָּרוּר, הַיְּחָסִים בֵּינֵיהֶן אֲפִילּוּ זָכוּ לְחִיקּוּי בְּאֶרֶץ נֶהְדֶּרֶת הַשָּׁנָה.
גַּלִּית הִיא מְאוּפֶּקֶת, אֲסוּפָה, זְקוּפָה; אִילָנִית הִיא יוֹתֵר מֵעוֹלְמוֹת הָרוּחַ וְהַטּוֹרְנָדוֹ.
קֶצֶב הַדִּיבּוּר שֶׁלָּהּ מָהִיר, הִיא נוֹגַעַת יוֹתֵר, מִתְקָרֶבֶת.
גַּם בּוֹכָה לֹא מְעַט.

מִמָּה אַתְּ יוֹתֵר נֶהֱנֵית הַיּוֹם, לְדַגְמֵן אוֹ לְהַנְחוֹת?
"אֲנִי אוֹהֶבֶת אֶת שְׁנֵי הָעוֹלָמוֹת, אֲבָל לְהַנְחוֹת אֲנִי מַמָּשׁ אוֹהֶבֶת.
לְדַגְמֵן זֶה יוֹתֵר קָשֶׁה.
הַרְבֵּה יוֹתֵר קָשֶׁה.
אֲנִי סוֹבְיֵיטִית בָּעֲבוֹדָה, לָמַדְתִּי הַכֹּל מִגָּלִית גּוּטְמָן (צוֹחֶקֶת) וַעֲדַיִין, זֶה קָשֶׁה לַעֲמוֹד עַל עֲקֵבִים 12 שָׁעוֹת בְּרֶצֶף.
בְּהַנְחָיָה אֲנִי קוֹרֵאת תַּחְקִירִים, מְדַבֶּרֶת עִם אֲנָשִׁים וְחוֹקֶרֶת עַל כָּל נוֹשֵׂא וְאָדָם שֶׁמַּגִּיעַ לַתָּכְנִית.
זֶה פּוֹתֵחַ אֶצְלִי מָלֵא דְּבָרִים בָּרֹאשׁ".

יוֹצֵא לָכֶן לְדַבֵּר עַל פּוֹלִיטִיקָה בֵּינֵיכֶן? אֶת אִמְרַת שֶׁהִצְבַּעְתָּ לִיכּוּד, הִיא, אֲנִי מַנִּיחַ, לֹא מַצְבִּיעַת לִיכּוּד.

"בָּרוּר שֶׁיּוֹצֵא.
אֲנַחְנוּ אוּלַי לֹא מַסְכִּימוֹת עַל דְּבָרִים, אֲבָל זֶה לֹא יַגִּיעַ לְרָמַת וִיכּוּחַ.
אֲנַחְנוּ מְדַבְּרוֹת עַל הַכֹּל.
אָמְנָם אֲנִי מַצְבִּיעַת יָמִין וְהִיא שְׂמֹאל וַעֲדַיִין, יֵשׁ הֲמוֹן קַוֵּוי דִּמְיוֹן.
אַגַּב, בְּעֵינַי, הַכֹּל עֲרָפֶל עַכְשָׁיו.
אֵין צְדָדִים יוֹתֵר, הַכֹּל מִתְעַרְבֵּב.
אֲנִי וְגָלִית שׁוֹנוֹת בֶּהָמוֹן דְּבָרִים, לְיָדָהּ אֲנִי נִרְאֵית כְּאִילּוּ אֲנִי בְּהִילּוּךְ מָהִיר.
אֲנַחְנוּ מְדַבְּרוֹת שָׂפָה שׁוֹנָה, אֲבָל אֲנַחְנוּ אוֹתוֹ בֶּן אָדָם.
הַמֶּרְכָּז שֶׁלָּנוּ בְּחַיִּים מְאוֹד דּוֹמֶה, אֲבָל הַשָּׂפָה שׁוֹנָה.
זֶה חֵלֶק מִסּוֹד הַקֶּסֶם בַּתָּכְנִית שֶׁלָּנוּ וְאַחַת הַסִּיבּוֹת שֶׁכֵּיף לָנוּ בְּיַחַד".

מֵהַצַּד שֶׁל הַדּוּגְמָנוּת, נִבְחֶרֶת לְהוֹבִיל אֶת הַקַּמְפֵּייְן שֶׁל גּוֹלְבֶּרִי.
אֵיךְ זֶה לְהִיכָּנֵס לְנַעֲלַיִים שֶׁל אֶסְתִּי גִּינְזְבּוּרְג? בְּכָל זֹאת, הִיא הָייְתָה הַפְּרֵזֶנְטוֹרִית בְּמֶשֶׁךְ 8 עוֹנוֹת.
"אֲנִי לֹא מִתְחַבֶּרֶת לַמּוּנָּח שֶׁל 'לְהִכָּנֵס לְנַעֲלַיִים שֶׁל מִישֶׁהוּ'.
גַּם לֹא לַנַּעֲלַיִים שֶׁלִּי.
זוֹ הָעֲבוֹדָה שֶׁלָּנוּ, אִם זֶה לֹא הָיָה עוֹבֵד כָּכָה - לֹא הָיִינוּ עוֹבְדִים.
כָּל אַחַת עוֹשֶׂה כַּמָּה וְכַמָּה עוֹנוֹת וְאַחַר כָּךְ יֵשׁ רִעֲנוּן וְתַחֲלוּפָה.
מַחֲלִיפִים אוֹתִי, הֶחֱלַפְתִּי וְיַחֲלִיפוּ אוֹתִי גַּם בֶּעָתִיד.
זֶה מַשֶּׁהוּ שֶׁהוּא מְאוֹד לֵגִיטִימִי בְּעֵינִי.
מוּזָר לִי שֶׁזֶּה עֲדַיִין אִישׁוּ.
יֵשׁ אֶת עִנְיַין הַמְּבוּגָּרוֹת וְהַצְּעִירוֹת שֶׁמּוּחְלָפוֹת.
אֲנִי יוֹתֵר מְבוּגֶּרֶת מֵאֶסְתִּי וְיֵשׁ לִי גַּם יְלָדִים גְּדוֹלִים, אֲנִי נִתְפֶּסֶת כְּיוּתֵּר מְבוּגֶּרֶת וַאֲנָשִׁים חוֹשְׁבִים שֶׁאֲנִי כְּבָר עָבַרְתִּי מִזְּמַן אֶת גִּיל 40 לַמְרוֹת שֶׁאֲנִי בַּת 37.
גַּם כְּשֶׁאֲנִי אֶעֱשֶׂה קַמְפֵּייְנִים בְּגִיל מְבוּגָּר יוֹתֵר - זֶה לֹא יַעֲלִיב אוֹתִי שֶׁיַּחֲלִיפוּ אוֹתִי בְּמִישֶׁהִי צְעִירָה יוֹתֵר.
כָּכָה זֶה עוֹבֵד".

"אֲנָשִׁים לֹא יוֹדְעִים לְנַקּוֹת, אֲנִי בְּהֶלֶם מִזֶּה"
בְּמֶשֶׁךְ כַּמָּה שָׁנִים, בְּכָל פַּעַם שֶׁמִּישֶׁהוּ רָצָה לְתַיֵּיג אֶת לֵוִי בְּאִינְסְטַגְרָם הוּא נֶאֱלַץ לְהִשְׁתַּמֵּשׁ בְּהָאשְׁטָאג #לְאִילָנִית אֵין אִינְסְטָה.
הִיא הָייְתָה בֵּין הָאַיִּי-לִיסְטִיּוּת הַבּוֹדְדוֹת בְּלִי חֶשְׁבּוֹן עָמוּס וּמְתוּחְזָק.
לִפְנֵי קְצָת פָּחוֹת מִשְׁנָה הִיא סוֹף כָּל סוֹף פָּתְחָה אֶחָד, אֲבָל בִּמְקוֹם לְשַׁתֵּף אֶת הָעוֹקְבִים שֶׁלָּהּ בִּתְמוּנוֹת שֶׁל גְּלָאם, שֶׁל מִשְׁפָּחָה מוּשְׁלֶמֶת, הִיא מַעֲלָה לְשֵׁם בְּעִיקָּר הֲצָצוֹת לָאַלְטֶר אֵגוֹ שֶׁלָּהּ.
אִישָּׁה עִם חִיבָּה לְמַגֵּב.

יֵשׁ לְךָ כְּבָר לְמַעְלָה מִ-200 אֶלֶף עוֹקְבִים, מְפוּרְסָמִים אֲחֵרִים נִמְצָאִים שָׁם שָׁנִים וְלֹא הִגִּיעוּ לַמִּסְפָּר הַזֶּה.
הָאַהֲבָה הַזֹּאת מַגִּיעָה אֵלַיִיךְ, אַתְּ מַרְגִּישָׁה אוֹתָהּ?
"מַמָּשׁ מַרְגִּישָׁה.
אֲנִי גַּם יוֹדַעַת שֶׁטּוֹקְבֵּקִיסְטִים שֶׁכּוֹתְבִים עָלַיי דְּבָרִים פָּחוּת נְעִימִים יָבוֹאוּ אֵלַיי בָּרְחוֹב וִיחַבְּקוּ אוֹתִי.
זֶה נוֹרָא טְרִיקִי.
אֲנִי חוֹשֶׁבֶת שֶׁאֲנִי בֶּאֱמֶת אוֹהֶבֶת אָדָם.
אֲנָשִׁים נִיסּוּ לְהַסְבִּיר לִי בַּחַיִּים שֶׁזֶּה לֹא נָכוֹן, אֲבָל אֲנִי עַד הַיּוֹם חוֹשֶׁבֶת שֶׁאָדָם בַּבָּסִיס שֶׁלּוֹ הוּא טוֹב.
אֲנִי לֹא עִיוֶּורֶת, אֲבָל בּוֹחֶרֶת לְהַאֲמִין בְּטוֹב.
אֲנִי בֶּטַח לֹא תְּמִימָה, כֵּן?".

אַתְּ מַרְגִּישָׁה פִסְפוּס שֶׁהִצְטָרַפְתְּ לָרְשָׁתוֹת הַחֶבְרָתִיּוֹת בְּאִיחוּר? בְּכָל זֹאת, נִרְאֶה שֶׁכֵּיף לְךָ שָׁם.
"אֶת הָאֱמֶת שֶׁהֵבַנְתִּי אֶת הַאִינְסְטַגְרָם הָפוּךְ מֵאֲנָשִׁים.
כְּשֶׁנִּכְנַסְתִּי לְזֶה רַק לִפְנֵי קְצָת יוֹתֵר מִשְׁנֶה בְּקֶטַע שֶׁל צְחוֹק, הִתְחַלְתִּי לְצַלֵּם אֶת הַחַיִּים שֶׁלִּי כְּמוֹ שֶׁהֵם.
אֲנִי מַעֲלֶה סְטוֹרִיז בְּלִי אִיפּוּר, בַּבַּיִת, עוֹשֶׂה צְחוֹקִים וּדְמוּיוֹת בְּשֵׁם חֶדְוָוה וְצִיפִּי וּמְצַלֶּמֶת אֶת עַצְמִי מְנַקָּה אֶת הַתַּנּוּר, אֲבָל אֲנָשִׁים עוֹשִׂים בְּאִינְסְטַגְרָם בְּדִיּוּק אֶת הַהֶפֶךְ וּמַעֲלִים תְּמוּנוֹת שֶׁלָּהֶם עִם פִילְטֵרִים וּבוֹחֲרִים כָּל תְּמוּנָה יוֹתֵר מִשָּׁעָה.
מָה זֶה? הִשְׁתַּגַּעְתֶּם? לָמָּה לֹא לְהַרְאוֹת אֶת הַחַיִּים כְּמוֹ שֶׁהֵם? מָלֵא כָּתְבוּ לִי שֶׁנָּתַתִּי לָהֶם אוֹמֶץ לְתַעֵד בָּרְשָׁתוֹת הַחֶבְרָתִיּוֹת אֶת עַצְמָם בְּלִי מַסֵּכוֹת.
הִתְאַהַבְתִּי בַּזֶּה שֶׁאֲנִי יְכוֹלָה לְהַרְאוֹת לְלִיאָם וּלְאָלֵין שֶׁגַּם אֲנִי, אִילָנִית, מַרְאָה אֶת עַצְמִי כְּמוֹ שֶׁאֲנִי".

אִם כָּכָה, כּוּלָּם צְרִיכִים לְפַרְסֵם אֶת עַצְמָם מְנַקִּים אֶת הַמִּיקְרוֹגַל וְהַתַּנּוּר.
לְפִי אֵיךְ שֶׁאַתְּ פּוֹעֶלֶת, זוֹ הַדֶּרֶךְ הַמְּהִירָה לְמֵאוֹת אַלְפֵי עוֹקְבִים.
"אֲנִי פִּתְאוֹם הֵבַנְתִּי שֶׁאֲנָשִׁים לֹא יוֹדְעִים לְנַקּוֹת כְּמוֹ שֶׁצָּרִיךְ, אֲנִי בְּהֶלֶם מִזֶּה.
יֵשׁ לִי אַלְפֵי הוֹדָעוֹת בְּאִינְסְטַגְרָם מֵאֲנָשִׁים שֶׁמְּבַקְּשִׁים מִמֶּנִּי עוֹד וְעוֹד טִיפִּים.
הֶעֱלִיתִי וִידֵאוֹ שֶׁל אֵיךְ אֲנִי מְקַפֶּלֶת מַצָּעִים וְאֵיךְ אֲנִי מְנַקָּה אֶת הַמִּיקְרוֹגַל וַאֲנָשִׁים הִשְׁתַּגְּעוּ.
אֲנִי אֲלַמֵּד אֶת כָּל הַטְּרִיקִים, אֲנִי מְאוֹד רוֹצָה שֶׁאֲנָשִׁים יָדְעוּ".
