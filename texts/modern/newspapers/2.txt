יוֹ"ר כָּחוֹל לָבָן בֵּנִי גַּנְץ צָפוּי לִפְנוֹת מָחָר (יוֹם א') לִנְשִׂיא הַמְּדִינָה רְאוּבֵן רִיבְלִין וּלְבַקֵּשׁ מִמֶּנּוּ אַרְכָּה לַמַּנְדָּט לְהַרְכָּבַת הַמֶּמְשָׁלָה.
הַמַּנְדָּט שֶׁהוּעֲנַק לְגַנְץ יִסְתַּיֵּים בְּיוֹם שֵׁנִי בַּחֲצוֹת, וְלַנָּשִׂיא יֵשׁ אֶפְשָׁרוּת לְהַעֲנִיק לְגַנְץ אַרְכָּה שֶׁל עַד 14 יָמִים נוֹסָפִים, אוֹ לְהַעֲבִיר אֶת הַמַּנְדָּט לִנְתַנְיָהוּ.
בֵּינְתַיִים, הָעֶרֶב פָּנָה נְתַנְיָהוּ לְגַנְץ וְקָרָא לוֹ לְהַגִּיעַ לִפְגִישָׁה.
מֵהַלִּיכּוּד נִמְסַר: "רֹאשׁ הַמֶּמְשָׁלָה נְתַנְיָהוּ פָּנָה  לְלִשְׁכַּת חַ"כּ בְּנֵי גַּנְץ כְּדֵי לְקַיֵּים עוֹד הָעֶרֶב פְּגִישָׁה בִּמְעוֹן רֹאשׁ הַמֶּמְשָׁלָה בֵּינֵיהֶם וּבֵין צִוְותֵי הַמּוּ"מ, מִתּוֹךְ רָצוֹן טוֹב וְאַחְרָיוּת לְאוּמִּית לַהֲקָמַת מֶמְשֶׁלֶת אַחְדוּת בַּהֶקְדֵּם הָאֶפְשָׁרִי".

לְפִי שָׁעָה, מֵאָז פִּיצוּץ הַמּוּ"מ בֵּין מִפְלַגְתּוֹ שֶׁל גַּנְץ לַלִּיכּוּד, לֹא נִרְשְׁמָה הִתְקַדְּמוּת בְּמַגָּעִים וְאַף לֹא נֶעֶרְכוּ שִׂיחוֹת אוֹ פְּגִישׁוֹת בְּאֶמְצָעוּת זוּם.
אִם תָּחוּל הִתְקַדְּמוּת דְּרָמָטִית בְּמוּ"מ בֵּין הַצְּדָדִים, יִיתָּכֵן שֶׁנְּתַנְיָהוּ וְגַנְץ יַסְכִּימוּ לִפְנוֹת בִּמְשׁוּתָּף לִנְשִׂיא רִיבְלִין.
גַּם בְּהִינָּתֵן אֶפְשָׁרוּת שֶׁלֹּא תָּחוּל הִתְקַדְּמוּת – יוֹ"ר כָּחוֹל לָבָן מִתְכַּוֵּון לִפְנוֹת בְּכָל מִקְרֶה לַנָּשִׂיא רִיבְלִין.
אֶפְשָׁרוּת נוֹסֶפֶת שֶׁעוֹמֶדֶת בִּפְנֵי הַנָּשִׂיא הִיא הַעֲבָרַת הַמַּנְדָּט לַכְּנֶסֶת.
אִם אֶחָד הַמּוּעֲמָדִים יִזְכֶּה לְ-61 חֲתִימוֹת שֶׁל חַבְרֵי כְּנֶסֶת, הוּא זֶה שֶׁיְּקַבֵּל אֶת הַמַּנְדָּט לְהַרְכָּבַת הַמֶּמְשָׁלָה.

בְּכָחוֹל לָבָן הִבְהִירוּ כִּי לֹא יִהְיוּ מוּכָנִים לִפְתּוֹחַ מֵחָדָשׁ אֶת הַהֶסְכֵּם בְּכָל הַנּוֹגֵעַ לְמִינּוּי שַׂר הַמִּשְׁפָּטִים וְהַוַּועֲדָה לִבְחִירַת שׁוֹפְטִים.
"אֵין מוּ"מ בֵּין צִוְותֵי הַמּוּ"מ אֲבָל יֵשׁ רַק דִּיבּוּרִים מִסָּבִיב", אָמַר גּוֹרֵם בְּכָחוֹל לָבָן.
גּוֹרֵם בְּצֶוֶות הַמּוּ"מ צִייֵּן כִּי "לִנְתַנְיָהוּ יֵשׁ אֶפְשָׁרוּת לַחֲזוֹר עַכְשָׁיו לְשׁוּלְחַן הַמּוּ"מ וְלִסְגּוֹר הֶסְכֵּם תּוֹךְ 24 שָׁעוֹת".

מִנֶּגֶד, שַׂר הָאֵנֶרְגִּיָּה יוּבָל שְׁטַיְינִיץ (הַלִּיכּוּד) קָרָא לַנָּשִׂיא לְהַעֲנִיק אֶת הַמַּנְדָּט לְרֹאשׁ הַמֶּמְשָׁלָה נְתַנְיָהוּ בְּיוֹם שְׁלִישִׁי הַקָּרוֹב, "כְּדֵי לְקַדֵּם וּלְזָרֵז אֶת הֲקָמָתָהּ שֶׁל מֶמְשֶׁלֶת אַחְדוּת", לִדְבָרָיו.
שְׁטַיְינִיץ צִייֵּן כִּי "מֵאַחַר שֶׁמּוּסְכָּם שֶׁנְּתַנְיָהוּ יִהְיֶה רֹאשׁ הַמֶּמְשָׁלָה הָרִאשׁוֹן בְּרוֹטַצְיָה - כָּל עִיכּוּב בַּהֲטָלַת הַמַּנְדָּט עָלָיו, עָלוּל לְעַכֵּב אֶת הֲקָמָתָהּ שֶׁל הַמֶּמְשָׁלָה הַחֲדָשָׁה, שֶׁעַם יִשְׂרָאֵל כָּל כָּךְ זָקוּק לָהּ בְּמַצַּב הַחֵירוּם הַנּוֹכְחִי".
 
נַזְכִּיר כִּי בַּיּוֹם שֵׁנִי הָאַחֲרוֹן הוּפְסַק הַמַּשָּׂא וּמַתָּן הַקּוֹאָלִיצְיוֹנִי בֵּין הַצְּדָדִים, אַף שֶׁשְּׁנֵיהֶם אוֹתְתוּ עַל חֲתִימָה קְרוֹבָה עַל הֶסְכֵּם.
בְּהוֹדָעָה שֶׁנִּמְסְרָה מִטַּעַם כָּחוֹל לָבָן נִכְתַּב כִּי הַמּוּ"מ הוּפְסַק מֵאַחַר שֶׁבַּלִּיכּוּד דָּרְשׁוּ לָדוּן מֵחָדָשׁ בְּאוֹפֶן פְּעוּלָּתָהּ שֶׁל הַוַּועֲדָה לִבְחִירַת שׁוֹפְטִים.
 
"לֹא נְאַפְשֵׁר שׁוּם שִׁינּוּי בְּתִפְקוּד הַוַּועֲדָה לִבְחִירוֹת שׁוֹפְטִים וּפְגִיעָה בַּדֵּמוֹקְרַטְיָה", אָמְרוּ בְּכָחוֹל לָבָן.
לִפְנֵי כֵן מָסְרוּ גּוֹרְמִים בִּשְׁתֵּי הַמִּפְלָגוֹת כִּי גַּנְץ וִיתֵּר עַל וֵטוֹ בְּנוֹשֵׂא הֶחָלַת הָרִיבּוֹנוּת וּנְתַנְיָהוּ מִצִּדּוֹ וִיתֵּר עַל וֵטוֹ בְּנוֹשֵׂא בְּחִירַת הַשּׁוֹפְטִים, מָה שֶׁהוֹבִיל לְבִיקּוֹרֶת חֲרִיפָה מִיָּמִין.

עִם זֹאת, בְּיוֹם רְבִיעִי שׂוֹחֲחוּ נְתַנְיָהוּ וְגַנְץ וְסִיכְּמוּ לְהַמְשִׁיךְ אֶת הַהִידָּבְרוּת בְּצֵאת הַחַג.
בְּהוֹדָעָה מְשׁוּתֶּפֶת לִשְׁתֵּי הַמִּפְלָגוֹת נִמְסַר כִּי נְתַנְיָהוּ הִתְקַשֵּׁר לְגַנְץ וְאִיחֵל לוֹ חַג שָׂמֵחַ.
"שְׁנֵיהֶם דִּיבְּרוּ עַל הַצּוֹרֶךְ בְּמֶמְשֶׁלֶת חֵירוּם לְאוּמִּית לְטוֹבַת עַם יִשְׂרָאֵל בָּעֵת הַזּוֹ", נִכְתַּב.
 
זְמַן קָצָר לִפְנֵי כֵן הִתְיַיחֵס גַּנְץ לְמוּ"מ שֶׁפּוּצַּץ בֵּין הַצְּדָדִים.
"הָיִינוּ קְרוֹבִים מְאוֹד לְבָרֵךְ עַל הַמּוּגְמָר, וְאָז הִגִּיעוּ נְתַנְיָהוּ וַאֲנָשָׁיו עִם דְּרִישָׁה לִפְגּוֹעַ בָּעֲבוֹדָה הַתַּקִּינָה שֶׁל הַוַּועֲדָה לִבְחִירַת שׁוֹפְטִים בְּנִיגּוּד לְמָה שֶׁכְּבָר סוּכַּם.
הוֹדַעְנוּ שֶׁאָנוּ חֲפָצִים בַּהֲקָמַת מֶמְשֶׁלֶת חֵירוּם לְאוּמִּית - אֲבָל לֹא בְּכָל מְחִיר.
אִיחוּד כּוֹחוֹת בִּשְׁעַת מַשְׁבֵּר הוּא חָשׁוּב – אֲבָל דַּוְוקָא בְּשָׁעוֹת כָּאֵלּוּ שְׁמִירָה עַל הַדֶּמוֹקְרַטְיָה וְשִׁלְטוֹן הַחוֹק, חֲשׁוּבָה קְרִיטִית", אָמַר גַּנְץ.
 
גּוֹרְמִים בְּכָחוֹל לָבָן אַף צִייְּנוּ כִּי נְתַנְיָהוּ נָסוֹג בָּרֶגַע הָאַחֲרוֹן מֵהַהֶסְכֵּם, שֶׁכְּבָר הוּכַן וְהוּדְפַּס בְּאַרְבָּעָה עוֹתָקִים בִּמְעוֹנוֹ.
"אוֹ שֶׁהֵם (בַּלִּיכּוּד – י"ק) שָׁמְעוּ אֶת הַתְּגוּבוֹת בַּיָּמִין בְּעִנְיַין מַעֲרֶכֶת הַמִּשְׁפָּט וְנִבְהֲלוּ, אוֹ שֶׁהֵם תִּכְנְנוּ תַּרְגִּיל מְלוּכְלָךְ", אָמְרוּ הַגּוֹרְמִים.

בַּלִּיכּוּד הֵגִיבוּ עַל דִּבְרִי גַּנְץ: "מֵהָרֶגַע הָרִאשׁוֹן הוּסְכַּם כִּי מֶמְשֶׁלֶת הָאַחְדוּת הַשִּׁוְויוֹנִית תָּקוּם עַל שְׁנֵי קַוֵּוי יְסוֹד בְּרוּרִים: קַבָּלַת הַחְלָטוֹת מְשׁוּתֶּפֶת בְּכָל הַנּוֹשְׂאִים וְקִידּוּם הַחַלַּת הָרִיבּוֹנוּת.
לְצַעֲרֵנוּ בָּרֶגַע הָאַחֲרוֹן כָּחוֹל לָבָן הִיא שֶׁחָזְרָה בָּהּ מֵהַסְכָּמוֹת אֵלּוּ, שֶׁהֵן בָּסִיס הֶכְרֵחִי לְכָל מֶמְשָׁלָה שִׁוְויוֹנִית.
בְּרֶגַע שֶׁכָּחוֹל לָבָן תַּחֲזוֹר לַהַסְכָּמוֹת הַמְּקוֹרִיּוֹת נִיתָּן יִהְיֶה לְהַשְׁלִים אֶת הַהֶסְכֵּם וּלְהָקִים מֶמְשֶׁלֶת אַחְדוּת.
סְפִּינִים תִּקְשׁוֹרְתִּיִּים כּוֹזְבִים שֶׁל כָּחוֹל לָבָן אֵינָם מְקָרְבִים אֶת הָאַחְדוּת אֶלָּא מַרְחִיקִים אוֹתָהּ".
