אִמָּא נִכְנְסָה לַחֶדֶר שֶׁלִי וְרָאֲתָה אוֹתִי עִם הַטֵלֵוִיזְיָה מְפֹרֶקֶת לַחֲתִיכוֹת עַל הַשָּׁטִיחַ.
הִיא נֶאֶנְחָה אֲנָחָה עֲמֻקָה.
"תִצְטָרֵךְ לְהַרְכִּיב אוֹתָה בַּחֲזָרָה עַד הַחֲדָשׁוֹת בָּעֶרֶב.
תַסְפִּיק?"
מִגִיל אַרְבַּע אֲנִי מְפָרֵק כָּל מָה שֶׁיֵשׁ בַּבַּיִת.
הִתְחַלְתִי בְּקָטָן עִם הַטוֹסְטֶר.
כְּשֶׁאִמָּא רָאֲתָה אוֹתִי עִם כָּל הַקְּפִיצִים מִסְבִיבִי, הִיא קָפְצָה עַד הַתִקְרָה.
"מָה עָשִׂיתָ? הִשְׁתַגַעְתָ? אָסוּר לְךָ לָגַעַת בְּמַכְשִׁירֵי חַשְׁמַל!" הָעֹנֶשׁ שֶׁקִבַּלְתִי הָיָה שָׁבוּעַ שָׁלֵם בְּלִי מַמְתָקִים, אֲבָל זֶה הָיָה שָׁוֶה אֶת זֶה.
כַּמּוּבָן שֶׁלֹא הִרְכַּבְתִי מֵחָדָשׁ אֶת הַטוֹסְטֶר.
הָיִיתִי רַק בֶּן אַרְבַּע! לֹא הָיָה לִי מֻשָׂג אֵיךְ לַעֲשׂוֹת אֶת זֶה.
אֲבָל לְאַט לְאַט הִתְחַלְתִי לְהָבִין אֵיךְ הַדְּבָרִים עוֹבְדִים.
הִתְחַלְתִי לְנַסוֹת לְהַרְכִּיב אֶת מָה שֶׁפֵּרַקְתִי.
בַּהַתְחָלָה אֲפִלוּ לֹא הִשְׁתַמַשְׁתִי בְּמַבְרֵג כְּדֵי לִפְתֹחַ וְלִסְגֹּר בְּרָגִים.
אִלְתַרְתִי עִם דְבָרִים שֶׁהָיוּ מִסְבִיבִי.
לְגִיל חָמֵשׁ כְּבָר קִבַּלְתִי בְּמַתָנָה אַרְגַז כְּלֵי עֲבוֹדָה וְזֶה הָיָה הַיּוֹם הַמְּאֻשָׁר בְּחַיַי (וְהָאֻמְלָל בְּחַיֵי הַהוֹרִים שֶׁלִי).
מֵאוֹתוֹ רֶגַע לֹא הָיָה צַעֲצוּעַ אֶחָד שֶׁהִשְׁאַרְתִי שָׁלֵם, שׁוּם מַכְשִׁיר חַשְׁמַלִי לֹא הָיָה מוּגָן מִפָּנַי.
הַיּוֹם אֲנִי בֶּן עֶשֶׂר וְהַהוֹרִים שֶׁלִי כְּבָר לֹא מִתְרַגְשִׁים כְּשֶׁהֵם רוֹאִים אֶת אַחַד הַמַּכְשִׁירִים הַיְּקָרִים שֶׁלָהֶם מְפֹרָק.
הֵם יוֹדְעִים שֶׁהֵם יְקַבְּלוּ אוֹתוֹ בַּחֲזָרָה.
לֹא תָמִיד בְּדִיוּק בְּאוֹתוֹ מַצָב שֶׁהָיָה לִפְנֵי הַפֵּרוּק.
הָרַדְיוֹ שֶׁל הַמְּכוֹנִית, לְמָשָׁל, מְשַׁדֵר רַק תָכְנִיוֹת בִּסְפָרַדִית מֵאָז שֶׁפֵּרַקְתִי אוֹתוֹ.
אֲנִי מְתַכְנֵן לְפָרֵק אוֹתוֹ שׁוּב בַּשָּׁבוּעַ הַבָּא כְּדֵי לְהָבִין מָה עָשִׂיתִי לֹא נָכוֹן.
הִסְתַכַּלְתִי בַּשָּׁעוֹן.
הַשָּׁעָה הָיְתָה שֵׁשׁ.
הָיוּ לִי שְׁעָתַיִים לְהַרְכִּיב מֵחָדָשׁ אֶת הַטֵלֵוִיזְיָה עַד הַחֲדָשׁוֹת.
הָיִיתִי צָרִיךְ לָגֶשֶׁת לָעֲבוֹדָה.
הַמּוּזִיקָה שֶׁל תְחִלַת הַחֲדָשׁוֹת נִשְׁמְעָה בְּדִיוּק אַחֲרֵי שֶׁתָלִיתִי (אַבָּא עָזַר לִי) אֶת הַטֵלֵוִיזְיָה, חִבַּרְתִי בִּמְהִירוּת אֶת הַחוּטִים וְלָחַצְתִי עַל כַּפְתוֹר הַשַּׁלָט.
בּוּל בַּזְּמַן.
הָיִיתִי צָרִיךְ לְמַהֵר לְהִכָּנֵס לָאַמְבַּטְיָה.
לֹא רָצִיתִי שֶׁהַהוֹרִים שֶׁלִי יִמְצְאוּ סִבָּה לְהַפְסִיק לָתֵת לִי לְאַרְגֵן מֵחָדָשׁ אֶת מַכְשִׁירֵי הַבַּיִת.
"הֵי, תִרְאֶה אֵיזֶה קֶטַע," שָׁמַעְתִי אֶת אִמָּא אוֹמֶרֶת לְאַבָּא לִפְנֵי שֶׁנִּכְנַסְתִי לַחֲדַר הָאַמְבַּטְיָה.
"הֵם הִתְבַּלְבְּלוּ בַּתַאֲרִיךְ.
הֵם שָׂמוּ אֶת הַתַאֲרִיךְ שֶׁל מָחָר."
כְּשֶׁיָּצָאתִי מֵהָאַמְבַּטְיָה, נָקִי וּמְצֻחְצָח, הֵם קָרְאוּ לִי לַסָלוֹן.
לְפִי הַפָּנִים שֶׁלָהֶם עָשִׂיתִי מַשֶׁהוּ לֹא טוֹב.
"יוֹתָם, מָה עָשִׂיתָ לַטֵלֵוִיזְיָה?"
אוֹי, לֹא.
"לֹא עָשִׂיתִי שׁוּם דָבָר מְיֻחָד.
פֵּרַקְתִי וְהִרְכַּבְתִי.
כְּמוֹ תָמִיד." זִיַפְתִי פִּהוּק.
"אֲנִי מַמָשׁ עָיֵף, אֲנִי יָכוֹל לָלֶכֶת לַמִּטָה?"
"יוֹתָמִי, אוּלַי בְּכָל זֹאת עָשִׂיתָ מַשֶׁהוּ אַחֶרֶת? תְנַסֶה לִזְכֹּר.
זֶה חָשׁוּב."
בֶּאֱמֶת שֶׁנִּסִיתִי לַחְשֹׁב, אֲבָל לֹא עָלָה לִי בָּרֹאשׁ שׁוּם דָבָר.
מָשַׁכְתִי בִּכְתֵפַי.
"לָמָה? מָה קָרָה? נִרְאֶה שֶׁהַטֵלֵוִיזְיָה עוֹבֶדֶת בְּסֵדֶר," אָמַרְתִי.
"אֲנַחְנוּ לֹא בְּטוּחִים…" הִתְחִילָה אִמָּא.
"אֲבָל אֲנַחְנוּ חוֹשְׁבִים…" הִמְשִׁיךְ אַבָּא, "אֲנַחְנוּ חוֹשְׁבִים שֶׁהָרֶגַע רָאִינוּ אֶת מַהֲדוּרַת הַחֲדָשׁוֹת שֶׁל מָחָר."
הַטֵלֵוִיזְיָה שֶׁלָנוּ שִׁדְרָה אֶת הַתָּכְנִיוֹת שֶׁל מָחָר.
בַּהַתְחָלָה זֶה הָיָה דַוְקָא נֶחְמָד.
הִזְמַנְתִי חֲבֵרִים לִרְאוֹת אֶת הַפֶּרֶק הַבָּא שֶׁל הַסִּדְרָה שֶׁאֲנַחְנוּ אוֹהֲבִים יוֹם לִפְנֵי שֶׁהוּא יָצָא.
סִפַּרְתִי לָהֶם שֶׁאִמָּא שֶׁלִי מַכִּירָה מִישֶׁהוּ שֶׁעוֹבֵד בַּטֵלֵוִיזְיָה, וְלָכֵן אֲנַחְנוּ מְקַבְּלִים אֶת הַפְּרָקִים הַחֲדָשִׁים לִפְנֵי כֻּלָם.
אַבָּא שֶׁלִי רָאָה בַּחֲדָשׁוֹת דִוּוּחַ עַל פְּקָק נוֹרָאִי בַּדֶּרֶךְ לְאַשְׁדוֹד.
פְּקָק שֶׁיִקְרֶה, בְּעֶצֶם, מָחָר.
"אֵיזֶה מַזָל, מָחָר יֵשׁ לִי פְּגִישָׁה בְּאַשְׁדוֹד," שָׁמַעְתִי אוֹתוֹ אוֹמֵר לְאִמָּא וּמְצַלְצֵל כְּדֵי לִדְחוֹת אֶת הַפְּגִישָׁה שֶׁלוֹ.
בְּדֶרֶךְ כְּלָל הָיָה אָסוּר לִי לִרְאוֹת חֲדָשׁוֹת, אֲבָל יוֹם אֶחָד שָׁמַעְתִי אֶת הַהוֹרִים שֶׁלִי צוֹעֲקִים מוּל הַטֵלֵוִיזְיָה.
רַצְתִי אֲלֵיהֶם מֵהַחֶדֶר שֶׁלִי לָבוּשׁ בְּפִּיגָ'מָה.
"מָה קָרָה?" שָׁאַלְתִי.
הֵם יָשְׁבוּ חִוְרִים מוּל הַחֲדָשׁוֹת.
בַּחֲדָשׁוֹת דִבְּרוּ עַל רְעִידַת אֲדָמָה וְעַל בַּיִת שֶׁנֶהֱרַס מִמֶנָה.
יֶלֶד נִכְלָא בֵּין הַהֲרִיסוֹת וְרָאִינוּ תְמוּנוֹת שֶׁל מְכַבֵּי אֵשׁ שֶׁמְנַסִים לְהַגִיעַ אֵלָיו וּלְהַצִּיּל אוֹתוֹ.
"אֲנַחְנוּ חַיָבִים לְהַגִיד לָהֶם," אָמְרָה אִמָּא, "שֶׁיֵצְאוּ מֵהַבַּיִת לִפְנֵי רְעִידַת הָאֲדָמָה."
"הֵם יַחְשְׁבוּ שֶׁאֲנַחְנוּ מְשֻׁגָעִים," אָמַר אַבָּא.
"אֵיךְ נַסְבִּיר לָהֶם שֶׁאֲנַחְנוּ יוֹדְעִים דְבָרִים לִפְנֵי שֶׁהֵם קָרוּ?"
"יֵשׁ לִי רַעְיוֹן," אָמַרְתִי.
"בּוֹאוּ נִסַע לַמָּקוֹם הַזֶּה לִפְנֵי רְעִידַת הָאֲדָמָה וַאֲנִי אֲנַסֶה לְהוֹצִיא אֶת הַיֶּלֶד מֵהַבַּיִת שֶׁלוֹ בְּלִי לְסַפֵּר לָהֶם מָה יִהְיֶה."
אִמָּא נִעְנְעָה בְּרֹאשָׁה.
"זֶה מְסֻכָּן מִדַי."
"אֵיפֹה זֶה? אֵיפֹה הַבַּיִת הַזֶּה?" שָׁאַלְתִי.
"מַסְפִּיק.
לֵךְ לִישֹׁן," אָמַר אַבָּא בְּפָנִים רְצִינִיוֹת, וַאֲנִי יָדַעְתִי שֶׁלֹא כְּדַאי לְהִתְוַכֵּחַ אִתוֹ.
אֲבָל שְׁנִיָה לִפְנֵי שֶׁנִּכְנַסְתִי לַחֶדֶר שָׁמַעְתִי אֶת הַשַּׁדְרָנִית אוֹמֶרֶת שֶׁהַבַּיִת הֶהָרוּס נִמְצָא בִּרְחוֹב הַכַּלָנִית 7 בָּעִיר חֲדֵרָה.
בְּדֶרֶךְ כְּלָל אֲנִי לֹא הַיֶּלֶד הֲכִי אַמִיץ בָּעוֹלָם.
אֲפִלוּ לֹא הֲכִי אַמִיץ בַּכִּתָה.
אֲבָל בְּאוֹתוֹ יוֹם הֶחְלַטְתִי שֶׁאֲנִי חַיָב לְהַצִּיּל אֶת הַיֶּלֶד הַזֶּה.
בַּדֶּרֶךְ לְבֵית הַסֵּפֶר לָקַחְתִי קְצָת כֶּסֶף מִקֻפַּת הַחִסָכוֹן שֶׁלִי.
בַּטֵלֵוִיזְיָה אָמְרוּ שֶׁרְעִידַת הָאֲדָמָה הָיְתָה אַחֲרֵי הַצָּהֳרַיִם, כָּךְ שֶׁתִכְנַנְתִי לִנְסֹעַ לַחֲדֵרָה אַחֲרֵי הַלִּמוּדִים.
כְּשֶׁיָּצָאתִי מֵהַבַּיִת רָאִיתִי אֶת אִמָּא מִסְתַכֶּלֶת בְּחַדְשׁוֹת הַבֹּקֶר.
הִיא אַף פַּעַם לֹא עָשְׂתָה אֶת זֶה עַד הַיּוֹם.
"יֵשׁ חָדָשׁ עִם הַיֶּלֶד?" שָׁאַלְתִי לִפְנֵי שֶׁיָּצָאתִי.
"לֹא," נֶאֶנְחָה אִמָּא.
"עֲדַיִן מְנַסִים לְפַנוֹת אֶת הַהֲרִיסוֹת.
מִסְתַבֵּר שֶׁזֶה בִּכְלָל לֹא…" הִמְשִׁיכָה לְהַגִיד, אֲבָל לֹא שָׁמַעְתִי אֶת הֶמְשֵׁךְ דְבָרֶיהָ.
כְּבָר הָיִיתִי בַּחוּץ.
אַחֲרֵי הַלִּמוּדִים עָלִיתִי עַל אוֹטוֹבּוּס לַתַחֲנָה הַמֶּרְכָּזִית וּמִשָׁם לָקַחְתִי אוֹטוֹבּוּס לַחֲדֵרָה.
כָּל הַזְּמַן הִסְתַכַּלְתִי בַּשָּׁעוֹן.
הָיִיתִי צָרִיךְ לְהַסְפִּיק לְהַגִיעַ לִרְחוֹב הַכַּלָנִית 7 לִפְנֵי חָמֵשׁ אַחַר הַצָּהֳרַיִם וּלְהוֹצִיא אֵיכְשֶׁהוּ אֶת הַיֶּלֶד מֵהַבַּיִת.
לִפְנֵי שֶׁהִגַעְנוּ לַחֲדֵרָה שָׁאַלְתִי אֶת הַנֶּהָג אֵיפֹה זֶה רְחוֹב הַכַּלָנִית וְהוּא הִסְבִּיר לִי בְּדִיוּק אֵיךְ לְהַגִיעַ.
קְצָת הִתְבַּלְבַּלְתִי בַּדֶּרֶךְ.
כְּשֶׁהִגַעְתִי הָיָה כְּבָר אַרְבַּע וָחֵצִי.
הָיִיתִי צָרִיךְ לְהִזְדָרֵז.
הַבַּיִת בִּרְחוֹב הַכַּלָנִית 7 הָיָה בַּיִת פְּרָטִי קָטָן וְהוּא נִרְאָה עוֹמֵד וְשָׁלֵם לַחֲלוּטִין.
שׁוּם דָבָר לֹא רָמַז שֶׁבְּתוֹךְ חֲצִי שָׁעָה הוּא יַהֲפֹךְ לָחֳרָבוֹת.
שַׁעַר הַגִּנָה הָיָה פָּתוּחַ.
"הָלוֹ? יֵשׁ כָּאן מִישֶׁהוּ?" קָרָאתִי כְּשֶׁעָבַרְתִי בַּשַּׁעַר.
רֹאשׁ שֶׁל יֶלֶד בְּגִילִי הוֹפִיעַ בַּחַלוֹן.
"מִי אַתָה?" הוּא קָרָא לְעֶבְרִי.
"אַתָה צָרִיךְ לָצֵאת הַחוּצָה! אֲנִי צָרִיךְ לְהַגִיד לְךָ מַשֶׁהוּ!" קָרָאתִי אֵלָיו.
"לֹא יָכוֹל.
אֲנִי צָרִיךְ לְהָכִין שִׁעוּרִים," הוּא אָמַר לִי.
"אַתָה חַיָב לָצֵאת!" צָעַקְתִי.
"מָה פִּתְאוֹם?" אָמַר הַיֶּלֶד.
הִתְקַדַמְתִי וְדָפַקְתִי עַל דֶלֶת הַכְּנִיסָה.
לֹא יָדַעְתִי מָתַי בְּדִיוּק תִקְרֶה רְעִידַת הָאֲדָמָה, אֲבָל זֶה יָכוֹל לִקְרוֹת בְּכָל רֶגַע.
הַדֶּלֶת נִפְתְחָה וּפַרְצוּפוֹ הַכּוֹעֵס שֶׁל הַיֶּלֶד שָׁאַג עָלַי: "אָמַרְתִי לְךָ שֶׁאֲנִי עָסוּק עַכְשָׁו!"
"אַתָה חַיָב לִרְאוֹת מַשֶׁהוּ בַּגִנָה," הִתְחַנַנְתִי אֵלָיו.
הוּא רַק נִרְאָה יוֹתֵר כּוֹעֵס וְהָלַךְ שְׁלוֹשָׁה צְעָדִים אֲחוֹרָה אֶל תוֹךְ הַבַּיִת.
נִכְנַסְתִי אַחֲרָיו.
בְּעוֹד כַּמָּה דַקוֹת הוּא יִהְיֶה קָבוּר בֵּין הַהֲרִיסוֹת אִם לֹא אֶדְחַף אוֹתוֹ הַחוּצָה.
"מָה אַתָה עוֹשֶׂה?" פִּתְאוֹם הוּא נִרְאָה קְצָת מְפֻחָד.
"לָמָה אַתָה נִכְנָס אֵלַי הַבַּיְתָה? אֲנִי לֹא מַכִּיר אוֹתְךָ בִּכְלָל."
הֶחְלַטְתִי שֶׁאֵין לִי מָה לְהַפְסִיד וְאָמַרְתִי אֶת הָאֱמֶת: "עוֹד כַּמָּה דַקוֹת תִהְיֶה רְעִידַת אֲדָמָה.
אַתָה חַיָב לָצֵאת מֵהַבַּיִת."
"אֵיזֶה תַרְגִיל אַתָה עוֹשֶׂה לִי?" שָׁאַל בְּחַשְׁדָנוּת, אֲבָל אָז הִרְגַשְׁנוּ אֶת הָרַעַד בָּרַגְלַיִם.
בַּהַתְחָלָה חַלָשׁ כְּמוֹ דִגְדוּג, אֲבָל לְאַט הוּא הִתְגַבֵּר.
הַזְּגוּגִיוֹת בַּחַלוֹן הִשְׁמִיעוּ קוֹל מַפְחִיד.
הַיֶּלֶד הִסְתַכֵּל בִּי וּפִתְאוֹם רָץ הַחוּצָה, תוֹךְ שֶׁהוּא נִתְקָל בִּכְתֵפִי.
נָפַלְתִי עַל בִּרְכַּי וּבִזְמַן שֶׁרְעִידַת הָאֲדָמָה הִתְגַבְּרָה וְהַקּוֹרוֹת הֵחֵלוּ לִפֹּל, הֵבַנְתִי שֶׁהַיֶּלֶד הַכָּלוּא בֵּין הַהֲרִיסוֹת הוּא אֲנִי.
רַק אַחֲרֵי יוֹמַיִים הִצְלִיחוּ לְחַלֵץ אוֹתִי מִבֵּין הֲרִיסוֹת הַבַּיִת.
יוֹמַיִים שֶׁבָּהֶם הַהוֹרִים שֶׁלִי נִטְרְפוּ מִדְאָגָה.
בַּהַתְחָלָה הֵם לֹא הֵבִינוּ לָמָה לֹא חָזַרְתִי מִבֵּית הַסֵּפֶר.
רַק אַחַר כָּךְ, כְּשֶׁהַיֶּלֶד הוֹפִיעַ בַּטֵלֵוִיזְיָה וְסִפֵּר מָה בָּאֱמֶת קָרָה, הֵם הֵבִינוּ.
מִילְיוֹן מַצְלֵמוֹת טֵלֵוִיזְיָה סִנְוְרוּ אוֹתִי כְּשֶׁהִצְלִיחוּ לְחַלֵץ אוֹתִי, עָטְפוּ אוֹתִי בִּשְׂמִיכָה וְנָתְנוּ לִי לִשְׁתוֹת.
הַפֶּה שֶׁלִי הָיָה יָבֵשׁ וּמָלֵא בְּעָפָר.
הִצְלַחְתִי לְהִתְחַמֵק וְלֹא לְגַלוֹת לָהֶם כְּלוּם לַמְרוֹת הַשְׁאֵלוֹת שֶׁהִמְטִירוּ עָלַי: "אֵיךְ יָדַעְתָ לְהַזְהִיר אֶת הַיֶּלֶד שֶׁגָר בַּבַּיִת?" "אֵיךְ הִגַעְתָ בְּדִיוּק לַבַּיִת הַיָּחִיד שֶׁנֶהֱרַס בִּרְעִידַת הָאֲדָמָה?"
אַחֲרֵי שֶׁסוֹף-סוֹף נִפְטַרְנוּ מֵהַכַּתָבִים וּמֵהָעִתוֹנָאִים וְאַחֲרֵי שֶׁקִבַּלְתִי טִפּוּל רְפוּאִי קָצָר, הִגַעְנוּ הַבַּיְתָה.
הַדָּבָר הָרִאשׁוֹן שֶׁרָאִיתִי הָיָה אַרְגָז גָדוֹל לְיַד הַפַּחִים בַּכְּנִיסָה לַבַּיִת.
כְּשֶׁנִּכְנַסְתִי לַסָלוֹן וְרָאִיתִי אֶת הַמָּקוֹם הָרֵיק עַל הַקִּיר, הֵבַנְתִי מָה הָיָה בָּאַרְגָז.
"אַתֶּם זוֹרְקִים אֶת הַטֵלֵוִיזְיָה שֶׁהִמְצֵאתִי? הַטֵלֵוִיזְיָה שֶׁמְשַׁדֶרֶת אֶת הַשִּׁדוּרִים שֶׁל מָחָר?" שָׁאַלְתִי.
"הֶחְלַטְנוּ שֶׁאֲנַחְנוּ לֹא רוֹצִים לָדַעַת מָה יִהְיֶה מָחָר," אָמְרָה אִמָּא.
"וְאִם נִרְצֶה לָדַעַת," הוֹסִיף אַבָּא, "פָּשׁוּט נְחַכֶּה לְמָחָר."
