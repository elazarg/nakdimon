
בְּיָם שֶׁל עֲשָׂבִים, מְכֻסֶה בְּטַל מִשָׁמַיִם, שׁוֹכֵב לוֹ נַעַר וְרוֹעֶה אֶת עֵדֶר הַצֹּאן שֶׁל אָבִיו.
כֻּלָם קוֹרְאִים לוֹ יַאשֶׁק.
אֶת שְׁמוֹ הַמָּלֵא הוּא עַצְמוֹ אֵינוֹ יוֹדֵעַ.
כֻּלָם יוֹדְעִים שֶׁהוּא אִכָּר.
בּוּר.
אָבִיו, יִצְחָק הַחַלְבָּן, כְּבָר הִתְיָאֵשׁ מִמֶנוּ.
אֲבָל בְּכָל זֹאת נִסָה לְהָבִיא אֵלָיו מְלַמֵד מֵהָעִיר.
הַמְּלַמֵד נִסָה לְלַמֵד אֶת יַאשֶׁק לִקְרֹא.
הַרְבֵּה פְּעָמִים חָזַר אִתוֹ עַל כָּל מִלָּה וּמִלָּה.
וְהוּא, יַאשֶׁק, הָאוֹתִיוֹת פּוֹרְחוֹת מִמֶנוּ כְּאִלוּ אֵינָן נוֹגְעוֹת לוֹ.
– אִי אֶפְשָׁר לְלַמֵד חֲמוֹר – אוֹמֵר הַמְּלַמֵד וּמֵנִיד בְּרֹאשׁוֹ, וְאִמּוֹ שֶׁל יַאשֶׁק מִתְבּוֹנֶנֶת בִּבְנָה וְנֶאֱנַחַת.
הוֹרָיו אֲפִלוּ אֵינָם מַכְרִיחִים אוֹתוֹ לָקַחַת סִדוּר וּלְהִתְפַּלֵל.
לְהִתְפַּלֵל? הוּא? יַאשֶׁק יִתְפַּלֵל? מָה לוֹ וְלֶאֱלֹהִים? חֶרְפָּה לְבוֹרֵא עוֹלָם שֶׁיַּאשֶׁק יִתְפַּלֵל לְפָנָיו.
וּבְכָל זֹאת מֵבִין יַאשֶׁק אֶת הָאֱלֹהִים וּמַרְגִישׁ אוֹתוֹ בְּלִבּוֹ.
הוּא רוֹאֶה אֶת אֱלֹהִים בְּכָל מָקוֹם, בְּכָל אֲשֶׁר יִפְנֶה.
הוּא רוֹאֶה אוֹתוֹ כְּנַחַל הַזּוֹרֵם וּמְסַפֵּר אֶת סוֹדוֹתָיו לָעֵשֶׂב הַצּוֹמֵחַ עַל גְּדוֹתָיו.
וּכְשֶׁהוּא רוֹאֶה עָנָן בַּשָּׁמַיִם, הוּא מַרְגִישׁ, הוּא מִתְאַוֶה לְאֵיזֶה דָּבָר, מִתְגַעְגֵעַ… כַּאֲשֶׁר עַנְנֵי גֶּשֶׁם מוֹפִיעִים בַּשָּׁמַיִם, הָרַעַם רוֹעֵם וְהַבָּרָק מִתְלַקֵחַ כְּמוֹ אֵשׁ… כַּאֲשֶׁר יוֹרֵד גֶּשֶׁם אוֹ עֲלָטָה מְכַסָה אֶת הַכְּפָר… אָז הוּא מַרְגִישׁ בְּלִבּוֹ אֶת הָאֱלֹהִים… לֹא בַּשָּׁמַיִם הוּא יוֹשֵׁב.
יַאשֶׁק לֹא צָרִיךְ לְהָרִים רֹאשׁוֹ לְמַעְלָה כְּדֵי לִרְאוֹת אוֹתוֹ.
לֹא, לְיַאשֶׁק נִרְאֶה שֶׁאֱלֹהִים גָּר בְּאֶרֶץ רְחוֹקָה, בְּעִיר גְּדוֹלָה, וּסְבִיבוֹ שָׂרִים גְּדוֹלִים וְכֻלָם עֲבָדִים שֶׁלוֹ כְּמוֹ הָעֲבָדִים שֶׁיֵשׁ לְרֹאשׁ הַכְּפָר שֶׁל יַאשֶׁק.
וְאֵיךְ יָכוֹל יַאשֶׁק עַצְמוֹ לְהִתְקָרֵב לֶאֱלֹהִים? אֲפִלוּ רוֹעֵה הַצֹּאן שֶׁל אֱלֹהִים – חוֹשֵׁב יַאשֶׁק בְּלִבּוֹ- צָרִיךְ לִהְיוֹת אָדוֹן גָּדוֹל מְאֹד.
וּמִי יוֹדֵעַ אִם אֲפִלוּ רֹאשׁ הַכְּפָר שֶׁלוֹ יוּכַל לִהְיוֹת נַהַג הַמֶּרְכָּבָה שֶׁל אֱלֹהִים?
וְאֵיךְ יִתְנוּ לוֹ, לְיַאשֶׁק, לָגֶשֶׁת אֵלָיו?
אֲבָל לִפְעָמִים, כַּאֲשֶׁר פְּנֵי הַשָּׁמַיִם מִלְמַעְלָה טְהוֹרִים וּבְהִירִים וְחוֹלְמִים בִּמְנוּחָה בִּצְעִיף תְּכֵלֶת, וְהָעֲשָׂבִים צוֹמְחִים מִתַחְתָם בִּדְמָמָה, וְהַצְמָחִים מַעֲמִיקִים שֹׁרֶשׁ בָּאֲדָמָה וּמַבִּיטִים בִּשְׁתִיקָה לְמַעְלָה, וּמִמוּל עוֹמֵד הַיַּעַר כְּמוֹ זָקָן יָרֹק שֶׁל הַכְּפָר וּמִישֶׁהוּ הוֹלֵךְ בַּדֶּרֶךְ מֵרָחוֹק, וְהַשָּׁמַיִם מִלְמַעְלָה יוֹרְדִים לְאַט-לְאַט עַד שֶׁכִּמְעַט מַגִיעִים אֶל הָאָרֶץ, וְיַאשֶׁק יוֹשֵׁב לוֹ וּמַבִּיט – אָז נִדְמֶה לוֹ שֶׁאֱלֹהִים עָזַב לְרֶגַע אֶת כְּבוֹד הַמְּלָכִים וְהִגִיעַ לְבַדוֹ אֶל הַשַָּׂדֶה, הִשְׂתָרֵעַ עַל הָאָרֶץ לְצִדוֹ שֶׁל יַאשֶׁק וְיוֹשֵׁב לוֹ וּמְחַכֶּה…
וְיַאשֶׁק רוֹצֶה לְהוֹדוֹת וּלְהַלֵּל אֶת ה' עַל כָּל אֲשֶׁר פֹּה מִסָבִיב.
וְלִפְעָמִים הוּא רוֹצֶה לָלֶכֶת וְלָלֶכֶת עַד אֲשֶׁר יַגִיעַ לְשָׁם, אֶל הָעִיר הַגְּדוֹלָה, אֶל מְקוֹם מוֹשָׁבוֹ שֶׁל הָאֱלֹהִים, אֶל הֵיכָלוֹ.
– אָז אֶגַשׁ וַאֲנַשֵׁק אֶת הָאֱלֹהִים – הוּא חוֹשֵׁב בְּלִבּוֹ.
וְתֵכֶף הוּא נֶאֱנָח: הַאִם יִתְנוּ לִי לָגֶשֶׁת אֵלָיו?
אוּלַי יוּכַל לְהַגִיעַ אֶל הָאַרְמוֹן – הוּא אוֹמֵר בְּלִבּוֹ – אֲבָל עוֹמְדִים שָׁם שׁוֹמְרִים, אֲנָשִׁים גְּבוֹהִים לְבוּשִׁים בִּבְגָדִים מְפֹאָרִים, וְהֵם יְגָרְשׁוּ אוֹתוֹ מִשָׁם…
וּמִיָד הוּא מִתְחָרֵט כִּי אֵינֶנוּ צָרִיךְ לָלֶכֶת, כִּי אֱלֹהִים נִמְצָא פֹּה, בַּשָּׂדֶה הָרָחָב, וְהַשּׁוֹמְרִים הַגְּבוֹהִים שׁוֹמְרִים רַק עַל אַרְמוֹן רֵיק.
"אֵל מֶלֶךְ נֶאֱמָן" מְלַמֵד אוֹתוֹ הָרַבִּי מִתוֹךְ סִדוּר הַתְפִלָה.
וְהַמִּלִים "אֵל מֶלֶךְ נֶאֱמָן" נִרְאוֹת לוֹ כְּמוֹ לַחַשׁ, כְּמוֹ סִיסְמָה לְהַגִיד לַשּׁוֹמְרִים כְּדֵי שֶׁיִתְנוּ לוֹ לְהִכָּנֵס לְהֵיכַל הָאֱלֹהִים.
אֲבָל יֵשׁ לוֹ מִין תְּפִלָה אַחֶרֶת אֶל הָאֱלֹהִים, תְּפִלָה בְּלִי מִלִים.
תְפִלָה הַצּוֹמַחַת מִקִרְבּוֹ וּמְמַלֵאת אֶת לִבּוֹ וְהִיא מִתְפָּרֶצֶת מִשָׁם הַחוּצָה בִּשְׁרִיקָה…
כְּשֶׁהוּא רוֹצֶה לְהִתְפַּלֵל הוּא מְקָרֵב שְׁתֵי אֶצְבָּעוֹת אֶל פִּיו, מְנַפֵּח אֶת שְׂפָתָיו וְנוֹשֵׁף – וּתְפִלָתוֹ מַשְׁמִיעָה קוֹל הֵד בְּכָל הַיַּעַר…
וֶאֱלֹהִים מֵבִין אוֹתוֹ, מֵבִין הֵיטֵב אֶת תְּפִלָתוֹ.
וְהוּא שׁוֹרֵק כָּךְ רַק בְּשָׁעָה שֶׁהוּא רוֹצֶה לִשְׁרֹק אֶל הָאֱלֹהִים, כְּשֶׁהוּא מַרְגִישׁ צרֶֹךְ בִּשְׁרִיקָה.
וֶאֱלֹהִים, כָּךְ נִדְמֶה לוֹ, פָּשַׁט אֶת בְּגָדָיו וְהוּא מִשְׂתָרֵעַ עַכְשָׁו עַל כָּל מֶרְחֲבֵי הַשַָּׂדֶה וְשׁוֹמֵעַ אֶת שְׁרִיקָתוֹ וְנֶהֱנֶה מִמֶנָה וְשָׂמֵחַ בָּה.
לְיַאשֶׁק כְּבָר מָלְאוּ שְׁלוֹשׁ-עֶשְׂרֵה שָׁנָה וְהוּא לֹא יָכוֹל לִקְרֹא "שְׁמַע" וְרֹאשׁ הַשָּׁנָה מִתְקָרֵב וּבָא.
וְהָרַבִּי מְנַסֶה לְלַמֵד אוֹתוֹ כְּדֵי שֶׁלְפָחוֹת יֵדַע אֶת "שְׁמַע" בְּרֹאשׁ הַשָּׁנָה וְיוּכְלוּ לָקַחַת אוֹתוֹ הָעִירָה.
וְלִכְבוֹד הַחַג קָנוּ לוֹ, לְיַאשֶׁק, חֲלִיפַת בְּגָדִים חֲדָשָׁה, נַעֲלַיִם וְכוֹבַע חָדָשׁ… לְפָחוֹת שֶׁיִקְרָא אֶת "שְׁמַע"…
בְּבֵית הַכְּנֶסֶת עוֹלֶה קוֹל בְּכִי חָנוּק מֵעֶזְרַת הַנָּשִׁים, וְיַאשֶׁק, לָבוּשׁ בִּבְגָדָיו הַחֲדָשִׁים, עוֹמֵד וְשַׂעֲרוֹתָיו נוֹפְלוֹת מִתַחַת לְכוֹבָעוֹ עַל פָּנָיו.
פִּתְאֹם הוּא עוֹזֵב אֶת מְקוֹמוֹ וּפוֹרֵץ לוֹ דֶּרֶךְ וּמִתְקָרֵב אֶל אֲרוֹן הַקֹדֶשׁ וְעוֹמֵד לְיַד הַחַזָן.
הַסִּדוּר אֲשֶׁר נָתַן לוֹ אָבִיו כְּדֵי שֶׁיַּחְשְׁבוּ שֶׁהוּא מִתְפַּלֵל מִמֶנוּ נוֹפֵל מִיָדָיו וְהוּא עוֹמֵד בְּעֵינַיִם פְּעוּרוֹת וּמַבִּיט.
נְעָרִים שׁוֹבָבִים רוֹאִים אֶת יַאשֶׁק, רוֹמְזִים אֶחָד לַשֵׁנִי, מַצְבִּיעִים עָלָיו, צוֹחֲקִים, וְאֶחָד מֵהֶם נִגָשׁ אֵלָיו וְסוֹטֵר בְּאֶצְבָּעוֹ עַל אַפּוֹ שֶׁל יַאשֶׁק.
וְיַאשֶׁק אֵינוֹ שׁוֹמֵעַ, אֵינוֹ מַרְגִישׁ.
הוּא עוֹמֵד וּמַבִּיט עַל הַחַזָן, עַל הַמִּתְפַּלְּלִים הַבּוֹכִים.
הִנֵה לְפָנָיו הַפָּרֹכֶת הַמְּכַסָה אֶת אֲרוֹן הַקֹדֶשׁ וְעָלֶיהָ כָּתוּב בְּאוֹתִיוֹת זָהָב "קֹדֶשׁ לְה".
-אֱלֹהִים בְּוַדַאי שָׁם מֵאֲחוֹרֵי הַפָּרֹכֶת – הוּא חוֹשֵׁב בְּלִבּוֹ.
אִמּוֹ מַבִּיטָה עָלָיו מֵעֶזְרַת הַנָּשִׁים.
הִיא רוֹאָה אוֹתוֹ עוֹמֵד וְלוֹטֵשׁ עֵינַיִם מִסָבִיב, וְהַסִּדוּר נוֹפֵל מִיָדָיו, וְהִיא נֶאֱנַחַת וְאוֹמֶרֶת בְּלִבָּה: "אִכָּר, בּוּר גָּמוּר!"
וְאָבִיו רוֹאֶה אוֹתוֹ מִתַחַת לְטַלִיתוֹ וְנֶאֱנָח מָרָה: "רִבּוֹנוֹ שֶׁל עוֹלָם! זְכרֹ גַּם יְלָדִים כְּמוֹ יַאשֶׁק!"
וְיַאשֶׁק עוֹמֵד וּמִתְבּוֹנֵן.
כֻּלָם בּוֹכִים, כֻּלָם מִתְפַּלְלִים, כֻּלָם קוֹרְאִים לֶאֱלֹהִים בְּחָזְקָה וְגַם הוּא רוֹצֶה לַעֲשׂוֹת כְּמוֹהֶם.
גַם הוּא רוֹצֶה לְהִתְפַּלֵל.
גַם הוּא רוֹצֶה – לֹא לִבְכּוֹת, לֹא לִצְעֹק, אֲבָל לְהוֹדוֹת וּלְהַלֵּל אֶת ה' הוּא רוֹצֶה! עַל הַכֹּל הוּא רוֹצֶה לְהוֹדוֹת! הוּא חוֹטֵף אֶת הַסִּדוּר, פּוֹתֵחַ אוֹתוֹ בַּמָּקוֹם הַמְּסֻמָן לוֹ, בִּ"קְרִיאַת שְׁמַע", וּמַתְחִיל לוֹמַר אֶת הַמִּלִים.
"אֵ-ל מֶ-לֶ-ךְ נֶ-אֱ-מָ-ן…"
אֲבָל הַתְפִלָה הַזֹּאת אֵינָה מְלַבֶּבֶת אוֹתוֹ.
הוּא לֹא מֵבִין אוֹתָה.
זֶה רַק לַחַשׁ, לַחַשׁ יָבֵשׁ…
הוּא רוֹצֶה לְהוֹדוֹת וּלְהַלֵּל בֶּאֱמוּנָה, מֵעֹמֶק לִבּוֹ – – –
הוּא פּוֹחֵד מֵהָאֲנָשִׁים הַמִּתְפַּלְּלִים, מֵהַקָּהָל הַגָּדוֹל הַמִּתְפַּלֵל בְּאֹפֶן אַחֵר.
אֲבָל הוּא לֹא יָכוֹל לְהִתְאַפֵּק- ה' הוּא גָּבוֹהַ מֵעַל כֻּלָם! הוּא מֻכְרָח!
וְהוּא מְקָרֵב אֶת אֶצְבְּעוֹתָיו אֶל פִּיו – – –
וּשְׁרִיקָה חַדָּה עוֹבֶרֶת עַל פְּנֵי בֵּית הַכְּנֶסֶת הַבּוֹכֶה.
כָּל קְהַל הַמִּתְפַּלְּלִים נִבְהָל.
מִי הוּא זֶה? מָה זֶה? מִי שׁוֹרֵק בְּמָקוֹם קָדוֹשׁ? אָבִיו הִתְקָרֵב אֵלָיו כְּדֵי לִתְפֹּס בְּצַוָארוֹ, כֻּלָם הִקִיפוּ אוֹתוֹ כְּדֵי לְהַכּוֹתוֹ.
פִּתְאֹם פּוֹנֶה הָרַב הַקָּדוֹשׁ מֵאֲרוֹן הַקֹדֶשׁ אֶל הַקָּהָל וְשׁוֹאֵל "אֵיפֹה הַצַּדִיק אֲשֶׁר נָקַב וְחָדַר מִבַּעַד לְעַנְנֵי הָעוֹפֶרֶת עַד הַשָּׁמַיִם וּפָרַץ דֶּרֶךְ לִתְפִלוֹתֵינוּ לַעֲבוֹר שָׁמָה?"
אֲבָל הַצַּדִיק כְּבָר לֹא הָיָה בְּבֵית הַכְּנֶסֶת.
הוּא בָּרַח מִשָׁם, חָלַץ אֶת נַעֲלָיו, שָׂם אוֹתָן עַל שִׁכְמוֹ וְהוּא צוֹעֵד בַּדֶּרֶךְ אֶל הַכְּפָר…
