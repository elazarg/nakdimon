בְּיָמִים רְגִילִים אֲנִי לֹא אוֹהֶבֶת לְבַלּוֹת יוֹתֵר מִדַּי זְמַן בַּמִּטְבָּח. כְּלוֹמַר, בִּישּׁוּל זֶה הַמִּקְצוֹעַ שֶׁלִּי וְגַם אַהֲבָה גְּדוֹלָה מְאוֹד, אֲבָל הַמְּשִׁיכָה הַגְּדוֹלָה שֶׁלִּי הִיא לֹא לְמַתְכּוֹנִים מוּרְכָּבִים וְרַבֵּי שְׁלַבִּים שֶׁדּוֹרְשִׁים שָׁעוֹת שֶׁל הֲכָנוֹת, אֶלָּא לְמַתְכּוֹנִים בֵּיתִיִּים קְלִילִים וּמְהִירִים שֶׁגּוֹרְפִים מַחְמָאוֹת (בְּלִי לְגַלּוֹת כַּמָּה קַל הָיָה לְהָכִין אוֹתָם, כֵּן?)

בַּיָּמִים הָאֵלֶּה, בָּהֶם כּוּלָּנוּ מִתְכַּנְּסִים בַּבָּתִּים עִם הַמִּשְׁפָּחָה וְהַיְּלָדִים, הַמִּסְעָדוֹת נִסְגָּרוֹת וַאֲנַחְנוּ חוֹזְרִים לְבַלּוֹת יוֹתֵר בֵּין הַסִּירִים, אֲנִי דַּוְוקָא מוֹצֵאת מִפְלָט בַּמִּטְבָּח הַבֵּיתִי שֶׁלִּי. כְּשֶׁיֵּשׁ זְמַן לְבַשֵּׁל, פִּתְאוֹם כְּבָר נֶהֱנִים יוֹתֵר מִלְּהַשְׁקִיעַ בְּאוֹכֶל טָעִים וּמֵזִין לַאֲהוּבִים עָלֵינוּ, אֲנַחְנוּ לֹא מְמַהֲרִים לְשׁוּם מָקוֹם וְאִם גַּם אַתֶּם נִתְקַפְתֶּם בְּקַדַּחַת קְנִיּוֹת, הֲרֵי שֶׁהַבַּיִת שֶׁלָּכֶם מָלֵא בִּרְכִיבִים מִכָּל הַסּוּגִים שֶׁפּוֹתְחִים בִּפְנֵיכֶם שֶׁפַע שֶׁל אֶפְשָׁרוּיוֹת טְעִימוֹת. וּבֵינֵינוּ, מְדוּבָּר בְּיוֹפִי שֶׁל תְּרַאפְּיָה לְהַרְגָּעַת הָעֲצַבִּים הַמְּתוּחִים.

הַשְּׁהִיָּיה הַמִּשְׁפַּחְתִּית בַּבַּיִת לֹא רַק מַחֲזִירָה אוֹתָנוּ לִישִׁיבָה מְשׁוּתֶּפֶת סְבִיב שׁוּלְחַן הָאוֹכֶל הַבֵּיתִי, הִיא גַּם הִזְדַּמְּנוּת לְהַזְמִין אֶת הַיְּלָדִים לְהִשְׁתַּתֵּף בַּהֲכָנוֹת – לְלַמֵּד אוֹתָם כְּלָלֵי בִּישּׁוּל בְּסִיסִיִּים, לְהֵיעָזֵר בָּהֶם בַּמַּטָּלוֹת פְּשׁוּטוֹת שֶׁל בְּחִישָׁה וְעִרְבּוּבִים וּלְמַנּוֹת אוֹתָם לִמְתַבְּלִים הַלְּאוּמִּיִּים.

לְצַד הַכֵּיף וְהִטְעִים, בִּמְיוּחָד בַּיָּמִים הָאֵלֶּה אָסוּר לִשְׁכּוֹחַ אֶת הַסִּיבָּה הָרִאשׁוֹנִית לְכָךְ שֶׁאֲנַחְנוּ אוֹכְלִים – כְּדֵי לְסַפֵּק לַגּוּף שֶׁלָּנוּ חוֹמְרֵי בִּנְיָין חִיּוּנִיִּים וְרָצוּי שֶׁיִּהְיוּ עַתִּירִים בָּחוֹמָרִים מְזִינִים כְּמוֹ חֶלְבּוֹנִים אֵיכוּתִיִּים, וִיטָמִינִים וּמִינֵרָלִים, סִיבִים תְּזוּנָתִיִּים שֶׁעוֹזְרִים בִּתְחוּשַׁת הַשּׂוֹבַע וְכָל מָה שֶׁיָּכוֹל לְהַגְבִּיר אֶת הָאֵנֶרְגִּיָּה וּלְחַזֵּק אֶת הַמַּעֲרֶכֶת הַחִיסּוּנִית, בִּמְיוּחָד בַּיָּמִים הַנּוֹכְחִיִּים.

לֹא פָּחוֹת חָשׁוּב - לְהִימָּנַע מִלְּהַעֲמִיס עַל הַגּוּף מַאֲכָלִים מַכְבִּידִים וּמַזִּיקִים. זֶה שֶׁאֲנַחְנוּ תְּקוּעִים בַּבַּיִת לֹא אוֹמֵר שֶׁעַכְשָׁיו הַזְּמַן לְהִתְפָּרֵעַ עִם מַמְתַּקִּים וַחֲטִיפִים אוֹ לֶאֱכוֹל מָזוֹן עַתִּיר שׁוּמָּן רַווּי, כּוֹלֶסְטֵרוֹל וּשְׁאָר מַזִּיקִים. בִּמְיוּחָד בִּתְקוּפָה כָּזוֹ, הַגּוּף שֶׁלָּכֶם יוֹדֶה לָכֶם עַל אַסְפָּקָה סְדִירָה שֶׁל מָזוֹן טָרִי וְאֵיכוּתִי מֵהַצּוֹמֵחַ, עִם מִינִימוּם מוּצָרִים מְעוּבָּדִים וּמַקְסִימוּם אוֹכֶל טִבְעִי וְטָעִים.

הִנֵּה כַּמָּה רַעֲיוֹנוֹת לְמַתְכּוֹנִים קַלִּים וּטְעִימִים שֶׁכְּדַאי לָכֶם לְאַמֵּץ:

בּוּלוֹנֵז עֲדָשִׁים כְּתוּמּוֹת:  מְכִינִים בִּמְהִירוּת וּבְקַלּוּת מֵעֲדָשִׁים כְּתוּמּוֹת, שֶׁמִּתְבַּשְּׁלוֹת מַהֵר וּלְלֹא הַשְׁרָיָיה מוּקְדֶּמֶת וּמְכִילוֹת שֶׁפַע שֶׁל חֳומָרִים מְזִינִים. אֶפְשָׁר לְהַגִּישׁ עִם פַּסְטָה אוֹ אוֹרֶז (הֲכִי טוֹב אוֹרֶז מָלֵא, יְלָדִים אוֹהֲבִים אוֹרֶז מָלֵא עָגוֹל) וּלְקַבֵּל אֲרוּחָה מְלֵאָה וּמְזִינָה.
קְצִיצוֹת יֶרֶק: קְצִיצוֹת בֵּיתִיּוֹת הֵן מִצְרָךְ מְבוּקָּשׁ בְּכָל מִשְׁפָּחָה. נָסוּ לְהַחֲלִיף אֶת הַקְּצִיצוֹת הַסְּטַנְדַּרְטִיּוֹת בִּקְצִיצוֹת שֶׁמְּכִינִים מַעֲלִים יְרוּקִּים אוֹ מִיְּרָקוֹת בְּשִׁילּוּבִים שׁוֹנִים. מְאַחֲדִים אוֹתָן בְּעֶזְרַת פֵּירוּרֵי לֶחֶם אוֹ קֶמַח (נָסוּ לְהִשְׁתַּמֵּשׁ בְּקִמְחֵי קִטְנִיּוֹת כְּמוֹ חוּמוּס אוֹ עֲדָשִׁים, שֶׁהֵם עֲשִׁירִים בְּחֶלְבּוֹן וּבְבַרְזֶל) וְאוֹפִים בַּתַּנּוּר לְהִתְיַיצְּבוּת. לֹא תַּאֲמִינוּ בְּאֵיזֶה לָהִיט מְדוּבָּר.
שְׁנִיצֵל תִּירָס בֵּיתִי: נָכוֹן שֶׁאֶפְשָׁר לִקְנוֹת אוֹתוֹ מוּכָן, אֲבָל כְּשֶׁתָּכִינוּ אֶת הַשְּׁנִיצֵל הַזֶּה בְּעַצְמְכֶם תֵּדְעוּ בְּדִיּוּק מָה יֵשׁ בּוֹ, תּוּכְלוּ לִשְׁלוֹט בְּכַמּוּת הַשֶּׁמֶן, לְהַכְנִיס אֵלָיו חֶלְבּוֹן (בְּצוּרַת טוֹפוּ טָחוּן, זֶה טָעִים מְאֹד!), לְהַגְנִיב פְּנִימָה עוֹד יְרָקוֹת (מִי אָמַר כְּרוּבִית) וְכַמּוּבָן לֶאֱפוֹת וְלֹא לְטַגֵּן.
גְּלִידוֹנִיּוֹת בָּנָנָה: בָּנָנוֹת בְּשֵׁלוֹת הֵן בָּסִיס לִגְלִידָהּ מְעוּלָּה שֶׁהַיְּלָדִים יִטְרְפוּ בְּלִי לְהָבִין שֶׁהֵם בְּעֶצֶם אוֹכְלִים פֵּירוֹת. אֶפְשָׁר לְקַלֵּף, לִפְרוֹס וּלְהַקְפִּיא וְאָז לִטְחוֹן לִקְרֵם גְּלִידָתִי עִם אֵיזֶה תּוֹסָפוֹת שֶׁאוֹהֲבִים וְאֶפְשָׁר לְיַיצֵּר מֵהַבָּנָנוֹת גְּלִידוֹנִיּוֹת עִם חֶמְאַת בּוֹטְנִים, לְצַפּוֹת בְּשׁוֹקוֹלָד מוּמָס וְלִזְלוֹל בְּיַחַד.
