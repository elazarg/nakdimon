מַשְׁבֵּר הַקּוֹרוֹנָה עֲדַיִין בְּעִיצּוּמוֹ בָּעוֹלָם כּוּלּוֹ וְגַם בְּיִשְׂרָאֵל, וְהִשְׁפִּיעַ עַל שִׁכְבוֹת הַחֶבְרָה וּמִגְּזָרֶיהָ.
אֶחָד הַצִּיבּוּרִים שֶׁנִּמְצָא בְּמוֹקֵד הַדִּיּוּנִים כְּבָר מֵרֵאשִׁית הַמַּשְׁבֵּר הוּא הַצִּיבּוּר הַחֲרֵדִי, שֶׁסָּבַל מִכַּמָּה מִמּוֹקְדֵי הַהִתְפָּרְצוּת הַקָּשִׁים וְהִתְמוֹדֵד עִם אֶתְגָּרִים לֹא פְּשׁוּטִים מִבְּחִינַת סֶגֶר וּבִידּוּד.
יְהוּדָה שְׁלֵזִינְגֶּר, הַכְּתָב הַפּוֹלִיטִי וְהַכְּתָב לְעִנְייְנֵי חֲרֵדִים שֶׁל 'יִשְׂרָאֵל הַיּוֹם' וּמַדְרִיךְ סִיּוּרִים בִּבְנֵי-בְּרַק, מְסַפֵּר בְּשִׂיחָה עִם 'מִידָּהּ' כִּי "בָּעוֹלָם הַחֲרֵדִי בְּהֶחְלֵט הִפְנִימוּ אֶת הַמַּצָּב מְאוּחָר מִדַּי".

לִדְבָרָיו, יֵשׁ שְׁנֵי מַרְכִּיבִים מֶרְכָּזִיִּים שֶׁגָּרְמוּ לְהִתְעוֹרְרוּת הַמְּאוּחֶרֶת.
"רֵאשִׁית כֹּל, יֶשְׁנוֹ חוֹסֶר הַחִיבּוּר לְעוֹלָם הַטֶּכְנוֹלוֹגְיָה: אָמְנָם בַּמִּגְזָר הַחֲרֵדִי בְּהֶחְלֵט צוֹרְכִים אֶת כְּלִי תִּקְשׁוֹרֶת הַמִּגְזָרִית, אֲבָל אֵין לָהֶם טֵלֵוִויזְיָה וְעִדְכּוּנִים תְּמִידִיִּים בַּטֵּלֵפוֹן הַנַּיָּיד, וְהֵם לֹא קִיבְּלוּ אֶת הַמֵּידָע שֶׁקִּיבֵּל הַצִּיבּוּר הַכְּלָלִי.
בְּנוֹסָף, הָייְתָה תַּקָּלָה גְּדוֹלָה מְאוֹד בִּתְחִילַּת הַדֶּרֶךְ שֶׁכִּמְעַט אַף אֶחָד לֹא מְדַבֵּר עָלֶיהָ, וְזוֹ הַהוֹרָאָה שֶׁל הָרַב קַנְיֶיבְסְקִי לְהַמְשִׁיךְ וְלִלְמוֹד לַמְרוֹת תַּקָּנוֹת הַבְּרִיאוּת שֶׁהוֹרוּ עַל סְגִירַת כָּל מוֹסְדוֹת הַלִּימּוּד.
אַחַר כָּךְ יָצָא אֵיזֶה סִרְטוֹן שֶׁמַּסְבִּיר שֶׁהָרַב כַּנִּרְאֶה לֹא הֵבִין אוֹ לֹא יָדַע עַל מָה הוּא פּוֹסֵק, אוֹ שֶׁהוּא בְּעֶצֶם זִיהָה אֶת הַסַּכָּנָה וְאָמַר שֶׁאֶפְשָׁר לְהַמְשִׁיךְ לְלַמֵּד אֲבָל לְפִי הַכְּלָלִים וְלִפְנֵי שֶׁהֶחֱמִירוּ אֶת הַנְּהָלִים.
בְּכָל מִקְרֶה עָלוּ כָּאן שְׁאֵלוֹת קָשׁוֹת מְאוֹד מִבְּחִינַת אֱמוּנָה.
אֵצֶל הַחֲרֵדִי הַמְּמוּצָּע, קָשֶׁה מְאוֹד לִחְיוֹת עִם הַטַּעֲנָה שֶׁהָרַב לֹא יָדַע כִּי מְדוּבָּר בְּקוֹדֶשׁ הַקּוֹדָשִׁים מִבְּחִינָתוֹ.
מִבְּחִינָה הַזּוֹ אִם הָיִינוּ מְצַפִּים לְחֶשְׁבּוֹן נֶפֶשׁ עָמוֹק מֵהָעוֹלָם הַחֲרֵדִי, לֹא נִרְאֶה שֶׁהוּא אִיבֵּד אֶת הָאֱמוּנָה בַּחֲכָמִים לַמְרוֹת הַתַּקָּלָה הָעֲצוּמָה שֶׁהָייְתָה פֹּה".

מֵהֶם הָאֶתְגָּרִים הָעִיקָּרִיִּים שֶׁעוֹמְדִים בִּפְנֵי הַצִּיבּוּר הַחֲרֵדִי בְּמַהֲלַךְ הַמַּשְׁבֵּר?

"מָה שֶׁמַּטְרִיד אֶת הָאָדָם הַחֲרֵדִי זֶה כַּמּוּבָן הַחוֹלִי וְהַסַּכָּנָה שֶׁל הַקּוֹרוֹנָה, אֲבָל מֵעֵבֶר לְזֶה גַּם הַתַּדְמִית הַנּוֹרָאִית שֶׁחֲלָקִים בַּצִּיבּוּר הַכְּלָלִי הִדְבִּיקוּ לְמִגְזָר שָׁלֵם.
אִם בְּגִבְעָתַיִים לְמָשָׁל הָייְתָה מִתְרַחֶשֶׁת חָלִילָה הִתְפָּרְצוּת שֶׁל הַמַּגֵּפָה, אַף אֶחָד לֹא הָיָה כּוֹתֵב טוֹקְבֵּקִים עַל זֶה שֶׁתּוֹשָׁבֵי הָעִיר מְזוֹהָמִים וְצָרִיךְ לִסְגּוֹר אוֹתָם בַּגֵּטָאוֹת".

מֶה גָּרַם לְהִיוָּוצְרוּת שֶׁל תַּדְמִית כְּזֹאת?

"אֵלֶּה תַּדְמִיּוֹת שֶׁמְּשׁוּיָּכוֹת לַחֲרֵדִים תָּמִיד, וּבַשָּׁנִים הָאַחֲרוֹנוֹת הָפְכוּ גַּם לְחֵלֶק מֵהַהֲסָתָה שֶׁפּוֹלִיטִיקָאִים מְנַהֲלִים נֶגְדָּם.
תִּרְאֶה לְמָשָׁל אֶת אֲבִיגְדוֹר לִיבֶּרְמָן וּלְמָה הוּא הָפַךְ בְּמַעֲרָכוֹת הַבְּחִירוֹת הָאַחֲרוֹנוֹת בִּזְכוּת שִׂנְאַת הַחֲרֵדִים.
יֵשׁ קְהַל יַעַד מְסוּיָּם שֶׁסּוּגִיּוֹת כְּמוֹ שִׁוְויוֹן בַּנֵּטֶל וְגִיּוּס חֲרֵדִים מְנִיעוֹת אוֹתוֹ, וְהַקּוֹרוֹנָה הִתְיַישְּׁבָה בְּדִיּוּק עַל הַנְּקוּדָּה הַזּוֹ שֶׁאוּלַי הָייְתָה קְצָת חֲבוּיָה".

אַתָּה חוֹשֵׁב שֶׁהַבִּיקּוֹרֶת עַל הַמִּגְזָר הַחֲרֵדִי הָייְתָה מוּגְזֶמֶת?

"אֵין סָפֵק שֶׁלְּפָחוֹת בִּתְחִילַּת הַמַּשְׁבֵּר הַצִּיבּוּר הַחֲרֵדִי חָטַף הַרְבֵּה מְאוֹד מֵעֵבֶר לַמְּקוּבָּל.
זֶה קְצָת כְּמוֹ שֶׁשַּׂר הַבְּרִיאוּת לִיצְמָן חָטַף רַק בִּגְלַל שֶׁהוּא חֲרֵדִי, כִּי לָאָדָם הַמְּמוּצָּע עֲדַיִין קָשֶׁה לִרְאוֹת שַׂר עִם זָקָן וּסְפּוֹדִיק וּלְקַבֵּל מִמֶּנּוּ אֶת הַהוֹרָאוֹת.
יֵשׁ לִי תְּחוּשָׁה שֶׁגַּם מְקַבְּלֵי הַהַחְלָטוֹת הֵבִינוּ אֶת זֶה וְלָכֵן רָאִינוּ אֶת לִיצְמָן נֶעֱלַם מִמְּסִיבּוֹת הָעִיתּוֹנָאִים עַל הַקּוֹרוֹנָה, לַמְרוֹת שֶׁהוּא הָיָה בֵּין הָרִאשׁוֹנִים שֶׁדָּרַשׁ לְהַפְסִיק אֶת הַטִּיסוֹת.
שֶׁכּוּלָּם אָמְרוּ אָז שֶׁהוּא זוֹרֵעַ הִיסְטֶרְיָה וּבְסוֹךְ מִסְתַּבֵּר שֶׁזֶּה מָה שֶׁהִצִּיל אוֹתָנוּ.
זֶה עוֹד חֵלֶק מֵהַכַּעַס עַל הָעוֹלָם הַחֲרֵדִי: אִם יִהְיוּ אַלְפֵי הֲרוּגִים מֵהַקּוֹרוֹנֶה אָז יַאֲשִׁימוּ אֶת לִיצְמָן, וְאִם בְּסוֹפוֹ שֶׁל דְּבַר הַתּוֹצָאוֹת יִהְיוּ טוֹבוֹת אָז הַקְּרֵדִיט יֵלֵךְ לְמִישֶׁהוּ אַחֵר".

אַךְ לְצַד הַבְּעָיוֹת וְהַמַּחֲלוֹקוֹת, שְׁלֵזִינְגֶּר מְזַהֶה גַּם נְקוּדּוֹת לְאוֹפְּטִימִיּוּת בַּנְּסִיבּוֹת שֶׁנּוֹצְרוּ.
"קוֹדֶם אִי אֶפְשָׁר לְהִתְעַלֵּם מֵהַנּוֹכְחוּת הַנִּרְחֶבֶת שֶׁל כּוֹחוֹת צָבָא בָּרְחוֹבוֹת, מֵהַחַיָּילִים שֶׁמַּגִּיעִים לְכָל מִשְׁפָּחָה חֲרֵדִית וְדוֹפְקִים לָהּ עַל הַדֶּלֶת וּמִבִּיטּוּיֵי הָאַהֲבָה שֶׁל הַצִּיבּוּר לְסִיּוּעַ שֶׁהוּא מְקַבֵּל מִצַּהַ"ל", הוּא אוֹמֵר וּמַסְבִּיר: "חָשׁוּב לְהָבִין שֶׁהַצִּיבּוּר הַחֲרֵדִי לֹא שׂוֹנֵא חַיָּילִים.
זֶה אוּלַי לֹא יִגְרוֹם לְכָל הַצִּיבּוּר לְהִתְגַּיֵּיס מָחָר בַּבּוֹקֶר אֲבָל אֲנִי מַאֲמִין שֶׁזֶּה מַשֶּׁהוּ שֶׁיְּחַלְחֵל לַחֶבְרָה וְיִתְרוֹם בְּסַךְ הַכֹּל בְּאוֹפֶן חִיּוּבִי בְּתוֹךְ כָּל הַמַּשְׁבֵּר הַזֶּה.
מִבְּחִינַת הָאֱמוּנָה הַמּוּחְלֶטֶת בְּרַבָּנִים אֲנִי לֹא רוֹאֶה שִׁינּוּי.
זֶה חֵלֶק מַשְׁמָעוּתִי בַּחֶבְרָה הַחֲרֵדִית שֶׁמַּתְחִיל כְּבָר מִגִּיל שָׁנָה וְיִהְיֶה קָשֶׁה מְאוֹד לְשַׁנּוֹת.
בְּסוֹפוֹ שֶׁל דָּבָר כּוּלָּנוּ, גַּם חֲרֵדִים וְגַם חִילּוֹנִיִּים, נִתְרַגֵּל לִחְיוֹת אֶת הַחַיִּים שֶׁאַחֲרֵי הַקּוֹרוֹנָה".

הַאִם יֵשׁ קוֹשִׁי מְיוּחָד לִשְׁמוֹר עַל סֶגֶר בַּחֶבְרָה הַחֲרֵדִית?

"הַקּוֹשִׁי הָעִיקָּרִי הוּא אוֹפֶן הַחַיִּים שֶׁל הָאָדָם הַחֲרֵדִי, שֶׁבְּמֶרְכָּזָם עוֹמֶדֶת הַקְּהִילָּתִיּוּת.
חֲרֵדִי מִתְפַּלֵּל שָׁלוֹשׁ פְּעָמִים בַּיּוֹם בַּמִּנְיָין, בְּמַהֲלַךְ הַיּוֹם הוּא נִמְצָא בַּכּוֹלֵל עִם חַבְרוּתוֹת וְאַחֲרֵי הַצָּהֳרַיִים לוֹמֵד אֶת הַדַּף הַיּוֹמִי.
בָּעֶרֶב יוֹצְאִים בְּדֶרֶךְ כְּלָל לְאֵירוּעִים מִשְׁפַּחְתִּיִּים וּבְשִׁישִּׁי-שַׁבָּת נֶעֱרָכִים טִישִׁים בַּחֲצֵרוֹת הַגְּדוֹלוֹת.
הַקְּהִילָּה הִיא הַתַּשְׁתִּית שֶׁל הָעוֹלָם הַזֶּה, וּמִכָּאן גַּם נִגְזֶרֶת צוּרַת הַחַיִּים הַצְּפוּפָה בָּעִיר כְּמוֹ בְּנֵי-בְּרַק.
בַּסִּיּוּרִים שֶׁאֲנִי עוֹרֵךְ בָּעִיר אֲנַחְנוּ עוֹבְרִים בַּשְּׁכוּנוֹת הַיְּשָׁנוֹת שֶׁאֲנִי קוֹרֵא לָהֶן 'פָּבֶלוֹת', שָׁם אִי אֶפְשָׁר לִמְצוֹא מֶטֶר אֶחָד שֶׁלֹּא בְּנוּיָה עָלָיו דִּירָה.
זֹאת חָבִית שֶׁל חוֹמֶר נֶפֶץ שֶׁבָּרֶגַע שֶׁמַּשֶּׁהוּ מִתְפָּרֵץ שָׁם אָז כִּמְעַט בִּלְתִּי נִמְנָע שֶׁתִּהְיֶה הַדְבָּקָה נִרְחֶבֶת כְּמוֹ שֶׁקָּרָה".

אֶחָד הַוִּויכּוּחִים הַגְּדוֹלִים שֶׁל תְּקוּפַת הַקּוֹרוֹנָה מִתְנַהֵל סְבִיב הַסִּיקּוּר הֲתִקְשׁוֹרְתִּי שֶׁנִּיתָּן לְהִתְפָּרְצוּת בַּמִּגְזָר הַחֲרֵדִי וּלְטִיפּוּל בָּהּ, וְהַתְּגוּבוֹת מִכָּל הַצְּדָדִים הָיוּ סוֹעֲרוֹת בִּמְיוּחָד.
לְדִבְרֵי שְׁלֵזִינְגֶּר, הַבְּעָיָה הָעִיקָּרִית בִּתְחוּם הַתִּקְשׁוֹרֶת הָייְתָה שֶׁפָּשׁוּט לֹא נִיתָּן יִיצּוּג מַסְפִּיק לַחֲרֵדִים עַל הַמָּסַכִּים.
"לֹא רָאִיתִי מַסְפִּיק חֲרֵדִים בָּאוּלְפָּנִים וְזֹאת אַחַת הַתַּקָּלוֹת שֶׁל כָּל הָעֲרוּצִים.
בְּכָל יוֹם מְשַׁדְּרִים שִׁידּוּרִים חַיִּים רְצוּפִים מֵהַצָּהֳרַיִם וְעַד חֲצוֹת, וּכְמוֹ שֶׁיּוֹשֵׁב רוֹפֵא בְּכָל פַּאנֶל אָז הָיָה צָרִיךְ לָשֶׁבֶת גַּם חֲרֵדִי, וּבִמְיוּחָד בְּדִיּוּנִים שֶׁעוֹסְקִים בַּצִּיבּוּר הַזֶּה.
גַּם נְצִיגֵי הַצִּיבּוּר כִּמְעַט וְלֹא הִתְרַאְייְנוּ, וְהַקּוֹל הַזֶּה כִּמְעַט וְלֹא נִשְׁמַע בַּשָּׁבוּעוֹת הָרִאשׁוֹנִים לְמַעֵט חֲרֵדִים שֶׁהִתְרַאְייְנוּ בָּרְחוֹב בְּכַתָּבוֹת שׁוֹנוֹת".

מָה דַּעְתְּךָ עַל סַעֲרַת רִינָּה מַצְלִיחַ וְהַהִתְבַּטְּאוּיוֹת שֶׁהִשְׁמִיעָה נֶגֶד הַצִּיבּוּר הַחֲרֵדִי?

"אֲנִי דַּוְוקָא מְעוֹדָד מִמָּה שֶׁקָּרָה עִם רִינָּה מַצְלִיחַ, בִּגְלַל הַבִּיקּוֹרֶת הָרְחָבָה שֶׁהִיא סָפְגָה אַחֲרֵי הַמִּקְרֶה וְלֹא רַק מֵהַחֲרֵדִים אֶלָּא אֲפִילּוּ בְּתוֹךְ הָאוּלְפָּן שֶׁלָּהּ.
מִמָּה שֶׁרָאִיתִי בַּטְּוִויטֶר הָיוּ כַּמָּה שֶׁהֵגֵנּוּ עָלֶיהָ וְעַל הַזְּכוּת שֶׁלָּהּ לְבַטֵּא אֶת דַּעְתָּהּ, אֲבָל הָייְתָה כִּמְעַט אֶפֶס תְּמִיכָה בְּתוֹכֶן הַדְּבָרִים שֶׁלָּהּ גַּם בְּקֶרֶב אַנְשֵׁי תִּקְשׁוֹרֶת.
הָיָה פֹּה מֶסֶר בָּרוּר שֶׁאִי אֶפְשָׁר יוֹתֵר לְהָסִית נֶגֶד מָה שֶׁמַּצְלִיחַ כִּינְּתָה "רוֹב הַחֲרֵדִים" וְעוֹד לְגַלּוֹת בּוּרוּת וּלְהִשְׁתַּמֵּשׁ בְּדוּגְמָאוֹת מִלִּפְנֵי 25 שָׁנָה.
זֶה הָיָה כָּל כָּךְ נָמוּךְ וְלָכֵן גַּם גָּרַר תְּגוּבוֹת קָשׁוֹת".

יָכוֹל לִהְיוֹת שֶׁפָּשׁוּט הַמִּגְזָר הַחֲרֵדִי לֹא רָגִיל לְכָזֶה סִיקּוּר תִּקְשׁוֹרְתִּי נִרְחָב?

"בְּסוֹפוֹ שֶׁל דָּבָר יֵשׁ פֹּה מַשְׁבֵּר עֲנָק, וַהֲכִי קַל תָּמִיד לָלֶכֶת לַסְּטֵרֵיאוֹטִיפִּים.
מִצַּד שְׁנִי רָאִיתִי גַּם הַרְבֵּה דְּבָרִים חִיּוּבִיִּים כְּמוֹ הַסִּיקּוּר שֶׁל גַּיְא לֵרֶר בַּתָּכְנִית 'הַצִּינּוֹר' וּבַעֲבוֹדָה הַמְּצוּיֶּנֶת שֶׁעוֹשִׂים כְּתָבִים כְּמוֹ יָאִיר שֶׁרְקִי.
כְּשֶׁיֵּשׁ אֲנָשִׁים שֶׁבָּאִים וּבֶאֱמֶת מַסְבִּירִים וּמַרְאִים אֶת הָאֱמֶת, אָז הַתְּמוּנָה נִרְאֵית אַחֶרֶת.
בַּשָּׁבוּעוֹת הָאַחֲרוֹנִים אֶפְשָׁר לִרְאוֹת כְּבָר שִׁינּוּי גַּם בַּדִּיוּוּחִים מֵהַשֶּׁטַח שֶׁמַּרְאִים אֶת הָרְחוֹבוֹת הָרֵיקִים בִּבְנֵי-בְּרַק אוֹ בְּכַתָּבוֹת כְּמוֹ שֶׁל אַיָּילָה חָסוֹן עַל תַּעֲשִׂיַּית הַחֶסֶד.
אֲנִי חוֹשֵׁב שֶׁרוֹב הַתִּקְשׁוֹרֶת הִתְבַּגְּרָה בַּיַּחַס שֶׁלָּהּ לַמִּגְזָר, וְדַוְוקָא אֲנָשִׁים כְּמוֹ רִינָּה מַצְלִיחַ שֶׁהַדֵּעוֹת שֶׁלָּהּ הָיוּ פַּעַם בְּקוֹנְצֶנְזוּס הָפְכוּ הַיּוֹם לְמִיעוּט".

אֵיךְ הַתִּקְשׁוֹרֶת הַחֲרֵדִית מְסַקֶּרֶת אֶת מַשְׁבֵּר הַקּוֹרוֹנָה?

"בְּאוֹפֶן כְּלָלִי הַתִּקְשׁוֹרֶת הַחֲרֵדִית עוֹשָׂה עֲבוֹדָה נֶהְדֶּרֶת, בְּעִיקָּר מִבְּחִינַת הַסְבָּרָה.
עִיתּוֹן 'הַמְּבַשֵּׂר' לְמָשָׁל מְפַרְסֵם כִּמְעַט כָּל יוֹם אֶת הַתְּמוּנוֹת וְהַשֵּׁמוֹת שֶׁל הַנִּפְטָרִים מֵהַקּוֹרוֹנָה מֵחוּ"ל כְּדֵי לְזַעְזֵעַ אֶת הַצִּיבּוּר הַחֲרֵדִי, וּבְעִיתּוֹנִים אֲחֵרִים מִתְפַּרְסְמוֹת גִּרְפִיקוֹת וְטַבְלָאוֹת שֶׁמַּסְבִּירוֹת אֶת הַמַּצָּב בְּאוֹפֶן מִקְצוֹעִי.
גַּם אַנְשֵׁי הַתִּקְשׁוֹרֶת הַחֲרֵדִים בִּטְוִויטֶר אוֹ כָּאֵלֶּה שֶׁהִתְרַאְייְנוּ לַכַּתָּבוֹת כָּל הַזְּמַן הִזְהִירוּ וְהִפְצִירוּ בַּצִּיבּוּר לְהִישָּׁמַע לְהוֹרָאוֹת כְּדֵי לְהַצִּיל חַיִּים.
הַמַּטָּרָה הָייְתָה לִגְרוֹם לְזַעֲזוּעַ וְלַהֲבָנָה שֶׁל גּוֹדֶל הָאִיּוּם, וּמִי שֶׁעָבַר עַל כְּלֵי הַתִּקְשׁוֹרֶת הַחֲרֵדִים הָיָה יָכוֹל לְהַרְגִּישׁ אֶת זֶה הֵיטֵב".
