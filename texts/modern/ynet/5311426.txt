רֵיאָל מַדְרִיד כְּבָר מָצְאָה אֵיךְ לְהוֹצִיא אֶת חֵלֶק מֵהַכֶּסֶף, אוֹתוֹ הִרְוִויחָה מִמְּכִירָתוֹ שֶׁל כְּרִיסְטְיָאנוֹ  רוֹנַאלְדּוֹ לְיוּבֶנְטוּס.
עַל פִּי דִּיוּוּחַ הָעֶרֶב (שְׁלִישִׁי) בְּצָרְפַת, הַבְּלָאנְקוֹס וְצֶ'לְסִי הִגִּיעוּ לְסִיכּוּם  עַל הַעֲבָרָתוֹ שֶׁל תִּיבוֹ קוֹרְטוֹאָה לְסַנְטִיָּאגוֹ בָּרְנַבֵּאוּ.
הַבְּלוּז יְקַבְּלוּ מֵרֵיאָל 35 מִילְיוֹן אֵירוֹ בִּלְבַד  וְהַשּׁוֹעֵר הַבֶּלְגִּי בֶּן הַ-26 יַחְתּוֹם עַל חוֹזֶה לִ-4 שָׁנִים.
אַחֲרֵי שֶׁהוּשְׁאַל לְאַתְלֵטִיקוֹ מַדְרִיד מִצֶּ'לְסִי  בְּ-2011, בִּילָּה קוֹרְטוּאָה שָׁלוֹשׁ שָׁנִים בִּסְפָרַד שָׁם זָכָה בְּאַלִּיפוּת, גְּבִיעַ הַמֶּלֶךְ וּבַלִּיגָה הָאֵירוֹפִּית.
 בְּ-2014 חָזַר לַבְּלוּז וְזָכָה אִיתָּהּ בִּשְׁתֵּי אַלִּיפוּיוֹת וּבַגָּבִיעַ הָאַנְגְּלִי.
רֵיאָל הוֹלֶכֶת בְּכָל הַכּוֹחַ גַּם  עַל אֶדֶן הָזָאר, רַק שֶׁבַּמִּקְרֶה הַזֶּה צֶ'לְסִי פָּשׁוּט לֹא מוּכָנָה לְוַותֵּר עָלָיו.
עַל פִּי הַדִּיוּוּחִים בַּיָּמִים  הָאַחֲרוֹנִים, הַבְּלָאנְקוֹס הִצִּיעוּ מִסְפַּר הַצָּעוֹת, כּוֹלֵל אַחַת בְּסַךְ 225 מִילְיוֹן אֵירוֹ עַל הַקֶּשֶׁר הַבֶּלְגִּי,  אַךְ הַבְּלוּז מַמְשִׁיכִים לְהַקְפִּיץ אֶת הַמְּחִיר.
הָזָאר עַצְמוֹ כְּבָר הִתְבַּטֵּא רֶגַע אַחֲרֵי הַמִּשְׂחָק עַל הַמָּקוֹם  הַשְּׁלִישִׁי, שֶׁהוּא שׁוֹקֵל לַעֲזוֹב אֶת צֶ'לְסִי וְ"כּוּלָּם יוֹדְעִים אֵיפֹה אֲנִי רוֹצָה לְשַׂחֵק" - רֶמֶז עָבֶה  מְאוֹד לָעִנְיָין שֶׁל רֵיאָל בּוֹ, אֲבָל עִסְקַת הָעֲנָק לֹא מִתְקַדֶּמֶת כָּרֶגַע לְשׁוּם מָקוֹם.
