אַט אַט הוּא מִתְקָרֵב אֶל הַמִּגְדָּל.
הַדְּמוּת נֶעֱצֶרֶת בְּמֶרְחַק 200  מֶטֶר וְנִשְׁכֶּבֶת.
שָׁלוֹשׁ שְׁנִיּוֹת אַחֲרֵי זֶה, קָלִיעַ מוּשְׁתָּק מְפַלֵּחַ אֶת שְׁמֵי הַלַּיְלָה וְאֶת הַשּׁוֹמֵר בְּמִגְדַּל.
 לְחִישָׁה שְׁקֵטָה לַמִּיקְרוֹפוֹן - "הַמַּטָּרָה נוּטְרְלָה, הַשֶּׁטַח נָקִי!".

הַמָּטוֹס נֶעֱצַר מֵעַל הַבִּנְיָין  עָלָיו הֵגֵן מִגְדַּל הַשְּׁמִירָה וְנוֹחֵת בְּרַכּוּת כְּמוֹ מַסּוֹק.
מִתּוֹךְ הַמָּטוֹס יוֹצְאִים שְׁלוֹשָׁה אֲנָשִׁים.
מְפַקְּדָם  מְסַמֵּן לָהֶם וְהֵם נִכְנָסִים לְבִנְיָן מִשְּׁלוֹשָׁה כִּיוּוּנִים.
כַּעֲבוֹר שָׁלוֹשׁ דַּקּוֹת, שְׁמוֹנָה שׁוֹמְרִים מֵתִים  וּמִטְעַן נֶפֶץ מוּטְמָן אֶחָד, יוֹצְאִים הַשְּׁלוֹשָׁה מִן הַבִּנְיָין, עוֹלִים עַל הַמָּטוֹס וְנֶעֱלָמִים, לֹא לִפְנֵי  שֶׁהֵם אוֹסְפִים אֶת הַצַּלָּף הַשָּׁקֵט בִּשְׂדֵה הַסְּלָעִים.
עֶשֶׂר דַּקּוֹת נוֹסָפוֹת, וְהַדְּמָמָה נִקְטַעַת בְּהִתְפּוֹצְצוּת  אַדִּירָה הַמְּאִירָה בְּאוֹר אָדוֹם אֶת כָּל הָאֵזוֹר.
עוֹד לַיְלָה שֶׁל עֲבוֹדָה שִׁגְרָתִית בָּא לְקִיצּוֹ.
פֵייד אָאוּט  לְמוּזִיקָה.
בְּמַהֲלַךְ שְׁנוֹת הָאַלְפַּיִים הַמּוּקְדָּמוֹת (בָּאֵזוֹר 2010) מַגִּיעוֹת מַעַצְמוֹת הָעָל שֶׁל הָעוֹלָם  לַמַּסְקָנָה הַמִּתְבַּקֶּשֶׁת.
נִמְאַס לָהֶן לִבְחוֹשׁ בְּצָרוֹת שֶׁל אֲחֵרִים וּלְשַׂחֵק אוֹתָהּ מוֹשִׁיעֵי הָעוֹלָם.
אַרְצוֹת  הַבְּרִית וְרוּסְיָה מַחֲלִיטוֹת שֶׁאֶת כָּל הַכּוֹחַ שֶׁלָּהֶן יַשְׁקִיעוּ בַּהֲגָנָה עַל עַצְמָן וְעַל עַצְמָן בִּלְבַד.
דַּי  לִתְמִיכָה בְּכָל מִינֵי מְדִינוֹת קְטַנּוֹת וְחַלָּשׁוֹת, בִּמְיוּחָד כָּאֵלֶּה מֵהַמִּזְרָח הַתִּיכוֹן.
אִם יֵשׁ לָהֶן בְּעָיָה  שֶׁיִּפְתְּרוּ אוֹתָהּ בְּעַצְמָן (הֶעָרַת הַכּוֹתֵב - הַלְּוַואי עָלֵינוּ).
(הֶעָרַת הָעוֹרֵךְ: כֵּן, כְּמוֹ שֶׁאֲנַחְנוּ  פּוֹתְרִים אוֹתָן בְּעַצְמֵנוּ כְּבָר שָׁנִים....) וְאָכֵן, מִמַּחְשָׁבָה לְמַעֲשֶׂה, נוֹכְחוּתָם וְהַשְׁפָּעָתָם שֶׁל  הָאֲמֵרִיקָאִים וְהָרוּסִים נֶעֱלָמוֹת כִּמְעַט לַחֲלוּטִין מִכָּל רַחֲבֵי הָעוֹלָם הַחוֹפְשִׁי וּבְעִיקָּר מֵרַחֲבֵי הָעוֹלָם  הַלֹּא חוֹפְשִׁי.
כְּמוֹ דַּרְדַּקִּים שֶׁהַמְּלַמֵּד שֶׁלָּהֶם הִתְפַּטֵּר מִתַּפְקִידוֹ, שְׁאַר מְדִינוֹת הָעוֹלָם מַתְחִילוֹת בְּחֶדְווֹה  לְהִילָּחֵם בֵּינָן לְבֵין עַצְמָם.
אֲבָל מִכֵּיוֹוֹן שֶׁהָעֲלוּיוֹת לְמִלְחָמָה בָּעֵת הַחֲדָשָׁה מַגִּיעוֹת לְמִסְפָּרִים  אַסְטְרוֹנוֹמִיִּים, הִגִּיעוּ הַצְּדָדִים הַלּוֹחֲמִים לְמַסְקָנָה שֶׁצָּרִיךְ לִנְקוֹט בַּחֲשִׁיבָה שׁוֹנָה לַחֲלוּטִין.
בִּמְקוֹם  לְיַיצֵּר מֵאוֹת טַנְקִים וְלַעֲרוֹךְ Tank Rush עַצְבָּנִי וְיָקָר עַד מְאֹד, הַכֶּסֶף מוּשְׁקָע בִּיחִידוֹת קְטַנּוֹת  וּמְאוּמָּנוֹת הֵיטֵב שֶׁפּוֹעֲלוֹת בְּעוֹמֶק הָאוֹיֵב וְגוֹרְמוֹת לַנֶּזֶק הַמַּקְסִימָלִי בְּמִינִימוּם הַשְׁקָעָה.
זִירוֹת  לְחִימָה חֲדָשׁוֹת צָצוֹת עַל מַפַּת הָעוֹלָם כְּפִטְרִיּוֹת אַחֲרֵי הַגֶּשֶׁם.
שְׂדֵה הַקְּרָב הַגָּדוֹל בְּיוֹתֵר נוֹצַר בְּחֵלֶק  מֵהָאוֹקְיָאנוֹס הַשֶּׁקֶט בֵּין אָלַסְקָה וּמִזְרָח סִיבִּיר.
מְדוּבָּר בְּשֶׁטַח נִרְחָב מָלֵא אִיִּים קְטַנִּים וְגוֹבֵל  בְּחוֹפֵי אֲרָצוֹת רַבּוֹת.
אָלַסְקָה וְסִיבִּיר הִינָּן, גַּם הַיּוֹם, חֶבְלֵי-אֶרֶץ מְלַאי מַשְׁאַבֵּי טֶבַע שֶׁעוֹד לֹא  נוּצְּלוּ.
יִהְיֶה זֶה רַק טִבְעִי שֶׁמְּדִינוֹת רַבּוֹת הַקְּרוֹבוֹת אֲלֵיהֶם יִשְׂמְחוּ לִתְקוֹעַ יָתֵד בְּקַרְקַע בְּתוּלָה  זוֹ.
מִלְחָמָה עַל הַשֶּׁטַח הָיְיתָה רַק עִנְיָין שֶׁל זְמַן, וְאָכֵן, עִם הֵיעָלְמוּתָן שֶׁל מַעַצְמוֹת-הָעָל, הַבִּלְתִּי  נִמְנָע אָכֵן לֹא נִמְנַע וְסוֹפוֹת הַמִּלְחָמָה מַתְחִילוֹת לְנַשֵּׁב בְּעַרְבוֹת סִיבִּיר...
עַל מִלְחָמָה הִיפּוֹתֵטִית זוֹ  מְבוּסָּס הַמִּשְׂחָק הֶחָדָשׁ שֶׁל Wings Simulation, הַנִּקְרָא Soldner: Secret Wars.
בִּיטּוּי נָכוֹן שֶׁל  הַשֵּׁם יִשְׁמַע, בְּתוֹסֶפֶת מִבְטָא גֶּרְמָנִי כָּבֵד, בְּעֵרֶךְ כָּךְ: זוֹלַדְנֵר: זִיקְרֶט ווֹרְז.
הַסִּיבָּה לַמִּבְטָא הַזָּר  הִיא שֶׁהַמְּפַתְּחִים הַמּוּזְכָּרִים לְעֵיל הִינָּם גֶּרְמָנִים בִּרְמַ"ח אֵיבְרֵיהֶם וּשְׁסָ"ה גִּידֵיהֶם.
הַמִּשְׂחָק הַקּוֹדֵם  שֶׁלָּהֶם, סִימוּלָטוֹר הַטַּנְק Panzer Elite, (לוֹמַר בְּאוֹתוֹ מִבְטָא יֵקֶה) זָכָה לְבִיקּוֹרוֹת אוֹהֲדוֹת לְמַדַּי  מִשַּׂחְקָנִים וְסוֹקְרִים כְּאֶחָד.

אֶת Soldner הֵם מְפַתְּחִים לְ-PC בִּלְבַד.

שַׂחְקָנִים שְׁקוּלִים? כְּבָר צִמְצַמְתֶּם אֶת קְהַל  הַיַּעַד שֶׁלָּכֶם בְּ-90 אֲחוּזִים לְפָחוֹת...

 הַתּוֹצָאָה תִּהְיֶה שֶׁכְּשֶׁבְּיַפָּן, לְדוּגְמָא, תֵּיחָשְׁבוּ לאח"מים, בְּסִין יִדְאֲגוּ לַעֲשׂוֹת מִכֶּם אוֹרֶז בָּרֶגַע שֶׁכַּף  רַגְלְכֶם תָּעֵז לִדְרוֹךְ שָׁם.
כָּל זֶה יוּכַל כַּמּוּבָן לְהִשְׁתַּנּוֹת אִם רַק תִּבְחֲרוּ לָלֶכֶת עַל הַמְּשִׂימוֹת שֶׁיּוּצְּעוּ  לָכֶם ע"י הַסִּינִים.
הַמְּשִׂימוֹת לֹא יִהְיוּ בְּנוּיוֹת מֵרֹאשׁ, דְּהַיְינוּ, מִמִּשְׂחָק לְמִשְׂחָק יִשְׁתַּנּוּ הַמַּטָּרוֹת,  פְּנֵי הַשֶּׁטַח וְכוֹחוֹת הָאוֹיֵב.
מַטְּרוֹת מְשִׂימָה יָנוּעוּ בֵּין נוֹהַל סְרוֹק וְהַשְׁמֵד פָּשׁוּט לִמְשִׂימוֹת  מוּרְכָּבוֹת יוֹתֵר הַכּוֹלְלוֹת חִילּוּץ, הַצָּלָה וְחִיפּוּי.
לְדִבְרֵי טָאוּט (אֵיזֶה שֵׁמוֹת יֵשׁ לַגֶּרְמָנִים הָאֵלֶּה,  ה' יִשְׁמוֹר), שַׂחְקָן מְמוּצָּע יַעֲבוֹר בְּעֵרֶךְ 50 מְשִׂימוֹת לִפְנֵי שֶׁהוּא יַתְחִיל לְקַבֵּל מְשִׂימוֹת דּוֹמוֹת  לְכָאֵלֶּה שֶׁכְּבָר עָבַר.
מְשִׂימוֹת מְיוּחָדוֹת יְשׁוּלְּבוּ בְּתוֹךְ הַקַּמְפֵּיְין בִּשְׁבִיל לְקַדֵּם אֶת הָעֲלִילָה וְאוֹתָן  יִצְטָרֵךְ כָּל שַׂחְקָן לַעֲבוֹר, אֲבָל רוֹב הַמְּשִׂימוֹת יִוּוֹצְרוּ On-The-Fly עַל יְדֵי הַמִּשְׂחָק.
בְּמַהֲלַךְ  הַקַּמְפֵּיְין תְּשַׁלְּטוּ עַל צֶוֶות שֶׁל עַד 32 לוֹחֲמִים.
כָּל הַלּוֹחֲמִים בְּצֶוֶות, חוּץ מִכֶּם, יִישָּׁלְטוּ ע"י  הָאִינְטֵלִיגֶנְצִיָּה הַמְּלָאכוּתִית, וְאַתֶּם תַּחְלְקוּ לָהֶם פְּקוּדּוֹת.
הַמִּמְשָׁק דַּרְכּוֹ תְּנַהֲלוּ אֶת הַצֶּוֶות שֶׁלָּכֶם  עֲדַיִין לֹא לוּטַּשׁ, וּבְהֶחְלֵט יֵשׁ עוֹד הַרְבֵּה עֲבוֹדָה לַצֶּוֶות לְהַשְׁקִיעַ בַּנּוֹשֵׂא.
הַדְּמוּת אוֹתָהּ תְּשַׂחֲקוּ  תּוּכַל לְהִתְחַמֵּשׁ בְּמֵיטַב הַטֶּכְנוֹלוֹגְיָה שֶׁל מִיגּוּן אִישִׁי, נְשָׁקִים וּמִיכְשׁוּר טֶכְנִי.
הַצִּיּוּד שֶׁתִּבְחֲרוּ  יַגְדִּיר אֶת תַּפְקִידְכֶם בַּמִּשְׂחָק.
"הֶחְלַטְנוּ לֹא לְהָכִין עֶרְכַּת צִיּוּד מוּכָנָה מֵרֹאשׁ שֶׁתַּגְדִּיר אֵיזֶה סוּג  לוֹחֵם יִהְיֶה הַשַּׂחְקָן, מִכֵּיוֹוֹן שֶׁזֶּה יַגְבִּיל אוֹתוֹ.
בִּמְקוֹם זֶה, הַשַּׂחְקָן בּוֹחֵר מָה שֶׁהוּא רוֹצֶה מִכָּל  הַמְּלַאי וּלְפִי מָה שֶׁבָּחַר יִתְבָּרֵר תַּפְקִידוֹ בְּכוֹחַ", מַסְבִּיר טָאוּט.
עֲדַיִין, כְּדֵי לְהָקֵל עַל שַׂחְקָנִים,  הֵכִינוּ הַיּוֹצְרִים עֶרְכוֹת צִיּוּד מוּכָנוֹת שֶׁיַּעַזְרוּ לְמִי שֶׁבְּכָל זֹאת רוֹצָה לִהְיוֹת בַּעַל תַּפְקִיד מוּגְדָּר  אֶחָד, כְּמוֹ חוֹבֵשׁ לְדוּגְמָא, לְהִיכָּנֵס לַגּ'וֹבּ הָרָצוּי בְּקַלּוּת.
אַנְשֵׁי הַמֶּחְקָר שֶׁל Wings אָסְפוּ לְתוֹךְ  הַמִּשְׂחָק בִּסְבִיבוֹת 50 כְּלֵי נֶשֶׁק שׁוֹנִים.
כָּל אֶחָד מֵהֶם נִבְנָה בִּמְפוֹרָט בַּמַּחְשֵׁב ע"פ תְּמוּנוֹת  וּסְטָטִיסְטִיקוֹת מְדוּיָּקוֹת מֵהָאִינְטֶרְנֶט.
בֵּין הַנְּשָׁקִים שֶׁנֶּחְשְׂפוּ נִיתָּן לִמְצוֹא אֶת הַגָּלִיל שֶׁלָּנוּ - כּוֹלֵל  הַהִיסְטוֹרְיָה הַמַּקִּיפָה שֶׁלּוֹ - בְּקָטֵגוֹרְיָת רוֹבֵי הַסַּעַר, וְכֵן אֶת נֶשֶׁר הַמִּדְבָּר, גַּם כֵּן שֶׁלָּנוּ,  בָּאֶקְדָּחִים.
אַחֲרֵי שֶׁנִּיפַּחְנוּ אֶת הֶחָזֶה וְאֶת הַגַּאֲווֹה הַלְּאוּמִּית, נִגְלָה שֶׁיֶּשְׁנָם עוֹד כְּלֵי נֶשֶׁק רַבִּים  בְּאַרְסֶנַל הַמַּרְשִׁים שֶׁל Soldner, חֶלְקָם עֲדַיִין בְּפִיתּוּחַ וַאֲמוּרִים לָצֵאת רַק בְּעוֹד 4-5 שָׁנִים.
 בֵּינֵיהֶם הַגִּרְסָה הַחֲדָשָׁה שֶׁל הַ-M16 הַוּוֹתִיק - הָ-M4 Carbine, Heckler & Koch OICW, Benelli  M4 Super90 בְּרוֹבֵי סַעַר, וְ-RG 6 Grenade Launcher לְצַד כְּלִי מַשְׁחִית אֲחֵרִים בְּקָטֵגוֹרִית הַכֵּלִים  הַכְּבֵדִים.
קָטֵגוֹרְיוֹת נוֹסָפוֹת כּוֹלְלוֹת רוֹבֵי צַלָּפִים, תַּתֵּי-מַקְלֵעַ, נִשְׁקֵי לש"ב (לְחִימָה בְּשֶׁטַח בָּנוּי)  וְכָל מִינֵי תּוֹסָפוֹת הַיְטֵק לְמִינֵיהֶן, כְּמוֹ הַ-RCS: מַחְשֵׁב נַיָּיד שִׁינַּתֶּן לְכָל לוֹחֵם, יְקַשֵּׁר בֵּין חַבְרֵי  הַקְּבוּצָה וְיַעֲבִיר לָהֶם מֵידָע מֵהַשֶּׁטַח בִּזְמַן אֱמֶת.
לֹא רַק בִּנְשָׁקִים הִשְׁקִיעוּ אַנְשֵׁי הַמֶּחְקָר שֶׁל  Soldner, הֵם עָבְדוּ קָשָׁה גַּם בְּהוֹסָפַת כְּלִי רֶכֶב לְמִשְׂחָק, לֹא פָּחוֹת מִ-80 מֵהֶם, מִסְפָּר שֶׁמַּבְהִיר אֶת  נְטִיָּיתָם שֶׁל הַמְּפַתְּחִים לְזַ'אֲנֵר הַסִּימוּלַצְיָה.
לְעוּמַּת הַנְּשָׁקִים, שֶׁנֶּאֶסְפוּ מִכָּל הָעוֹלָם, כְּלֵי הָרֶכֶב  יִהְיוּ בְּעִיקָּר מִתּוֹצֶרֶת רוּסְיָה וְסִין וְיִכְלְלוּ, מֵעֵבֶר לְמַשָּׂאִיּוֹת, טַנְקִים, שִׁרְיוֹנִיּוֹת קַלּוֹת וּשְׁאָר  יְרָקוֹת, גַּם מַסּוֹקֵי תּוֹבָלָה כְּמוֹ Mi-24 HIND וּמַסּוֹקֵי קְרָב כְּמוֹ הַ-Ka-50 Hokum.
גַּם מְקוֹמָם שֶׁל  הַמְּטוֹסִים לֹא נִפְקַד וְיִכְלוֹל בֵּין הַשְּׁאָר אֶת הַ-Mig-29, Su-28 וּמְטוֹסֵי VTOL שֶׁיְּכוֹלִים לְהַמְרִיא  וְלִנְחוֹת כְּמוֹ הֵלִיקוֹפְּטֵר.

אֶת כָּל הַכֵּלִים נִיתָּן יִהְיֶה לְהַלְבִּישׁ בְּצִבְעִי  הַסְווֹאָה.
יוֹתֵר לֹא תִּרְאוּ טַנְק בְּצֶבַע חוֹם מְכוֹעָר בּוֹלֵט בְּאֶמְצַע נוֹף מוּשְׁלָג כְּמוֹ מִתְנַחֵל שֶׁנִּקְלַע  לַהַפְגָּנָה שֶׁל מֶרֶ"צ.
בְּנוֹסָף לִבְחִירַת הַצִּיּוּד, יוּכַל הַשַּׂחְקָן לְהַלְבִּישׁ אֶת הַדְּמוּת שֶׁלּוֹ כִּרְצוֹנוֹ.
 מִבְּחִירַת הַבְּגָדִים וּמִיקּוּם סֵמֶל הַקְּבוּצָה וְעַד דְּפוּס הַהַסְווֹאָה, הַכֹּל בִּידֵי הַשַּׂחְקָן.
"חָשַׁבְנוּ שֶׁלְּכוּלָּם  צְרִיכָה לִהְיוֹת הָאֶפְשָׁרוּת לִבְחוֹר אֵיךְ לְהִתְלַבֵּשׁ, גַּם אִם זֶה בָּאִינְטֶרְנֵט" אוֹמֵר טָאוּט, וּמוֹסִיף  בִּקְרִיצָה: "לְכָל הַשּׁוֹאֲלִים - כֵּן, יִהְיוּ לוֹחֲמוֹת בַּמִּשְׂחָק, וְלֹא, אִי אֶפְשָׁר לְהָסִיר אֶת הַבְּגָדִים  הַתַּחְתּוֹנִים".
כָּכָה, הָא? קְהִילַּת הַמּוֹדִים בָּאִינְטֶרְנֵט - לְטִיפּוּלְכֶם...
הַיּוֹצְרִים עֲדַיִן בְּדִיּוּנִים אִם  לְשַׁלֵּב בַּמִּשְׂחָק הִתְקַדְּמוּת אִישִׁית וְשִׁיפּוּר יְכוֹלוֹת ע"י צְבִירַת נִסָּיוֹן.
כָּרֶגַע הוּחְלַט שֶׁלֹּא לְאַפְשֵׁר אֶת  זֶה, אֲבָל הַמְּפַתְּחִים מַשְׁאִירִים אֶת הָאוֹפְּצִיָּה הַזּוֹ פְּתוּחָה לַמִּקְרֶה שֶׁבַּבֶּטָא יִרְאוּ שֶׁיֵּשׁ דְּרִישָׁה לָעִנְיָין.

לֹא רֵיאָלִיסְטִי אֲבָל בְּהֶחְלֵט יוֹתֵר כֵּיף.

לַמְרוֹת הַשָּׁלָב הַמִּתְקַדֵּם בּוֹ נִמְצָא פִּיתּוּחַ הַמִּשְׂחָק, חֲסֵרִים עֲדַיִין  מִסְפַּר נְתוּנִים חֲשׁוּבִים: אֵיךְ פּוֹעֲלִים בְּצַוְותָּא עִם הַשַּׂחְקָנִים הַמְּמוּחְשָׁבִים? מָה יִהְיֶה מִמְשַׁק הַתִּקְשׁוֹרֶת  בְּמַצָּב מְרוּבֵּה מִשְׁתַּתְּפִים? מָה יִקְרֶה אִם לַמָּטוֹס יִגָּמֵר הַדֶּלֶק בְּאֶמְצַע סִיבִּיר, מָה עוֹשִׂים? הוֹלְכִים  בָּרֶגֶל? הַמְּפַתְּחִים מוּדָעִים לִבְעָיוֹת אֵלּוּ, אַךְ מוֹדִיעִים שֶׁשְּׁאֵלוֹת אֵלּוּ וַאֲחֵרוֹת יְקַבְּלוּ תְּשׁוּבָה רַק  לְאַחַר יְצִיאַת הַבֶּטָא מָתַיְשֶׁהוּ בֶּעָתִיד הַקָּרוֹב.
אִם נַעֲצוֹר לְסִיכּוּם, נִגְלָה שֶׁיֵּשׁ לָנוּ בַּיָּדַיִים מִשְׂחָק  שֶׁיָּכוֹל לְשַׁנּוֹת אֶת עוֹלַם מִשְׂחֲקֵי הָאֶקְשֵׁן הַטַּקְטִי בַּמַּחְשֵׁב הָאִישִׁי, יוֹתֵר מֵאֲשֶׁר עָשָׂה כָּל מִשְׂחָק בַּשָּׁנָה  הַחוֹלֶפֶת.
הָאֶפְשָׁרוּת לְשַׁנּוֹת כִּמְעַט כָּל דָּבָר, מִמַּרְאֵה הַשַּׂחְקָן וְעַד מְאַפְיְינֵי שְׂדֵה הַקְּרָב, וְהַהַבְטָחָה שֶׁל  Wings Simulation לְהוֹסִיף מִדֵּי שָׁבוּעַ נְשָׁקִים וּרְכָבִים חֲדָשִׁים דֶּרֶךְ הָאֲתָר שֶׁלָּהֶם מוֹסִיפִים אֶת  מֵימַד הַגִּיוּוּן הַדָּרוּשׁ לְהַצְלָחָתוֹ לְאוֹרֶךְ זְמַן.
טָאוּט מְיוּדָּעֵנוּ מְסַכֵּם גַּם הוּא: "לְאַחַר שֶׁתַּחֲווּ אֶת  שְׂדוֹת הַקְּרָב הַדִּינָאמִיִּים וְהַנִּיתָּנִים לַהֲרִיסָה שֶׁל Soldner, תְּגַלּוּ מָה זֶה הָיָה שֶׁחָסֵר לָכֶם עַד הַיּוֹם  בְּמִשְׂחֲקֵי הַמּוּלְטִי פְּלֶיֵיר הֲכִי טוֹבִים".
אִם לְהִסְתַּמֵּךְ עַל זֶה שֶׁהַגֶּרְמָנִים יְדוּעִים בְּחוּשׁ הַדִּיּוּק שֶׁלָּהֶם,  נִיתָּן לִקְבּוֹעַ שֶׁתַּאֲרִיךְ הַיְּצִיאָה שֶׁל הַמִּשְׂחָק יִהְיֶה אָכֵן בְּסוֹף הַחֵלֶק הַשֵּׁנִי שֶׁל 2003.
