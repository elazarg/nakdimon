בְּעֶרֶב יוֹם הַכִּפּוּרִים יָשְׁבוּ שְׁנֵי בַחוּרִים בְּבֵית הַכְּנֶסֶת וַיִּלְמְדוּ בְּחֵשֶׁק גָּדוֹל, וַיִּשְׁכְּחוּ אֶת הָעוֹלָם וּמְלֹאוֹ וַיָּשִׂימוּ אֶת כָּל מַעְיָּנָם בַּלִּמּוּדִים, וְגַם אֹכֶל לֹא בָּא אֶל פִּיהֶם כָּל הַיּוֹם. וַיִּמְצְאוּ בְּאַחַד מִסִּפְרֵי הַתַּלְמוּד מָקוֹם קָשֶׁה לְהָבִין אֲשֶׁר נִשְׂגָּב הָיָה מִבִּינָתָם, וַיְחַפְּשׂוּ בִסְפָרִים פֵּרוּשִׁים לְבָאֵר אֶת הַסָּתוּם. בֵּינְתַּיִם יָרַד הַיּוֹם וְהַקָּהָל הֵחֵל לְהִתְאַסֵּף לִתְפִלַּת “כָּל נִדְרֵי”, וְהַבַּחוּרִים בִּרְאוֹתָם שֶׁעוֹד שָׁעָה תַּעֲבֹר עַד הָחֵל הַקָּהָל לְהִתְפַּלֵּל, וַיְמָאֲנוּ לַחֲדֹל מִלִּמּוּדָם וַיִּקְּחוּ לָהֶם נֵר קָטָן וַיֵּלְכוּ אֶל חֶדֶר קָטָן הַנִּשְׁעָן אֶל בֵּית הַכְּנֶסֶת, וַיֹּאמְרוּ: לְאוֹר הַנֵּר הַקָּטָן הַזֶּה נֵשֵׁב וְנַעֲסוֹק בְּלִמּוּד, וְהָיָה כַּאֲשֶׁר יִתַּם הַנֵּר אַחֲרֵי חֲצִי שָׁעָה אוֹ שָׁעָה וְשַׁבְנוּ אֶל בֵּית הַתְּפִלָּה וְהִתְפַּלַּלְנוּ עִם כָּל הַקָּהָל. וַיָּחֵלוּ לִלְמוֹד וַיִּשְׁתַּדְלוּ לְבָאֵר אֶת הַמָּקוֹם הַסָּתוּם כָּל אֶחָד לְפִי הֲבָנָתוֹ וַיָּבִיאוּ רְאָיוֹת מִסְּפָרִים שׁוֹנִים, כְּדֵי לְהוֹכִיחַ אֶת צִדְקָתָם, וַיִּתְוַכְּחוּ וַיִּשְׁכְּחוּ כִּי עֶרֶב יוֹם הַכִּפּוּרִים הוּא, וְהַנֵּר בָּעַר וְלֹא כָבָה. כִּי חָפֵץ הַשָּׂטָן לְהָבִיא רָעָה עַל שְׁנֵי הַבַּחוּרִים, אֲשֶׁר תַּחַת לְהִתְפַּלֵּל בַּיּוֹם הַקָּדוֹשׁ הַזֶּה וּלְבַקֵּשׁ מֵאֱלֹהֵי הַשָּׁמַיִם סְלִיחָה וּמְחִילָה עַל חַטָּאתָם כָּל הַשָּׁנָה, יְבַלּוּ אֶת זְמַנָם בְּוִכּוּחִים, וְלָכֵן הֵסֵב כִּי לֹא יִכְבֶּה הַנֵּר הַדּוֹלֵק. וַיּוֹסִיפוּ לִלְמוֹד וּלְהִתְוַכֵּחַ בְּלִי הֶרֶף עַד כְּבוֹת הַנֵּר, וַיִּסְגְּרוּ אֶת סִפְרֵיהֶם וַיָּרוּצוּ אֶל בֵּית הַכְּנֶסֶת לִתְפִלַּת “כָּל נִדְרֵי” עִם כָּל הַקָּהָל, וְהִנֵּה לְתִמְהוֹנָם רָאוּ הֲמוֹן אֲנָשִׁים יוֹצְאִים מִפִּתְחֵי בֵית הַכְּנֶסֶת, טַלִּיתוֹתֵיהֶם וְסִדּוּרֵיהֶם תַּחַת זְרוֹעָם וְהֵם מְבָרְכִים אִישׁ אֶת רֵעֵהוּ בְּשָׁנָה טוֹבָה וּמְמַהֲרִים לָלֶכֶת הַבַּיְתָה לֶאֱכֹל אֶת אֲרוּחָתָם, כִּי יוֹם הַכִּפּוּרִים חָלַף. וַיְהִי כַּאֲשֶׁר הִתְבּוֹנְנוּ הַבַּחוּרִים וַיָּבִינוּ כִּי עֶשְׂרִים וְאַרְבַּע שָׁעוֹת יָשְׁבוּ בְחַדְרָם וְלָמְדוּ, וְאֶת הַיּוֹם הַקָּדוֹשׁ שָׁכְחוּ, וַיִּבָּהֲלוּ מְאֹד וַיִּתְעַצְּבוּ אֶל לִבָּם וַיְמַהֲרוּ אֶל מוֹרָם וַיְסַפְּרוּ לוֹ אֵת כָּל הַקּוֹרוֹת אוֹתָם וְעֵינֵיהֶם זוֹלְגוֹת דְּמָעוֹת בְּדַבְּרָם, וַיְבַקְּשׁוּ מִמֶּנּוּ עֵצָה מַה לַּעֲשׂוֹת, כְּדֵי לְהַעֲבִיר אֶת הַחֵטְא הַגָּדוֹל הַזֶּה מֵעֲלֵיהֶם, וַיֹּאמֶר אֲלֵיהֶם מוֹרָם: הֵרָגְעוּ וְאַל תִּפְחֲדוּ, אוּלַי נִמְצָא עוֹד שִׁבְעָה אֲנָשִׁים אֲשֶׁר יֵאוֹתוּ לְהִלָּוֹת אֵלֵינוּ, וְהָיָה כַּאֲשֶׁר יִהְיֶה מִנְיַן אֲנָשִׁים, וְהִתְפַּלַּלְנוּ אֶת כָּל הַתְּפִלּוֹת מֵרֹאשׁ, וְהָיָה לָנוּ יוֹם כִּפּוּרִים שֵׁנִי, וְסָר עֲוֹנְכֶם וְחַטַּאתְכֶם תְּכֻפָּר. וַיֵּלְכוּ הַבַּחוּרִים מֵאִישׁ לְאִישׁ וַיַּפְצִירוּ בָהֶם לְהִשָּׁאֵר אִתָּם וּלְהִתְפַּלֵּל יַחַד עִמָּהֶם, אֲבָל הָאֲנָשִׁים לֹא שָׂמוּ לֵב לְבַקָּשָׁתָם וַיֹּאמְרוּ: יוֹם וָלַיְלָה לֹא בָא אֹכֶל אֶל פִינוּ וַנִּתְפַּלֵּל בְּלִי הֶרֶף וַאֲנַחְנוּ עֲיֵפִים וִיגֵעִים וְאֵיךְ זֶה נָחֵל לְהִתְפַּלֵּל מֵרֹאשׁ? וַיֵּלְכוּ לְדַרְכָּם. בֵּין כַּךְ וְכַךְ עָזְבוּ כָל הָאֲנָשִׁים אֶת בֵּית הַכְּנֶסֶת, וְרַק שְׁנֵי הַבַּחוּרִים וּמוֹרָם נִשְׁאֲרוּ עֲצֵבִים וַעֲגֻמִּים, מִבְּלִי דַעַת מַה לַעֲשׂוֹת. וַיִּשְּׂאוּ אֶת עֵינֵיהֶם וַיִּרְאוּ וְהִנֵּה בְּפִנַּת בֵּית הַכְּנֶסֶת עוֹמֵד נִשְׁעָן אֶל הַקִּיר זָקֵן אֶחָד קְטַן קוֹמָה, רָזֶה וְדַל בָּשָׂר וְעֵינָיו עֲצוּמוֹת לְמֶחֱצָה, וּפָנָיו כִּפְנֵי אִישׁ אֲשֶׁר תַּרְדֵּמָה נָפְלָה עָלָיו, וַיִּגְּשׁוּ אֵלָיו וַיַּגִּידוּ לוֹ אֶת חֶפְצָם, וַיֹּאמֶר לָהֶם הַזָּקֵן:
— כַּתְּרוּ לִי זְעֵיר וְאָנֹכִי אָבִיא לָכֶם עוֹד שִׁשָּׁה אֲנָשִׁים וּבְהִתְקַבֵּץ מִנְיַן אֲנָשִׁים וְהִתְפַּלַּלְנוּ יַחַד.
אַחֲרֵי רְגָעִים אֲחָדִים שָׁב הַזָּקֵן בְּלִוְיַת שֵׁשֶׁת אֲנָשִׁים וְהַתְּפִלָּה הֵחֵלָּה. וַיְזַמְּרוּ וַיָּרֹנּוּ אֶת תְּפִלּוֹת יוֹם הַכִּפּוּרִים וְאַף כִּי עֲיֵפִים וִיגֵעִים הָיוּ כֻלָּם, בְּכָל זֹאת אִמְּצוּ אֶת שְׁאֵרִית כֹּחוֹתֵיהֶם לְמַלֵּא אֶת חוֹבָתָם. וּבְהַגִּיעַ חֲצוֹת הַלַּיְלָה וְהֵם גָּמְרוּ אֶת תְּפִלַּת הָעֶרֶב וְהִנֵּה הִתְרוֹמֵם הַזָּקֵן, אֲשֶׁר לִפְנֵי רֶגַע נִרְאֶה כִּקְטַן הַקּוֹמָה וְצָנוּם מְאֹד, וַיְהִי לְאִישׁ גָּבוֹהַּ, יְפֵה תֹאַר וִיפֵה מַרְאֶה, אֲשֶׁר שַׂעֲרוֹת זְקָנוֹ לְבָנוֹת כַּכֶּסֶף וּבְגָדָיו צְחוֹרִים וְצַחִים מִשֶּׁלֶג. וַיִּפֹּל עַל הַבַּחוּרִים וּמוֹרָם פַּחַד גָּדוֹל וַיֶּחְרְדוּ חֲרָדָה גְדוֹלָה, וַיֹּאמֶר הַזָּקֵן:
— אַל תִּירְאוּ וְאַל תִּפְחְדוּ! אֵלִיָּהוּ הַנָּבִיא אָנֹכִי. בָּאתִי לְנַסּוֹתְכֶם וְלִרְאוֹת אִם בֶּאֱמֶת וּבְתָמִים חֲפֵצִים אַתֶּם לְהִתְפַּלֵּל וְלָצוּם עוֹד יוֹם תָּמִים וּלְבַקֵּשׁ סְלִיחָה מֵאֵת ה' אֱלֹהֵי הַשָּׁמַיִם וְהָאָרֶץ, וָאָבִיא לָכֶם אֶת אֲבוֹתֵיכֶם הַמֵּתִים הֵנָּה לְהַשְׁלִים אֶת הַמִּנְיָן, וְעַתָּה בִרְאוֹתִי כִּי תָמִים לִבְּכֶם עִם ה' הִנְנִי מְבַשֵּׂר לָכֶם כִּי סָלַח ה' לַעֲוֹנְכֶם. לְכוּ לְבָתֵּיכֶם, אִכְלוּ וּשְׁתוּ וְהֵיטִיבוּ אֶת לִבְּכֶם.
