שִׁמְשׁוֹן, אֱלִיל שֶׁמֶשׁ
דְּלִילָה, אֱלִילַת מַיִם
פְּלִשְׁתִּים, עַנְנֵי חֹרֶף
סַרְנֵי עֶקְרוֹן
סַרְנֵי גַת           מַרְאוֹת שְׁקִיעָה בָּעֲנָנִים
סַרְנֵי אַשְׁקְלוֹן
סַרְנֵי עַזָּה
בְּבֵית-דָּגוֹן. עֵירֹם, רַק חֲגוֹרָה צְהֻבָּה לְמָתְנָיו, עוֹמֵד שִׁמְשׁוֹן בֵּין שְׁנֵי עַמּוּדִים: אֵלֶּה עֲשׂוּיִים אֲבָנִים לֹא-מְהֻקְצָעוֹת, מְשׁוּחוֹת בְּצֶבַע אָדֹם, מֻנָּחוֹת זוֹ עַל גַבֵּי זוֹ בִּמְאֻנָּךְ. פְּלִשְׁתִּים מִתְגָּרִים בּוֹ.
פְּלִשְׁתִּים:
אַיֵּה חַרְבְּךָ, הַגִּבּוֹר?
אֵי קַשְׁתְּךָ וְחִצֶּיךָ?
הַבֵּט אֵלֵינוּ בְּזַעְמֶךָ. הָהּ, יָרֵאנוּ מִמְּאוֹר-עֵינֶיךָ! חֲנִיתוֹת וּבְרָקִים תַּחַת עַפְעַפֶּיךָ!
הַצְלֵף עָלֵינוּ בְּשַׂעֲרוֹתֶיךָ! קָשְׁרֵנוּ בַּעֲבוֹתוֹת תַּלְתַּלֶּיךָ!
הֲזֹאת גְּבוּרָתֶךָ? אִשָּׁה עֲכָרָתְךָ, נָזִיר! לֹא בַּקְּרָב, בַּקֶּטֶל, כָּשָׁלְתָּ. אִשָּׁה הִקְרִיחָה שַׂעֲרוֹתֶיךָ, וּמֵרֵעֶיהָ חִלְּלוּ בְּנוֹת-עֵינֶיךָ.
אֵיפֹה דְּלִילָה? לָמָּה יִפָּקֵד מְקוֹמָהּ בֵּין הַמִּתְעַלְּלִים? לָמָּה תֵשְׁבִי לְמֵרָחוֹק כַּעֲצוּבָה, דְּלִילָה?
דְּלִילָה, אִמֵּנוּ, שָׂרָתֵנוּ. בְּחֵיקָהּ שָׁכַב וַיֵּרָדַם, מָתְקוּ לוֹ חֲלוֹמוֹתָיו. וְהִיא לַתַּעַר שִׁלְּחָה יָדָהּ.
שִׁמְשׁוֹן
בָּאֲפֵלָה, בְּרֶחֶם אִמִּי, בְּהֵרָקֵם עַצְמוֹתַי וְגִידַי, רֻקְּמוּ אִתִּי עֲתִידוֹתַי, וּלְפִתְחֵי לֵדָתִי עָמְדוּ הַדְּחִיפָה וְהַפְּקֻדָּה אֲשֶׁר הִצִּיבוּנִי בִּמְסִלָּתִי, וְתַקִּיף לַחֲשָׁן לִי: נְסוֹר! הֵנָּה בֵּרְכוּנִי, וַתְּעַטֵּרְנָה רֹאשִׁי שַׂעֲרוֹת-גְּבוּרָה וְנָתוֹן אוֹתִי גְּבִיר-עָם, לְהַבְרִיחַ גְּדוּדֵי-אֹפֶל, וּלְהַפְרוֹת אֶרֶץ וְהַעֲשֵׁר יוֹשְׁבֶיהָ – וְהֵנָּה לִהֲטוּ תְשׁוּקָתִי לִשְזוֹר בְּמַיִם כּוֹזְבִים זָהֳרֵי-קְוֻצּוֹתַי וְלִשְׁפּוֹךְ אוֹנִי עַל סְלָעִים לֹא-יָצִיצוּ. עַתָּה, בְּשִׁית עָלַי אוֹיְבִים סָבִיב, וְחָשְׁכוּ מְאוֹרֵי-עֵינַי, וְלֹא אַבִּיט הֲדַר-הָרִים, וְיֶרֶק-מוֹרָד, וְחוֹלוֹת-יָם, וְלִבְנַת-קֶצֶף זוֹחֶלֶת עַל מִשְׁבָּרִים יְרֻקִּים, – מָה אֶפְשַׁע בְּגוֹרָלִי וְאֶתְאוֹנָן? אַבְלִיג עַל מְרִירוּתִי וְאֶדּוֹם, כִּי קָרוֹב מוֹעֵד שְׁקִיעָתִי, וּמַעֲגָּלִי יִתַּם בִּמְעוֹנוֹת-שַׁלְוָה.
מִשְׁלְחוֹת-סְרָנִים נוֹשְׂאוֹת-מִנְחָה צוֹעֲדוֹת אֶל דְּלִילָה בִּמְחוֹל-טֶקֶס. הֵן מִתְיַצְּבוֹת לְפָנֶיהָ.
סַרְנֵי עֶקְרוֹן
עֶקְרוֹן שְׁלָחַתְנוּ. מֵרוּחוֹת-הָרִים הִדְבִּיק אָבִיב אֶת-חֹרֶף. לְבָנָה קִרְיָתֵנוּ בַּנִּצָּה, וְרֻדָּה בַּשּׁוֹשַׁנִּים. נֵפֶץ נִשְׁמַע בַּדֻּבְדְּבָן, לֹא עָמְדוּ כַּפּוֹתָיו מִלַּחַץ נִצָּנִים כִּי כָּבֵדוּ. הוֹלֵךְ רֵיחַ הַחֲבַצָּלוֹת לְמֵרָחוֹק, מַרְחִיב נְחִירֵי הַגְּמַלִּים. גֶּשֶׁם נְבָטִים צְהַבְהַבִּים נִתָּךְ-נוֹטֵף מִן הַשְּׁזִיפִים, מְרַחֵף בָּרוּחַ, נִצְבָּר שִׁכְבוֹת-זָהָב עַל יַד הַגְּדֵרוֹת. עֲנַף-תַּפּוּחַ קָטַפְנוּ, טוֹבֵל בְּלַבְנוּנִית-פְּרִיחָה, אַגָּס בְּלִבְלוּבוֹ שָׁדַדְנוּ, בְּזֵרֵי-לוֹט קְשַׁרְנוּם וּבְמִבְחַר כּוֹכָבִיּוֹת כְּחַלְחַלּוֹת: מַתְּנַת עֶקְרוֹן הִיא בִּידֵי שְׁלוּחֶיהָ אֵלַיִךְ, דְּלִילָה, חֶבֶל אָסַר שִׁמְשׁוֹן וּמְשָׁכָהוּ בָּטֵל לְחֶרְפָּתוֹ.
סַרְנֵי גַת:
נַחְנוּ מִגַּת. עִנְּבֵי-כְּרָמֵינוּ בְּשֵׁלוֹת וּמִתְבַּקְּעוֹת מֵעָסִיס. תְּמָרֵינוּ כְּבֵדִים, הוֹלְכִים דְּבָשׁ. דְּשֵׁנוֹת תְּאֵנֵינוּ עַל זַלְזַלֵּיהֶן. עוֹבְרוֹת מִיץ וָמֶגֶד. מֵאֵלֶּה, דְּלִילָה, בְּסַלֵּי-עֲרָבִים, מְרֻפָּדִים אֵשֶׁל וְדַפְנָה, הֵבֵאנוּ מַתַּת-מִזְעָר לָךְ, רֶשֶׁת לָכְדָה שִׁמְשׁוֹן וְהֶעֱלַתְהוּ מְפַרְפֵּר לַמְעַנִּים.
סַרְנֵי אַשְׁקְלוֹן
מֵאַשְׁקְלוֹן אֲנַחְנוּ. אַדְמָתֵנוּ אוֹצֶרֶת זָהָב מַבְרִיק כְּלַפִּידִים. נְחֻשְׁתֵּנוּ קָשָׁה, לוֹהֲטָה וּמְצַלְצֶלֶת. כַּסְפֵּנוּ עָמוּם, מְנַצְנֵץ בְּהִשָּׁבְרוֹ. מַטְבְּעוֹת-זָהָב מִנְחָתֵנוּ לָךְ, לִתְנוּךְ-הָאֹזֶן וְלַגַּרְגֶּרֶת מַטְבְּעוֹת מְנֻקָּבִים, עִם נְטִיפֵי-כֶּסֶף מָאֳרָכִים. אַף שַׁרְשְׁרוֹת פַּעֲמוֹנִים קְטַנִּים, עִנְבְּלֵי-בַּעַץ בִּגְבִיעֵי-נְחשֶׁת, לְאֶזְרוֹעֵךְ בְּעֵת רַקְּדֵךְ. אֵלֶּה לָךְ, דְּלִילָה, חֶרֶב מָחֲצָה שִׁמְשׁוֹן וּבִתְּקָה כַּדּוּרֵי-עֵינָיו.
סַרְנֵי עַזָּה
בְּנֵי עַזָּה אָנוּ. בַּשּׁוּק, בְּטַבּוּר עִירֵנוּ, נֶעֱרָמוֹת עֲרֵמוֹת אֶרֶג וְרִקְמָה: מֶשִׁי מִקִּדְמַת-קֶדֶם עַל גְּמַלִּים יוּבָא, פָּרוֹכוֹת מִפָּרַס, מַעֲשֵׂי אַרְגָּמָן וִירַקְרַק מִכּוּשׁ וּמִשְּׁבָא, הִכְבִּידוּ אֳנִיּוֹת-צוֹר. נִפְרוֹשׂ עַל בִּרְכַּיִךְ רַעֲלַת-רֹךְ, רוֹעֶפֶת פַּעַם כֶּסֶף וּפַעַם תְּכֵלֶת כְּהֵהָפְכָהּ לִפְנֵי אוֹר. זֹה מִיָּדֵינוּ לָךְ, עִם נַעֲלֵי אַלְמֻגִּים פְּצוּעוֹת-תַּרְשִׁישׁ, עִם לְשׁוֹנוֹת אָדְמָה וּשְׁחוֹר, תְּלַחֵכְנָה בְּשָׂרֵךְ כְּלֶהָבוֹת בַּלַּיְלָה. שְׁעִי אֶל מִנְחָתֵנוּ, דְּלִילָה, שֵׁבֶט דִּכֵּא עֹרֶף שִׁמְשׁוֹן וְהוֹלִיכָהוּ עִוֵּר לִפְנֵי נוֹגְשָׂיו.
דְּלִילָה
מַתְּנוֹתֵיכֶם תּוֹעֵבָה לִי, מְרוֹרוֹת בִּכּוּרֵיכֶם, הַפְּרָחִים – בָּאְשָׁה, וְשַׂלְמוֹתֵיכֶם סָס וְקוּרֵי-עַכָּבִישׁ בְּלִי זֶרַח אוֹר פְּנֵי שִׁמְשׁוֹן. – שִׁמְשׁוֹן, זָכַרְתִּי פַּזֶּרְךָ תַּלְתַּלֶּיךָ עַל מִצְחִי, וּפַזֵּז אֶצְבְּעוֹתֶיךָ עַל יְצוּרַי, רַעֲנַנִּים וְרוֹטְטִים לִקְרַאת חֻמָּן. זָכַרְתִּי מַבָּטְךָ, לִטְּפַנִי מִפָּז וּמִמֶּשִׁי רִחֲמָנִי. וְעַתָּה מוּבָס אֶרְאֶךָּ, רֹאשְׁךָ קָרְחָה וְלֹא זֹהַר, עֵינֶיךָ בּוֹרוֹת נְקוּרִים, וְלִבְּךָ מִדְרָס לְלַעַג בְּנֵי-בְּלִי-כֹחַ. מָה הֵעַבְתִּי חַיַּי, שִׁמְשׁוֹן, בְּמָכְרִי הוֹדְךָ לְמַחֲשַׁכֵּי צִבְאוֹת פְּלָשֶׁת. הֵן תָּקוּם בְּעֻזְּךָ וּתְפַזְּרֵם! לְמַעַן צַעֲרִי כִּי עָז וַחֲרָטָתִי כִּי נֶאֱמָנָה, עוּרָה, שִׁמְשׁוֹן, עוֹרֵר גְּבוּרָתְךָ וַהֲפִיצֵם!
פְּלִשְׁתִּים:
לֹא יְנַגַּח הַשּׁוֹר, קָצוּץ קַרְנָיו.
לֹא יִדְרוֹס הָאֲרִי, נָתַן לְגוֹזְזִים רַעְמָתוֹ.
הַנָּחָשׁ הַמָּחוּץ לֹא יִנְשׁוֹךְ.
דְּלִילָה
לֹא כְּבֶן-תְּמוּתָה שִׁמְשׁוֹן וְיִיעָף. הֵן גְּבוּרָתְךָ, שִׁמְשׁוֹן, מִתְּהוֹם, מִפְּנִים, מֵאוֹצָר לֹא יִגָּרַע. וּמַה-כִּי קֻצְּרוּ שַׂעֲרוֹתֶיךָ-זִיו וְסֻמְּאוּ עֵינֶיךָ-נֹגַהּ? עוֹד נַפְשְׁךָ מָעוֹן לְכֹחַ עַתִּיק, גִּלְגֵּל רַעֲמוֹ מִסִּינַי, לָשׁ מֵרֹאשׁ-יָמִים קִמְטֵי-הֶהָרִים, וְרָבַץ בֵּין עַרְפִלִּים בְּטֶרֶם כּוֹכָבִים קֹרָצוּ, וְלִבְּךָ עוֹד יְרֵשָׁה מִגֶּזַע, שָׁחַק בִּצְעָדָיו צְחִיחֵי מִדְבָּר לְמַחֲנוֹת עַנְנֵי-אָבָק, דּוֹר מִפַּחַד מִלְּפָנָיו רָתַח יָם וְעָף כֹּבֶד-קִיטוֹר. שָׂא זְרוֹעֲךָ, שִׁמְשׁוֹן, אַחַת פְּרוֹץ בִּקְהַל אוֹיְבֶיךָ, וְיֵרָאֶה לַכֹּל כִּי כֹּחֲךָ נְצָחִים וְלֹא-אַכְזָב!
שִׁמְשׁוֹן
(לוֹפֵת הָעַמּוּדִים) אֵל-אֱלֹהִים הַבּוֹעֵר בְּדָמִי, מִקְוֵה-לַהַב הַגּוֹעֵשׁ בְּלִבִּי וְשׁוֹלֵחַ זְרָמָיו-אֵשׁ בְּמוֹ עוֹרְקַי, רֹגֶז-בְּשָׂרִי, אֹנֶס אֲשֶׁר לְקָחַנִי מֵאֹהֶל-אִמִּי וּדְפָקַנִי לָצֵאת נוֹצֵץ בִּמְסִלָּתִי, לִפְסוֹעַ עַל קָדְקֹד זֵדִים, גְּדָרוּנִי, לְשַׁלֵּחַ צָרֶבֶת בְּקָמַת אוֹיְבַי, וּלְמַגֵּר עַזִּים, פָּעֲרוּ פֶּה לְבַלְעֵנִי, וְהָפֹךְ אוֹתָם לִמְקוֹרוֹת מֹתֶק וּמִחְיָה? גּוֹאֲלִי, אֲשֶׁר בְּכָל הֱיוֹתִי שְׂרוּט פְּצָעִים, נְגוּעַ יִסּוּרִים וְיֵאוּשׁ, קְדוּר סְעָרוֹת, שְׁסוּעַ נִגּוּדִים, סְפֵקוֹת וְלִקּוּיִים, תָּמִיד רִפֵּאתַנִי וַתּוֹצִיאֵנִי לְדַרְכִּי שָׁלֵם, עָגֹל, גִּבּוֹר וְקוֹרֵן; אַתָּה, עֵדוּת אֲשֶׁר בִּקְרָבַי, וְשָׁלוֹם רָחוּשׁ לִי מִכּוֹכָבִים מֵרָחוֹק, עָזְרֵנִי בְּעֵת שָׁקְעִי, עֲנֵנִי אַךְ הַפָּעַם, וִיעֹרַב דָּמִי עִם דַּם פְּלִשְׁתִּים, בְּטֶרֶם אֶגְוַע וְאֵינֶנִּי.
נוֹטֶה בְּכֹחַ. הָעַמּוּדִים מִתְפַּלְּצִים. הַבַּיִת מָט לִנְפּוֹל. הַפְּלִשְׁתִּים —תַּדְהֵמָה וּבֶהָלָה. מַפֹּלֶת וְאָבָק. שִׁמְשׁוֹן בַּנּוֹפְלִים.
דְּלִילָה
בְּזֹאת אֵדַע כִּי קוֹם תָּקוּם וְתָשׁוּב-תִּחְיֶה, שִׁמְשׁוֹן! פְּלִיאָה יִפְעָתְךָ, חוֹתַם-עוֹלָמִים עָלֶיהָ. גַּם כִּי תוּעַב, לֹא תַחֲלִיד וְלֹא תִכְלֶה. יְנַבֵּא לִי מוֹרָאֲךָ הָאַחֲרוֹן כִּי אַךְ לְמוֹעֵד לָבַשְׁתָּ אֹפֶל וְהָכְרַעְתָּ חָלָל. וּבְבוֹא תוֹר צְפִירָתְךָ, שֵׁנִית תּוֹפִיעַ, נַעַר וְעַלִּיז לַקְּרָב. כְּקֶדֶם תַּשֵּׁב מַעַרְכוֹת קָמֶיךָ; תְּנַגְּחֵם בְּקַרְנֵי-אוֹן וְתַצְמִיתֵם. סַלְעֵי-מִדְבָּר יֵדְעוּ קַדַּחַת-צַעֲדֶךָ. אָז תִּפְקְדֵנִי גַּם אָנִי, שִׁמְשׁוֹן. שׁוּב תַּעַקְדֵנִי בְּנַהֲרַת מַחְלְפוֹתֶיךָ, וְתַרְקִיד מַבָּטֶיךָ זַהֲרוּרִים-זַהֲרוּרִים עַל גִּבְעוֹת-גוּפִי. תְּחַמֵּם נוֹזְלֵי-לִבִּי מִכִּמּוּרֵי רַחֲמֶיךָ עָלָי… בֵּינְתַיִם אֵרֵד אֶל גְּבוּל עַמְּךָ, אֲלַוֶּךָ אֶל קֶבֶר מָנוֹחַ אָבִיךָ, וְאֶאֱבַל יַחַד עִם אַחֶיךָ. יַשְׁחִירוּ וְיִתְאַבְּלוּ גַּם שָׁמַיִם עִמָּנוּ… וַאֲנִי, כְּעוּרָה וְרוֹגֶשֶׁת, אַפְקִיר צַלְעוֹתַי לַסּוּפָה וּלְחָיַּי לְנִטְפֵי-גֶשֶׁם, יִמָּהֲלוּ בְּדִמְעוֹתַי. כִּי הַרְבֵּה נָהַמְתִּי אֵלֶיךָ, שִׁמְשׁוֹן, וְנִגְנַז מִמֶּנִּי מַרְפֵּא-אֶצְבְּעוֹתֶיךָ.
