עַל-פִּי וִילְיַם בַּטְלֶר יֵיטְס

מַעֲשֶׂה וְעוֹבְדֵי-הָאֲדָמָה שֶׁל כְּפָר אֶחָד בְּאֶרֶץ-אַיְרֶה בְּהוּלִים הָיוּ לִגְמוֹר אֶת מְלֶאכֶת-הַדַּיִשׁ בַּגֹּרֶן הַכְּלָלִית אֲשֶׁר תַּחַת כִּפַּת-הַשָּׁמָיִם. לְאוֹר-הַלְּבָנָה עָבְדוּ, כִּי הַסְּתָו הִקְדִּים לָבוֹא, וְעִם שְׁקִיעָה נִרְאוּ בָּאֹפֶק הַצְּפוֹנִי רָאשֵׁי-עֲנָנִים שְׁחוֹרִים וּמְאַיְּמִים, וּבְרָקִים בּוֹדְדִים מְקַפְּצִים עַל קְצוֹתָם; וְגֶשֶׁם כִּי יַךְ בַּתְּבוּאָה, יַשְׁחִיתֶנָּה.

הֵמָּה עוֹבְדִים בְּקַדַּחְתָּנוּת, וַאֲפֵלָה כִּסְתָה אֶת כָּל הַשָׁמַיִם – בָּלְעָה אֶת הַלְּבָנָה, וּמֵרָחוֹק נִשְׁמַע רִטּוּן-רְעָמִים עָמוּם. וַתְּהִי עֲלָטָה, עַד לֹא רָאָה אִישׁ אֶת יַד עַצְמוֹ. אֵיכָה יַעֲבֹדוּ?

וּבַיָּמִים הַתְּמִימִים הָהֵם לֹא חָשְׁבוּ מְלָכִים לִפְחִיתוּת-כָּבוֹד לָהֶם לַחֲרוֹשׁ וְלִזְרוֹעַ וְלָדוּשׁ עִם כָּל אַנְשֵׁי-הַשָּׂדֶה – כְּמוֹ שָׁאוּל אֲשֶׁר, אַחֲרֵי הֱיוֹתוֹ מָשׁוּחַ מֶלֶךְ, הָלַךְ אַחַר הַבָּקָר וְהַמַּחֲרֵשָׁה; וּכְמוֹ אוֹדִיסֶס, אֲשֶׁר שִׁנָּה טַעְמוֹ, וְחָרַשׁ בַּחֲמוֹר וָשׁוֹר יַחַד, וּבַמַּעֲנִית הַחֲרוּשָׁה זָרַע מֶלֶח תַּחַת גַּרְעִינִים, לְעֵינֵי שְׁלִיחֵי אֲגֵמֶמְנוֹן, כְּדֵי לְהִשְׁתַּמֵּט מִנְּסוֹעַ לַמִּלְחָמָה עַל אִילְיוֹן, הִיא טְרוֹיָה, בִּגְלַל אֵשֶׁת-אִישׁ סוֹרֶרֶת.

וְכֵן עָבַד שָׁם הַמֶּלֶךְ הָאַיְרִי עִם הַכַּפְרִיִּים הָהֵם בְּדַיִשׁ בְּעֵת-לָיְלָה. וְכַאֲשֶׁר נִמְחֲקָה הַלְּבָנָה, וְהֶחְשִׁיךְ פִּתְאוֹם עַד אֵין עֲבוֹד, שָׁלַח בְּיַד נַעַר אֶל הַמַּלְכָּה אֲשֶׁר בַּבָּיִת: “הָבִי לָנוּ תַּלְתַּל מִתַּלְתַּלָּיִךְ”. גָּזְזָה הַמַּלְכָּה תַּלְתַּל אֶחָד מִשְּׂעַר-רֹאשָׁהּ וֶהֱרִיצַתְהוּ אֶל בַּעֲלָהּ וְהָאֲנָשִׁים אֲשֶׁר עִמוֹ בְּגֹרֶן-הַשָּׂדֶה. תָּלוּ הָאֲנָשִׁים אֶת הַתַּלְתַּל עַל עֵץ אֶחָד, וּלְזִיו חֶסֶד-זְהָבוֹ כִּלּוּ אֶת הַדַּיִשׁ בְּטֶרֶם יַדְבִּיקֵם הַגֶּשֶׁם.
